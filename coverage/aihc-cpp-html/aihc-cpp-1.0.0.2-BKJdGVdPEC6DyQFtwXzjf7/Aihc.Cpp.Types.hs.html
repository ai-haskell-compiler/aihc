   40 -- | Configuration for the C preprocessor.
   41 data Config = Config
   42   { -- | The name of the input file, used in @#line@ directives and
   43     -- @__FILE__@ expansion.
   44     configInputFile :: FilePath,
   45     -- | User-defined macros. These are expanded as object-like macros.
   46     -- Note that the values should include any necessary quoting. For
   47     -- example, to define a string macro, use @"\"value\""@.
   48     configMacros :: !(Map Text Text)
   49   }
   50 
   51 data MacroDef
   52   = ObjectMacro !Text
   53   | FunctionMacro ![Text] !Text
   54   deriving (Eq, Show)
   55 
   56 -- | Default configuration with sensible defaults.
   57 --
   58 -- * 'configInputFile' is set to @\"\<input\>\"@
   59 -- * 'configMacros' includes @__DATE__@ and @__TIME__@ set to the Unix epoch
   60 --
   61 -- To customize the date and time macros:
   62 --
   63 -- >>> import qualified Data.Map.Strict as M
   64 -- >>> let cfg = defaultConfig { configMacros = M.fromList [("__DATE__", "\"Mar 15 2026\""), ("__TIME__", "\"14:30:00\"")] }
   65 -- >>> configMacros cfg
   66 -- fromList [("__DATE__","\"Mar 15 2026\""),("__TIME__","\"14:30:00\"")]
   67 --
   68 -- To add additional macros while keeping the defaults:
   69 --
   70 -- >>> import qualified Data.Map.Strict as M
   71 -- >>> let cfg = defaultConfig { configMacros = M.insert "VERSION" "42" (configMacros defaultConfig) }
   72 -- >>> M.lookup "VERSION" (configMacros cfg)
   73 -- Just "42"
   74 defaultConfig :: Config
   75 defaultConfig =
   76   Config
   77     { configInputFile = "<input>",
   78       configMacros =
   79         M.fromList
   80           [ ("__DATE__", "\"Jan  1 1970\""),
   81             ("__TIME__", "\"00:00:00\"")
   82           ]
   83     }

   85 -- | The kind of @#include@ directive.
   86 data IncludeKind = IncludeLocal | IncludeSystem deriving (Eq, Show, Generic, NFData)

   88 -- | Information about a pending @#include@ that needs to be resolved.
   89 data IncludeRequest = IncludeRequest
   90   { -- | The path specified in the include directive.
   91     includePath :: !FilePath,
   92     -- | Whether this is a local (@\"...\"@) or system (@\<...\>@) include.
   93     includeKind :: !IncludeKind,
   94     -- | The file that contains the @#include@ directive.
   95     includeFrom :: !FilePath,
   96     -- | The line number of the @#include@ directive.
   97     includeLine :: !Int
   98   }
   99   deriving (Eq, Show, Generic, NFData)

  101 -- | Severity level for diagnostics.
  102 data Severity = Warning | Error deriving (Eq, Show, Generic, NFData)
  103 
  104 -- | A diagnostic message emitted during preprocessing.
  105 data Diagnostic = Diagnostic
  106   { -- | The severity of the diagnostic.
  107     diagSeverity :: !Severity,
  108     -- | The diagnostic message text.
  109     diagMessage :: !Text,
  110     -- | The file where the diagnostic occurred.
  111     diagFile :: !FilePath,
  112     -- | The line number where the diagnostic occurred.
  113     diagLine :: !Int
  114   }
  115   deriving (Eq, Show, Generic, NFData)

  117 -- | The result of preprocessing.
  118 data Result = Result
  119   { -- | The preprocessed output text.
  120     resultOutput :: !Text,
  121     -- | Any diagnostics (warnings or errors) emitted during preprocessing.
  122     resultDiagnostics :: ![Diagnostic]
  123   }
  124   deriving (Eq, Show, Generic, NFData)
  125 
  126 -- | A step in the preprocessing process. Either preprocessing is complete
  127 -- ('Done') or an @#include@ directive needs to be resolved ('NeedInclude').
  128 data Step
  129   = -- | Preprocessing is complete.
  130     Done !Result
  131   | -- | An @#include@ directive was encountered. The caller must provide
  132     -- the contents of the included file (or 'Nothing' if not found),
  133     -- and preprocessing will continue.
  134     NeedInclude !IncludeRequest !(Maybe ByteString -> Step)
  135 
  136 data EngineState = EngineState
  137   { stMacros :: !(Map Text MacroDef),
  138     stOutput :: !TB.Builder,
  139     stOutputLineCount :: {-# UNPACK #-} !Int,
  140     stDiagnosticsRev :: ![Diagnostic],
  141     stPragmaOnceFiles :: !(Set FilePath),
  142     stSkippingDanglingElse :: !Bool,
  143     stHsBlockCommentDepth :: !Int,
  144     stCBlockCommentDepth :: !Int,
  145     stCurrentFile :: !FilePath,
  146     stCurrentLine :: !Int
  147   }
  148 
  149 emptyState :: FilePath -> EngineState
  150 emptyState filePath =
  151   EngineState
  152     { stMacros = M.empty,
  153       stOutput = mempty,
  154       stOutputLineCount = 0,
  155       stDiagnosticsRev = [],
  156       stPragmaOnceFiles = S.empty,
  157       stSkippingDanglingElse = False,
  158       stHsBlockCommentDepth = 0,
  159       stCBlockCommentDepth = 0,
  160       stCurrentFile = filePath,
  161       stCurrentLine = 1
  162     }

  164 data CondFrame = CondFrame
  165   { frameOuterActive :: !Bool,
  166     frameConditionTrue :: !Bool,
  167     frameInElse :: !Bool,
  168     frameCurrentActive :: !Bool
  169   }
  170 
  171 currentActive :: [CondFrame] -> Bool
  172 currentActive [] = True
  173 currentActive (f : _) = frameCurrentActive f
  174 
  175 mkFrame :: Bool -> Bool -> CondFrame
  176 mkFrame outer cond =
  177   CondFrame
  178     { frameOuterActive = outer,
  179       frameConditionTrue = cond,
  180       frameInElse = False,
  181       frameCurrentActive = outer && cond
  182     }
  183 
  184 type Continuation = EngineState -> Step

  186 data LineContext = LineContext
  187   { lcFilePath :: !FilePath,

  189     lcLineSpan :: !Int,
  190     lcNextLineNo :: !Int,
  191     -- | Cursor positioned after the current line (past its newline).
  192     lcRestCursor :: !Cursor,
  193     lcStack :: ![CondFrame],
  194     lcContinue :: EngineState -> Step,
  195     lcContinueWith :: [CondFrame] -> EngineState -> Step,
  196     lcDone :: Continuation
  197   }
