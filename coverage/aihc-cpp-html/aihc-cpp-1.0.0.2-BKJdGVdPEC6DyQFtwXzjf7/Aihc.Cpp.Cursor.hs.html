    4 -- Module      : Aihc.Cpp.Cursor
    5 -- Description : Zero-copy cursor over ByteString for efficient scanning
    6 -- License     : Unlicense

    8 -- A lightweight cursor abstraction over a strict 'ByteString'. The cursor
    9 -- tracks a position into a shared buffer, enabling O(1) peeking and
   10 -- zero-copy slicing. All CPP-significant bytes are ASCII (0x00-0x7F),
   11 -- so byte-level operations are safe; non-ASCII bytes (>= 0x80) can be
   12 -- bulk-copied without decoding.
   13 module Aihc.Cpp.Cursor
   14   ( Cursor (..),
   15     fromByteString,
   16     fromText,
   17     toText,
   18     null,
   19     peekByte,

   21     advance,
   22     advance2,
   23     sliceText,
   24     sliceSince,
   25     skipWhile,
   26     skipToInteresting,
   27     bufLength,

   39 import Data.ByteString (ByteString)
   40 import qualified Data.ByteString as BS
   41 import Data.Text (Text)
   42 import qualified Data.Text.Encoding as TE
   43 import Data.Word (Word8)

   46 -- | A position-based cursor into a 'ByteString' buffer.
   47 -- The buffer is shared across all cursors derived from the same input,
   48 -- enabling zero-copy slicing.
   49 data Cursor = Cursor
   50   { curBuf :: {-# UNPACK #-} !ByteString,
   51     curPos :: {-# UNPACK #-} !Int
   52   }
   53   deriving (Show)
   54 
   55 -- | Create a cursor at the start of a 'ByteString'.
   56 fromByteString :: ByteString -> Cursor
   57 fromByteString bs = Cursor bs 0
   58 
   59 -- | Create a cursor from 'Text' by encoding to UTF-8.
   60 fromText :: Text -> Cursor
   61 fromText = fromByteString . TE.encodeUtf8
   62 
   63 -- | Decode the remaining bytes from the cursor position as UTF-8 'Text'.
   64 toText :: Cursor -> Text
   65 toText (Cursor buf pos) = TE.decodeUtf8 (BS.drop pos buf)
   66 
   67 -- | Is the cursor at the end of input?
   68 null :: Cursor -> Bool
   69 null (Cursor buf pos) = pos >= BS.length buf
   70 {-# INLINE null #-}
   71 
   72 -- | Check whether the cursor is at the end of input (synonym for 'null').
   73 atEnd :: Cursor -> Bool
   74 atEnd = null
   75 {-# INLINE atEnd #-}
   76 
   77 -- | Peek at the current byte without advancing. Returns 'Nothing' at end
   78 -- of input.
   79 peekByte :: Cursor -> Maybe Word8
   80 peekByte (Cursor buf pos)
   81   | pos >= BS.length buf = Nothing
   82   | otherwise = Just (BS.index buf pos)
   83 {-# INLINE peekByte #-}
   84 
   85 -- | Peek at the current and next byte without advancing. Returns 'Nothing'
   86 -- if fewer than 2 bytes remain.
   87 peekByte2 :: Cursor -> Maybe (Word8, Word8)
   88 peekByte2 (Cursor buf pos)
   89   | pos + 1 >= BS.length buf = Nothing
   90   | otherwise = Just (BS.index buf pos, BS.index buf (pos + 1))
   91 {-# INLINE peekByte2 #-}
   92 
   93 -- | Peek at a byte at an absolute position in the buffer.
   94 peekByteAt :: Int -> Cursor -> Maybe Word8
   95 peekByteAt absPos (Cursor buf _)
   96   | absPos < 0 || absPos >= BS.length buf = Nothing
   97   | otherwise = Just (BS.index buf absPos)
   98 {-# INLINE peekByteAt #-}
   99 
  100 -- | Advance the cursor by one byte.
  101 advance :: Cursor -> Cursor
  102 advance (Cursor buf pos) = Cursor buf (pos + 1)
  103 {-# INLINE advance #-}
  104 
  105 -- | Advance the cursor by two bytes.
  106 advance2 :: Cursor -> Cursor
  107 advance2 (Cursor buf pos) = Cursor buf (pos + 2)
  108 {-# INLINE advance2 #-}
  109 
  110 -- | Extract a zero-copy 'Text' slice from position @start@ to position
  111 -- @end@ (exclusive) in the cursor's buffer.
  112 sliceText :: Int -> Int -> Cursor -> Text
  113 sliceText start end (Cursor buf _) =
  114   TE.decodeUtf8 (BS.take (end - start) (BS.drop start buf))
  115 {-# INLINE sliceText #-}
  116 
  117 -- | Extract a zero-copy 'Text' slice from the given start position to
  118 -- the cursor's current position.
  119 sliceSince :: Int -> Cursor -> Text
  120 sliceSince start cur = sliceText start (curPos cur) cur
  121 {-# INLINE sliceSince #-}
  122 
  123 -- | Advance the cursor while the predicate holds for the current byte.
  124 skipWhile :: (Word8 -> Bool) -> Cursor -> Cursor
  125 skipWhile p = go
  126   where
  127     go !cur = case peekByte cur of
  128       Just b | p b -> go (advance cur)
  129       _ -> cur
  130 {-# INLINE skipWhile #-}
  131 
  132 -- | Total length of the underlying buffer.
  133 bufLength :: Cursor -> Int
  134 bufLength (Cursor buf _) = BS.length buf
  135 {-# INLINE bufLength #-}
  136 
  137 -- | Advance the cursor past bytes that cannot start any CPP-significant
  138 -- two-character sequence. Stops at: @\"@ (0x22), @\'@ (0x27), @*@ (0x2A),
  139 -- @-@ (0x2D), @/@ (0x2F), @\\@ (0x5C), @{@ (0x7B), @}@ (0x7D),
  140 -- or end of input. This allows bulk-copying runs of plain text
  141 -- (identifiers, whitespace, operators, non-ASCII UTF-8) without
  142 -- per-byte dispatch.
  143 skipToInteresting :: Cursor -> Cursor
  144 skipToInteresting = go
  145   where
  146     go !cur = case peekByte cur of
  147       Nothing -> cur
  148       Just b
  149         | isInteresting b -> cur
  150         | otherwise -> go (advance cur)
  151 
  152     isInteresting :: Word8 -> Bool
  153     isInteresting b =
  154       b == 0x22 -- '"'
  155         || b == 0x27 -- '\''
  156         || b == 0x2A -- '*'
  157         || b == 0x2D -- '-'
  158         || b == 0x2F -- '/'
  159         || b == 0x5C -- '\\'
  160         || b == 0x7B -- '{'
  161         || b == 0x7D -- '}'
  162     {-# INLINE isInteresting #-}
  163 {-# INLINE skipToInteresting #-}
  164 
  165 -- | Find the next newline byte (0x0A) or EOF. Returns a cursor
  166 -- positioned at the newline (or at EOF). The bytes from the
  167 -- original position to the returned position form the line content
  168 -- (without the newline).
  169 findNewline :: Cursor -> Cursor
  170 findNewline = go
  171   where
  172     go !cur = case peekByte cur of
  173       Nothing -> cur
  174       Just 0x0A -> cur -- '\n'
  175       Just _ -> go (advance cur)
  176 {-# INLINE findNewline #-}
  177 
  178 -- | Advance past a newline byte if the cursor is currently on one.
  179 -- Returns 'Nothing' at EOF, 'Just cursor' after the newline otherwise.
  180 skipNewline :: Cursor -> Maybe Cursor
  181 skipNewline cur = case peekByte cur of
  182   Just 0x0A -> Just (advance cur) -- '\n'
  183   _ -> Nothing
  184 {-# INLINE skipNewline #-}
  185 
  186 -- | Create a cursor that views only the bytes from @curPos cur@ to @end@
  187 -- (exclusive) in the same buffer. This is a logical "line slice" — a
  188 -- sub-cursor bounded at @end@.
  189 --
  190 -- Implementation: creates a new cursor over a sub-ByteString. The
  191 -- sub-ByteString shares the underlying memory (zero-copy via 'BS.take'
  192 -- and 'BS.drop').
  193 lineSlice :: Int -> Cursor -> Cursor
  194 lineSlice end (Cursor buf start) =
  195   Cursor (BS.take (end - start) (BS.drop start buf)) 0
  196 {-# INLINE lineSlice #-}
  197 
  198 -- | Check if the byte at the cursor's current position matches the
  199 -- given byte.
  200 startsWithByte :: Word8 -> Cursor -> Bool
  201 startsWithByte b cur = peekByte cur == Just b
  202 {-# INLINE startsWithByte #-}
