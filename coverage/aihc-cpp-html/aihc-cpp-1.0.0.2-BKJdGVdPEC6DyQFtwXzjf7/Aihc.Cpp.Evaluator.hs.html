   38 -- | Expand macros in a single piece of text using the blue-paint algorithm.
   39 -- A single pass with a suppression set replaces the previous iterate-up-to-32
   40 -- fixpoint approach.
   41 expandMacros :: EngineState -> Text -> Text
   42 expandMacros st txt =
   43   builderToText (expandBlue st S.empty False False False (TB.fromText txt))

   45 -- | Expand macros with multi-line support. When a function-like macro call
   46 -- spans multiple lines, continuation lines are consumed from @moreLines@.
   47 -- Returns the expanded text and the number of extra lines consumed.
   48 expandMacrosMultiline :: EngineState -> Text -> [Text] -> (Text, Int)
   49 expandMacrosMultiline st txt moreLines =
   50   let extraNeeded = countExtraLinesConsumed st txt moreLines
   51    in if extraNeeded == 0
   52         then (expandMacros st txt, 0)
   53         else
   54           let combinedLines = txt : take extraNeeded moreLines
   55               combined = T.intercalate "\n" combinedLines
   56               expanded = expandMacros st combined
   57            in (expanded, extraNeeded)

   59 -- | Count how many extra lines a function macro call consumes.
   60 -- Scans the first line for an identifier that matches a function macro,
   61 -- then checks if parseCallArgs needs to span into continuation lines.
   62 countExtraLinesConsumed :: EngineState -> Text -> [Text] -> Int
   63 countExtraLinesConsumed st txt moreLines = scanForFunctionMacro False False False txt
   64   where
   65     macros = stMacros st
   66 
   67     scanForFunctionMacro :: Bool -> Bool -> Bool -> Text -> Int
   68     scanForFunctionMacro _ _ _ t | T.null t = 0
   69     scanForFunctionMacro inString inChar escaped t =
   70       case T.uncons t of
   71         Nothing -> 0
   72         Just (c, rest)
   73           | inString ->
   74               let escaped' = c == '\\' && not escaped
   75                   inString' = not (c == '"' && not escaped)
   76                in scanForFunctionMacro inString' False escaped' rest
   77           | inChar ->
   78               let escaped' = c == '\\' && not escaped
   79                   inChar' = not (c == '\'' && not escaped)
   80                in scanForFunctionMacro False inChar' escaped' rest
   81           | startsHsBlockComment t ->
   82               let (_, remaining) = consumeHsBlockComment t
   83                in scanForFunctionMacro False False False remaining
   84           | c == '"' -> scanForFunctionMacro True False False rest
   85           | c == '\'' -> scanForFunctionMacro False True False rest
   86           | isIdentStart c ->
   87               let (ident, rest') = T.span isIdentChar t
   88                in case M.lookup ident macros of
   89                     Just (FunctionMacro _ _) ->
   90                       case tryMultilineCallArgs rest' of
   91                         Just n -> n
   92                         Nothing -> scanForFunctionMacro False False False rest'
   93                     _ -> scanForFunctionMacro False False False rest'
   94           | otherwise -> scanForFunctionMacro False False False rest
   95 
   96     -- Try to parse function call args, potentially spanning multiple lines.
   97     -- Returns Just n if the call spans n extra lines, Nothing if no call.
   98     tryMultilineCallArgs :: Text -> Maybe Int
   99     tryMultilineCallArgs rest = seekOpenParen (T.dropWhile isSpace rest) 0
  100 
  101     seekOpenParen :: Text -> Int -> Maybe Int
  102     seekOpenParen remaining extraLines =
  103       case T.uncons remaining of
  104         Just ('(', afterOpen) ->
  105           findClosingParen 0 afterOpen extraLines
  106         Just _ ->
  107           Nothing
  108         Nothing ->
  109           case drop extraLines moreLines of
  110             [] -> Nothing
  111             (nextLine : _) ->
  112               seekOpenParen (T.dropWhile isSpace nextLine) (extraLines + 1)
  113 
  114     findClosingParen :: Int -> Text -> Int -> Maybe Int
  115     findClosingParen = goClosing False False False
  116       where
  117         goClosing :: Bool -> Bool -> Bool -> Int -> Text -> Int -> Maybe Int
  118         goClosing inString inChar escaped depth remaining extraLines =
  119           case T.uncons remaining of
  120             Nothing ->
  121               -- Need more lines
  122               case drop extraLines moreLines of
  123                 [] -> Nothing -- No more lines, unclosed call
  124                 (nextLine : _) ->
  125                   goClosing inString inChar escaped depth (T.cons '\n' nextLine) (extraLines + 1)
  126             Just (ch, rest)
  127               | inString ->
  128                   let escaped' = ch == '\\' && not escaped
  129                       inString' = not (ch == '"' && not escaped)
  130                    in goClosing inString' False escaped' depth rest extraLines
  131               | inChar ->
  132                   let escaped' = ch == '\\' && not escaped
  133                       inChar' = not (ch == '\'' && not escaped)
  134                    in goClosing False inChar' escaped' depth rest extraLines
  135               | startsHsBlockComment remaining ->
  136                   let (_, afterComment) = consumeHsBlockComment remaining
  137                    in goClosing False False False depth afterComment extraLines
  138               | ch == '"' -> goClosing True False False depth rest extraLines
  139               | ch == '\'' -> goClosing False True False depth rest extraLines
  140               | ch == '(' -> goClosing False False False (depth + 1) rest extraLines
  141               | ch == ')' && depth > 0 -> goClosing False False False (depth - 1) rest extraLines
  142               | ch == ')' -> Just extraLines
  143               | otherwise -> goClosing False False False depth rest extraLines

  145 -- | Blue-paint macro expansion engine. Uses a suppression set (@painted@)
  146 -- to prevent infinite recursion instead of iterating to a fixpoint.
  147 -- Output is accumulated via a lazy 'TB.Builder' for amortized O(n).
  148 expandBlue :: EngineState -> Set Text -> Bool -> Bool -> Bool -> TB.Builder -> TB.Builder
  149 expandBlue st painted inString inChar escaped input =
  150   let txt = builderToText input
  151    in goText st painted inString inChar escaped txt mempty
  152 
  153 -- | Walk the input text, expanding macros with blue-paint suppression.
  154 goText :: EngineState -> Set Text -> Bool -> Bool -> Bool -> Text -> TB.Builder -> TB.Builder
  155 goText _ _ _ _ _ txt acc | T.null txt = acc
  156 goText st painted inString inChar escaped txt acc =
  157   case T.uncons txt of
  158     Nothing -> acc
  159     Just (c, rest)
  160       | inString ->
  161           let escaped' = c == '\\' && not escaped
  162               inString' = not (c == '"' && not escaped)
  163            in goText st painted inString' False escaped' rest (acc <> TB.singleton c)
  164       | inChar ->
  165           let escaped' = c == '\\' && not escaped
  166               inChar' = not (c == '\'' && not escaped)
  167            in goText st painted False inChar' escaped' rest (acc <> TB.singleton c)
  168       | startsHsBlockComment txt ->
  169           let (commentText, remaining) = consumeHsBlockComment txt
  170            in goText st painted False False False remaining (acc <> TB.fromText commentText)
  171       | c == '"' ->
  172           goText st painted True False False rest (acc <> TB.singleton c)
  173       | c == '\'' ->
  174           goText st painted False True False rest (acc <> TB.singleton c)
  175       | isIdentStart c ->
  176           expandIdentBlue st painted txt acc
  177       | c == '-',
  178         Just ('-', _) <- T.uncons rest ->
  179           -- Haskell line comment: copy remainder verbatim without macro expansion
  180           acc <> TB.fromText txt
  181       | otherwise ->
  182           goText st painted False False False rest (acc <> TB.singleton c)

  184 -- | Handle an identifier during blue-paint expansion.
  185 expandIdentBlue :: EngineState -> Set Text -> Text -> TB.Builder -> TB.Builder
  186 expandIdentBlue st painted txt acc =
  187   let (ident, rest) = T.span isIdentChar txt
  188    in if S.member ident painted
  189         then -- Blue-painted: copy verbatim, don't expand
  190           goText st painted False False False rest (acc <> TB.fromText ident)
  191         else case ident of
  192           "__LINE__" ->
  193             goText st painted False False False rest (acc <> TB.fromString (show (stCurrentLine st)))
  194           "__FILE__" ->
  195             goText st painted False False False rest (acc <> TB.fromString (show (stCurrentFile st)))
  196           _ ->
  197             case M.lookup ident (stMacros st) of
  198               Just (ObjectMacro replacement) ->
  199                 let painted' = S.insert ident painted
  200                     replacement' = normalizeObjectReplacement replacement
  201                     expanded = builderToText (goText st painted' False False False replacement' mempty)
  202                  in goText st painted False False False rest (acc <> TB.fromText expanded)
  203               Just (FunctionMacro params body) ->
  204                 case parseCallArgs rest of
  205                   Nothing ->
  206                     goText st painted False False False rest (acc <> TB.fromText ident)
  207                   Just (args, restAfter)
  208                     | length args == length params ->
  209                         let body' = substituteParamsBuilder (M.fromList (zip params args)) body
  210                             painted' = S.insert ident painted
  211                             expanded = builderToText (goText st painted' False False False body' mempty)
  212                          in goText st painted False False False restAfter (acc <> TB.fromText expanded)
  213                     | otherwise ->
  214                         goText st painted False False False rest (acc <> TB.fromText ident)
  215               Nothing ->
  216                 goText st painted False False False rest (acc <> TB.fromText ident)

  218 -- | Normalize comments inside object-like macro replacement text while
  219 -- preserving string and char literals. cpphs replaces @/* ... */@ with spaces
  220 -- matching the width of the comment body, but treats empty @/**/@ as a token
  221 -- pasting hack with zero width.
  222 normalizeObjectReplacement :: Text -> Text
  223 normalizeObjectReplacement = T.stripEnd . go False False False mempty
  224   where
  225     go :: Bool -> Bool -> Bool -> TB.Builder -> Text -> Text
  226     go _ _ _ acc txt | T.null txt = builderToText acc
  227     go inString inChar escaped acc txt =
  228       case T.uncons txt of
  229         Nothing -> builderToText acc
  230         Just (c, rest)
  231           | inString ->
  232               let escaped' = c == '\\' && not escaped
  233                   inString' = not (c == '"' && not escaped)
  234                in go inString' False escaped' (acc <> TB.singleton c) rest
  235           | inChar ->
  236               let escaped' = c == '\\' && not escaped
  237                   inChar' = not (c == '\'' && not escaped)
  238                in go False inChar' escaped' (acc <> TB.singleton c) rest
  239           | c == '"' -> go True False False (acc <> TB.singleton c) rest
  240           | c == '\'' -> go False True False (acc <> TB.singleton c) rest
  241           | "/*" `T.isPrefixOf` txt ->
  242               let (commentText, remaining) = consumeCBlockComment txt
  243                   replacement = commentReplacement commentText
  244                in go False False False (acc <> TB.fromText replacement) remaining
  245           | otherwise ->
  246               go False False False (acc <> TB.singleton c) rest
  247 
  248 consumeCBlockComment :: Text -> (Text, Text)
  249 consumeCBlockComment txt =
  250   let afterOpen = T.drop 2 txt
  251       (inside, suffix) = T.breakOn "*/" afterOpen
  252    in if T.null suffix
  253         then (txt, "")
  254         else ("/*" <> inside <> "*/", T.drop 2 suffix)
  255 
  256 commentReplacement :: Text -> Text
  257 commentReplacement commentText
  258   | commentText == "/**/" = ""
  259   | otherwise = T.replicate (T.length (commentBody commentText)) " "
  260 
  261 commentBody :: Text -> Text
  262 commentBody commentText =
  263   if "/*" `T.isPrefixOf` commentText && "*/" `T.isSuffixOf` commentText
  264     then T.dropEnd 2 (T.drop 2 commentText)
  265     else T.drop 2 commentText

  267 -- | Parse function-like macro call arguments.
  268 parseCallArgs :: Text -> Maybe ([Text], Text)
  269 parseCallArgs input = do
  270   ('(', rest) <- T.uncons (T.dropWhile isSpace input)
  271   parseArgs False False False 0 [] mempty rest
  272 
  273 parseArgs :: Bool -> Bool -> Bool -> Int -> [Text] -> TB.Builder -> Text -> Maybe ([Text], Text)
  274 parseArgs inString inChar escaped depth argsRev current remaining =
  275   case T.uncons remaining of
  276     Nothing -> Nothing
  277     Just (ch, rest)
  278       | inString ->
  279           let escaped' = ch == '\\' && not escaped
  280               inString' = not (ch == '"' && not escaped)
  281            in parseArgs inString' False escaped' depth argsRev (current <> TB.singleton ch) rest
  282       | inChar ->
  283           let escaped' = ch == '\\' && not escaped
  284               inChar' = not (ch == '\'' && not escaped)
  285            in parseArgs False inChar' escaped' depth argsRev (current <> TB.singleton ch) rest
  286       | startsHsBlockComment remaining ->
  287           let (commentText, afterComment) = consumeHsBlockComment remaining
  288            in parseArgs False False False depth argsRev (current <> TB.fromText commentText) afterComment
  289       | ch == '"' ->
  290           parseArgs True False False depth argsRev (current <> TB.singleton ch) rest
  291       | ch == '\'' ->
  292           parseArgs False True False depth argsRev (current <> TB.singleton ch) rest
  293       | ch == '(' ->
  294           parseArgs False False False (depth + 1) argsRev (current <> TB.singleton ch) rest
  295       | ch == ')' && depth > 0 ->
  296           parseArgs False False False (depth - 1) argsRev (current <> TB.singleton ch) rest
  297       | ch == ')' && depth == 0 ->
  298           let arg = trimSpacesText (builderToText current)
  299               argsRev' =
  300                 if T.null arg && null argsRev
  301                   then [""]
  302                   else arg : argsRev
  303            in Just (reverse argsRev', rest)
  304       | ch == ',' && depth == 0 ->
  305           let arg = trimSpacesText (builderToText current)
  306            in parseArgs False False False depth (arg : argsRev) mempty rest
  307       | ch == '-' && depth == 0,
  308         Just ('-', afterDash) <- T.uncons rest ->
  309           -- Haskell line comment inside arg list: close the arg, find ')' in comment
  310           let commentText = "--" <> afterDash
  311            in case findLastCloseParen commentText of
  312                 Nothing -> Nothing
  313                 Just (commentPrefix, afterClose) ->
  314                   let currentText = builderToText current
  315                       arg = trimSpacesText currentText
  316                       trailingWS = T.takeWhileEnd isSpace currentText
  317                       argsRev' = if T.null arg && null argsRev then [""] else arg : argsRev
  318                    in Just (reverse argsRev', trailingWS <> commentPrefix <> afterClose)
  319       | otherwise ->
  320           parseArgs False False False depth argsRev (current <> TB.singleton ch) rest
  321 
  322 -- | Find the last ')' in text and split before it.
  323 findLastCloseParen :: Text -> Maybe (Text, Text)
  324 findLastCloseParen txt =
  325   case T.findIndex (== ')') (T.reverse txt) of
  326     Nothing -> Nothing
  327     Just revIdx ->
  328       let idx = T.length txt - revIdx - 1
  329        in Just (T.take idx txt, T.drop (idx + 1) txt)
  330 
  331 startsHsBlockComment :: Text -> Bool
  332 startsHsBlockComment txt =
  333   case T.uncons txt of
  334     Just ('{', rest) ->
  335       case T.uncons rest of
  336         Just ('-', rest') ->
  337           case T.uncons rest' of
  338             Just ('#', _) -> False
  339             _ -> True
  340         _ -> False
  341     _ -> False
  342 
  343 consumeHsBlockComment :: Text -> (Text, Text)
  344 consumeHsBlockComment = go 0 mempty
  345   where
  346     go :: Int -> TB.Builder -> Text -> (Text, Text)
  347     go depth acc txt =
  348       case T.uncons txt of
  349         Nothing -> (builderToText acc, "")
  350         Just (c, rest) ->
  351           case T.uncons rest of
  352             Just ('-', rest')
  353               | c == '{' ->
  354                   go (depth + 1) (acc <> TB.fromText "{-") rest'
  355             Just ('}', rest')
  356               | c == '-' && depth <= 1 ->
  357                   (builderToText (acc <> TB.fromText "-}"), rest')
  358             Just ('}', rest')
  359               | c == '-' ->
  360                   go (depth - 1) (acc <> TB.fromText "-}") rest'
  361             _ ->
  362               go depth (acc <> TB.singleton c) rest

  364 data Piece
  365   = PieceWhitespace !Text
  366   | PiecePaste
  367   | PieceRaw !Text
  368   | PieceParam !Text
  369 
  370 substituteParams :: Map Text Text -> Text -> Text
  371 substituteParams = substituteParamsBuilder

  373 -- | Builder-based parameter substitution. Replaces identifiers found
  374 -- in the substitution map, respecting string and char literals.
  375 substituteParamsBuilder :: Map Text Text -> Text -> Text
  376 substituteParamsBuilder subs = renderPieces . collapseTokenPastes . collapseStringizing . tokenizeReplacementList
  377   where
  378     tokenizeReplacementList :: Text -> [Piece]
  379     tokenizeReplacementList txt =
  380       case T.uncons txt of
  381         Nothing -> []
  382         Just (c, rest)
  383           | isSpace c ->
  384               let (spaces, remaining) = T.span isSpace txt
  385                in PieceWhitespace spaces : tokenizeReplacementList remaining
  386           | c == '"' ->
  387               let (literal, remaining) = scanQuoted '"' txt
  388                in PieceRaw literal : tokenizeReplacementList remaining
  389           | c == '\'' ->
  390               let (literal, remaining) = scanQuoted '\'' txt
  391                in PieceRaw literal : tokenizeReplacementList remaining
  392           | "/*" `T.isPrefixOf` txt ->
  393               let (commentText, remaining) = consumeCBlockComment txt
  394                   piece = if commentText == "/**/" then PiecePaste else PieceWhitespace (commentReplacement commentText)
  395                in piece : tokenizeReplacementList remaining
  396           | "##" `T.isPrefixOf` txt ->
  397               PiecePaste : tokenizeReplacementList (T.drop 2 txt)
  398           | isIdentStart c ->
  399               let (ident, remaining) = T.span isIdentChar txt
  400                   piece = if M.member ident subs then PieceParam ident else PieceRaw ident
  401                in piece : tokenizeReplacementList remaining
  402           | otherwise ->
  403               PieceRaw (T.singleton c) : tokenizeReplacementList rest
  404 
  405     scanQuoted :: Char -> Text -> (Text, Text)
  406     scanQuoted quote = go False mempty
  407       where
  408         go escaped acc remaining =
  409           case T.uncons remaining of
  410             Nothing -> (builderToText acc, "")
  411             Just (c, rest)
  412               | c == quote && not escaped ->
  413                   (builderToText (acc <> TB.singleton c), rest)
  414               | c == '\\' ->
  415                   go (not escaped) (acc <> TB.singleton c) rest
  416               | otherwise ->
  417                   go False (acc <> TB.singleton c) rest
  418 
  419     collapseStringizing :: [Piece] -> [Piece]
  420     collapseStringizing [] = []
  421     collapseStringizing (PieceRaw "#" : PieceParam name : rest) =
  422       PieceRaw (stringizeArgument (lookupParam name)) : collapseStringizing rest
  423     collapseStringizing (PieceRaw "#" : rest) =
  424       PieceRaw "#" : collapseStringizing rest
  425     collapseStringizing (piece : rest) = piece : collapseStringizing rest
  426 
  427     collapseTokenPastes :: [Piece] -> [Piece]
  428     collapseTokenPastes = go []
  429       where
  430         go acc [] = acc
  431         go acc (piece : rest) =
  432           case piece of
  433             PiecePaste ->
  434               let (accNoSpace, _) = trimTrailingWhitespace acc
  435                   (leadingSpace, restAfterSpace) = span isWhitespacePiece rest
  436                in case (unsnoc accNoSpace, restAfterSpace) of
  437                     (Just (accInit, leftPiece), rightPiece : remaining) ->
  438                       go (accInit <> [PieceRaw (renderPiece leftPiece <> renderPiece rightPiece)]) remaining
  439                     _ -> go (acc <> [PieceRaw "##"] <> leadingSpace) restAfterSpace
  440             _ -> go (acc <> [piece]) rest
  441 
  442     trimTrailingWhitespace :: [Piece] -> ([Piece], [Piece])
  443     trimTrailingWhitespace pieces =
  444       let (trailingRev, restRev) = span isWhitespacePiece (reverse pieces)
  445        in (reverse restRev, reverse trailingRev)
  446 
  447     unsnoc :: [a] -> Maybe ([a], a)
  448     unsnoc [] = Nothing
  449     unsnoc [x] = Just ([], x)
  450     unsnoc (x : xs) = do
  451       (init', last') <- unsnoc xs
  452       pure (x : init', last')
  453 
  454     isWhitespacePiece :: Piece -> Bool
  455     isWhitespacePiece (PieceWhitespace _) = True
  456     isWhitespacePiece _ = False
  457 
  458     lookupParam :: Text -> Text
  459     lookupParam name = M.findWithDefault name name subs
  460 
  461     renderPieces :: [Piece] -> Text
  462     renderPieces = T.concat . map renderPiece
  463 
  464     renderPiece :: Piece -> Text
  465     renderPiece piece =
  466       case piece of
  467         PieceWhitespace txt -> txt
  468         PiecePaste -> "##"
  469         PieceRaw txt -> txt
  470         PieceParam name -> lookupParam name
  471 
  472     stringizeArgument :: Text -> Text
  473     stringizeArgument arg =
  474       let normalized = normalizeWhitespace arg
  475           escaped = T.concatMap escapeStringChar normalized
  476        in T.cons '"' (T.snoc escaped '"')
  477 
  478     normalizeWhitespace :: Text -> Text
  479     normalizeWhitespace = T.unwords . T.words
  480 
  481     escapeStringChar :: Char -> Text
  482     escapeStringChar '"' = "\\\""
  483     escapeStringChar '\\' = "\\\\"
  484     escapeStringChar c = T.singleton c

  486 evalCondition :: EngineState -> Text -> Bool
  487 evalCondition st expr = eval expr /= 0
  488   where
  489     macros = stMacros st
  490     eval = evalNumeric . replaceRemainingWithZero . expandMacros st . replaceDefined macros

  492 evalNumeric :: Text -> Integer
  493 evalNumeric input =
  494   let tokens = tokenize input
  495    in case parseExpr tokens of
  496         (val, _) -> val
  497 
  498 data Token = TOp Text | TNum Integer | TIdent Text | TOpenParen | TCloseParen deriving (Show)
  499 
  500 tokenize :: Text -> [Token]
  501 tokenize input =
  502   case T.uncons input of
  503     Nothing -> []
  504     Just (c, rest)
  505       | isSpace c ->
  506           tokenize (T.dropWhile isSpace rest)
  507       | isDigit c ->
  508           let (num, remaining) = T.span isDigit input
  509            in case TR.decimal num of
  510                 Right (value, _) -> TNum value : tokenize remaining
  511                 Left _ -> tokenize remaining
  512       | isIdentStart c ->
  513           let (ident, remaining) = T.span isIdentChar input
  514            in TIdent ident : tokenize remaining
  515       | c == '(' ->
  516           TOpenParen : tokenize rest
  517       | c == ')' ->
  518           TCloseParen : tokenize rest
  519       | otherwise ->
  520           let (op, remaining) = T.span isOpChar input
  521            in if T.null op
  522                 then tokenize rest
  523                 else TOp op : tokenize remaining

  525 parseExpr :: [Token] -> (Integer, [Token])
  526 parseExpr = parseOr
  527 
  528 binary :: ([Token] -> (Integer, [Token])) -> [Text] -> [Token] -> (Integer, [Token])
  529 binary next ops ts =
  530   let (v1, ts1) = next ts
  531    in go v1 ts1
  532   where
  533     go v1 (TOp op : ts2)
  534       | op `elem` ops =
  535           let (v2, ts3) = next ts2
  536            in go (apply op v1 v2) ts3
  537     go v1 ts2 = (v1, ts2)
  538 
  539     apply "||" a b = if a /= 0 || b /= 0 then 1 else 0
  540     apply "&&" a b = if a /= 0 && b /= 0 then 1 else 0
  541     apply "==" a b = if a == b then 1 else 0
  542     apply "!=" a b = if a /= b then 1 else 0
  543     apply "<" a b = if a < b then 1 else 0
  544     apply ">" a b = if a > b then 1 else 0
  545     apply "<=" a b = if a <= b then 1 else 0
  546     apply ">=" a b = if a >= b then 1 else 0
  547     apply "+" a b = a + b
  548     apply "-" a b = a - b
  549     apply "*" a b = a * b
  550     apply "/" a b = if b == 0 then 0 else a `div` b
  551     apply "%" a b = if b == 0 then 0 else a `mod` b
  552     apply _ a _ = a
  553 
  554 parseOr, parseAnd, parseEq, parseRel, parseAdd, parseMul :: [Token] -> (Integer, [Token])
  555 parseOr = binary parseAnd ["||"]
  556 parseAnd = binary parseEq ["&&"]
  557 parseEq = binary parseRel ["==", "!="]
  558 parseRel = binary parseAdd ["<", ">", "<=", ">="]
  559 parseAdd = binary parseMul ["+", "-"]
  560 parseMul = binary parseUnary ["*", "/", "%"]

  562 parseUnary :: [Token] -> (Integer, [Token])
  563 parseUnary (TOp "!" : ts) = let (v, ts') = parseUnary ts in (if v == 0 then 1 else 0, ts')
  564 parseUnary (TOp "-" : ts) = let (v, ts') = parseUnary ts in (-v, ts')
  565 parseUnary ts = parseAtom ts

  567 parseAtom :: [Token] -> (Integer, [Token])
  568 parseAtom (TNum n : ts) = (n, ts)
  569 parseAtom (TIdent _ : ts) = (0, ts)
  570 parseAtom (TOpenParen : ts) =
  571   let (v, ts1) = parseExpr ts
  572    in case ts1 of
  573         TCloseParen : ts2 -> (v, ts2)
  574         _ -> (v, ts1)
  575 parseAtom ts = (0, ts)

  577 replaceDefined :: Map Text MacroDef -> Text -> Text
  578 replaceDefined macros = go
  579   where
  580     go txt =
  581       case T.uncons txt of
  582         Nothing -> ""
  583         Just (c, rest)
  584           | "defined" `T.isPrefixOf` txt && not (nextCharIsIdent (T.drop 7 txt)) ->
  585               expandDefined (T.dropWhile isSpace (T.drop 7 txt))
  586           | otherwise ->
  587               T.cons c (go rest)
  588 
  589     expandDefined rest =
  590       case T.uncons rest of
  591         Just ('(', restAfterOpen) ->
  592           let rest' = T.dropWhile isSpace restAfterOpen
  593               (name, restAfterName0) = T.span isIdentChar rest'
  594               restAfterName = T.dropWhile isSpace restAfterName0
  595            in case T.uncons restAfterName of
  596                 Just (')', restAfterClose) ->
  597                   boolLiteral (M.member name macros) <> go restAfterClose
  598                 _ ->
  599                   boolLiteral False <> go restAfterName
  600         _ ->
  601           let (name, restAfterName) = T.span isIdentChar rest
  602            in if T.null name
  603                 then boolLiteral False <> go rest
  604                 else boolLiteral (M.member name macros) <> go restAfterName
  605 
  606     boolLiteral True = " 1 "
  607     boolLiteral False = " 0 "
  608 
  609     nextCharIsIdent remaining =
  610       case T.uncons remaining of
  611         Just (c, _) -> isIdentChar c
  612         Nothing -> False

  614 replaceRemainingWithZero :: Text -> Text
  615 replaceRemainingWithZero = go
  616   where
  617     go txt =
  618       case T.uncons txt of
  619         Nothing -> ""
  620         Just (c, rest)
  621           | isIdentStart c ->
  622               let (_, remaining) = T.span isIdentChar txt
  623                in " 0 " <> go remaining
  624           | otherwise ->
  625               T.cons c (go rest)
