   43 -- | Expand macros in a list of line spans (single-line, no lookahead).
   44 expandLineBySpan :: EngineState -> [LineSpan] -> Text
   45 expandLineBySpan st =
   46   T.concat . map expandSpan
   47   where
   48     expandSpan lineChunk
   49       | lineSpanInBlockComment lineChunk = lineSpanText lineChunk
   50       | otherwise = expandMacros st (lineSpanText lineChunk)

   52 -- | Expand macros in a list of line spans with multi-line lookahead.
   53 -- When a function macro call spans multiple lines, continuation lines
   54 -- are consumed from the @futureCursor@ (positioned after the current line).
   55 -- Returns (expanded text, number of extra lines consumed).
   56 --
   57 -- Multi-line expansion is only attempted for lines that consist entirely
   58 -- of code spans (no inline comments). Mixed code/comment lines use
   59 -- single-line expansion to preserve comment span positions.
   60 expandLineBySpanMultiline :: EngineState -> [LineSpan] -> Cursor -> (Text, Int)
   61 expandLineBySpanMultiline st spans futureCursor =
   62   let commentSpans = filter lineSpanInBlockComment spans
   63       hasLineComment = any (\s -> "--" `T.isPrefixOf` lineSpanText s) commentSpans
   64       hasCBlockComment = any (T.all (== ' ') . lineSpanText) commentSpans
   65       hasHsComment = case commentSpans of
   66         [] -> False
   67         _ -> not hasCBlockComment
   68    in if hasLineComment || hasHsComment
   69         then -- Haskell comments stay in the token stream, so expand the full line.
   70           let fullText = T.concat [lineSpanText s | s <- spans]
   71            in (expandMacros st fullText, 0)
   72         else
   73           if hasCBlockComment
   74             then -- C comments are stripped to spaces, so preserve per-span handling.
   75               (expandLineBySpan st spans, 0)
   76             else -- Pure code line: try multi-line expansion
   77               let codeText = T.concat [lineSpanText s | s <- spans]
   78                   futureCodeLines = cursorToLines futureCursor
   79                in expandMacrosMultiline st codeText futureCodeLines

   81 -- | Extract lines from a cursor as a lazy list of Text values.
   82 -- Each line is the text up to the next newline (or EOF).
   83 cursorToLines :: Cursor -> [Text]
   84 cursorToLines !cur
   85   | null cur = []
   86   | otherwise =
   87       let eol = findNewline cur
   88           lineText = sliceText (curPos cur) (curPos eol) cur
   89        in lineText : maybe [] cursorToLines (skipNewline eol)

   91 -- | Lightweight scan that only tracks block comment depth changes.
   92 -- Does not build 'LineSpan' segments or track string/char literals.
   93 -- Used for inactive conditional branches where only comment depth
   94 -- tracking is needed (no macro expansion or span splitting).
   95 --
   96 -- Accepts a 'Cursor' positioned at the start of the line content.
   97 -- The cursor should be bounded to the line (e.g., via 'lineSlice').
   98 scanLineDepthOnly :: Int -> Int -> Cursor -> (Int, Int)
   99 scanLineDepthOnly = goDepth
  100   where
  101     goDepth :: Int -> Int -> Cursor -> (Int, Int)
  102     goDepth !hsDepth !cDepth !cur
  103       | null cur = (hsDepth, cDepth)
  104       | otherwise =
  105           case peekByte2 cur of
  106             Nothing ->
  107               -- One byte left, no two-char sequence possible
  108               (hsDepth, cDepth)
  109             Just (b1, b2)
  110               | cDepth > 0 ->
  111                   if b1 == 0x2A && b2 == 0x2F -- '*/'
  112                     then goDepth hsDepth 0 (advance2 cur)
  113                     else goDepth hsDepth cDepth (advance cur)
  114               | hsDepth > 0 && b1 == 0x2D && b2 == 0x7D -> -- '-}'
  115                   goDepth (hsDepth - 1) cDepth (advance2 cur)
  116               | b1 == 0x7B && b2 == 0x2D -> -- '{-'
  117                   let cur' = advance2 cur
  118                    in case peekByte cur' of
  119                         Just 0x23 ->
  120                           -- {-# is a pragma, not a comment
  121                           goDepth hsDepth cDepth (advance cur)
  122                         _ ->
  123                           goDepth (hsDepth + 1) cDepth cur'
  124               | hsDepth == 0 && b1 == 0x2F && b2 == 0x2A -> -- '/*'
  125                   goDepth hsDepth 1 (advance2 cur)
  126               | hsDepth == 0 && b1 == 0x2D && b2 == 0x2D -> -- '--' line comment
  127                   (hsDepth, cDepth)
  128               | otherwise ->
  129                   goDepth hsDepth cDepth (advance cur)

  131 -- | Scan a line, tracking comment depths and splitting into spans that are
  132 -- either inside or outside block comments. Uses a byte-level cursor for
  133 -- efficient scanning instead of character-by-character T.uncons/T.cons.
  134 --
  135 -- Accepts a 'Cursor' positioned at the start of the line content.
  136 -- The cursor should be bounded to the line (e.g., via 'lineSlice').
  137 --
  138 -- The scanner splits the line into 'LineSpan' segments. Each segment is
  139 -- tagged with whether it is inside a block comment. Code spans (outside
  140 -- comments) are zero-copy slices of the UTF-8 encoded input. C89 comment
  141 -- content is replaced with spaces to preserve column alignment.
  142 scanLine :: Int -> Int -> Cursor -> LineScan
  143 scanLine hsDepth0 cDepth0 cursor0 =
  144   let (spans, finalHsDepth, finalCDepth) =
  145         go
  146           hsDepth0
  147           cDepth0
  148           False
  149           False
  150           False
  151           []
  152           (curPos cursor0)
  153           (hsDepth0 > 0 || cDepth0 > 0)
  154           cursor0
  155    in LineScan
  156         { lineScanSpans = reverse spans,
  157           lineScanFinalHsDepth = finalHsDepth,
  158           lineScanFinalCDepth = finalCDepth
  159         }
  160   where
  161     -- \| Emit a span from @start@ to @end@ if non-empty, prepending to @acc@.
  162     emit :: [LineSpan] -> Int -> Int -> Cursor -> Bool -> [LineSpan]
  163     emit acc start end cur inComment
  164       | start >= end = acc
  165       | otherwise = LineSpan inComment (sliceText start end cur) : acc
  166     {-# INLINE emit #-}
