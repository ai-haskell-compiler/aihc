   25 data Directive
   26   = DirDefineObject !Text !Text
   27   | DirDefineFunction !Text ![Text] !Text
   28   | DirUndef !Text
   29   | DirInclude !IncludeKind !Text
   30   | DirIf !Text
   31   | DirIfDef !Text
   32   | DirIfNDef !Text
   33   | DirElif !Text
   34   | DirElse
   35   | DirEndIf
   36   | DirLine !Int !(Maybe FilePath)
   37   | DirPragmaOnce
   38   | DirWarning !Text
   39   | DirError !Text
   40   | DirUnsupported !Text
   41 
   42 parseDirective :: Text -> Maybe Directive
   43 parseDirective raw =
   44   let trimmed = T.stripStart raw
   45    in if "#" `T.isPrefixOf` trimmed
   46         then
   47           let body = T.stripStart (T.drop 1 trimmed)
   48            in case T.uncons body of
   49                 Just (c, _) | isLetter c || isDigit c -> parseDirectiveBody body
   50                 _ -> Nothing
   51         else Nothing
   52 
   53 parseDirectiveBody :: Text -> Maybe Directive
   54 parseDirectiveBody body =
   55   let (name, rest0) = T.span isIdentChar body
   56       rest = T.stripStart rest0
   57    in if T.null name
   58         then case T.uncons body of
   59           Just (c, _) | isDigit c -> parseLineDirective body
   60           _ -> Nothing
   61         else case name of
   62           "define" -> parseDefine rest
   63           "undef" -> DirUndef <$> parseIdentifier rest
   64           "include" -> parseInclude rest
   65           "if" -> Just (DirIf rest)
   66           "ifdef" -> DirIfDef <$> parseIdentifier rest
   67           "ifndef" -> DirIfNDef <$> parseIdentifier rest
   68           -- Keep `#isndef` as explicitly unsupported for diagnostics on common typo input.
   69           "isndef" -> Just (DirUnsupported "isndef")
   70           "elif" -> Just (DirElif rest)
   71           "elseif" -> Just (DirElif rest)
   72           "else" -> Just DirElse
   73           "endif" -> Just DirEndIf
   74           "line" -> parseLineDirective rest
   75           "pragma" -> parsePragma rest
   76           "warning" -> Just (DirWarning rest)
   77           "error" -> Just (DirError rest)
   78           _ -> Nothing
   79 
   80 parseLineDirective :: Text -> Maybe Directive
   81 parseLineDirective body =
   82   case TR.decimal body of
   83     Left _ -> Nothing
   84     Right (lineNumber, rest0) ->
   85       let rest = T.stripStart rest0
   86        in case parseQuotedText rest of
   87             Nothing -> Just (DirLine lineNumber Nothing)
   88             Just path -> Just (DirLine lineNumber (Just (T.unpack path)))
   89 
   90 parsePragma :: Text -> Maybe Directive
   91 parsePragma body =
   92   case T.words body of
   93     ["once"] -> Just DirPragmaOnce
   94     _ -> Nothing
   95 
   96 parseDefine :: Text -> Maybe Directive
   97 parseDefine rest = do
   98   let (name, rest0) = T.span isIdentChar rest
   99   if T.null name
  100     then Nothing
  101     else case T.uncons rest0 of
  102       Just ('(', afterOpen) ->
  103         let (params, restAfterParams) = parseDefineParams afterOpen
  104          in case params of
  105               Nothing -> Just (DirUnsupported "define-function-macro")
  106               Just names -> Just (DirDefineFunction name names (T.stripStart restAfterParams))
  107       _ -> Just (DirDefineObject name (T.stripStart rest0))
  108 
  109 parseDefineParams :: Text -> (Maybe [Text], Text)
  110 parseDefineParams input =
  111   let (inside, suffix) = T.breakOn ")" input
  112    in if T.null suffix
  113         then (Nothing, "")
  114         else
  115           let rawParams = T.splitOn "," inside
  116               params = map (T.takeWhile isIdentChar . T.strip) rawParams
  117            in if T.null (T.strip inside)
  118                 then (Just [], T.drop 1 suffix)
  119                 else
  120                   if any T.null params
  121                     then (Nothing, T.drop 1 suffix)
  122                     else (Just params, T.drop 1 suffix)
  123 
  124 parseIdentifier :: Text -> Maybe Text
  125 parseIdentifier txt =
  126   let ident = T.takeWhile isIdentChar (T.stripStart txt)
  127    in if T.null ident then Nothing else Just ident
  128 
  129 parseInclude :: Text -> Maybe Directive
  130 parseInclude txt =
  131   case T.uncons (T.stripStart txt) of
  132     Just ('"', rest) ->
  133       let (path, suffix) = T.breakOn "\"" rest
  134        in if T.null suffix then Nothing else Just (DirInclude IncludeLocal path)
  135     Just ('<', rest) ->
  136       let (path, suffix) = T.breakOn ">" rest
  137        in if T.null suffix then Nothing else Just (DirInclude IncludeSystem path)
  138     _ -> Nothing
  139 
  140 parseQuotedText :: Text -> Maybe Text
  141 parseQuotedText txt = do
  142   ('"', rest) <- T.uncons txt
  143   let (path, suffix) = T.breakOn "\"" rest
  144   if T.null suffix then Nothing else Just path
  145 
  146 isIdentStart :: Char -> Bool
  147 isIdentStart c = c == '_' || isLetter c
  148 
  149 isIdentChar :: Char -> Bool
  150 isIdentChar c = c == '_' || isAlphaNum c
  151 
  152 isOpChar :: Char -> Bool
  153 isOpChar c =
  154   c == '+'
  155     || c == '-'
  156     || c == '*'
  157     || c == '/'
  158     || c == '%'
  159     || c == '&'
  160     || c == '|'
  161     || c == '!'
  162     || c == '='
  163     || c == '<'
  164     || c == '>'
