   92 -- | Preprocess C preprocessor directives in the input.
   93 --
   94 -- This function handles:
   95 --
   96 -- * Macro definitions (@#define@) and expansion
   97 -- * Conditional compilation (@#if@, @#ifdef@, @#ifndef@, @#elif@, @#else@, @#endif@)
   98 -- * File inclusion (@#include@)
   99 -- * Diagnostics (@#warning@, @#error@)
  100 -- * Line control (@#line@)
  101 -- * Predefined macros (@__FILE__@, @__LINE__@, @__DATE__@, @__TIME__@)
  102 --
  103 -- === Macro expansion
  104 --
  105 -- Object-like macros are expanded in the output:
  106 --
  107 -- >>> let Done r = preprocess defaultConfig "#define FOO 42\nThe answer is FOO"
  108 -- >>> T.putStr (resultOutput r)
  109 -- #line 1 "<input>"
  110 -- <BLANKLINE>
  111 -- The answer is 42
  112 --
  113 -- Function-like macros are also supported:
  114 --
  115 -- >>> let Done r = preprocess defaultConfig "#define MAX(a,b) ((a) > (b) ? (a) : (b))\nMAX(3, 5)"
  116 -- >>> T.putStr (resultOutput r)
  117 -- #line 1 "<input>"
  118 -- <BLANKLINE>
  119 -- ((3) > (5) ? (3) : (5))
  120 --
  121 -- === Conditional compilation
  122 --
  123 -- Conditional directives control which sections of code are included:
  124 --
  125 -- >>> :{
  126 -- let Done r = preprocess defaultConfig
  127 --       "#define DEBUG 1\n#if DEBUG\ndebug mode\n#else\nrelease mode\n#endif"
  128 -- in T.putStr (resultOutput r)
  129 -- :}
  130 -- #line 1 "<input>"
  131 -- <BLANKLINE>
  132 -- <BLANKLINE>
  133 -- debug mode
  134 -- <BLANKLINE>
  135 -- <BLANKLINE>
  136 -- <BLANKLINE>
  137 --
  138 -- === Include handling
  139 --
  140 -- When an @#include@ directive is encountered, 'preprocess' returns a
  141 -- 'NeedInclude' step. The caller must provide the contents of the included
  142 -- file as a 'ByteString':
  143 --
  144 -- >>> :{
  145 -- let NeedInclude req k = preprocess defaultConfig "#include \"header.h\"\nmain code"
  146 --     Done r = k (Just "-- header content")
  147 -- in T.putStr (resultOutput r)
  148 -- :}
  149 -- #line 1 "<input>"
  150 -- #line 1 "./header.h"
  151 -- -- header content
  152 -- #line 2 "<input>"
  153 -- main code
  154 --
  155 -- If the include file is not found, pass 'Nothing' to emit an error:
  156 --
  157 -- >>> :{
  158 -- let NeedInclude _ k = preprocess defaultConfig "#include \"missing.h\""
  159 --     Done r = k Nothing
  160 -- in do
  161 --   T.putStr (resultOutput r)
  162 --   mapM_ print (resultDiagnostics r)
  163 -- :}
  164 -- #line 1 "<input>"
  165 -- Diagnostic {diagSeverity = Error, diagMessage = "missing include: missing.h", diagFile = "<input>", diagLine = 1}
  166 --
  167 -- === Diagnostics
  168 --
  169 -- The @#warning@ directive emits a warning:
  170 --
  171 -- >>> :{
  172 -- let Done r = preprocess defaultConfig "#warning This is a warning"
  173 -- in do
  174 --   T.putStr (resultOutput r)
  175 --   mapM_ print (resultDiagnostics r)
  176 -- :}
  177 -- #line 1 "<input>"
  178 -- <BLANKLINE>
  179 -- Diagnostic {diagSeverity = Warning, diagMessage = "This is a warning", diagFile = "<input>", diagLine = 1}
  180 --
  181 -- The @#error@ directive emits an error and stops preprocessing:
  182 --
  183 -- >>> :{
  184 -- let Done r = preprocess defaultConfig "#error Build failed\nthis line is not processed"
  185 -- in do
  186 --   T.putStr (resultOutput r)
  187 --   mapM_ print (resultDiagnostics r)
  188 -- :}
  189 -- #line 1 "<input>"
  190 -- <BLANKLINE>
  191 -- Diagnostic {diagSeverity = Error, diagMessage = "Build failed", diagFile = "<input>", diagLine = 1}
  192 preprocess :: Config -> ByteString -> Step
  193 preprocess cfg input =
  194   let cursor = fromByteString input
  195    in processFile (configInputFile cfg) cursor False [] 1 initialState finish
  196   where
  197     initialState =
  198       let st0 = emitLine (linePragma 1 (configInputFile cfg)) (emptyState (configInputFile cfg))
  199        in st0
  200             { stMacros = M.map ObjectMacro (configMacros cfg)
  201             }
  202 
  203     finish st =
  204       let out = TL.toStrict (TB.toLazyText (stOutput st))
  205           outWithTrailingNewline =
  206             if T.null out
  207               then out
  208               else out <> "\n"
  209        in Done
  210             Result
  211               { resultOutput = outWithTrailingNewline,
  212                 resultDiagnostics = reverse (stDiagnosticsRev st)
  213               }

  215 -- | Find the next line in the cursor, handling backslash-continuation
  216 -- for directive lines. Returns (lineCursor, lineSpan, restCursor) where:
  217 --   * lineCursor is a sub-cursor bounded to the logical line content
  218 --   * lineSpan is the number of physical lines consumed (>= 1)
  219 --   * restCursor is positioned after the line (past the newline)
  220 --
  221 -- Backslash-continuation is only applied when the line starts with '#'
  222 -- (after optional whitespace), matching CPP semantics.
  223 -- For continuation lines, a new ByteString is allocated with the
  224 -- backslash-newline sequences removed.
  225 nextLine :: Cursor -> (Cursor, Int, Cursor)
  226 nextLine cur =
  227   let eol = findNewline cur
  228       lineStart = curPos cur
  229       lineEnd = curPos eol
  230       rest = fromMaybe eol (skipNewline eol)
  231    in if lineEnd > lineStart
  232         && peekByteAt (lineEnd - 1) cur == Just 0x5C -- '\' at end
  233         && isDirectiveLine cur lineEnd
  234         then -- Backslash continuation: join lines, stripping '\' and '\n'
  235           joinContinuationLines cur lineStart lineEnd rest
  236         else
  237           let lineText = sliceText lineStart lineEnd cur
  238            in if hasGccStringContinuation emptyQuoteState lineText
  239                 then joinStringContinuationLines cur lineStart lineEnd rest
  240                 else (lineSlice lineEnd cur, 1, rest)
  241 
  242 -- | Check if the bytes from curPos to lineEnd start with '#' after
  243 -- optional whitespace. This determines whether backslash-continuation
  244 -- applies.
  245 isDirectiveLine :: Cursor -> Int -> Bool
  246 isDirectiveLine cur lineEnd =
  247   let trimmed = skipWhile isSpaceOrTab cur
  248    in curPos trimmed < lineEnd
  249         && peekByte trimmed == Just 0x23 -- '#'
  250   where
  251     isSpaceOrTab b = b == 0x20 || b == 0x09
  252 
  253 -- | Join backslash-continuation lines into a single logical line.
  254 -- Builds a new ByteString with '\<newline>' sequences removed.
  255 -- Returns (joinedCursor, physicalLineCount, restCursor).
  256 joinContinuationLines :: Cursor -> Int -> Int -> Cursor -> (Cursor, Int, Cursor)
  257 joinContinuationLines origCur lineStart firstLineEnd firstRest =
  258   let buf = curBuf origCur
  259       -- First segment: from lineStart to firstLineEnd - 1 (exclude '\')
  260       firstSegment = BSB.byteString (sliceBS lineStart (firstLineEnd - 1) buf)
  261    in go firstSegment 1 firstRest
  262   where
  263     go !acc !spanCount !rest
  264       | atEnd rest =
  265           -- No more input; finalize
  266           let joined = BSL.toStrict (BSB.toLazyByteString acc)
  267            in (fromByteString joined, spanCount, rest)
  268       | otherwise =
  269           let eol = findNewline rest
  270               segStart = curPos rest
  271               segEnd = curPos eol
  272               rest' = fromMaybe eol (skipNewline eol)
  273            in if segEnd > segStart
  274                 && peekByteAt (segEnd - 1) origCur == Just 0x5C -- ends with '\'
  275                 then
  276                   -- Another continuation line: append without the trailing '\'
  277                   let segment = BSB.byteString (sliceBS segStart (segEnd - 1) (curBuf origCur))
  278                    in go (acc <> segment) (spanCount + 1) rest'
  279                 else
  280                   -- Last line of continuation
  281                   let segment = BSB.byteString (sliceBS segStart segEnd (curBuf origCur))
  282                       joined = BSL.toStrict (BSB.toLazyByteString (acc <> segment))
  283                    in (fromByteString joined, spanCount + 1, rest')

  285 -- | Slice a ByteString from position @start@ to @end@ (exclusive).
  286 sliceBS :: Int -> Int -> ByteString -> ByteString
  287 sliceBS start end bs = BS.take (end - start) (BS.drop start bs)

  290 data QuoteState = QuoteState
  291   { qsInString :: !Bool,
  292     qsInChar :: !Bool,
  293     qsEscaped :: !Bool
  294   }
  295 
  296 emptyQuoteState :: QuoteState
  297 emptyQuoteState = QuoteState False False False

  299 -- | GCC removes @\<newline>@ before the Haskell lexer sees the file.  Some
  300 -- Haskell packages rely on that inside string gaps by writing lines that end
  301 -- in @\\@: one backslash remains as the Haskell string-gap opener and the
  302 -- final backslash is the CPP continuation marker.  GHC's default CPP-like
  303 -- handling also accepts the single-backslash spelling, so only the double
  304 -- spelling is spliced here.
  305 hasGccStringContinuation :: QuoteState -> Text -> Bool
  306 hasGccStringContinuation st lineText =
  307   qsInString (scanQuoteState st lineText) && "\\\\" `T.isSuffixOf` lineText

  309 joinStringContinuationLines :: Cursor -> Int -> Int -> Cursor -> (Cursor, Int, Cursor)
  310 joinStringContinuationLines origCur lineStart firstLineEnd firstRest =
  311   let buf = curBuf origCur
  312       firstSegment = sliceText lineStart (firstLineEnd - 1) origCur
  313       firstBytes = BSB.byteString (sliceBS lineStart (firstLineEnd - 1) buf)
  314    in go firstBytes 1 firstRest (scanQuoteState emptyQuoteState firstSegment)
  315   where
  316     go !acc !spanCount !rest !quoteState
  317       | atEnd rest =
  318           let joined = BSL.toStrict (BSB.toLazyByteString acc)
  319            in (fromByteString joined, spanCount, rest)
  320       | otherwise =
  321           let eol = findNewline rest
  322               segStart = curPos rest
  323               segEnd = curPos eol
  324               rest' = fromMaybe eol (skipNewline eol)
  325               segmentText = sliceText segStart segEnd origCur
  326            in if hasGccStringContinuation quoteState segmentText
  327                 then
  328                   let segmentBytes = BSB.byteString (sliceBS segStart (segEnd - 1) (curBuf origCur))
  329                       scannedText = sliceText segStart (segEnd - 1) origCur
  330                    in go (acc <> segmentBytes) (spanCount + 1) rest' (scanQuoteState quoteState scannedText)
  331                 else
  332                   let segmentBytes = BSB.byteString (sliceBS segStart segEnd (curBuf origCur))
  333                       joined = BSL.toStrict (BSB.toLazyByteString (acc <> segmentBytes))
  334                    in (fromByteString joined, spanCount + 1, rest')

  336 scanQuoteState :: QuoteState -> Text -> QuoteState
  337 scanQuoteState = go
  338   where
  339     go st txt =
  340       case T.uncons txt of
  341         Nothing -> st
  342         Just (c, rest)
  343           | qsInString st ->
  344               if qsEscaped st && isSpace c
  345                 then go st {qsEscaped = False} (dropStringGapClose rest)
  346                 else
  347                   go
  348                     st
  349                       { qsInString = qsEscaped st || c /= '"',
  350                         qsEscaped = c == '\\' && not (qsEscaped st)
  351                       }
  352                     rest
  353           | qsInChar st ->
  354               go
  355                 st
  356                   { qsInChar = qsEscaped st || c /= '\'',
  357                     qsEscaped = c == '\\' && not (qsEscaped st)
  358                   }
  359                 rest
  360           | c == '"' -> go (QuoteState True False False) rest
  361           | c == '\'' -> go (QuoteState False True False) rest
  362           | otherwise -> go st {qsEscaped = False} rest
  363 
  364     dropStringGapClose txt =
  365       let afterSpace = T.dropWhile isSpace txt
  366        in case T.uncons afterSpace of
  367             Just ('\\', rest) -> rest
  368             _ -> afterSpace

  370 -- | Process a file from a cursor. The @trailingNewline@ flag controls
  371 -- whether a trailing newline in the input produces an extra empty line
  372 -- (used for include files to match @splitOn \"\\n\"@ semantics).
  373 processFile :: FilePath -> Cursor -> Bool -> [CondFrame] -> Int -> EngineState -> Continuation -> Step
  374 processFile _ cursor trailingNl _ _ st k
  375   | atEnd cursor =
  376       if trailingNl
  377         then -- Include file: trailing newline produces one more empty line
  378           k (emitBlankLines 1 st)
  379         else k st
  380 processFile filePath cursor trailingNl stack !lineNo st k =
  381   let (lineCur, lineSpan, restCursor) = nextLine cursor
  382       -- Detect if this line was followed by a newline (vs EOF).
  383       -- If so and restCursor is at EOF, the file had a trailing newline.
  384       hasTrailingNl = trailingNl && not (atEnd restCursor) || (trailingNl && atEnd restCursor)
  385       -- Actually: trailingNl flag is set at processFile entry for includes.
  386       -- We just propagate it. The check at atEnd above handles the final empty line.
  387       lineText = toText lineCur
  388       startsInBlockComment = stHsBlockCommentDepth st > 0 || stCBlockCommentDepth st > 0
  389       parsedDirective =
  390         if startsInBlockComment
  391           then Nothing
  392           else parseDirective lineText
  393       isActive = currentActive stack
  394       nextLineNo = lineNo + lineSpan
  395    in if not isActive && not (stSkippingDanglingElse st)
  396         then -- === Fast path for inactive branches ===
  397         -- Only track comment depth; skip full span scanning and macro expansion.
  398           let (finalHs, finalC) = scanLineDepthOnly (stHsBlockCommentDepth st) (stCBlockCommentDepth st) lineCur
  399               advanceSt st' =
  400                 st'
  401                   { stCurrentLine = nextLineNo,
  402                     stHsBlockCommentDepth = finalHs,
  403                     stCBlockCommentDepth = finalC
  404                   }
  405               continueInactive st' = processFile filePath restCursor hasTrailingNl stack nextLineNo (advanceSt st') k
  406               continueInactiveWith stack' st' = processFile filePath restCursor hasTrailingNl stack' nextLineNo (advanceSt st') k
  407            in case parsedDirective of
  408                 Nothing ->
  409                   continueInactive (emitBlankLines lineSpan st)
  410                 Just directive ->
  411                   let ctx =
  412                         LineContext
  413                           { lcFilePath = filePath,
  414                             lcLineNo = lineNo,
  415                             lcLineSpan = lineSpan,
  416                             lcNextLineNo = nextLineNo,
  417                             lcRestCursor = restCursor,
  418                             lcStack = stack,
  419                             lcContinue = continueInactive,
  420                             lcContinueWith = continueInactiveWith,
  421                             lcDone = k
  422                           }
  423                    in handleDirective ctx st directive
  424         else -- === Normal path: full scan + expansion ===
  425           let lineScan = scanLine (stHsBlockCommentDepth st) (stCBlockCommentDepth st) lineCur
  426               advanceLineState st' =
  427                 st'
  428                   { stCurrentLine = nextLineNo,
  429                     stHsBlockCommentDepth = lineScanFinalHsDepth lineScan,
  430                     stCBlockCommentDepth = lineScanFinalCDepth lineScan
  431                   }
  432               continue st' = processFile filePath restCursor hasTrailingNl stack nextLineNo (advanceLineState st') k
  433               continueWith stack' st' = processFile filePath restCursor hasTrailingNl stack' nextLineNo (advanceLineState st') k
  434               ctx =
  435                 LineContext
  436                   { lcFilePath = filePath,
  437                     lcLineNo = lineNo,
  438                     lcLineSpan = lineSpan,
  439                     lcNextLineNo = nextLineNo,
  440                     lcRestCursor = restCursor,
  441                     lcStack = stack,
  442                     lcContinue = continue,
  443                     lcContinueWith = continueWith,
  444                     lcDone = k
  445                   }
  446            in if stSkippingDanglingElse st
  447                 then recoverDanglingElse ctx parsedDirective st
  448                 else case parsedDirective of
  449                   Nothing ->
  450                     -- Try multi-line expansion: look ahead at future lines via cursor
  451                     let (expanded, extraConsumed) =
  452                           expandLineBySpanMultiline st (lineScanSpans lineScan) restCursor
  453                      in if extraConsumed > 0
  454                           then
  455                             -- Skip consumed continuation lines by advancing the cursor
  456                             let (remainingCursor, totalExtraSpan) =
  457                                   skipNLines extraConsumed restCursor
  458                                 -- Scan consumed lines to update comment depths
  459                                 (finalHs, finalC) =
  460                                   scanConsumedLines
  461                                     (lineScanFinalHsDepth lineScan)
  462                                     (lineScanFinalCDepth lineScan)
  463                                     restCursor
  464                                     extraConsumed
  465                                 nextLineNo' = nextLineNo + totalExtraSpan
  466                                 st' =
  467                                   (emitLine expanded st)
  468                                     { stCurrentLine = nextLineNo',
  469                                       stHsBlockCommentDepth = finalHs,
  470                                       stCBlockCommentDepth = finalC
  471                                     }
  472                              in processFile filePath remainingCursor hasTrailingNl stack nextLineNo' st' k
  473                           else continue (emitLine expanded st)
  474                   Just directive ->
  475                     handleDirective ctx st directive

  477 -- | Skip N lines from a cursor, returning (rest cursor, total lines skipped).
  478 skipNLines :: Int -> Cursor -> (Cursor, Int)
  479 skipNLines 0 cur = (cur, 0)
  480 skipNLines n cur = go 0 0 cur
  481   where
  482     go !count !consumed !c
  483       | count >= n = (c, consumed)
  484       | atEnd c = (c, consumed)
  485       | otherwise =
  486           let eol = findNewline c
  487               rest = fromMaybe eol (skipNewline eol)
  488            in go (count + 1) (consumed + 1) rest
  489 
  490 -- | Scan consumed lines to update comment depths.
  491 scanConsumedLines :: Int -> Int -> Cursor -> Int -> (Int, Int)
  492 scanConsumedLines !hsDepth !cDepth _cur 0 = (hsDepth, cDepth)
  493 scanConsumedLines !hsDepth !cDepth cur remaining
  494   | atEnd cur = (hsDepth, cDepth)
  495   | otherwise =
  496       let eol = findNewline cur
  497           lineCur = lineSlice (curPos eol) cur
  498           (hsDepth', cDepth') = scanLineDepthOnly hsDepth cDepth lineCur
  499           rest = fromMaybe eol (skipNewline eol)
  500        in scanConsumedLines hsDepth' cDepth' rest (remaining - 1)

  502 recoverDanglingElse :: LineContext -> Maybe Directive -> EngineState -> Step
  503 recoverDanglingElse ctx parsedDirective st =
  504   case parsedDirective of
  505     Just DirEndIf ->
  506       lcContinue
  507         ctx
  508         ( addDiag
  509             Warning
  510             "unmatched #endif"
  511             (lcFilePath ctx)
  512             (lcLineNo ctx)
  513             (st {stSkippingDanglingElse = False})
  514         )
  515     Just _ ->
  516       lcContinue ctx st
  517     Nothing ->
  518       lcContinue ctx (emitDirectiveBlank ctx st)

  520 handleDirective :: LineContext -> EngineState -> Directive -> Step
  521 handleDirective ctx st directive =
  522   case directive of
  523     DirDefineObject name value ->
  524       mutateMacrosWhenActive ctx st (M.insert name (ObjectMacro value))
  525     DirDefineFunction name params body ->
  526       mutateMacrosWhenActive ctx st (M.insert name (FunctionMacro params body))
  527     DirUndef name ->
  528       mutateMacrosWhenActive ctx st (M.delete name)
  529     DirInclude kind includeTarget ->
  530       handleIncludeDirective ctx st kind includeTarget
  531     DirIf expr ->
  532       pushConditionalFrame ctx st (evalCondition st expr)
  533     DirIfDef name ->
  534       pushConditionalFrame ctx st (M.member name (stMacros st))
  535     DirIfNDef name ->
  536       pushConditionalFrame ctx st (not (M.member name (stMacros st)))
  537     DirElif expr ->
  538       handleElifDirective ctx st expr
  539     DirElse ->
  540       handleElseDirective ctx st
  541     DirEndIf ->
  542       case lcStack ctx of
  543         [] ->
  544           lcContinue
  545             ctx
  546             (addDiag Warning "unmatched #endif" (lcFilePath ctx) (lcLineNo ctx) st)
  547         _ : rest ->
  548           continueBlankWithStack ctx rest st
  549     DirWarning msg ->
  550       addDiagnosticWhenActive ctx Warning msg st
  551     DirError msg ->
  552       if currentActive (lcStack ctx)
  553         then
  554           lcDone
  555             ctx
  556             (emitDirectiveBlank ctx (addDiag Error msg (lcFilePath ctx) (lcLineNo ctx) st))
  557         else continueBlank ctx st
  558     DirLine n mPath ->
  559       handleLineDirective ctx st n mPath
  560     DirPragmaOnce ->
  561       handlePragmaOnceDirective ctx st
  562     DirUnsupported name ->
  563       addDiagnosticWhenActive ctx Warning ("unsupported directive: " <> name) st

  565 emitDirectiveBlank :: LineContext -> EngineState -> EngineState
  566 emitDirectiveBlank ctx = emitBlankLines (lcLineSpan ctx)
  567 
  568 continueBlank :: LineContext -> EngineState -> Step
  569 continueBlank ctx st = lcContinue ctx (emitDirectiveBlank ctx st)
  570 
  571 continueBlankWithStack :: LineContext -> [CondFrame] -> EngineState -> Step
  572 continueBlankWithStack ctx stack st = lcContinueWith ctx stack (emitDirectiveBlank ctx st)
  573 
  574 mutateMacrosWhenActive :: LineContext -> EngineState -> (Map Text MacroDef -> Map Text MacroDef) -> Step
  575 mutateMacrosWhenActive ctx st mutate =
  576   if currentActive (lcStack ctx)
  577     then continueBlank ctx (st {stMacros = mutate (stMacros st)})
  578     else continueBlank ctx st

  580 addDiagnosticWhenActive :: LineContext -> Severity -> Text -> EngineState -> Step
  581 addDiagnosticWhenActive ctx severity message st =
  582   if currentActive (lcStack ctx)
  583     then continueBlank ctx (addDiag severity message (lcFilePath ctx) (lcLineNo ctx) st)
  584     else continueBlank ctx st

  586 handlePragmaOnceDirective :: LineContext -> EngineState -> Step
  587 handlePragmaOnceDirective ctx st =
  588   if currentActive (lcStack ctx)
  589     then continueBlank ctx (st {stPragmaOnceFiles = S.insert (lcFilePath ctx) (stPragmaOnceFiles st)})
  590     else continueBlank ctx st

  592 pushConditionalFrame :: LineContext -> EngineState -> Bool -> Step
  593 pushConditionalFrame ctx st cond =
  594   let frame = mkFrame (currentActive (lcStack ctx)) cond
  595    in continueBlankWithStack ctx (frame : lcStack ctx) st

  597 handleElifDirective :: LineContext -> EngineState -> Text -> Step
  598 handleElifDirective ctx st expr =
  599   case lcStack ctx of
  600     [] ->
  601       continueBlank
  602         ctx
  603         (addDiag Error "#elif without matching #if" (lcFilePath ctx) (lcLineNo ctx) st)
  604     f : rest ->
  605       if frameInElse f
  606         then
  607           continueBlank
  608             ctx
  609             (addDiag Error "#elif after #else" (lcFilePath ctx) (lcLineNo ctx) st)
  610         else
  611           let anyTaken = frameConditionTrue f
  612               newCond = not anyTaken && evalCondition st expr
  613               f' =
  614                 f
  615                   { frameConditionTrue = anyTaken || newCond,
  616                     frameCurrentActive = frameOuterActive f && newCond
  617                   }
  618            in continueBlankWithStack ctx (f' : rest) st
  619 
  620 handleElseDirective :: LineContext -> EngineState -> Step
  621 handleElseDirective ctx st =
  622   case lcStack ctx of
  623     [] ->
  624       lcContinue ctx (st {stSkippingDanglingElse = True})
  625     f : rest ->
  626       if frameInElse f
  627         then
  628           continueBlank
  629             ctx
  630             (addDiag Error "duplicate #else in conditional block" (lcFilePath ctx) (lcLineNo ctx) st)
  631         else
  632           let newCurrent = frameOuterActive f && not (frameConditionTrue f)
  633               f' =
  634                 f
  635                   { frameInElse = True,
  636                     frameCurrentActive = newCurrent
  637                   }
  638            in continueBlankWithStack ctx (f' : rest) st

  640 handleIncludeDirective :: LineContext -> EngineState -> IncludeKind -> Text -> Step
  641 handleIncludeDirective ctx st kind includeTarget
  642   | not (currentActive (lcStack ctx)) = continueBlank ctx st
  643   | S.member includeFilePath (stPragmaOnceFiles st) = continueBlank ctx st
  644   | otherwise = NeedInclude includeReq nextStep
  645   where
  646     includePathText = T.unpack includeTarget
  647     includeFilePath =
  648       case kind of
  649         IncludeLocal -> takeDirectory (lcFilePath ctx) </> includePathText
  650         IncludeSystem -> includePathText
  651     includeReq =
  652       IncludeRequest
  653         { includePath = includePathText,
  654           includeKind = kind,
  655           includeFrom = lcFilePath ctx,
  656           includeLine = lcLineNo ctx
  657         }
  658 
  659     nextStep Nothing =
  660       lcContinue
  661         ctx
  662         (addDiag Error ("missing include: " <> includeTarget) (lcFilePath ctx) (lcLineNo ctx) st)
  663     nextStep (Just includeBytes) =
  664       let includeCursor = fromByteString includeBytes
  665           -- Include files treat trailing newlines as producing an extra
  666           -- empty line (matching splitOn "\n" semantics).
  667           includeHasTrailingNl = not (BS.null includeBytes) && BS.last includeBytes == 0x0A
  668           stWithIncludePragma =
  669             emitLine (linePragma 1 includeFilePath) (st {stCurrentFile = includeFilePath, stCurrentLine = 1})
  670           resumeParent stAfterInclude =
  671             processFile
  672               (lcFilePath ctx)
  673               (lcRestCursor ctx)
  674               False
  675               (lcStack ctx)
  676               (lcNextLineNo ctx)
  677               ( emitLine
  678                   (linePragma (lcNextLineNo ctx) (lcFilePath ctx))
  679                   (stAfterInclude {stCurrentFile = lcFilePath ctx, stCurrentLine = lcNextLineNo ctx})
  680               )
  681               (lcDone ctx)
  682        in processFile includeFilePath includeCursor includeHasTrailingNl [] 1 stWithIncludePragma resumeParent

  684 handleLineDirective :: LineContext -> EngineState -> Int -> Maybe FilePath -> Step
  685 handleLineDirective ctx st lineNumber maybePath
  686   | not (currentActive (lcStack ctx)) = continueBlank ctx st
  687   | otherwise =
  688       let stWithFile =
  689             case maybePath of
  690               Just path -> st {stCurrentFile = path}
  691               Nothing -> st
  692           stWithLinePragma =
  693             emitLine
  694               (linePragma lineNumber (stCurrentFile stWithFile))
  695               (stWithFile {stCurrentLine = lineNumber})
  696        in processFile
  697             (lcFilePath ctx)
  698             (lcRestCursor ctx)
  699             False
  700             (lcStack ctx)
  701             lineNumber
  702             (stWithLinePragma {stCurrentLine = lineNumber})
  703             (lcDone ctx)
  704 
  705 emitLine :: Text -> EngineState -> EngineState
  706 emitLine line st =
  707   let sep = if stOutputLineCount st > 0 then TB.singleton '\n' else mempty
  708    in st
  709         { stOutput = stOutput st <> sep <> TB.fromText line,
  710           stOutputLineCount = stOutputLineCount st + 1
  711         }
  712 
  713 emitBlankLines :: Int -> EngineState -> EngineState
  714 emitBlankLines n st
  715   | n <= 0 = st
  716   | otherwise =
  717       let newlines = mconcat (replicate n (TB.singleton '\n'))
  718        in st
  719             { stOutput = stOutput st <> newlines,
  720               stOutputLineCount = stOutputLineCount st + n
  721             }

  723 addDiag :: Severity -> Text -> FilePath -> Int -> EngineState -> EngineState
  724 addDiag sev msg filePath lineNo st =
  725   st
  726     { stDiagnosticsRev =
  727         Diagnostic
  728           { diagSeverity = sev,
  729             diagMessage = msg,
  730             diagFile = filePath,
  731             diagLine = lineNo
  732           }
  733           : stDiagnosticsRev st
  734     }
  735 
  736 linePragma :: Int -> FilePath -> Text
  737 linePragma n path = "#line " <> T.pack (show n) <> " \"" <> T.pack path <> "\""
