    6 -- Description : Reclassify Expr trees as Pattern trees

    8 --
    9 -- Convert an expression AST into a pattern AST. This implements the
   10 -- \"parse as expression, then reclassify\" strategy described in
   11 -- @docs\/unified-expr-pattern-parsing.md@.
   12 --
   13 -- The core function 'checkPattern' performs a structural recursion over
   14 -- 'Expr', converting each expression node into its pattern counterpart.
   15 -- Expression-only constructs (if, case, do, lambda, etc.) are rejected
   16 -- with a descriptive error message.
   17 module Aihc.Parser.Internal.CheckPattern
   18   ( checkPattern,
   19     checkPatterns,
   20   )
   21 where
   22 
   23 import Aihc.Parser.Internal.Common (isConLikeName, nameToUnqualified)
   24 import Aihc.Parser.Syntax
   25 import Data.Maybe (isJust, isNothing)
   26 import Data.Text (Text)

   28 -- | Convert an expression tree into a pattern.
   29 -- Returns @Left@ with a diagnostic message if the expression cannot be
   30 -- interpreted as a valid pattern.
   31 checkPattern :: Expr -> Either Text Pattern
   32 checkPattern expr = case expr of
   33   EAnn ann sub -> fmap (PAnn ann) (checkPattern sub)
   34   EPragma pragma _ -> checkPragmaPattern pragma
   35   -- Variables and constructors
   36   EVar name
   37     | nameText name == "_" -> Right PWildcard
   38     | isConLikeName name -> Right (PCon name [] [])
   39     | isJust (nameQualifier name) -> Left "unexpected qualified name in pattern"
   40     | otherwise -> Right (PVar (nameToUnqualified name))
   41   ETypeSyntax form ty -> Right (PTypeSyntax form ty)
   42   -- Parenthesized expression
   43   -- When the inner expression is a view-pattern arrow (@expr -> expr@),
   44   -- produce @PParen (PView f pat)@ to preserve the explicit parens in
   45   -- the AST, matching the shape produced by the dedicated pattern parser.
   46   EParen inner
   47     | Just vp <- asViewPat inner -> Right (PParen vp)
   48     | otherwise -> PParen <$> checkPattern inner
   49   -- Tuple
   50   ETuple fl elems
   51     | Just tupleCon <- tupleConstructorPattern fl elems -> Right tupleCon
   52     | otherwise -> do
   53         pats <- traverse checkTupleElement elems
   54         Right (PTuple fl pats)
   55   -- List
   56   EList elems -> PList <$> traverse checkPattern elems
   57   -- Unboxed sum
   58   EUnboxedSum i n e -> PUnboxedSum i n <$> checkPattern e
   59   -- Infix: only constructor operators (starting with ':') or the view-pattern
   60   -- arrow @->@ are valid in patterns.
   61   EInfix l op r
   62     | renderName op == "->" -> do
   63         rPat <- checkPattern r
   64         Right (PView l rPat)
   65     | isConLikeOp op -> do
   66         lPat <- checkPattern l
   67         rPat <- checkPattern r
   68         Right (PInfix lPat op rPat)
   69     | otherwise -> Left ("unexpected variable operator '" <> renderName op <> "' in pattern")
   70   -- Type signature
   71   ETypeSig e ty -> do
   72     pat <- checkPattern e
   73     Right (PTypeSig pat ty)
   74   -- Negation (must be a literal)
   75   ENegate inner -> checkNegLitPattern inner
   76   -- Application: accumulate arguments into PCon
   77   EApp f x -> do
   78     fPat <- checkPattern f
   79     xPat <- checkPattern x
   80     case peelPatternAnn fPat of
   81       PCon name typeArgs args -> Right (PCon name typeArgs (args ++ [xPat]))
   82       _ -> Left "invalid pattern: application of non-constructor"
   83   -- Record construction -> record pattern
   84   ERecordCon name fields wc -> do
   85     patFields <-
   86       traverse
   87         ( \field -> do
   88             pat <- checkPattern (recordFieldValue field)
   89             pure field {recordFieldValue = pat}
   90         )
   91         fields
   92     Right (PRecord name patFields wc)
   93   -- Literals
   94   EInt n nt repr -> Right (PLit (LitInt n nt repr))
   95   EFloat x ft repr -> Right (PLit (LitFloat x ft repr))
   96   EChar c repr -> Right (PLit (LitChar c repr))
   97   ECharHash c repr -> Right (PLit (LitCharHash c repr))
   98   EString s repr -> Right (PLit (LitString s repr))
   99   EStringHash s repr -> Right (PLit (LitStringHash s repr))
  100   EOverloadedLabel {} -> Left "unexpected overloaded label in pattern"
  101   -- TH splice
  102   ETHSplice body -> Right (PSplice body)
  103   -- Quasi-quote
  104   EQuasiQuote q b -> Right (PQuasiQuote q b)
  105   -- Expression-only constructs: clear errors
  106   EIf {} -> Left "unexpected if-then-else in pattern"
  107   EMultiWayIf {} -> Left "unexpected multi-way if in pattern"
  108   ECase {} -> Left "unexpected case expression in pattern"
  109   EDo {} -> Left "unexpected do expression in pattern"
  110   ELambdaPats {} -> Left "unexpected lambda in pattern"
  111   ELambdaCase {} -> Left "unexpected lambda-case in pattern"
  112   ELambdaCases {} -> Left "unexpected lambda-cases in pattern"
  113   ELetDecls {} -> Left "unexpected let expression in pattern"
  114   EArithSeq {} -> Left "unexpected arithmetic sequence in pattern"
  115   EListComp {} -> Left "unexpected list comprehension in pattern"
  116   EListCompParallel {} -> Left "unexpected parallel list comprehension in pattern"
  117   ESectionL {} -> Left "unexpected left section in pattern"
  118   ESectionR {} -> Left "unexpected right section in pattern"
  119   ERecordUpd {} -> Left "unexpected record update in pattern"
  120   EGetField {} -> Left "unexpected record field access in pattern"
  121   EGetFieldProjection {} -> Left "unexpected projection section in pattern"
  122   ETypeApp fun ty -> do
  123     funPat <- checkPattern fun
  124     case peelPatternAnn funPat of
  125       PCon name typeArgs args -> Right (PCon name (typeArgs ++ [ty]) args)
  126       _ -> Left "unexpected type application in pattern"
  127   ETHExpQuote {} -> Left "unexpected Template Haskell expression quote in pattern"
  128   ETHTypedQuote {} -> Left "unexpected Template Haskell typed quote in pattern"
  129   ETHDeclQuote {} -> Left "unexpected Template Haskell declaration quote in pattern"
  130   ETHTypeQuote {} -> Left "unexpected Template Haskell type quote in pattern"
  131   ETHPatQuote {} -> Left "unexpected Template Haskell pattern quote in pattern"
  132   ETHNameQuote {} -> Left "unexpected Template Haskell name quote in pattern"
  133   ETHTypeNameQuote {} -> Left "unexpected Template Haskell type name quote in pattern"
  134   ETHTypedSplice {} -> Left "unexpected typed Template Haskell splice in pattern"
  135   EProc {} -> Left "unexpected proc expression in pattern"
  136 
  137 checkPragmaPattern :: Pragma -> Either Text Pattern
  138 checkPragmaPattern pragma =
  139   case pragmaType pragma of
  140     PragmaSCC {} -> Left "unexpected SCC pragma in pattern"
  141     _ -> Left "unexpected pragma in pattern"
  142 
  143 -- | Convert a list of expressions into patterns.
  144 checkPatterns :: [Expr] -> Either Text [Pattern]
  145 checkPatterns = traverse checkPattern
  146 
  147 -- | Check that a tuple element is not a tuple section (Nothing).
  148 checkTupleElement :: Maybe Expr -> Either Text Pattern
  149 checkTupleElement Nothing = Left "unexpected tuple section in pattern"
  150 checkTupleElement (Just e) = checkPattern e
  151 
  152 tupleConstructorPattern :: TupleFlavor -> [Maybe Expr] -> Maybe Pattern
  153 tupleConstructorPattern fl elems
  154   | null elems = Nothing
  155   | all isNothing elems = Just (PCon (tupleConstructorName fl (length elems)) [] [])
  156   | otherwise = Nothing
  157 
  158 tupleConstructorName :: TupleFlavor -> Int -> Name
  159 tupleConstructorName fl arity =
  160   qualifyName Nothing (mkUnqualifiedName NameConSym symbol)
  161   where
  162     symbol = case fl of
  163       Boxed -> tupleCommas
  164       Unboxed -> "#" <> tupleCommas <> "#"
  165     tupleCommas
  166       | arity == 1 = ""
  167       | otherwise = mconcat (replicate (arity - 1) ",")
  168 
  169 -- | Check that a negated expression is a literal (for PNegLit patterns).
  170 checkNegLitPattern :: Expr -> Either Text Pattern
  171 checkNegLitPattern inner = case inner of
  172   EInt n nt repr -> Right (PNegLit (LitInt n nt repr))
  173   EFloat x ft repr -> Right (PNegLit (LitFloat x ft repr))
  174   EAnn ann sub -> fmap (PAnn ann) (checkNegLitPattern sub)
  175   _ -> Left "negation in pattern requires a numeric literal"
  176 
  177 -- | Check whether an operator is a constructor operator (starts with ':').
  178 -- Constructor operators and backtick-quoted constructors are valid in patterns;
  179 -- variable operators like @+@ or @*@ are not.
  180 isConLikeOp :: Name -> Bool
  181 isConLikeOp = isConLikeName
  182 
  183 -- | Try to interpret an expression as a view pattern @expr -> expr@.
  184 -- Returns 'Just' the corresponding 'PView' when the expression is an
  185 -- 'EInfix' with @->@; 'Nothing' otherwise.  Used by the 'EParen' case of
  186 -- 'checkPattern' to strip the outer parentheses and produce @PView@
  187 -- directly (matching the AST shape that the dedicated pattern parser
  188 -- produces).
  189 asViewPat :: Expr -> Maybe Pattern
  190 asViewPat (EInfix l op r)
  191   | renderName op == "->" = case checkPattern r of
  192       Right rPat -> Just (PView l rPat)
  193       Left _ -> Nothing
  194 asViewPat _ = Nothing
