    6 -- Internal module containing the core module parser.
    7 -- This is used by both 'Aihc.Parser' and 'Aihc.Parser.Internal.FromTokens'.
    8 module Aihc.Parser.Internal.Module
    9   ( moduleParser,
   10   )
   11 where
   12 
   13 import Aihc.Parser.Internal.Common (TokParser, braces, expectedTok, skipSemicolons, withSpan)
   14 import Aihc.Parser.Internal.Decl (declParser)
   15 import Aihc.Parser.Internal.Import (importDeclParser, languagePragmaParser, moduleHeaderParser)
   16 import Aihc.Parser.Lex (LexTokenKind (..), lexTokenKind)
   17 import Aihc.Parser.Syntax (Decl, ImportDecl, Module (..), mkAnnotation)
   18 import Control.Monad (void)
   19 import Text.Megaparsec qualified as MP
   20 
   21 data RecoverParseStep a
   22   = RecoverDone
   23   | RecoverParsed !a
   24   | RecoverFailed
   25 
   26 moduleParser :: TokParser Module
   27 moduleParser = withSpan $ do
   28   languagePragmas <- MP.many (languagePragmaParser <* MP.many (expectedTok TkSpecialSemicolon))
   29   mHeader <- MP.optional (moduleHeaderParser <* MP.many (expectedTok TkSpecialSemicolon))
   30   (imports, decls) <- moduleBodyParser
   31   pure $ \span' ->
   32     Module
   33       { moduleAnns = [mkAnnotation span'],
   34         moduleHead = mHeader,
   35         moduleLanguagePragmas = concat languagePragmas,
   36         moduleImports = imports,
   37         moduleDecls = decls
   38       }
   39 
   40 moduleBodyParser :: TokParser ([ImportDecl], [Decl])
   41 moduleBodyParser = braces $ do
   42   skipSemicolons
   43   imports <- importDeclsWithRecovery
   44   decls <- declsWithRecovery
   45   skipSemicolons
   46   pure (imports, decls)
   47 
   48 importDeclsWithRecovery :: TokParser [ImportDecl]
   49 importDeclsWithRecovery = recoverDeclLike importDeclParser
   50 
   51 declsWithRecovery :: TokParser [Decl]
   52 declsWithRecovery = recoverDeclLike declParser
   53 
   54 recoverDeclLike :: TokParser a -> TokParser [a]
   55 recoverDeclLike parser = go []
   56   where
   57     go acc = do
   58       skipSemicolons
   59       step <- MP.withRecovery recoverParseError parseStep
   60       case step of
   61         RecoverDone -> pure (reverse acc)
   62         RecoverParsed parsed -> do
   63           skipSemicolons
   64           go (parsed : acc)
   65         RecoverFailed -> do
   66           skipSemicolons
   67           go acc
   68 
   69     parseStep = do
   70       mParsed <- MP.optional parser
   71       pure $ maybe RecoverDone RecoverParsed mParsed
   72 
   73     recoverParseError err = do
   74       MP.registerParseError err
   75       skipUntilDeclBoundary
   76       pure RecoverFailed
   77 
   78 skipUntilDeclBoundary :: TokParser ()
   79 skipUntilDeclBoundary = do
   80   _ <-
   81     MP.takeWhileP
   82       Nothing
   83       ( \tok ->
   84           let kind = lexTokenKind tok
   85            in kind /= TkSpecialSemicolon && kind /= TkSpecialRBrace
   86       )
   87   void (MP.optional (expectedTok TkSpecialSemicolon))
