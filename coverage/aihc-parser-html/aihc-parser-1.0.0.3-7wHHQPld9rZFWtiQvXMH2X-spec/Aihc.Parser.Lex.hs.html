  117 scanTokens :: LexerEnv -> LexerState -> [LexToken]
  118 scanTokens env st0 =
  119   case skipTrivia st0 of
  120     SkipToken tok st ->
  121       tok : scanTokens env (recordScannedToken tok st)
  122     SkipDone st
  123       | T.null (lexerInput st) -> [eofToken st]
  124       | otherwise ->
  125           let (tok, st') = nextToken env st
  126            in tok : scanTokens env (recordScannedToken tok st')

  128 data SkipResult = SkipDone !LexerState | SkipToken !LexToken !LexerState
  129 
  130 -- | Skip whitespace, line comments, block comments, and control pragmas ({-# LINE/COLUMN #-}).
  131 -- Regular pragmas ({-# ... #-}) are left for 'nextToken' to handle.
  132 -- Returns 'SkipToken' only for error tokens from malformed/unterminated constructs.
  133 skipTrivia :: LexerState -> SkipResult
  134 skipTrivia = go
  135   where
  136     go st =
  137       let inp = lexerInput st
  138        in case inp of
  139             Empty -> SkipDone st
  140             c :< _
  141               | isHaskellWhitespace c ->
  142                   go (markHadTrivia (consumeWhile isHaskellWhitespace st))
  143             _
  144               | Just rest <- T.stripPrefix "--" inp,
  145                 isLineComment rest ->
  146                   let (tok, st') = consumeLineCommentToken st
  147                    in SkipToken tok (markHadTrivia st')
  148             -- Check {-# before {- so control pragmas are handled first and
  149             -- block comment handler does not eat pragma tokens.
  150             _
  151               | "{-#" `T.isPrefixOf` inp ->
  152                   case tryConsumeControlPragma st of
  153                     Just (Nothing, st') -> go (markHadTrivia st')
  154                     Just (Just tok, st') -> SkipToken tok (markHadTrivia st')
  155                     Nothing -> SkipDone st -- not a control pragma; let nextToken handle it
  156               | "{-" `T.isPrefixOf` inp ->
  157                   case consumeBlockCommentTokenOrError st of
  158                     Right (tok, st') -> SkipToken tok (markHadTrivia st')
  159                     Left (tok, st') -> SkipToken tok (markHadTrivia st')
  160             _ ->
  161               case tryConsumeLineDirective st of
  162                 Just (Nothing, st') -> go (markHadTrivia st')
  163                 Just (Just tok, st') -> SkipToken tok (markHadTrivia st')
  164                 Nothing -> SkipDone st
  165 
  166 markHadTrivia :: LexerState -> LexerState
  167 markHadTrivia st = st {lexerHadTrivia = True}

  169 isCommentTokenKind :: LexTokenKind -> Bool

  176 recordScannedToken :: LexToken -> LexerState -> LexerState
  177 recordScannedToken tok st
  178   | isCommentTokenKind (lexTokenKind tok) = st {lexerHadTrivia = True}
  179   | otherwise = st {lexerPrevTokenKind = Just (lexTokenKind tok), lexerHadTrivia = False}

  181 -- | Lex a regular pragma token ({-# ... #-}).
  182 -- Control pragmas (LINE, COLUMN) are handled in 'skipTrivia' and never reach here.
  183 -- Falls back to a 'TkError' token when the pragma has no closing "#-}".
  184 lexPragma :: LexerState -> Maybe (LexToken, LexerState)
  185 lexPragma st
  186   | "{-#" `T.isPrefixOf` lexerInput st =
  187       Just $ case tryParsePragma st of
  188         Just result -> result
  189         Nothing ->
  190           -- Malformed pragma with no closing "#-}"
  191           let consumed = lexerInput st
  192               st' = advanceChars consumed st
  193            in (mkToken st st' consumed (TkError "malformed pragma"), st')
  194   | otherwise = Nothing

  196 nextToken :: LexerEnv -> LexerState -> (LexToken, LexerState)
  197 nextToken env st =
  198   -- Inline chain of alternatives with no intermediate list or closure allocation.
  199   -- (<|>) for Maybe short-circuits on the first Just without allocating.
  200   fromMaybe (lexErrorToken st "unexpected character") $
  201     lexPragma st
  202       <|> lexTHQuoteBracket env st
  203       <|> lexQuasiQuote st
  204       <|> lexHexFloat env st
  205       <|> lexFloat env st
  206       <|> lexIntBase env st
  207       <|> lexInt env st
  208       <|> lexTHNameQuote env st
  209       <|> lexPromotedQuote env st
  210       <|> lexChar env st
  211       <|> lexString env st
  212       <|> lexTHCloseQuote env st
  213       <|> lexSymbol env st
  214       <|> lexIdentifier env st
  215       <|> lexNegativeLiteralOrMinus env st
  216       <|> lexBangOrTildeOperator st
  217       <|> lexTypeApplication env st
  218       <|> lexOverloadedLabel env st
  219       <|> lexPrefixDollar env st
  220       <|> lexImplicitParam env st
  221       <|> lexOperator env st

  223 stepNextToken :: LexerEnv -> LexerState -> LayoutState -> Maybe (LexToken, LexerState, LayoutState)
  224 stepNextToken env lexSt laySt =
  225   case layoutBuffer laySt of
  226     tok : rest ->
  227       Just (tok, lexSt, laySt {layoutBuffer = rest})
  228     [] ->
  229       case scanOneToken env lexSt of
  230         Nothing -> Nothing
  231         Just (rawTok, lexSt') ->
  232           let (allToks, laySt') = layoutTransition laySt rawTok
  233            in case allToks of
  234                 [] -> Just (rawTok, lexSt', laySt')
  235                 [first] -> Just (first, lexSt', laySt')
  236                 first : rest -> Just (first, lexSt', laySt' {layoutBuffer = rest})

  238 scanOneToken :: LexerEnv -> LexerState -> Maybe (LexToken, LexerState)
  239 scanOneToken env st0 =
  240   case skipTrivia st0 of
  241     SkipToken tok st ->
  242       Just (tok, recordScannedToken tok st)
  243     SkipDone st
  244       | T.null (lexerInput st) ->
  245           case lexerPrevTokenKind st of
  246             Just TkEOF -> Nothing
  247             _ ->
  248               let tok = eofToken st
  249                   st' = st {lexerPrevTokenKind = Just TkEOF, lexerHadTrivia = False}
  250                in Just (tok, st')
  251       | otherwise ->
  252           let (tok, st') = nextToken env st
  253            in Just (tok, recordScannedToken tok st')

  255 scanAllTokens :: LexerEnv -> LexerState -> [LexToken]
  256 scanAllTokens env st =
  257   case scanOneToken env st of
  258     Nothing -> []
  259     Just (tok, st') -> tok : scanAllTokens env st'

  261 lexIdentifier :: LexerEnv -> LexerState -> Maybe (LexToken, LexerState)
  262 lexIdentifier env st =
  263   case lexerInput st of
  264     c :< rest
  265       | isIdentStart c ->
  266           let hasMagicHash = hasExt MagicHash env
  267               (seg, rest0) = consumeIdentTail hasMagicHash rest
  268               firstChunk = T.take (1 + T.length seg) (lexerInput st)
  269               (consumed, rest1, isQualified) = gatherQualified hasMagicHash firstChunk rest0
  270            in case (isQualified || isConIdStart c, rest1) of
  271                 (True, '.' :< dotRest@(opChar :< _))
  272                   | isSymbolicOpChar opChar ->
  273                       let opChars = T.takeWhile isSymbolicOpChar dotRest
  274                           fullOp = consumed <> "." <> opChars
  275                           (modName, opName) = splitQualified (consumed <> ".") opChars
  276                           kind =
  277                             if opChar == ':'
  278                               then TkQConSym modName opName
  279                               else TkQVarSym modName opName
  280                           st' = advanceChars fullOp st
  281                        in Just (mkToken st st' fullOp kind, st')
  282                 _ ->
  283                   let kind = classifyIdentifier c isQualified consumed
  284                       st' = advanceChars consumed st
  285                    in Just (mkToken st st' consumed kind, st')
  286     _ -> Nothing
  287   where
  288     gatherQualified :: Bool -> Text -> Text -> (Text, Text, Bool)
  289     gatherQualified hasMH acc chars =
  290       case chars of
  291         '.' :< dotRest@(c' :< more)
  292           | isIdentStart c',
  293             not (T.isSuffixOf "#" acc),
  294             isConIdStart (T.head acc) ->
  295               let (seg, rest) = consumeIdentTail hasMH more
  296                   segWithHead = T.take (1 + T.length seg) dotRest
  297                in gatherQualified hasMH (acc <> "." <> segWithHead) rest
  298         _ -> (acc, chars, T.any (== '.') acc)
  299 
  300     -- Split a qualified identifier into (module part, name part).
  301     -- E.g. "Data.Maybe." ++ "++" -> ("Data.Maybe", "++")
  302     splitQualified :: Text -> Text -> (Text, Text)
  303     splitQualified modWithDot name =
  304       (T.dropEnd 1 modWithDot, name)
  305 
  306     classifyIdentifier firstChar isQualified ident
  307       | isQualified =
  308           let rev = T.reverse ident
  309               (revName, revRest) = T.span (/= '.') rev
  310               modName = T.reverse (T.drop 1 revRest)
  311               name = T.reverse revName
  312            in case T.uncons name of
  313                 Just (c', _)
  314                   | isConIdStart c' -> TkQConId modName name
  315                   | name == "do" && hasExt QualifiedDo env -> TkQualifiedDo modName
  316                   | name == "mdo" && hasExt QualifiedDo env && hasExt RecursiveDo env -> TkQualifiedMdo modName
  317                 Just _ -> TkQVarId modName name
  318                 Nothing -> TkQVarId modName name
  319       | otherwise =
  320           case keywordTokenKind (lexerExtensions env) ident of
  321             Just kw -> kw
  322             Nothing
  323               | isConIdStart firstChar -> TkConId ident
  324               | otherwise -> TkVarId ident

  326 consumeIdentTail :: Bool -> Text -> (Text, Text)
  327 consumeIdentTail hasMH inp =
  328   let (tailPart, rest) = T.span isIdentTail inp
  329    in case rest of
  330         '#' :< _
  331           | hasMH ->
  332               let hashes = T.takeWhile (== '#') rest
  333                in (tailPart <> hashes, T.drop (T.length hashes) rest)
  334         _ -> (tailPart, rest)

  336 lexImplicitParam :: LexerEnv -> LexerState -> Maybe (LexToken, LexerState)
  337 lexImplicitParam env st
  338   | not (hasExt ImplicitParams env) = Nothing
  339   | otherwise =
  340       case lexerInput st of
  341         '?' :< rest0@(c :< _)
  342           | isVarIdentifierStartChar c ->
  343               let hasMagicHash = hasExt MagicHash env
  344                   (tailChars, _rest) = consumeIdentTail hasMagicHash (T.tail rest0)
  345                   txt = T.take (2 + T.length tailChars) (lexerInput st)
  346                   st' = advanceChars txt st
  347                in Just (mkToken st st' txt (TkImplicitParam txt), st')
  348         _ -> Nothing

  350 lexNegativeLiteralOrMinus :: LexerEnv -> LexerState -> Maybe (LexToken, LexerState)
  351 lexNegativeLiteralOrMinus env st
  352   | not hasNegExt && not hasMH = Nothing
  353   | not (isStandaloneMinus (lexerInput st)) = Nothing
  354   | otherwise =
  355       let prevAllows = allowsMergeOrPrefix (lexerPrevTokenKind st) (lexerHadTrivia st)
  356           rest = T.drop 1 (lexerInput st)
  357        in if hasExt NegativeLiterals env && prevAllows
  358             then case tryLexNumberAfterMinus env st of
  359               Just result -> Just result
  360               Nothing -> lexMinusOperator env st rest prevAllows
  361             else
  362               -- GHC merges minus into primitive unboxed literals (Int#, Word#,
  363               -- Float#, Double#, and ExtendedLiterals types) even without
  364               -- NegativeLiterals.  When only MagicHash is active, speculatively
  365               -- lex the number after the minus and keep the merged token only
  366               -- when the result carries a primitive suffix.
  367               if hasMH && prevAllows
  368                 then case tryLexNumberAfterMinus env st of
  369                   Just (tok, st')
  370                     | isPrimitiveNumericToken (lexTokenKind tok) -> Just (tok, st')
  371                   _ ->
  372                     if hasNegExt
  373                       then lexMinusOperator env st rest prevAllows
  374                       else Nothing
  375                 else
  376                   if hasNegExt
  377                     then lexMinusOperator env st rest prevAllows
  378                     else Nothing
  379   where
  380     hasNegExt =
  381       hasExt NegativeLiterals env
  382         || hasExt LexicalNegation env
  383     hasMH = hasExt MagicHash env

  385 isStandaloneMinus :: Text -> Bool
  386 isStandaloneMinus input =
  387   case input of
  388     '-' :< (c :< _) | isSymbolicOpChar c && c /= '-' -> False
  389     '-' :< _ -> True
  390     _ -> False
  391 
  392 tryLexNumberAfterMinus :: LexerEnv -> LexerState -> Maybe (LexToken, LexerState)
  393 tryLexNumberAfterMinus env st = do
  394   let stAfterMinus = advanceChars "-" st
  395   (numTok, stFinal) <-
  396     lexHexFloat env stAfterMinus
  397       <|> lexFloat env stAfterMinus
  398       <|> lexIntBase env stAfterMinus
  399       <|> lexInt env stAfterMinus
  400   Just (negateToken st numTok, stFinal)

  402 negateToken :: LexerState -> LexToken -> LexToken
  403 negateToken stBefore numTok =
  404   LexToken
  405     { lexTokenKind = negateKind (lexTokenKind numTok),
  406       lexTokenText = "-" <> lexTokenText numTok,
  407       lexTokenSpan = extendSpanLeft (lexTokenSpan numTok),
  408       lexTokenOrigin = lexTokenOrigin numTok,
  409       lexTokenAtLineStart = lexerAtLineStart stBefore
  410     }
  411   where
  412     negateKind k =
  413       case k of
  414         TkInteger n nt -> TkInteger (negate n) nt
  415         TkFloat n ft -> TkFloat (negate n) ft
  416         other -> other
  417 
  418     extendSpanLeft sp =
  419       case sp of
  420         SourceSpan {sourceSpanSourceName, sourceSpanEndLine = endLine, sourceSpanEndCol = endCol, sourceSpanEndOffset} ->
  421           SourceSpan
  422             { sourceSpanSourceName = sourceSpanSourceName,
  423               sourceSpanStartLine = lexerLine stBefore,
  424               sourceSpanStartCol = lexerCol stBefore,
  425               sourceSpanEndLine = endLine,
  426               sourceSpanEndCol = endCol,
  427               sourceSpanStartOffset = lexerByteOffset stBefore,
  428               sourceSpanEndOffset = sourceSpanEndOffset
  429             }
  430         NoSourceSpan -> NoSourceSpan

  432 -- | Does the token kind represent a primitive (unboxed) numeric literal?
  433 -- These are MagicHash types (Int#, Word#, Float#, Double#) and ExtendedLiterals
  434 -- types (Int8#, Int16#, etc.).  Plain boxed types (TInteger, TFractional) return
  435 -- False.
  436 isPrimitiveNumericToken :: LexTokenKind -> Bool
  437 isPrimitiveNumericToken k =
  438   case k of
  439     TkInteger _ nt -> nt /= TInteger
  440     TkFloat _ ft -> ft /= TFractional
  441     _ -> False

  443 lexMinusOperator :: LexerEnv -> LexerState -> Text -> Bool -> Maybe (LexToken, LexerState)
  444 lexMinusOperator env st rest prevAllows
  445   | not (hasExt LexicalNegation env) = Nothing
  446   | otherwise =
  447       let st' = advanceChars "-" st
  448           kind =
  449             if prevAllows && canStartNegatedAtom rest
  450               then TkPrefixMinus
  451               else TkMinusOperator
  452        in Just (mkToken st st' "-" kind, st')

  454 allowsMergeOrPrefix :: Maybe LexTokenKind -> Bool -> Bool
  455 allowsMergeOrPrefix prev hadTrivia =
  456   case prev of
  457     Nothing -> True
  458     Just _ | hadTrivia -> True
  459     Just prevKind -> prevTokenAllowsTightPrefix prevKind

  461 prevTokenAllowsTightPrefix :: LexTokenKind -> Bool
  462 prevTokenAllowsTightPrefix kind =
  463   case kind of
  464     TkTHTypeQuoteOpen -> True
  465     TkTHExpQuoteOpen -> True
  466     TkTHTypedQuoteOpen -> True
  467     TkTHDeclQuoteOpen -> True
  468     TkTHPatQuoteOpen -> True
  469     TkSpecialLParen -> True
  470     TkSpecialLBracket -> True
  471     TkSpecialLBrace -> True
  472     TkSpecialComma -> True
  473     TkSpecialSemicolon -> True
  474     TkVarSym _ -> True
  475     TkConSym _ -> True
  476     TkQVarSym _ _ -> True
  477     TkQConSym _ _ -> True
  478     TkMinusOperator -> True
  479     TkPrefixMinus -> True
  480     TkReservedEquals -> True
  481     TkReservedLeftArrow -> True
  482     TkReservedRightArrow -> True
  483     TkReservedDoubleArrow -> True
  484     TkReservedDoubleColon -> True
  485     TkReservedPipe -> True
  486     TkReservedBackslash -> True
  487     TkTypeApp -> True
  488     TkPragma _ -> True
  489     _ -> False

  491 -- | Returns True for tokens after which a '.' can begin a record field access
  492 -- or a projection section. This includes expression-ending tokens (the
  493 -- 'not prevTokenAllowsTightPrefix' cases) and '(' for projection sections
  494 -- like @(.field)@.
  495 prevTokenAllowsRecordDot :: LexTokenKind -> Bool
  496 prevTokenAllowsRecordDot TkSpecialLParen = True
  497 prevTokenAllowsRecordDot kind =
  498   not (tokenEndsWithHash kind) && not (prevTokenAllowsTightPrefix kind)
  499 
  500 -- | Bare @#.@ is an operator continuation, not record-dot syntax.  Require
  501 -- parentheses around hash-ending tokens before field access, e.g. @('a'#).x@.
  502 tokenEndsWithHash :: LexTokenKind -> Bool
  503 tokenEndsWithHash kind =
  504   case kind of
  505     TkInteger _ nt -> nt /= TInteger
  506     TkFloat _ ft -> ft /= TFractional
  507     TkCharHash {} -> True
  508     TkStringHash {} -> True
  509     TkVarId name -> T.isSuffixOf "#" name
  510     TkConId name -> T.isSuffixOf "#" name
  511     TkQVarId _ name -> T.isSuffixOf "#" name
  512     TkQConId _ name -> T.isSuffixOf "#" name
  513     _ -> False

  515 canStartNegatedAtom :: Text -> Bool
  516 canStartNegatedAtom rest =
  517   case rest of
  518     c :< _
  519       | isIdentStart c -> True
  520       | isDigit c -> True
  521       | c == '\'' -> True
  522       | c == '"' -> True
  523       | c == '(' -> True
  524       | c == '[' -> True
  525       | c == '\\' -> True
  526       | c == '-' -> True
  527       | otherwise -> False
  528     _ -> False

  530 lexTypeApplication :: LexerEnv -> LexerState -> Maybe (LexToken, LexerState)
  531 lexTypeApplication _env st =
  532   case lexerInput st of
  533     '@' :< rest
  534       | startsWithSplice rest -> Just (emitToken st "@" TkTypeApp)
  535       | not (startsWithSymOp rest) ->
  536           -- GHC lexes visible type application syntax based on layout, not on
  537           -- whether TypeApplications is enabled. Extension validity is checked
  538           -- later, after parsing.
  539           if lexerHadTrivia st
  540             then
  541               let kind
  542                     | canStartTypeAtomT rest = TkTypeApp
  543                     | otherwise = TkVarSym "@"
  544                in Just (emitToken st "@" kind)
  545             else Just (emitToken st "@" TkReservedAt)
  546     _ -> Nothing
  547   where
  548     canStartTypeAtomT t =
  549       case t of
  550         c :< _
  551           | isIdentStart c -> True
  552           | isDigit c -> True
  553           | c == '(' -> True
  554           | c == '[' -> True
  555           | c == '_' -> True
  556           | c == '\'' -> True
  557           | c == '"' -> True
  558         _ -> False
  559     startsWithSplice t =
  560       case t of
  561         '$' :< ('$' :< rest) -> canStartSpliceAtomT rest
  562         '$' :< rest -> canStartSpliceAtomT rest
  563         _ -> False
  564     canStartSpliceAtomT t =
  565       case t of
  566         c :< _ -> isIdentStart c || c == '('
  567         _ -> False

  569 lexOverloadedLabel :: LexerEnv -> LexerState -> Maybe (LexToken, LexerState)
  570 lexOverloadedLabel env st
  571   | not (hasExt OverloadedLabels env) = Nothing
  572   | otherwise =
  573       case lexerInput st of
  574         '#' :< rest
  575           | Just (label, raw) <- parseOverloadedLabel rest ->
  576               let fullRaw = "#" <> raw
  577                   st' = advanceChars fullRaw st
  578                in Just (mkToken st st' fullRaw (TkOverloadedLabel label fullRaw), st')
  579           | "\"" `T.isPrefixOf` rest ->
  580               let consumed = "#" <> takeMalformedString rest
  581                   st' = advanceChars consumed st
  582                in Just (mkErrorToken st st' consumed "invalid overloaded label", st')
  583         _ -> Nothing
  584   where
  585     parseOverloadedLabel chars =
  586       case chars of
  587         '"' :< rest ->
  588           case scanQuoted '"' rest of
  589             Right (body, _) ->
  590               let raw = "\"" <> body <> "\""
  591                   decoded =
  592                     case reads (T.unpack raw) of
  593                       [(str, "")] | not (null str) -> Just (T.pack str)
  594                       _ -> Nothing
  595                in (,raw) <$> decoded
  596             Left _ -> Nothing
  597         c :< rest
  598           | isVarIdentifierStartChar c ->
  599               let label = T.take (1 + T.length (T.takeWhile isIdentTail rest)) chars
  600                in Just (label, label)
  601         _ -> Nothing
  602 
  603     takeMalformedString chars =
  604       case scanQuoted '"' (T.drop 1 chars) of
  605         Right (body, _) -> "\"" <> body <> "\""
  606         Left raw -> "\"" <> raw

  608 lexBangOrTildeOperator :: LexerState -> Maybe (LexToken, LexerState)
  609 lexBangOrTildeOperator st =
  610   case lexerInput st of
  611     '!' :< rest -> lexPrefixSensitiveOp st '!' TkPrefixBang rest
  612     '~' :< rest -> lexPrefixSensitiveOp st '~' TkPrefixTilde rest
  613     '%' :< rest -> lexPrefixSensitiveOp st '%' TkPrefixPercent rest
  614     _ -> Nothing

  616 isPrefixPosition :: LexerState -> Bool
  617 isPrefixPosition st =
  618   case lexerPrevTokenKind st of
  619     Nothing -> True
  620     Just prevKind
  621       | lexerHadTrivia st -> True
  622       | otherwise -> prevTokenAllowsTightPrefix prevKind

  624 lexPrefixDollar :: LexerEnv -> LexerState -> Maybe (LexToken, LexerState)
  625 lexPrefixDollar env st
  626   | not (hasExt TemplateHaskellQuotes env || hasExt TemplateHaskell env) = Nothing
  627   | otherwise =
  628       case lexerInput st of
  629         '$' :< ('$' :< rest)
  630           | not (startsWithSymOp rest),
  631             isPrefixPosition st,
  632             canStartSpliceAtomT rest ->
  633               Just (emitToken st "$$" TkTHTypedSplice)
  634         '$' :< rest
  635           | not (startsWithSymOp rest),
  636             isPrefixPosition st,
  637             canStartSpliceAtomT rest ->
  638               Just (emitToken st "$" TkTHSplice)
  639         _ -> Nothing
  640   where
  641     canStartSpliceAtomT t =
  642       case t of
  643         c :< _ -> isIdentStart c || c == '('
  644         _ -> False

  646 lexPrefixSensitiveOp :: LexerState -> Char -> LexTokenKind -> Text -> Maybe (LexToken, LexerState)
  647 lexPrefixSensitiveOp st opChar prefixKind rest
  648   | startsWithSymOp rest = Nothing
  649   | isPrefixPosition st && canStartPrefixPatternAtom rest =
  650       Just (emitToken st (T.singleton opChar) prefixKind)
  651   | otherwise = Nothing
  652 
  653 canStartPrefixPatternAtom :: Text -> Bool
  654 canStartPrefixPatternAtom rest =
  655   case rest of
  656     c :< _
  657       | isIdentStart c -> True
  658       | isDigit c -> True
  659       | c == '\'' -> True
  660       | c == '"' -> True
  661       | c == '(' -> True
  662       | c == '[' -> True
  663       | c == '_' -> True
  664       | c == '!' -> True
  665       | c == '~' -> True
  666       | c == '$' -> True
  667       | otherwise -> False
  668     _ -> False

  670 lexOperator :: LexerEnv -> LexerState -> Maybe (LexToken, LexerState)
  671 lexOperator env st =
  672   let inp = lexerInput st
  673       opText = T.takeWhile isSymbolicOpChar inp
  674       hasArrows = hasExt Arrows env
  675    in case opText of
  676         Empty -> Nothing
  677         c :< _ ->
  678           case (hasArrows, opText, T.drop (T.length opText) inp) of
  679             (True, "|", ')' :< _) ->
  680               let bananaText = "|)"
  681                   st' = advanceChars bananaText st
  682                in Just (mkToken st st' bananaText TkBananaClose, st')
  683             _
  684               | opText == ".",
  685                 not (lexerHadTrivia st),
  686                 Just prevKind <- lexerPrevTokenKind st,
  687                 tokenEndsWithHash prevKind,
  688                 nextC :< _ <- T.drop 1 inp,
  689                 isVarIdentifierStartChar nextC ->
  690                   let st' = advanceChars opText st
  691                    in Just (mkErrorToken st st' opText "unexpected record-dot access after hash-ending token", st')
  692             _
  693               | hasExt OverloadedRecordDot env,
  694                 opText == ".",
  695                 not (lexerHadTrivia st),
  696                 Just prevKind <- lexerPrevTokenKind st,
  697                 prevTokenAllowsRecordDot prevKind,
  698                 nextC :< _ <- T.drop 1 inp,
  699                 isVarIdentifierStartChar nextC ->
  700                   let st' = advanceChars opText st
  701                    in Just (mkToken st st' opText TkRecordDot, st')
  702             _ ->
  703               let st' = advanceChars opText st
  704                   hasUnicode = hasExt UnicodeSyntax env
  705                   kind =
  706                     case reservedOpTokenKind opText of
  707                       Just reserved -> reserved
  708                       Nothing
  709                         | hasArrows, Just arrowKind <- arrowOpTokenKind opText -> arrowKind
  710                         | hasUnicode -> unicodeOpTokenKind hasArrows opText c
  711                         | c == ':' -> TkConSym opText
  712                         | otherwise -> TkVarSym opText
  713                in Just (mkToken st st' opText kind, st')

  715 unicodeOpTokenKind :: Bool -> Text -> Char -> LexTokenKind
  716 unicodeOpTokenKind hasArrows txt firstChar
  717   | txt == "∷" = TkReservedDoubleColon
  718   | txt == "⇒" = TkReservedDoubleArrow
  719   | txt == "→" = TkReservedRightArrow
  720   | txt == "←" = TkReservedLeftArrow
  721   | txt == "∀" = TkKeywordForall
  722   | txt == "⤙" = if hasArrows then TkArrowTail else TkVarSym "-<"
  723   | txt == "⤚" = if hasArrows then TkArrowTailReverse else TkVarSym ">-"
  724   | txt == "⤛" = if hasArrows then TkDoubleArrowTail else TkVarSym "-<<"
  725   | txt == "⤜" = if hasArrows then TkDoubleArrowTailReverse else TkVarSym ">>-"
  726   | txt == "⦇" = if hasArrows then TkBananaOpen else TkVarSym "(|"
  727   | txt == "⦈" = if hasArrows then TkBananaClose else TkVarSym "|)"
  728   | txt == "⟦" = TkVarSym "[|"
  729   | txt == "⟧" = TkVarSym "|]"
  730   | txt == "⊸" = TkLinearArrow
  731   | firstChar == ':' = TkConSym txt
  732   | otherwise = TkVarSym txt

  734 lexSymbol :: LexerEnv -> LexerState -> Maybe (LexToken, LexerState)
  735 lexSymbol env st =
  736   case lexerInput st of
  737     '(' :< ('#' :< _)
  738       | hasUnboxed -> Just (emitToken st "(#" TkSpecialUnboxedLParen)
  739     '#' :< (')' :< _)
  740       | hasUnboxed -> Just (emitToken st "#)" TkSpecialUnboxedRParen)
  741     '(' :< ('|' :< rest)
  742       | hasArrows,
  743         bananaOpenAllowed rest ->
  744           Just (emitToken st "(|" TkBananaOpen)
  745     '(' :< _ -> Just (emitToken st "(" TkSpecialLParen)
  746     ')' :< _ -> Just (emitToken st ")" TkSpecialRParen)
  747     '[' :< _ -> Just (emitToken st "[" TkSpecialLBracket)
  748     ']' :< _ -> Just (emitToken st "]" TkSpecialRBracket)
  749     '{' :< _ -> Just (emitToken st "{" TkSpecialLBrace)
  750     '}' :< _ -> Just (emitToken st "}" TkSpecialRBrace)
  751     ',' :< _ -> Just (emitToken st "," TkSpecialComma)
  752     ';' :< _ -> Just (emitToken st ";" TkSpecialSemicolon)
  753     '`' :< _ -> Just (emitToken st "`" TkSpecialBacktick)
  754     _ -> Nothing
  755   where
  756     hasUnboxed = hasExt UnboxedTuples env || hasExt UnboxedSums env
  757     hasArrows = hasExt Arrows env
  758 
  759     bananaOpenAllowed rest =
  760       case rest of
  761         c :< _ -> not (isSymbolicOpChar c)
  762         _ -> True

  764 isValidCharLiteral :: Text -> Bool
  765 isValidCharLiteral chars =
  766   case scanQuoted '\'' chars of
  767     Right (body, _) -> isJust (readMaybeChar ("'" <> body <> "'"))
  768     Left _ -> False

  770 lexPromotedQuote :: LexerEnv -> LexerState -> Maybe (LexToken, LexerState)
  771 lexPromotedQuote _env st =
  772   case lexerInput st of
  773     '\'' :< rest
  774       | isValidCharLiteral rest -> Nothing
  775       | isPromotionStart rest ->
  776           let st' = advanceChars "'" st
  777            in Just (mkToken st st' "'" (TkVarSym "'"), st')
  778       | otherwise -> Nothing
  779     _ -> Nothing
  780   where
  781     isPromotionStart chars =
  782       case skipHorizontalWhitespace chars of
  783         c :< _
  784           | c == '[' -> True
  785           | c == '(' -> True
  786           | c == ':' -> True
  787           | isConIdStart c -> True
  788           | isSymbolicOpChar c -> True
  789         _ -> False
  790 
  791     skipHorizontalWhitespace = T.dropWhile (\c -> c == ' ' || c == '\t')

  793 lexChar :: LexerEnv -> LexerState -> Maybe (LexToken, LexerState)
  794 lexChar env st =
  795   case lexerInput st of
  796     '\'' :< rest ->
  797       case scanQuoted '\'' rest of
  798         Right (body, _) ->
  799           let rawT = "'" <> body <> "'"
  800            in case readMaybeChar rawT of
  801                 Just c ->
  802                   let (tokTxt, tokKind, st') =
  803                         withOptionalMagicHashSuffix 1 env st rawT (TkChar c) (TkCharHash c)
  804                    in Just (mkToken st st' tokTxt tokKind, st')
  805                 Nothing ->
  806                   let st' = advanceChars rawT st
  807                    in Just (mkErrorToken st st' rawT "invalid char literal", st')
  808         Left raw ->
  809           let full = "'" <> raw
  810               st' = advanceChars full st
  811            in Just (mkErrorToken st st' full "unterminated char literal", st')
  812     _ -> Nothing

  814 lexString :: LexerEnv -> LexerState -> Maybe (LexToken, LexerState)
  815 lexString env st =
  816   let inp = lexerInput st
  817    in case T.stripPrefix "\"\"\"" inp of
  818         Just restText | hasExt MultilineStrings env ->
  819           case scanMultilineString restText of
  820             Right (body, _) ->
  821               let raw = "\"\"\"" <> body <> "\"\"\""
  822                   decoded = T.pack (processMultilineString (T.unpack body))
  823                   (tokTxt, tokKind, st') = lexStringKind env st raw decoded
  824                in Just (mkToken st st' tokTxt tokKind, st')
  825             Left raw ->
  826               let full = "\"\"\"" <> raw
  827                   st' = advanceChars full st
  828                in Just (mkErrorToken st st' full "unterminated multiline string literal", st')
  829         _ ->
  830           case inp of
  831             '"' :< rest ->
  832               case scanQuoted '"' rest of
  833                 Right (body, _) ->
  834                   let rawT = "\"" <> body <> "\""
  835                       decoded = fromMaybe body (decodeStringBody body)
  836                       (tokTxt, tokKind, st') = lexStringKind env st rawT decoded
  837                    in Just (mkToken st st' tokTxt tokKind, st')
  838                 Left raw ->
  839                   let full = "\"" <> raw
  840                       st' = advanceChars full st
  841                    in Just (mkErrorToken st st' full "unterminated string literal", st')
  842             _ -> Nothing

  844 lexStringKind :: LexerEnv -> LexerState -> Text -> Text -> (Text, LexTokenKind, LexerState)
  845 lexStringKind env st raw decoded =
  846   case withOptionalMagicHashSuffix 1 env st raw (TkString decoded) (TkStringHash decoded) of
  847     (tokTxt, TkStringHash _ _, st')
  848       | T.any ((>= 256) . ord) decoded ->
  849           (tokTxt, TkError "primitive string literal contains a character outside Latin-1", st')
  850     result -> result

  852 lexQuasiQuote :: LexerState -> Maybe (LexToken, LexerState)
  853 lexQuasiQuote st =
  854   case lexerInput st of
  855     '[' :< rest ->
  856       case parseQuasiQuote rest of
  857         Just (quoter, body) ->
  858           let raw = "[" <> quoter <> "|" <> body <> "|]"
  859               st' = advanceChars raw st
  860            in Just (mkToken st st' raw (TkQuasiQuote quoter body), st')
  861         Nothing -> Nothing
  862     _ -> Nothing
  863   where
  864     parseQuasiQuote chars =
  865       let (quoter, rest0) = takeQuoter chars
  866        in if T.null quoter
  867             then Nothing
  868             else case rest0 of
  869               '|' :< rest1 ->
  870                 let (body, rest2) = T.breakOn "|]" rest1
  871                  in if T.null rest2
  872                       then Nothing
  873                       else Just (quoter, body)
  874               _ -> Nothing

  876 emitToken :: LexerState -> Text -> LexTokenKind -> (LexToken, LexerState)
  877 emitToken st raw kind =
  878   let st' = advanceChars raw st
  879    in (mkToken st st' raw kind, st')

  912 lexTHNameQuote :: LexerEnv -> LexerState -> Maybe (LexToken, LexerState)
  913 lexTHNameQuote env st
  914   | not (thNameQuotesEnabled env) = Nothing
  915   | otherwise =
  916       case lexerInput st of
  917         '\'' :< ('\'' :< rest)
  918           | Just c <- nextNonTriviaChar rest,
  919             isIdentStart c || c == '(' || c == '[' ->
  920               Just (emitToken st "''" TkTHTypeQuoteTick)
  921         '\'' :< rest0
  922           | not (isValidCharLiteral rest0) ->
  923               Just (emitToken st "'" TkTHQuoteTick)
  924         _ -> Nothing
  925   where
  926     nextNonTriviaChar chars =
  927       case skipTrivia (st {lexerInput = chars}) of
  928         SkipDone st' ->
  929           case lexerInput st' of
  930             c :< _ -> Just c
  931             _ -> Nothing
  932         SkipToken _ st' ->
  933           case lexerInput st' of
  934             c :< _ -> Just c
  935             _ -> Nothing

  974 takeQuoter :: Text -> (Text, Text)
  975 takeQuoter input =
  976   case input of
  977     c :< rest
  978       | isIdentStart c ->
  979           let tailChars = T.takeWhile isIdentTail rest
  980               firstSeg = T.take (1 + T.length tailChars) input
  981               rest0 = T.drop (T.length tailChars) rest
  982            in go (T.length firstSeg) rest0
  983     _ -> ("", input)
  984   where
  985     go !n chars =
  986       case chars of
  987         '.' :< (c' :< more)
  988           | isIdentStart c' ->
  989               let tailChars = T.takeWhile isIdentTail more
  990                   segLen = 1 + 1 + T.length tailChars
  991                in go (n + segLen) (T.drop (T.length tailChars) more)
  992         _ -> (T.take n input, chars)

  994 isIdentStart :: Char -> Bool

 1001 isIdentTail c = isIdentStart c || isIdentContinue c || c == '\''

 1013 isUniOtherLetter c = not (isAscii c) && generalCategory c == OtherLetter

 1018     || generalCategory c == DecimalNumber
