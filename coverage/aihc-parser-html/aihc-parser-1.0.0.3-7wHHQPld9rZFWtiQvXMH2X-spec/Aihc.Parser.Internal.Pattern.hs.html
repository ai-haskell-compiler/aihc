   24 -- | Report core:
   25 --
   26 -- > pat -> lpat qconop pat
   27 -- >     | lpat
   28 patternParser :: TokParser Pattern
   29 patternParser = patternParserWithTypeSigParser typeParser
   30 
   31 patternParserWithTypeSigParser :: TokParser Type -> TokParser Pattern
   32 patternParserWithTypeSigParser typeSigParser =
   33   label "pattern" $
   34     optionalSuffix
   35       (expectedTok TkReservedDoubleColon *> typeSigParser)
   36       PTypeSig
   37       patParser

   39 -- | Parse the pattern position of a case alternative.
   40 --
   41 -- GHC does not accept a top-level pattern type signature before the alternative
   42 -- arrow: @case x of _ :: T -> rhs@ is rejected, while @case x of (_ :: T) ->
   43 -- rhs@ is accepted because the signature is parenthesized into an atomic
   44 -- pattern.  Use the report @pat@ level here so the outer @::@ is left for the
   45 -- case RHS parser, which then rejects it.
   46 -- | Case alternatives intentionally use the report @pat@ level rather than
   47 -- the outer typed-pattern wrapper.
   48 caseAltPatternParser :: TokParser Pattern
   49 caseAltPatternParser = patParser

   51 -- | Parse a pattern (@pat@ in the Haskell Report).
   52 --
   53 -- @
   54 -- pat → lpat qconop pat
   55 --     | lpat
   56 -- @
   57 patParser :: TokParser Pattern
   58 patParser = do
   59   lhs <- lpatParser
   60   rest <- MP.many ((,) <$> conOperatorParser <*> lpatParser)
   61   pure (foldInfixL buildInfixPattern lhs rest)
   62 
   63 buildInfixPattern :: Pattern -> (Name, Pattern) -> Pattern
   64 buildInfixPattern lhs (op, rhs) =
   65   PInfix lhs op rhs
   66 
   67 conOperatorParser :: TokParser Name
   68 conOperatorParser =
   69   symbolicConOp <|> backtickConOp
   70   where
   71     symbolicConOp =
   72       tokenSatisfy "constructor operator" $ \tok ->
   73         case lexTokenKind tok of
   74           TkConSym op -> Just (qualifyName Nothing (mkUnqualifiedNameAt tok NameConSym op))
   75           TkQConSym modName op -> Just (mkNameAt tok (Just modName) NameConSym op)
   76           TkReservedColon -> Just (qualifyName Nothing (mkUnqualifiedNameAt tok NameConSym ":"))
   77           _ -> Nothing
   78     backtickConOp =
   79       MP.try $
   80         expectedTok TkSpecialBacktick *> constructorNameParser <* expectedTok TkSpecialBacktick
   81 
   82 -- | Parse a left pattern (@lpat@ in the Haskell Report).
   83 --
   84 -- @
   85 -- lpat → apat
   86 --      | - (integer | float)       (negative literal)
   87 --      | gcon apat₁ … apatₖ       (arity gcon = k, k ≥ 1)
   88 -- @
   89 --
   90 -- Negative literals and constructor application live here, not in
   91 -- 'apatParser' (@apat@).  This distinction is critical: since
   92 -- constructor arguments are @apat@s, @Con - 0@ cannot be misparsed as
   93 -- @Con (-0)@ — the @-@ is not valid inside an @apat@.
   94 lpatParser :: TokParser Pattern
   95 lpatParser =
   96   negativeLiteralPatternParser <|> do
   97     first <- apatParser
   98     if isPatternAppHead first
   99       then foldl buildPatternApp first <$> MP.many apatParser
  100       else pure first
  101 
  102 buildPatternApp :: Pattern -> Pattern -> Pattern
  103 buildPatternApp lhs rhs =
  104   case peelPatternAnn lhs of
  105     PCon name typeArgs args ->
  106       PAnn
  107         (mkAnnotation NoSourceSpan)
  108         (PCon name typeArgs (args <> [rhs]))
  109     _ -> lhs
  110 
  111 -- | Parse an atomic pattern (@apat@ in the Haskell Report).
  112 --
  113 -- This intentionally does NOT handle negative literals (@- integer@),
  114 -- which belong to the @lpat@ level ('lpatParser').
  115 apatParser :: TokParser Pattern
  116 apatParser =
  117   label "pattern atom" $
  118     asApatParser <|> nonAsApatParser
  119 
  120 -- | Parse an as-pattern as an atomic pattern: @binder\@apat@.
  121 --
  122 -- The binder may be a parenthesized operator, e.g. @(+)\@C@.  Trying this
  123 -- before the parenthesized-pattern parser keeps @(+)\@(+)@C@ in the as-pattern
  124 -- grammar instead of prematurely committing to the bare operator pattern @(+)@.
  125 asApatParser :: TokParser Pattern
  126 asApatParser =
  127   asPatternParser apatParser
  128 
  129 nonAsApatParser :: TokParser Pattern
  130 nonAsApatParser = do
  131   thAny <- thAnyEnabled
  132   explicitNamespacesEnabled <- isExtensionEnabled ExplicitNamespaces
  133   requiredTypeArgumentsEnabled <- isExtensionEnabled RequiredTypeArguments
  134   typeAbstractionsEnabled <- isExtensionEnabled TypeAbstractions
  135   tok <- lookAhead anySingle
  136   case lexTokenKind tok of
  137     TkTypeApp
  138       | typeAbstractionsEnabled -> typeBinderPatternParser
  139     TkPrefixBang -> strictPatternParser
  140     TkPrefixTilde -> irrefutablePatternParser
  141     TkKeywordType
  142       | explicitNamespacesEnabled || requiredTypeArgumentsEnabled -> explicitTypePatternParser
  143     TkQuasiQuote {} -> quasiQuotePatternParser
  144     TkTHSplice | thAny -> thSplicePatternParser
  145     TkKeywordUnderscore -> wildcardPatternParser
  146     TkInteger {} -> literalPatternParser
  147     TkFloat {} -> literalPatternParser
  148     TkChar {} -> literalPatternParser
  149     TkCharHash {} -> literalPatternParser
  150     TkString {} -> literalPatternParser
  151     TkStringHash {} -> literalPatternParser
  152     TkSpecialLBracket -> listPatternParser
  153     TkSpecialLParen -> parenOrTuplePatternParser
  154     TkSpecialUnboxedLParen -> parenOrTuplePatternParser
  155     _ -> varOrConPatternParser
  156 
  157 typeBinderPatternParser :: TokParser Pattern
  158 typeBinderPatternParser = withSpanAnn (PAnn . mkAnnotation) $ do
  159   expectedTok TkTypeApp
  160   PTypeBinder <$> visibleTypeBinderCoreParser

  162 explicitTypePatternParser :: TokParser Pattern
  163 explicitTypePatternParser = withSpanAnn (PAnn . mkAnnotation) $ do
  164   expectedTok TkKeywordType
  165   PTypeSyntax TypeSyntaxExplicitNamespace <$> typeParser
  166 
  167 strictPatternParser :: TokParser Pattern
  168 strictPatternParser = withSpanAnn (PAnn . mkAnnotation) $ do
  169   expectedTok TkPrefixBang
  170   PStrict <$> apatParser
  171 
  172 irrefutablePatternParser :: TokParser Pattern
  173 irrefutablePatternParser = withSpanAnn (PAnn . mkAnnotation) $ do
  174   expectedTok TkPrefixTilde
  175   PIrrefutable <$> apatParser

  177 negativeLiteralPatternParser :: TokParser Pattern
  178 negativeLiteralPatternParser = MP.try $ withSpanAnn (PAnn . mkAnnotation) $ do
  179   expectedTok (TkVarSym "-")
  180   PNegLit <$> numericLiteralParser

  182 -- | Parse only numeric literals (integer or float), used for negative literal
  183 -- patterns where GHC only allows @-@ before numeric literals, not strings or chars.
  184 numericLiteralParser :: TokParser Literal
  185 numericLiteralParser = intLiteralParser <|> floatLiteralParser

  187 wildcardPatternParser :: TokParser Pattern
  188 wildcardPatternParser =
  189   tokenSatisfy "wildcard" $ \tok ->
  190     case lexTokenKind tok of
  191       TkKeywordUnderscore -> Just (PAnn (mkAnnotation (lexTokenSpan tok)) PWildcard)
  192       _ -> Nothing

  194 literalPatternParser :: TokParser Pattern
  195 literalPatternParser =
  196   tokenSatisfy "literal" $ \tok -> do
  197     lit <- literalFromToken tok
  198     let ann = mkAnnotation (lexTokenSpan tok)
  199     pure (PAnn ann (PLit (LitAnn ann lit)))
  200 
  201 quasiQuotePatternParser :: TokParser Pattern
  202 quasiQuotePatternParser =
  203   tokenSatisfy "quasi quote" $ \tok ->
  204     case lexTokenKind tok of
  205       TkQuasiQuote quoter body -> Just (PAnn (mkAnnotation (lexTokenSpan tok)) (PQuasiQuote quoter body))
  206       _ -> Nothing

  208 literalParser :: TokParser Literal
  209 literalParser = intLiteralParser <|> floatLiteralParser <|> charLiteralParser <|> stringLiteralParser

  211 tokenLiteralParser :: String -> (LexToken -> Maybe Literal) -> TokParser Literal
  212 tokenLiteralParser expected matchToken =
  213   tokenSatisfy expected $ \tok ->
  214     LitAnn (mkAnnotation (lexTokenSpan tok)) <$> matchToken tok

  216 literalFromToken :: LexToken -> Maybe Literal
  217 literalFromToken tok =
  218   intLiteralFromToken tok

  220     <|> charLiteralFromToken tok
  221     <|> stringLiteralFromToken tok

  228 intLiteralFromToken tok =
  229   case lexTokenKind tok of

  238 floatLiteralFromToken tok =
  239   case lexTokenKind tok of

  259 stringLiteralFromToken tok =

  265 -- | Parse Template Haskell pattern splice: $pat or $(pat)
  266 thSplicePatternParser :: TokParser Pattern
  267 thSplicePatternParser = withSpanAnn (PAnn . mkAnnotation) $ do
  268   expectedTok TkTHSplice
  269   tok <- lookAhead anySingle
  270   case lexTokenKind tok of
  271     TkKeywordType -> MP.empty
  272     _ -> PSplice <$> atomExprParser

  274 visibleTypeBinderCoreParser :: TokParser TyVarBinder
  275 visibleTypeBinderCoreParser =
  276   withSpan $
  277     ( do
  278         ident <- tyVarNameParser
  279         pure (\span' -> TyVarBinder [mkAnnotation span'] ident Nothing TyVarBSpecified TyVarBInvisible)
  280     )
  281       <|> ( do
  282               expectedTok TkSpecialLParen
  283               ident <- tyVarNameParser
  284               expectedTok TkReservedDoubleColon
  285               kind <- typeParser
  286               expectedTok TkSpecialRParen
  287               pure (\span' -> TyVarBinder [mkAnnotation span'] ident (Just kind) TyVarBSpecified TyVarBInvisible)
  288           )

  290 varOrConPatternParser :: TokParser Pattern
  291 varOrConPatternParser = do
  292   (tok, name) <- identifierNameWithTokenParser
  293   mNextTok <- MP.optional (lookAhead anySingle)
  294   let ann = mkAnnotation (lexTokenSpan tok)
  295   case mNextTok of
  296     Just nextTok
  297       | isConLikeName name && lexTokenKind nextTok == TkSpecialLBrace -> do
  298           (fields, hasWildcard) <- braces recordPatternFieldListParser
  299           endInput <- MP.getInput
  300           let endSpan = maybe noSourceSpan lexTokenSpan (tokStreamPrevToken endInput)
  301               recordSpan = mergeSourceSpans (lexTokenSpan tok) endSpan
  302           pure (PAnn (mkAnnotation recordSpan) (PRecord name fields hasWildcard))
  303     _ ->
  304       pure $
  305         PAnn ann $
  306           if isConLikeName name
  307             then PCon name [] []
  308             else PVar (nameToUnqualified name)
  309 
  310 recordFieldPatternParser :: TokParser (RecordField Pattern)
  311 recordFieldPatternParser = do
  312   field <- recordFieldNameParser
  313   mEq <- MP.optional (expectedTok TkReservedEquals)
  314   case mEq of
  315     Just () -> do
  316       pat <- subpatternWithBareViewParser
  317       pure (RecordField field pat False)
  318     Nothing -> do
  319       -- NamedFieldPuns: just "field" means "field = field"
  320       pure (RecordField field (PVar (nameToUnqualified field)) True)

  322 -- | Parse the contents of record pattern braces, supporting RecordWildCards ".."
  323 recordPatternFieldListParser :: TokParser ([RecordField Pattern], Bool)
  324 recordPatternFieldListParser =
  325   recordFieldsWithWildcardsParser (recordFieldPatternParser `MP.sepEndBy` expectedTok TkSpecialComma)

  327 listPatternParser :: TokParser Pattern
  328 listPatternParser = withSpanAnn (PAnn . mkAnnotation) $ do
  329   expectedTok TkSpecialLBracket
  330   elems <- listPatternElementParser `MP.sepBy` expectedTok TkSpecialComma
  331   expectedTok TkSpecialRBracket
  332   pure (PList elems)
  333   where
  334     -- List elements can contain bare view patterns such as [id -> x].
  335     -- Try the expr -> pattern form first, then fall back to normal patterns.
  336     listPatternElementParser :: TokParser Pattern
  337     listPatternElementParser = subpatternWithBareViewParser
  338 
  339 -- | Parse a subpattern position that admits a bare view pattern @expr -> pat@.
  340 -- This is needed in delimited pattern contexts like list elements and record
  341 -- fields, where there is no surrounding pair of parens to disambiguate the
  342 -- view-pattern arrow from the enclosing syntax.
  343 --
  344 -- This parser is recursive so that deeply nested view patterns such as
  345 -- @expr1 -> expr2 -> pat@ are accepted without requiring explicit parentheses
  346 -- around each intermediate view pattern.
  347 subpatternWithBareViewParser :: TokParser Pattern
  348 subpatternWithBareViewParser = do
  349   mView <- MP.optional . MP.try $ do
  350     expr <- exprParser
  351     expectedTok TkReservedRightArrow
  352     PView expr <$> subpatternWithBareViewParser
  353   maybe patternParser pure mView
  354 
  355 parenOrTuplePatternParser :: TokParser Pattern
  356 parenOrTuplePatternParser = withSpanAnn (PAnn . mkAnnotation) $ do
  357   (tupleFlavor, closeTok) <- tupleDelimsParser
  358   mNextTok <- MP.optional (lookAhead anySingle)
  359   case fmap lexTokenKind mNextTok of
  360     Just nextKind
  361       | nextKind == closeTok -> unitPatternParser tupleFlavor closeTok
  362       | tupleFlavor == Unboxed && nextKind == TkReservedPipe -> parseUnboxedSumPatLeadingBars closeTok
  363     _ -> do
  364       -- For boxed parens, try parsing as a top-level view pattern first.
  365       -- View patterns like (expr -> pat) produce PParen(PView(...)) so that
  366       -- the explicit source parens are faithfully represented in the AST.
  367       mView <-
  368         if tupleFlavor == Boxed
  369           then viewPatternParser closeTok
  370           else pure Nothing
  371       case mView of
  372         Just mkView -> pure mkView
  373         Nothing -> tupleOrParenPatternParser tupleFlavor closeTok
  374   where
  375     -- When a bare operator like + or :+ appears directly inside parens, the
  376     -- parens serve as prefix notation (e.g. (+), (:+)), not grouping.
  377     -- prettyPrefixName already adds parens when printing symbolic names, so
  378     -- wrapping with PParen would produce double parens.
  379     --
  380     -- However, when the inner pattern came from expression reclassification
  381     -- (e.g. ((:+)) where the expression parser already consumed the inner
  382     -- prefix-notation parens), the outer parens ARE grouping and must produce
  383     -- PParen.
  384     --
  385     -- The @isBareOperator@ flag distinguishes these two cases:
  386     --   True  -> parens are prefix notation, strip PParen
  387     --   False -> parens are grouping, wrap with PParen
  388     parenOrSymConParser isBareOperator inner = do
  389       case peelPatternAnn inner of
  390         PVar name
  391           | isBareOperator,
  392             unqualifiedNameType name == NameVarSym ->
  393               pure inner
  394         PCon con [] []
  395           | isBareOperator,
  396             nameType con == NameConSym -> do
  397               mBrace <- MP.optional . lookAhead $ anySingle
  398               case fmap lexTokenKind mBrace of
  399                 Just TkSpecialLBrace -> do
  400                   (fields, hasWildcard) <- braces recordPatternFieldListParser
  401                   pure (PRecord con fields hasWildcard)
  402                 _ -> pure inner
  403         _ -> pure (PParen inner)
  404 
  405     unitPatternParser tupleFlavor closeTok = do
  406       expectedTok closeTok
  407       pure (PTuple tupleFlavor [])
  408 
  409     -- Try to parse the paren content as a view pattern: expr -> pat.
  410     -- Uses exprParser which stops before '->', then checks for the arrow.
  411     -- Returns Nothing if the content is not a view pattern.
  412     viewPatternParser :: LexTokenKind -> TokParser (Maybe Pattern)
  413     viewPatternParser closeTok = MP.optional . MP.try $ do
  414       expr <- exprParser <* lookAhead (expectedTok TkReservedRightArrow)
  415       expectedTok TkReservedRightArrow
  416       inner <- subpatternWithBareViewParser
  417       expectedTok closeTok
  418       pure (PParen (PView expr inner))
  419 
  420     -- Parse a single element inside a paren/tuple/unboxed-sum pattern.
  421     -- Uses "parse as expression, then reclassify" to avoid backtracking
  422     -- for the common case. Pattern-only prefixes (!, ~, @) are dispatched
  423     -- to patternParser directly. When exprParser fails (e.g., nested parens
  424     -- containing pattern-only syntax like as-patterns or view patterns),
  425     -- we fall back to patternParser.
  426     --
  427     -- Operator tokens (TkVarSym, TkConSym, etc.) are handled directly here
  428     -- because they are valid patterns (binding the operator as a variable or
  429     -- constructor) but may not parse as expressions on their own.
  430     --
  431     -- Returns (isBareOperator, pattern) where isBareOperator indicates the
  432     -- pattern was parsed as a bare operator token (e.g. :+ in (:+)), meaning
  433     -- the surrounding parens serve as prefix notation rather than grouping.
  434     parenPatElementParser :: TokParser (Bool, Pattern)
  435     parenPatElementParser = do
  436       tok <- lookAhead anySingle
  437       case lexTokenKind tok of
  438         TkPrefixBang -> (False,) <$> patternParser
  439         TkPrefixTilde -> (False,) <$> patternParser
  440         -- Operator tokens can be valid patterns when they appear alone in parentheses.
  441         -- Examples: (+) in "f (+) = ...", (??) in "foldl' (??) z xs = ..."
  442         -- We detect this by checking if an operator is followed by a closing delimiter.
  443         TkVarSym {} -> operatorOrExprPatternParser
  444         TkConSym {} -> operatorOrExprPatternParser
  445         TkQConSym {} -> operatorOrExprPatternParser
  446         TkReservedColon -> operatorOrExprPatternParser
  447         TkReservedAt -> operatorOrExprPatternParser
  448         _ -> do
  449           isAs <- startsWithAsPattern
  450           if isAs
  451             then (False,) <$> patternParser
  452             else (False,) <$> exprThenReclassify
  453       where
  454         -- Try to parse an operator as a pattern if it's alone (followed by closing delim),
  455         -- otherwise fall back to parsing as an expression.
  456         operatorOrExprPatternParser :: TokParser (Bool, Pattern)
  457         operatorOrExprPatternParser = do
  458           -- Look ahead to check what comes after the operator
  459           mNext <- MP.optional . lookAhead . MP.try $ do
  460             _ <- anySingle -- skip the operator token itself
  461             lookAhead anySingle
  462           case fmap lexTokenKind mNext of
  463             -- If followed by closing delimiters, parse as operator pattern
  464             Just TkSpecialRParen -> (True,) <$> operatorPatternParser
  465             Just TkSpecialUnboxedRParen -> (True,) <$> operatorPatternParser
  466             Just TkSpecialComma -> (True,) <$> operatorPatternParser
  467             Just TkReservedPipe -> (True,) <$> operatorPatternParser
  468             -- Otherwise, try parsing as expression (for cases like (x + y))
  469             _ -> (False,) <$> exprThenReclassify
  470 
  471         -- Parse an operator token as a variable or constructor pattern.
  472         operatorPatternParser :: TokParser Pattern
  473         operatorPatternParser = do
  474           tok' <- anySingle
  475           let ann = mkAnnotation (lexTokenSpan tok')
  476           case lexTokenKind tok' of
  477             TkVarSym op -> pure (PAnn ann (PVar (mkUnqualifiedNameAt tok' NameVarSym op)))
  478             TkConSym op -> pure (PAnn ann (PCon (qualifyName Nothing (mkUnqualifiedNameAt tok' NameConSym op)) [] []))
  479             TkQConSym modName op -> pure (PAnn ann (PCon (mkNameAt tok' (Just modName) NameConSym op) [] []))
  480             TkReservedColon -> pure (PAnn ann (PCon (qualifyName Nothing (mkUnqualifiedNameAt tok' NameConSym ":")) [] []))
  481             TkReservedAt -> pure (PAnn ann (PVar (mkUnqualifiedNameAt tok' NameVarSym "@")))
  482             _ ->
  483               MP.customFailure
  484                 UnexpectedTokenExpecting
  485                   { unexpectedFound = Just (mkFoundToken tok'),
  486                     unexpectedExpecting = "operator token",
  487                     unexpectedContext = []
  488                   }
  489 
  490     -- Try to parse as expression, then reclassify via checkPattern.
  491     -- When exprParser fails, does not consume the full element (e.g.,
  492     -- '@' from an as-pattern), or checkPattern rejects it (e.g., variable
  493     -- operator in infix position), fall back to patternParser.
  494     --
  495     -- View patterns within tuple elements are also handled here: if '->'
  496     -- follows the parsed expression, it is a view pattern.
  497     exprThenReclassify :: TokParser Pattern
  498     exprThenReclassify = do
  499       mResult <- MP.optional . MP.try $ do
  500         expr <- exprParser
  501         -- Verify the expression consumed the full element: the next token
  502         -- must be a valid delimiter in paren/tuple/sum context. If not
  503         -- (e.g., '@' from an as-pattern), the expression parser stopped
  504         -- too early and we should backtrack to patternParser.
  505         tok <- lookAhead anySingle
  506         case lexTokenKind tok of
  507           TkReservedRightArrow -> pure (Left expr) -- view pattern: defer arrow handling
  508           TkSpecialComma -> Right <$> liftCheck (checkPattern expr)
  509           TkSpecialRParen -> Right <$> liftCheck (checkPattern expr)
  510           TkSpecialUnboxedRParen -> Right <$> liftCheck (checkPattern expr)
  511           TkReservedPipe -> Right <$> liftCheck (checkPattern expr)
  512           _ -> fail "incomplete element parse"
  513       case mResult of
  514         Just (Left expr) -> do
  515           -- View pattern: expr -> pattern
  516           expectedTok TkReservedRightArrow
  517           PView expr <$> subpatternWithBareViewParser
  518         Just (Right pat) ->
  519           pure pat
  520         Nothing ->
  521           patternParser
  522 
  523     tupleOrParenPatternParser tupleFlavor closeTok = do
  524       (isBareOp, first) <- parenPatElementParser
  525       mComma <- MP.optional (expectedTok TkSpecialComma)
  526       case mComma of
  527         Nothing -> do
  528           -- Check for pipe (unboxed sum: pattern in first slot)
  529           mPipe <- if tupleFlavor == Unboxed then MP.optional (expectedTok TkReservedPipe) else pure Nothing
  530           case mPipe of
  531             Just () -> do
  532               -- (# pat | ... #) - pattern in first slot of sum
  533               trailingBars <- MP.many (expectedTok TkReservedPipe)
  534               expectedTok closeTok
  535               let arity = 2 + length trailingBars
  536               pure (PUnboxedSum 0 arity first)
  537             Nothing -> do
  538               expectedTok closeTok
  539               if tupleFlavor == Boxed
  540                 then parenOrSymConParser isBareOp first
  541                 else pure (PTuple Unboxed [first])
  542         Just () -> do
  543           (_, second) <- parenPatElementParser
  544           more <- MP.many (expectedTok TkSpecialComma *> (snd <$> parenPatElementParser))
  545           expectedTok closeTok
  546           pure (PTuple tupleFlavor (first : second : more))
  547 
  548     parseUnboxedSumPatLeadingBars closeTok = do
  549       -- Parse (# | | ... | pat | ... | #) where pattern is not in first slot
  550       _ <- expectedTok TkReservedPipe
  551       leadingBars <- MP.many (MP.try (expectedTok TkReservedPipe))
  552       let altIdx = 1 + length leadingBars
  553       (_, inner) <- parenPatElementParser
  554       trailingBars <- MP.many (expectedTok TkReservedPipe)
  555       expectedTok closeTok
  556       let arity = altIdx + 1 + length trailingBars
  557       pure (PUnboxedSum altIdx arity inner)

  559 isPatternAppHead :: Pattern -> Bool
  560 isPatternAppHead pat =
  561   case peelPatternAnn pat of
  562     PCon {} -> True
  563     PVar name -> isConLikeNameType (unqualifiedNameType name)
  564     _ -> False
