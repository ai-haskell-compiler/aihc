   14 ordinaryLayout :: ImplicitLayoutSpec
   15 ordinaryLayout =
   16   ImplicitLayoutSpec
   17     { implicitLayoutSemicolons = LayoutEmitSemicolons,
   18       implicitLayoutIndentPolicy = LayoutAllowNondecreasingIndent,
   19       implicitLayoutBaseline = LayoutCurrentIndent,
   20       implicitLayoutChildBaseline = LayoutCurrentIndent,

   23     }

   53 applyLayoutTokens :: Bool -> [Extension] -> [LexToken] -> [LexToken]
   54 applyLayoutTokens enableModuleLayout exts =
   55   go (mkInitialLayoutState enableModuleLayout exts)
   56   where
   57     go st toks =
   58       case toks of
   59         [] ->
   60           let eofAnchor = NoSourceSpan
   61               (moduleInserted, stAfterModule) = finalizeModuleLayoutAtEOF st eofAnchor
   62               (pendingInserted, stAfterPending) = flushPendingCaseLayoutAtEOF stAfterModule eofAnchor
   63            in moduleInserted <> pendingInserted <> closeAllImplicit (layoutContexts stAfterPending) eofAnchor
   64         tok : rest ->
   65           let (emitted, stNext) = layoutTransition st tok
   66            in emitted <> go stNext rest
   67 
   68 finalizeModuleLayoutAtEOF :: LayoutState -> SourceSpan -> ([LexToken], LayoutState)
   69 finalizeModuleLayoutAtEOF st anchor =
   70   case layoutModuleMode st of
   71     mode
   72       | mode == ModuleLayoutSeekStart || mode == ModuleLayoutAwaitBody ->
   73           ( [virtualSymbolToken "{" anchor, virtualSymbolToken "}" anchor],
   74             st {layoutModuleMode = ModuleLayoutDone, layoutPendingLayout = Nothing}
   75           )
   76     _ -> ([], st)
   77 
   78 {-# INLINE noteModuleLayoutBeforeToken #-}
   79 noteModuleLayoutBeforeToken :: LayoutState -> LexToken -> LayoutState
   80 noteModuleLayoutBeforeToken st tok =
   81   case layoutModuleMode st of
   82     ModuleLayoutAwaitBody -> st {layoutModuleMode = ModuleLayoutDone}
   83     ModuleLayoutSeekStart ->
   84       case lexTokenKind tok of
   85         TkPragma _ -> st
   86         TkLineComment -> st
   87         TkBlockComment -> st
   88         TkKeywordModule -> st {layoutModuleMode = ModuleLayoutAwaitWhere}
   89         _ -> st {layoutModuleMode = ModuleLayoutDone, layoutPendingLayout = Just (PendingImplicitLayout ordinaryLayout)}
   90     _ -> st
   91 
   92 {-# INLINE noteModuleLayoutAfterToken #-}
   93 noteModuleLayoutAfterToken :: LayoutState -> LexToken -> LayoutState
   94 noteModuleLayoutAfterToken st tok =
   95   case layoutModuleMode st of
   96     ModuleLayoutAwaitWhere
   97       | lexTokenKind tok == TkKeywordWhere ->
   98           st {layoutModuleMode = ModuleLayoutAwaitBody, layoutPendingLayout = Just (PendingImplicitLayout ordinaryLayout)}
   99     _ -> st

  101 {-# INLINE openPendingLayout #-}
  102 openPendingLayout :: LayoutState -> LexToken -> ([LexToken], LayoutState, Bool)
  103 openPendingLayout st tok =
  104   case layoutPendingLayout st of
  105     Nothing -> ([], st, False)
  106     Just pending ->
  107       case pending of
  108         PendingMaybeMultiWayIf ->
  109           case lexTokenKind tok of
  110             TkReservedPipe -> openImplicitLayout multiWayIfLayout st tok
  111             _ -> ([], noteClassicIfInAfterThenElse (st {layoutPendingLayout = Nothing}), False)
  112         PendingMaybeLambdaCases ->
  113           case lexTokenKind tok of
  114             TkSpecialLBrace -> ([], st {layoutPendingLayout = Nothing}, False)
  115             tkKind
  116               | closesDelimiter tkKind ->
  117                   let anchor = lexTokenSpan tok
  118                       openTok = virtualSymbolToken "{" anchor
  119                       closeTok = virtualSymbolToken "}" anchor
  120                    in ([openTok, closeTok], st {layoutPendingLayout = Nothing}, False)
  121             _ -> openImplicitLayout (caseAlternativeLayout st) st tok
  122         PendingImplicitLayout spec ->
  123           case lexTokenKind tok of
  124             TkSpecialLBrace -> ([], st {layoutPendingLayout = Nothing}, False)
  125             tkKind
  126               | closesDelimiter tkKind ->
  127                   -- Closing delimiter immediately after layout-opener (e.g. [d| |])
  128                   -- produces empty implicit layout {}, then the closer proceeds normally.
  129                   let anchor = lexTokenSpan tok
  130                       openTok = virtualSymbolToken "{" anchor
  131                       closeTok = virtualSymbolToken "}" anchor
  132                    in ([openTok, closeTok], st {layoutPendingLayout = Nothing}, False)
  133             _ -> openImplicitLayout spec st tok
  134 
  135 openImplicitLayout :: ImplicitLayoutSpec -> LayoutState -> LexToken -> ([LexToken], LayoutState, Bool)
  136 openImplicitLayout spec st tok =
  137   let col = tokenStartCol tok
  138       parentIndent =
  139         case implicitLayoutBaseline spec of
  140           LayoutCurrentIndent -> currentLayoutIndent (layoutContexts st)
  141           LayoutColumnZero -> 0
  142       openTok = virtualSymbolToken "{" (lexTokenSpan tok)
  143       closeTok = virtualSymbolToken "}" (lexTokenSpan tok)
  144       newContext =
  145         LayoutImplicit
  146           LayoutFrame
  147             { layoutFrameIndent = col,
  148               layoutFrameSemicolons = implicitLayoutSemicolons spec,
  149               layoutFrameChildBaseline = implicitLayoutChildBaseline spec,
  150               layoutFrameThenElseDepth = implicitLayoutThenElseDepth spec
  151             }
  152       opensEmpty =
  153         case implicitLayoutIndentPolicy spec of
  154           LayoutAllowNondecreasingIndent
  155             | layoutNondecreasingIndent st -> col < parentIndent
  156           _ -> col <= parentIndent
  157    in if opensEmpty
  158         then ([openTok, closeTok], st {layoutPendingLayout = Nothing}, False)
  159         else
  160           ( [openTok],
  161             st
  162               { layoutPendingLayout = Nothing,
  163                 layoutContexts = newContext : layoutContexts st
  164               },
  165             True
  166           )

  168 {-# INLINE closeBeforeToken #-}
  169 closeBeforeToken :: LayoutState -> LexToken -> ([LexToken], LayoutState)
  170 closeBeforeToken st tok =
  171   case lexTokenKind tok of
  172     kind
  173       | closesImplicitBeforeDelimiter kind ->
  174           let (pendingInserted, st0) = flushPendingImplicitLayout st anchor
  175               (inserted, ctxs') = closeImplicitLayouts anchor (\_ _ -> True) (layoutContexts st0)
  176            in (pendingInserted <> inserted, st0 {layoutContexts = ctxs'})
  177       | closesImplicitBeforeLayoutKeyword kind ->
  178           closeBeforeLayoutKeyword
  179     _ -> ([], st)
  180   where
  181     anchor = lexTokenSpan tok
  182 
  183     closeBeforeLayoutKeyword =
  184       let col = tokenStartCol tok
  185           (pendingInserted, st0) = flushPendingImplicitLayout st anchor
  186           (inserted, ctxs') = closeImplicitLayouts anchor (shouldClose col) (layoutContexts st0)
  187        in (pendingInserted <> inserted, st0 {layoutContexts = ctxs'})
  188 
  189     shouldClose col indent kind =
  190       col < indent || (col == indent && closesSameColumnLayout kind)
  191 
  192     closesSameColumnLayout kind =
  193       case (lexTokenKind tok, kind) of
  194         (TkKeywordThen, LayoutFrame {layoutFrameThenElseDepth = Just nestedIfs}) -> nestedIfs == 0
  195         (TkKeywordElse, LayoutFrame {layoutFrameThenElseDepth = Just nestedIfs}) -> nestedIfs == 0
  196         (TkKeywordThen, _) -> False
  197         (TkKeywordElse, _) -> False
  198         _ -> True
  199 
  200 flushPendingImplicitLayout :: LayoutState -> SourceSpan -> ([LexToken], LayoutState)
  201 flushPendingImplicitLayout st anchor =
  202   case layoutPendingLayout st of
  203     Just (PendingImplicitLayout _) ->
  204       ( [virtualSymbolToken "{" anchor, virtualSymbolToken "}" anchor],
  205         st {layoutPendingLayout = Nothing}
  206       )
  207     _ -> ([], st)
  208 
  209 flushPendingCaseLayoutAtEOF :: LayoutState -> SourceSpan -> ([LexToken], LayoutState)
  210 flushPendingCaseLayoutAtEOF st anchor =
  211   case layoutPendingLayout st of
  212     Just (PendingImplicitLayout spec)
  213       | implicitLayoutFlushEmptyAtEOF spec ->
  214           ( [virtualSymbolToken "{" anchor, virtualSymbolToken "}" anchor],
  215             st {layoutPendingLayout = Nothing}
  216           )
  217     Just PendingMaybeLambdaCases ->
  218       ( [virtualSymbolToken "{" anchor, virtualSymbolToken "}" anchor],
  219         st {layoutPendingLayout = Nothing}
  220       )
  221     _ -> ([], st)

  223 {-# INLINE bolLayout #-}
  224 bolLayout :: LayoutState -> LexToken -> ([LexToken], LayoutState)
  225 bolLayout st tok
  226   | not (isBOL st tok) = ([], st)
  227   | otherwise =
  228       let col = tokenStartCol tok
  229           (inserted, contexts') = closeImplicitLayouts (lexTokenSpan tok) (\indent _ -> col < indent) (layoutContexts st)
  230           semiAnchor = fromMaybe (lexTokenSpan tok) (layoutPrevTokenEndSpan st)
  231           eqSemi =
  232             case currentLayoutIndentMaybe contexts' of
  233               Just indent
  234                 | col == indent,
  235                   currentLayoutAllowsSemicolon contexts',
  236                   lexTokenKind tok /= TkKeywordWhere ->
  237                     [virtualSymbolToken ";" semiAnchor]
  238               _ -> []
  239        in (inserted <> eqSemi, st {layoutContexts = contexts'})
  240 
  241 currentLayoutAllowsSemicolon :: [LayoutContext] -> Bool
  242 currentLayoutAllowsSemicolon contexts =
  243   case contexts of
  244     LayoutImplicit LayoutFrame {layoutFrameSemicolons = LayoutSuppressSemicolons} : _ -> False
  245     LayoutImplicit _ : _ -> True
  246     _ -> False

  248 closeImplicitLayouts :: SourceSpan -> (Int -> LayoutFrame -> Bool) -> [LayoutContext] -> ([LexToken], [LayoutContext])
  249 closeImplicitLayouts anchor shouldClose = go []
  250   where
  251     closeTok = virtualSymbolToken "}" anchor
  252 
  253     go acc contexts =
  254       case contexts of
  255         LayoutImplicit frame : rest
  256           | shouldClose (layoutFrameIndent frame) frame -> go (closeTok : acc) rest
  257         _ -> (reverse acc, contexts)
  258 
  259 closeAllImplicit :: [LayoutContext] -> SourceSpan -> [LexToken]
  260 closeAllImplicit contexts anchor =
  261   [virtualSymbolToken "}" anchor | ctx <- contexts, isImplicitLayoutContext ctx]

  263 {-# INLINE stepTokenContext #-}
  264 stepTokenContext :: LayoutState -> LexToken -> LayoutState
  265 stepTokenContext st tok =
  266   case lexTokenKind tok of
  267     TkKeywordDo
  268       | layoutPrevTokenKind st == Just TkKeywordThen
  269           || layoutPrevTokenKind st == Just TkKeywordElse ->
  270           st {layoutPendingLayout = Just (PendingImplicitLayout thenElseDoLayout)}
  271       | otherwise -> st {layoutPendingLayout = Just (PendingImplicitLayout ordinaryLayout)}
  272     TkKeywordMdo -> st {layoutPendingLayout = Just (PendingImplicitLayout ordinaryLayout)}
  273     TkQualifiedDo {}
  274       | layoutPrevTokenKind st == Just TkKeywordThen
  275           || layoutPrevTokenKind st == Just TkKeywordElse ->
  276           st {layoutPendingLayout = Just (PendingImplicitLayout thenElseDoLayout)}
  277       | otherwise -> st {layoutPendingLayout = Just (PendingImplicitLayout ordinaryLayout)}
  278     TkQualifiedMdo {} -> st {layoutPendingLayout = Just (PendingImplicitLayout ordinaryLayout)}
  279     TkKeywordOf -> st {layoutPendingLayout = Just (PendingImplicitLayout (caseAlternativeLayout st))}
  280     TkKeywordCase
  281       | layoutPrevTokenKind st == Just TkReservedBackslash ->
  282           st {layoutPendingLayout = Just (PendingImplicitLayout (caseAlternativeLayout st))}
  283       | otherwise -> st
  284     TkVarId "cases"
  285       | layoutLambdaCase st && layoutPrevTokenKind st == Just TkReservedBackslash ->
  286           st {layoutPendingLayout = Just PendingMaybeLambdaCases}
  287       | otherwise -> st
  288     TkKeywordLet -> st {layoutPendingLayout = Just (PendingImplicitLayout strictLayout)}
  289     TkKeywordRec -> st {layoutPendingLayout = Just (PendingImplicitLayout ordinaryLayout)}
  290     TkKeywordWhere -> st {layoutPendingLayout = Just (PendingImplicitLayout strictLayout)}
  291     TkKeywordElse -> decrementAfterThenElseClassicIfDepth st
  292     TkKeywordIf -> st {layoutPendingLayout = Just PendingMaybeMultiWayIf}
  293     TkTHDeclQuoteOpen ->
  294       st
  295         { layoutContexts = LayoutDelimiter : layoutContexts st,
  296           layoutPendingLayout = Just (PendingImplicitLayout thDeclQuoteLayout)
  297         }
  298     kind
  299       | opensDelimiter kind ->
  300           st {layoutContexts = LayoutDelimiter : layoutContexts st}
  301     kind
  302       | closesDelimiter kind ->
  303           st {layoutContexts = popToDelimiter (layoutContexts st)}
  304     TkSpecialLBrace -> st {layoutContexts = LayoutExplicit : layoutContexts st}
  305     TkSpecialRBrace -> st {layoutContexts = popOneContext (layoutContexts st)}
  306     _ -> st

  308 mapAfterThenElse :: (Int -> Int) -> [LayoutContext] -> [LayoutContext]
  309 mapAfterThenElse f = go
  310   where
  311     go contexts =
  312       case contexts of
  313         LayoutImplicit frame@LayoutFrame {layoutFrameThenElseDepth = Just nestedIfs} : rest ->
  314           LayoutImplicit frame {layoutFrameThenElseDepth = Just (f nestedIfs)} : rest
  315         ctx : rest -> ctx : go rest
  316         [] -> []
  317 
  318 noteClassicIfInAfterThenElse :: LayoutState -> LayoutState
  319 noteClassicIfInAfterThenElse st = st {layoutContexts = mapAfterThenElse (+ 1) (layoutContexts st)}
  320 
  321 decrementAfterThenElseClassicIfDepth :: LayoutState -> LayoutState
  322 decrementAfterThenElseClassicIfDepth st = st {layoutContexts = mapAfterThenElse (max 0 . subtract 1) (layoutContexts st)}
  323 
  324 popToDelimiter :: [LayoutContext] -> [LayoutContext]
  325 popToDelimiter contexts =
  326   case contexts of
  327     LayoutDelimiter : rest -> rest
  328     _ : rest -> popToDelimiter rest

  352 isImplicitLayoutContext :: LayoutContext -> Bool
  353 isImplicitLayoutContext ctx =
  354   case ctx of
  355     LayoutImplicit _ -> True
  356     LayoutExplicit -> False
  357     LayoutDelimiter -> False

  359 -- | Tokens that open a delimiter context (parens, brackets, TH quotes,
  360 -- unboxed parens).  These push 'LayoutDelimiter' to suppress implicit-layout
  361 -- closures inside the delimited group.
  362 --
  363 -- Note: 'TkTHDeclQuoteOpen' is intentionally absent here because it is handled
  364 -- explicitly in 'stepTokenContext', where it both pushes 'LayoutDelimiter' and
  365 -- sets up a pending implicit layout for the declaration splice body.
  366 opensDelimiter :: LexTokenKind -> Bool
  367 opensDelimiter kind =
  368   case kind of
  369     TkSpecialLParen -> True
  370     TkSpecialLBracket -> True
  371     TkTHExpQuoteOpen -> True
  372     TkTHTypedQuoteOpen -> True
  373     TkTHTypeQuoteOpen -> True
  374     TkTHPatQuoteOpen -> True
  375     TkSpecialUnboxedLParen -> True
  376     _ -> False
  377 
  378 -- | Tokens that close a delimiter context (parens, brackets, TH quotes,
  379 -- unboxed parens).  These pop the context stack back to the matching
  380 -- 'LayoutDelimiter' via 'popToDelimiter' in 'stepTokenContext'.
  381 --
  382 -- Related: 'closesImplicitBeforeDelimiter' determines which closing tokens
  383 -- also force all intervening implicit layouts to emit virtual @}@ tokens
  384 -- before the delimiter is popped.  'TkSpecialRBrace' is present there (but
  385 -- absent here) because it closes 'LayoutExplicit', not 'LayoutDelimiter'.
  386 closesDelimiter :: LexTokenKind -> Bool
  387 closesDelimiter kind =
  388   case kind of
  389     TkSpecialRParen -> True
  390     TkSpecialUnboxedRParen -> True
  391     TkSpecialRBracket -> True
  392     TkTHExpQuoteClose -> True
  393     TkTHTypedQuoteClose -> True
  394     _ -> False
  395 
  396 -- | Tokens before which all intervening implicit layouts must be closed
  397 -- (emitting virtual @}@ tokens).  See 'closesDelimiter' for the relationship
  398 -- between these two predicates.
  399 closesImplicitBeforeDelimiter :: LexTokenKind -> Bool
  400 closesImplicitBeforeDelimiter kind =
  401   case kind of
  402     TkSpecialRParen -> True
  403     TkSpecialUnboxedRParen -> True
  404     TkSpecialRBracket -> True
  405     TkTHExpQuoteClose -> True
  406     TkTHTypedQuoteClose -> True
  407     TkSpecialRBrace -> True
  408     _ -> False
  409 
  410 -- | Reserved words that cannot start an item in the currently open layout list.
  411 -- At the same column as that layout they trigger the report's parse-error rule:
  412 -- inserting @}@ makes the keyword legal where inserting @;@ would not.
  413 closesImplicitBeforeLayoutKeyword :: LexTokenKind -> Bool
  414 closesImplicitBeforeLayoutKeyword kind =
  415   case kind of
  416     TkKeywordWhere -> True
  417     TkKeywordIn -> True
  418     TkKeywordThen -> True
  419     TkKeywordElse -> True
  420     _ -> False

  422 isBOL :: LayoutState -> LexToken -> Bool
  423 isBOL _ = lexTokenAtLineStart

  425 layoutTransition :: LayoutState -> LexToken -> ([LexToken], LayoutState)
  426 layoutTransition st tok =
  427   case lexTokenKind tok of
  428     TkLineComment -> ([tok], st)
  429     TkBlockComment -> ([tok], st)
  430     TkEOF ->
  431       let eofAnchor = fromMaybe (lexTokenSpan tok) (layoutPrevTokenEndSpan st)
  432           (moduleInserted, stAfterModule) = finalizeModuleLayoutAtEOF st eofAnchor
  433           (pendingInserted, stAfterPending) = flushPendingCaseLayoutAtEOF stAfterModule eofAnchor
  434        in ( moduleInserted <> pendingInserted <> closeAllImplicit (layoutContexts stAfterPending) eofAnchor <> [tok],
  435             stAfterPending {layoutContexts = [], layoutBuffer = []}
  436           )
  437     _ ->
  438       let stModule = noteModuleLayoutBeforeToken st tok
  439           (preInserted, stBeforePending) = closeBeforeToken stModule tok
  440           (pendingInserted, stAfterPending, skipBOL) = openPendingLayout stBeforePending tok
  441           (bolInserted, stAfterBOL) = if skipBOL then ([], stAfterPending) else bolLayout stAfterPending tok
  442           stAfterToken = noteModuleLayoutAfterToken (stepTokenContext stAfterBOL tok) tok
  443           newEndSpan =
  444             if lexTokenOrigin tok == FromSource
  445               then Just (lexTokenSpan tok)
  446               else layoutPrevTokenEndSpan stAfterToken
  447           stNext =
  448             stAfterToken
  449               { layoutPrevTokenKind = Just (lexTokenKind tok),
  450                 layoutPrevTokenEndSpan = newEndSpan,
  451                 layoutBuffer = []
  452               }
  453        in (preInserted <> pendingInserted <> bolInserted <> [tok], stNext)
  454 {-# INLINE layoutTransition #-}

  456 closeImplicitLayoutContext :: LayoutState -> Maybe LayoutState
  457 closeImplicitLayoutContext st =
  458   case layoutContexts st of
  459     LayoutImplicit _ : rest -> Just (closeWith rest)
  460     _ -> Nothing
  461   where
  462     anchor = fromMaybe noSpan (layoutPrevTokenEndSpan st)
  463     noSpan =
  464       SourceSpan
  465         { sourceSpanSourceName = "",
  466           sourceSpanStartLine = 0,
  467           sourceSpanStartCol = 0,
  468           sourceSpanEndLine = 0,
  469           sourceSpanEndCol = 0,
  470           sourceSpanStartOffset = 0,
  471           sourceSpanEndOffset = 0
  472         }
  473     closeWith rest =
  474       st
  475         { layoutContexts = rest,
  476           layoutBuffer = virtualSymbolToken "}" anchor : layoutBuffer st
  477         }
