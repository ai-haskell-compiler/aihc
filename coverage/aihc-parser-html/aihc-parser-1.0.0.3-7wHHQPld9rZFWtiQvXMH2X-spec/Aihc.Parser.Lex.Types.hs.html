  320 mkInitialLexerState :: FilePath -> [Extension] -> Text -> (LexerEnv, LexerState)
  321 mkInitialLexerState sourceName exts input =
  322   ( mkLexerEnv exts,
  323     LexerState
  324       { lexerInput = input,
  325         lexerLogicalSourceName = sourceName,
  326         lexerLine = 1,
  327         lexerCol = 1,
  328         lexerByteOffset = 0,
  329         lexerAtLineStart = True,
  330         lexerPrevTokenKind = Nothing,
  331         lexerHadTrivia = True
  332       }
  333   )

  335 mkInitialLayoutState :: Bool -> [Extension] -> LayoutState
  336 mkInitialLayoutState enableModuleLayout exts =
  337   LayoutState
  338     { layoutContexts = [],
  339       layoutPendingLayout = Nothing,
  340       layoutPrevTokenKind = Nothing,
  341       layoutModuleMode =
  342         if enableModuleLayout
  343           then ModuleLayoutSeekStart
  344           else ModuleLayoutOff,
  345       layoutPrevTokenEndSpan = Nothing,
  346       layoutBuffer = [],
  347       layoutNondecreasingIndent = Syntax.NondecreasingIndentation `elem` exts,
  348       layoutLambdaCase = Syntax.LambdaCase `elem` exts
  349     }

  364 mkSpan :: LexerState -> LexerState -> SourceSpan
  365 mkSpan start end =
  366   SourceSpan
  367     { sourceSpanSourceName = lexerLogicalSourceName start,
  368       sourceSpanStartLine = lexerLine start,
  369       sourceSpanStartCol = lexerCol start,
  370       sourceSpanEndLine = lexerLine end,
  371       sourceSpanEndCol = lexerCol end,
  372       sourceSpanStartOffset = lexerByteOffset start,
  373       sourceSpanEndOffset = lexerByteOffset end
  374     }

  376 advanceChars :: Text -> LexerState -> LexerState
  377 advanceChars consumed st =
  378   let !n = T.length consumed
  379       go (!line, !col, !byteOff, !atLineStart) ch =
  380         case ch of
  381           '\n' -> (line + 1, 1, byteOff + 1, True)
  382           '\t' ->
  383             let nextTabStop = 8 - ((col - 1) `mod` 8)
  384              in (line, col + nextTabStop, byteOff + 1, atLineStart)
  385           ' ' -> (line, col + 1, byteOff + 1, atLineStart)
  386           _
  387             | isSpace ch -> (line, col + 1, byteOff + utf8CharWidth ch, atLineStart)
  388             | otherwise -> (line, col + 1, byteOff + utf8CharWidth ch, False)
  389       (!finalLine, !finalCol, !finalByteOff, !finalAtLineStart) =
  390         T.foldl' go (lexerLine st, lexerCol st, lexerByteOffset st, lexerAtLineStart st) consumed
  391    in st
  392         { lexerInput = T.drop n (lexerInput st),
  393           lexerLine = finalLine,
  394           lexerCol = finalCol,
  395           lexerByteOffset = finalByteOff,
  396           lexerAtLineStart = finalAtLineStart
  397         }
  398 
  399 advanceN :: Int -> LexerState -> LexerState
  400 advanceN n st = advanceChars (T.take n (lexerInput st)) st

  402 consumeWhile :: (Char -> Bool) -> LexerState -> LexerState
  403 consumeWhile f st =
  404   let consumed = T.takeWhile f (lexerInput st)
  405    in advanceChars consumed st

  407 tokenStartCol :: LexToken -> Int
  408 tokenStartCol tok =
  409   case lexTokenSpan tok of
  410     SourceSpan {sourceSpanStartCol = col} -> col
  411     NoSourceSpan -> 1

  413 virtualSymbolToken :: Text -> SourceSpan -> LexToken
  414 virtualSymbolToken sym span' =
  415   LexToken
  416     { lexTokenKind = case sym of
  417         "{" -> TkSpecialLBrace
  418         "}" -> TkSpecialRBrace
  419         ";" -> TkSpecialSemicolon
  420         _ -> error ("virtualSymbolToken: unknown symbol: " ++ show sym),
  421       lexTokenText = sym,
  422       lexTokenSpan = span',
  423       lexTokenOrigin = InsertedLayout,
  424       lexTokenAtLineStart = False
  425     }

  427 utf8CharWidth :: Char -> Int
  428 utf8CharWidth ch =
  429   case ord ch of
  430     code
  431       | code <= 0x7F -> 1
  432       | code <= 0x7FF -> 2
  433       | code <= 0xFFFF -> 3
  434       | otherwise -> 4

  436 isSymbolicOpChar :: Char -> Bool
  437 isSymbolicOpChar c = c `elem` (":!#$%&*+./<=>?@\\^|-~" :: String) || isUnicodeSymbol c

  439 isUnicodeSymbol :: Char -> Bool
  440 isUnicodeSymbol c =
  441   isUnicodeSymbolCategory c
  442     || c == '∷'
  443     || c == '⇒'
  444     || c == '→'
  445     || c == '←'
  446     || c == '∀'
  447     || c == '⤙'
  448     || c == '⤚'
  449     || c == '⤛'
  450     || c == '⤜'
  451     || c == '⦇'
  452     || c == '⦈'
  453     || c == '⟦'
  454     || c == '⟧'
  455     || c == '⊸'

  457 isUnicodeSymbolCategory :: Char -> Bool
  458 isUnicodeSymbolCategory c =
  459   case generalCategory c of
  460     MathSymbol -> True
  461     CurrencySymbol -> True
  462     ModifierSymbol -> not (isAscii c)
  463     OtherSymbol -> True
  464     ConnectorPunctuation -> not (isAscii c)
  465     DashPunctuation -> not (isAscii c)
  466     OtherPunctuation -> not (isAscii c)
  467     _ -> False
