  346 mkExtensionSet :: [Extension] -> ExtensionSet
  347 mkExtensionSet = List.foldl' insertExtension (ExtensionSet 0 0 0)
  348   where
  349     insertExtension (ExtensionSet low middle high) ext =
  350       case fromEnum ext `quotRem` 64 of
  351         (0, bit) -> ExtensionSet (setBit low bit) middle high
  352         (1, bit) -> ExtensionSet low (setBit middle bit) high
  353         (2, bit) -> ExtensionSet low middle (setBit high bit)
  354         _ -> error "mkExtensionSet: extension enum exceeds bitset capacity"

  356 memberExtension :: Extension -> ExtensionSet -> Bool
  357 memberExtension ext (ExtensionSet low middle high) =
  358   case fromEnum ext `quotRem` 64 of
  359     (0, bit) -> testBit low bit
  360     (1, bit) -> testBit middle bit
  361     (2, bit) -> testBit high bit
  362     _ -> False

  411 parseExtensionName :: Text -> Maybe Extension
  412 parseExtensionName raw =
  413   Map.lookup trimmed extensionMap
  414   where
  415     extensionMap :: Map.Map Text Extension
  416     extensionMap = Map.fromList $ aliases ++ [(T.pack (show ext), ext) | ext <- [minBound .. maxBound]]
  417     trimmed = T.strip raw
  418     aliases =
  419       [ ("Cpp", CPP),
  420         ("DerivingVia", DerivingViaExtension),
  421         ("GeneralisedNewtypeDeriving", GeneralizedNewtypeDeriving),
  422         ("Safe", SafeHaskell),
  423         ("Unsafe", UnsafeHaskell)
  424       ]

  426 parseExtensionSettingName :: Text -> Maybe ExtensionSetting
  427 parseExtensionSettingName raw =
  428   case T.stripPrefix (T.pack "No") trimmed of
  429     Just rest
  430       | not (T.null rest) ->
  431           case parseExtensionName rest of
  432             Just ext -> Just (DisableExtension ext)
  433             Nothing -> EnableExtension <$> parseExtensionName trimmed
  434     _ -> EnableExtension <$> parseExtensionName trimmed
  435   where
  436     trimmed = T.strip raw

  647 applyExtensionSetting :: ExtensionSetting -> [Extension] -> [Extension]

  649   case setting of

  653 applyImpliedExtensions :: [Extension] -> [Extension]
  654 applyImpliedExtensions extensions =
  655   let settings = concat $ mapMaybe (`lookup` impliedExtensions) extensions
  656       newExtensions = foldr applyExtensionSetting extensions settings
  657    in if sort newExtensions == sort extensions then extensions else applyImpliedExtensions newExtensions

  664 -- | Source location metadata for parsed syntax.
  665 -- Example: the span covering @map@ in @map f xs@.
  666 data SourceSpan
  667   = -- | No location information is available.
  668     NoSourceSpan
  669   | -- | A concrete span such as the token range for @map@ in @map f xs@.
  670     SourceSpan
  671       { sourceSpanSourceName :: !FilePath,
  672         sourceSpanStartLine :: !Int,
  673         sourceSpanStartCol :: !Int,
  674         sourceSpanEndLine :: !Int,
  675         sourceSpanEndCol :: !Int,
  676         sourceSpanStartOffset :: !Int,
  677         sourceSpanEndOffset :: !Int
  678       }
  679   deriving (Data, Eq, Ord, Generic, NFData)
  680 
  681 instance Show SourceSpan where
  682   show NoSourceSpan = "NoSourceSpan"
  683   show SourceSpan {sourceSpanStartLine, sourceSpanStartCol, sourceSpanEndLine, sourceSpanEndCol} =
  684     "SourceSpan "
  685       ++ show sourceSpanStartLine
  686       ++ " "
  687       ++ show sourceSpanStartCol
  688       ++ " "
  689       ++ show sourceSpanEndLine
  690       ++ " "
  691       ++ show sourceSpanEndCol

  693 noSourceSpan :: SourceSpan
  694 noSourceSpan = NoSourceSpan

  696 mergeSourceSpans :: SourceSpan -> SourceSpan -> SourceSpan
  697 mergeSourceSpans left right =
  698   case (left, right) of
  699     ( SourceSpan name l1 c1 _ _ startOffset _,

  769 nameFromText :: Text -> Name
  770 nameFromText txt =
  771   let (qualifier, localName) = splitQualifiedIdentifierText txt
  772    in Name qualifier (inferNameType localName) localName []
  773   where
  774     splitQualifiedIdentifierText fullName =
  775       case T.splitOn "." fullName of
  776         [] -> (Nothing, fullName)
  777         [_] -> (Nothing, fullName)
  778         parts
  779           | all isModuleSegment (init parts) && isIdentifierSegment (last parts) ->
  780               (Just (T.intercalate "." (init parts)), last parts)
  781           | otherwise ->
  782               (Nothing, fullName)
  783 
  784     isModuleSegment segment =
  785       case T.uncons segment of
  786         Just (c, rest) -> isConIdentifierStartChar c && T.all isIdentChar rest
  787         Nothing -> False
  788 
  789     isIdentifierSegment segment =
  790       let magicHashes = T.takeWhileEnd (== '#') segment
  791           baseSegment = T.dropEnd (T.length magicHashes) segment
  792        in case T.uncons baseSegment of
  793             Just (c, rest) -> isIdentifierStartChar c && T.all isIdentChar rest
  794             Nothing -> False
  795 
  796 unqualifiedNameFromText :: Text -> UnqualifiedName
  797 unqualifiedNameFromText txt = UnqualifiedName (inferNameType txt) txt []
  798 
  799 inferNameType :: Text -> NameType
  800 inferNameType localName
  801   | isOperatorLikeText localName =
  802       if T.isPrefixOf ":" localName
  803         then NameConSym
  804         else NameVarSym
  805   | otherwise =
  806       case T.uncons localName of
  807         Just (c, _) | isConIdentifierStartChar c -> NameConId
  808         Just (c, _) | isIdentifierStartChar c -> NameVarId
  809         _ -> NameConId
  810 
  811 isIdentChar :: Char -> Bool
  812 isIdentChar c = isIdentifierStartChar c || isIdentifierContinueChar c || c == '\''
  813 
  814 isIdentifierStartChar :: Char -> Bool
  815 isIdentifierStartChar c = c == '_' || generalCategory c == LowercaseLetter || isConIdentifierStartChar c
  816 
  817 isConIdentifierStartChar :: Char -> Bool
  818 isConIdentifierStartChar c = generalCategory c `elem` [UppercaseLetter, TitlecaseLetter]
  819 
  820 isIdentifierNumberChar :: Char -> Bool
  821 isIdentifierNumberChar c =
  822   case generalCategory c of
  823     DecimalNumber -> True
  824     OtherNumber -> True
  825     _ -> False
  826 
  827 isIdentifierContinueChar :: Char -> Bool
  828 isIdentifierContinueChar c =
  829   case generalCategory c of
  830     LetterNumber -> True
  831     ModifierLetter -> True
  832     NonSpacingMark -> True
  833     _ -> isIdentifierNumberChar c
  834 
  835 isOperatorLikeText :: Text -> Bool
  836 isOperatorLikeText op =
  837   not (T.null op) && T.all isOperatorChar op
  838 
  839 isOperatorChar :: Char -> Bool
  840 isOperatorChar c = c `elem` (":!#$%&*+./<=>?@\\^|-~" :: String) || isUnicodeOperatorChar c
  841 
  842 isUnicodeOperatorChar :: Char -> Bool
  843 isUnicodeOperatorChar c =
  844   case generalCategory c of
  845     MathSymbol -> True
  846     CurrencySymbol -> True
  847     ModifierSymbol -> True
  848     OtherSymbol -> True
  849     ConnectorPunctuation -> c > '\x7f'
  850     DashPunctuation -> c > '\x7f'
  851     OtherPunctuation -> c > '\x7f'
  852     _ -> False

 1506 -- | Extract the class name from an instance head type by walking the
 1507 -- application spine down to the leftmost atom, peeling through 'TParen'
 1508 -- and 'TAnn' nodes.
 1509 --
 1510 -- >>> fmap renderName $ instanceHeadName (TApp (TCon "C" Unpromoted) (TVar "a"))
 1511 -- Just "C"
 1512 -- >>> fmap renderName $ instanceHeadName (TInfix (TVar "a") ":=>" Unpromoted (TVar "b"))
 1513 -- Just ":=>"
 1514 instanceHeadName :: Type -> Maybe Name
 1515 instanceHeadName = go
 1516   where
 1517     go (TAnn _ t) = go t
 1518     go (TApp f _) = go f
 1519     go (TTypeApp f _) = go f
 1520     go (TParen t) = go t
 1521     go (TCon name _) = Just name
 1522     go (TInfix _ name _ _) = Just name
 1523     go _ = Nothing
 1524 
 1525 -- | Extract the type arguments from an instance head type.
 1526 --
 1527 -- For prefix application spines like @TApp (TApp (TCon C) a) b@, returns
 1528 -- @[a, b]@.  For infix heads like @TInfix a op b@, returns @[a, b]@.
 1529 instanceHeadTypes :: Type -> [Type]
 1530 instanceHeadTypes = go []
 1531   where
 1532     go acc (TAnn _ t) = go acc t
 1533     go acc (TApp f arg) = go (arg : acc) f
 1534     go acc (TTypeApp f _) = go acc f
 1535     go acc (TParen t) = go acc t
 1536     go _ (TInfix lhs _ _ rhs) = [lhs, rhs]
 1537     go acc _ = acc
