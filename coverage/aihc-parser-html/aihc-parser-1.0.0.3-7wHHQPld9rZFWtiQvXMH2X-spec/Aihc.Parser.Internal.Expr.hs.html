   31 -- | Parse an expression, then optionally consume @<-@ and a right-hand side.
   32 -- If the arrow is present, the expression is converted to a pattern via
   33 -- 'checkPattern' and the result is a bind; otherwise it is an expression.
   34 exprOrPatternBindParser ::
   35   TokParser Expr ->
   36   TokParser Expr ->
   37   (Pattern -> Expr -> a) ->
   38   (Expr -> a) ->
   39   TokParser a
   40 exprOrPatternBindParser exprP rhsP bindCtor exprCtor = do
   41   expr <- exprP
   42   mArrow <- MP.optional (expectedTok TkReservedLeftArrow)
   43   case mArrow of
   44     Just () -> do
   45       pat <- liftCheck (checkPattern expr)
   46       bindCtor pat <$> rhsP
   47     Nothing -> pure (exprCtor expr)

  110 -- | Optionally consume a @->@ token and parse the right-hand side as a
  111 -- view-pattern expression.  Returns the original expression unchanged when
  112 -- no @->@ follows.
  113 maybeViewPattern :: Expr -> TokParser Expr
  114 maybeViewPattern lhs = do
  115   mArrow <- MP.optional (expectedTok TkReservedRightArrow)
  116   case mArrow of
  117     Just () -> EInfix lhs viewPatArrowName <$> texprParser
  118     Nothing -> pure lhs

  123 -- @(# … #)@.

  198 doStmtParser :: TokParser (DoStmt Expr)
  199 doStmtParser = do
  200   tok <- lookAhead anySingle
  201   case lexTokenKind tok of
  202     TkKeywordLet -> MP.try doLetStmtParser <|> doBindOrExprStmtParser
  203     TkKeywordRec -> doRecStmtParser
  204     _ -> MP.try doPatBindStmtParser <|> doBindOrExprStmtParser

  206 doBindOrExprStmtParser :: TokParser (DoStmt Expr)
  207 doBindOrExprStmtParser = withSpanAnn (DoAnn . mkAnnotation) $ do
  208   mExpr <- MP.optional . MP.try $ exprParser
  209   case mExpr of
  210     Nothing -> do
  211       pat <- patternParser
  212       expectedTok TkReservedLeftArrow
  213       rhs <- region "while parsing '<-' binding" exprParser
  214       pure (DoBind pat rhs)
  215     Just expr -> do
  216       tok <- lookAhead anySingle
  217       case lexTokenKind tok of
  218         TkReservedAt -> do
  219           pat <- patternParser
  220           expectedTok TkReservedLeftArrow
  221           rhs <- region "while parsing '<-' binding" exprParser
  222           pure (DoBind pat rhs)
  223         _ -> do
  224           mArrow <- MP.optional (expectedTok TkReservedLeftArrow)
  225           case mArrow of
  226             Just () -> do
  227               pat <- liftCheck (checkPattern expr)
  228               rhs <- region "while parsing '<-' binding" exprParser
  229               pure (DoBind pat rhs)
  230             Nothing ->
  231               pure (DoExpr expr)
  232 
  233 doPatBindStmtParser :: TokParser (DoStmt Expr)
  234 doPatBindStmtParser = withSpanAnn (DoAnn . mkAnnotation) $ do
  235   pat <- patternParser
  236   expectedTok TkReservedLeftArrow
  237   expr <- region "while parsing '<-' binding" exprParser
  238   pure (DoBind pat expr)

  263 -- | Shared infix-chain parser used by the report core and contextual
  264 -- expression variants such as TransformListComp.
  265 exprInfixChainParser :: TokParser Expr -> TokParser Expr
  266 exprInfixChainParser lexp = do
  267   lhs <- lexp
  268   rest <-
  269     MP.many
  270       ( (,)
  271           <$> infixOperatorParser
  272           <*> region "after infix operator" lexp
  273       )
  274   pure (foldInfixL buildInfix lhs rest)

  395 -- | Shared application parser used by the report core and contextual
  396 -- variants.  The caller chooses the @aexp@-like atom parser.
  397 appExprParserWith :: TokParser Expr -> TokParser Expr
  398 appExprParserWith atomParser = do
  399   startInput <- MP.getInput
  400   first <- atomParser
  401   rest <- MP.many appArg
  402   case rest of
  403     [] -> pure first
  404     _ -> do
  405       endInput <- MP.getInput
  406       let startSpan = inputStartSpan startInput
  407           endSpan = maybe noSourceSpan lexTokenSpan (tokStreamPrevToken endInput)
  408           appSpan = mergeSourceSpans startSpan endSpan
  409       pure (EAnn (mkAnnotation appSpan) (foldl applyArg first rest))
  410   where
  411     appArg :: TokParser (Either Type Expr)
  412     appArg = (Left <$> typeAppArg) <|> (Right <$> appExprArgParser)
  413 
  414     typeAppArg :: TokParser Type
  415     typeAppArg = MP.try $ do
  416       expectedTok TkTypeApp
  417       typeAtomParser
  418 
  419     appExprArgParser :: TokParser Expr
  420     appExprArgParser = do
  421       tok <- lookAhead anySingle
  422       case lexTokenKind tok of
  423         -- GHC rejects bare explicit namespace syntax as a function argument:
  424         -- @f type T@ is invalid, while @f (type T)@ is accepted syntactically.
  425         TkKeywordType -> MP.empty
  426         _ -> atomParser
  427 
  428     applyArg :: Expr -> Either Type Expr -> Expr
  429     applyArg fn (Left ty) = ETypeApp fn ty
  430     applyArg fn (Right arg) = EApp fn arg

  432 -- | Parse an atomic expression, including record construction/update syntax.
  433 --
  434 -- Haskell 2010 makes record construction and update part of the @aexp@
  435 -- production:
  436 --
  437 -- > aexp -> qcon { fbind_1, ..., fbind_n }
  438 -- > aexp -> aexp<qcon> { fbind_1, ..., fbind_n }
  439 --
  440 -- So record braces are not a suffix on every expression form that this parser
  441 -- accepts as an application atom.  Extension forms such as bare explicit
  442 -- namespace expressions stay in 'atomExprParser'; they can be record-update
  443 -- bases only after parentheses make them an @aexp@.
  444 atomOrRecordExprParser :: TokParser Expr
  445 atomOrRecordExprParser = atomOrRecordExprParserWith NormalExprAtom
  446 
  447 atomOrRecordExprParserWith :: AtomContext -> TokParser Expr
  448 atomOrRecordExprParserWith atomContext =
  449   recordExprParser <|> atomExprParserWith atomContext
  450   where
  451     recordExprParser :: TokParser Expr
  452     recordExprParser = do
  453       base <- recordBaseAtomExprParserWith atomContext
  454       applyRecordSuffixes base
  455 
  456     applyRecordSuffixes :: Expr -> TokParser Expr
  457     applyRecordSuffixes e = do
  458       mRecordFields <- MP.optional recordBracesParser
  459       case mRecordFields of
  460         Just (fields, hasWildcard) -> do
  461           let result = case peelExprAnn e of
  462                 EVar name
  463                   | isConLikeName name ->
  464                       ERecordCon name (map normalizeField fields) hasWildcard
  465                 _ ->
  466                   ERecordUpd e (map normalizeField fields)
  467           applyRecordSuffixes result
  468         Nothing -> applyRecordDotSuffixes e
  469 
  470     applyRecordDotSuffixes :: Expr -> TokParser Expr
  471     applyRecordDotSuffixes e = do
  472       recordDotEnabled <- isExtensionEnabled OverloadedRecordDot
  473       if not recordDotEnabled || not (recordDotMayFollow e)
  474         then pure e
  475         else do
  476           mDot <- MP.optional (expectedTok TkRecordDot)
  477           case mDot of
  478             Nothing -> pure e
  479             Just () -> do
  480               fieldName <- recordFieldNameParser
  481               applyRecordSuffixes (EGetField e fieldName)
  482 
  483     normalizeField :: (Name, Maybe Expr, SourceSpan) -> RecordField Expr
  484     normalizeField (fieldName, mExpr, sp) =
  485       case mExpr of
  486         Just expr' -> RecordField fieldName expr' False
  487         Nothing -> RecordField fieldName (EAnn (mkAnnotation sp) (EVar fieldName)) True
  488 
  489     recordDotMayFollow :: Expr -> Bool
  490     recordDotMayFollow expr =
  491       case expr of
  492         EAnn _ sub -> recordDotMayFollow sub
  493         -- GHC rejects unparenthesized explicit namespace expressions before
  494         -- record-dot syntax, e.g. @type (:+).a@.
  495         ETypeSyntax TypeSyntaxExplicitNamespace _ -> False
  496         _ -> True

  510 -- | Parse the expression forms that correspond to the report's @aexp@
  511 -- production and can therefore be record construction/update bases.
  512 --
  513 -- > aexp -> qvar
  514 -- >      | gcon
  515 -- >      | literal
  516 -- >      | '(' exp ')'
  517 -- >      | '(' exp_1 ',' ... ',' exp_k ')'
  518 -- >      | '[' exp_1 ',' ... ',' exp_k ']'
  519 -- >      | '[' exp_1 [',' exp_2] '..' [exp_3] ']'
  520 -- >      | '[' exp '|' qual_1 ',' ... ',' qual_n ']'
  521 -- >      | '(' infixexp qop ')'
  522 -- >      | '(' qop infixexp ')'
  523 -- >      | qcon '{' fbind_1 ',' ... ',' fbind_n '}'
  524 -- >      | aexp<qcon> '{' fbind_1 ',' ... ',' fbind_n '}'
  525 recordBaseAtomExprParserWith :: AtomContext -> TokParser Expr
  526 recordBaseAtomExprParserWith atomContext = do
  527   tok <- lookAhead anySingle
  528   case lexTokenKind tok of
  529     TkImplicitParam {} -> implicitParamExprParser
  530     _ -> simpleAtomExprParserWith atomContext

  532 -- | Parse an atom without record construction/update syntax.
  533 --
  534 -- > aexp -> qvar | gcon | literal | '(' exp ')' | ...
  535 --
  536 -- This variant also admits extension-only atoms such as block arguments and
  537 -- explicit namespace syntax when the corresponding extensions are enabled.
  538 atomExprParser :: TokParser Expr
  539 atomExprParser = atomExprParserWith NormalExprAtom
  540 
  541 atomExprParserWith :: AtomContext -> TokParser Expr
  542 atomExprParserWith atomContext = do
  543   tok <- lookAhead anySingle
  544   case lexTokenKind tok of
  545     TkImplicitParam {} -> implicitParamExprParser
  546     TkKeywordType -> do
  547       explicitNamespacesEnabled <- isExtensionEnabled ExplicitNamespaces
  548       requiredTypeArgumentsEnabled <- isExtensionEnabled RequiredTypeArguments
  549       if explicitNamespacesEnabled || requiredTypeArgumentsEnabled
  550         then explicitTypeExprParser
  551         else simpleAtomExprParserWith atomContext
  552     TkReservedBackslash -> lambdaExprParser
  553     TkKeywordLet -> blockAtom letExprParser
  554     TkKeywordDo -> blockAtom doExprParser
  555     TkKeywordMdo -> blockAtom mdoExprParser
  556     TkQualifiedDo {} -> blockAtom qualifiedDoExprParser
  557     TkQualifiedMdo {} -> blockAtom qualifiedMdoExprParser
  558     TkKeywordCase -> blockAtom caseExprParser
  559     TkKeywordIf -> blockAtom ifExprParser
  560     TkKeywordProc
  561       | atomContext == NormalExprAtom -> blockAtom procExprParser
  562     _ -> simpleAtomExprParserWith atomContext
  563   where
  564     blockAtom parser = do
  565       blockArgsEnabled <- isExtensionEnabled BlockArguments
  566       if blockArgsEnabled
  567         then parser
  568         else simpleAtomExprParserWith atomContext

  570 simpleAtomExprParserWith :: AtomContext -> TokParser Expr
  571 simpleAtomExprParserWith atomContext = do
  572   tok <- lookAhead anySingle
  573   case lexTokenKind tok of
  574     TkPrefixMinus -> prefixNegateAtomExprParserWith atomContext
  575     TkSpecialLParen -> MP.try parenOperatorExprParser <|> parenExprParser
  576     TkSpecialUnboxedLParen -> parenExprParser
  577     TkSpecialLBracket -> listExprParser
  578     TkInteger {} -> intExprParser
  579     TkFloat {} -> floatExprParser
  580     TkChar {} -> charExprParser
  581     TkCharHash {} -> charExprParser
  582     TkString {} -> stringExprParser
  583     TkStringHash {} -> stringExprParser
  584     TkOverloadedLabel {} -> overloadedLabelExprParser
  585     TkKeywordUnderscore -> wildcardExprParser
  586     TkVarId {} -> varExprParser
  587     TkConId {} -> varExprParser
  588     TkQVarId {} -> varExprParser
  589     TkQConId {} -> varExprParser
  590     TkQuasiQuote {} -> quasiQuoteExprParser
  591     TkTHExpQuoteOpen -> thAtom thQuoteExprParser

  811 parenExprParser :: TokParser Expr
  812 parenExprParser = withSpanAnn (EAnn . mkAnnotation) $ do
  813   (tupleFlavor, closeTok) <- tupleDelimsParser
  814   mClosed <- MP.optional (expectedTok closeTok)
  815   case mClosed of
  816     Just () -> pure (ETuple tupleFlavor [])
  817     Nothing ->
  818       if tupleFlavor == Boxed
  819         then MP.try (parseNegateParen closeTok) <|> parseBoxedContent closeTok
  820         else MP.try (parseUnboxedSumExprLeadingBars closeTok) <|> parseTupleOrParen tupleFlavor closeTok
  821   where
  822     parseNegateParen closeTok = do
  823       minusTok <- minusTokenValueParser
  824       guard (parenNegateAllowed minusTok)
  825       -- Parse only the application-level expression as the negation's
  826       -- immediate operand, plus the block forms GHC permits after prefix
  827       -- negation.  Negation still binds tighter than any infix operator, so
  828       -- @(-l - 1)@ is @((negate l) - 1)@, not @(negate (l - 1))@.
  829       negOperand <- negateOperandParser
  830       let negBase = ENegate negOperand
  831       -- Continue with any infix operator chain, type signature, and view
  832       -- pattern that may follow the negated expression inside the parens.
  833       rest <-
  834         MP.many
  835           ( MP.try
  836               ( (,)
  837                   <$> infixOperatorParser
  838                   <*> region "after infix operator" lexpParser
  839               )
  840           )
  841       let withInfix = foldInfixL buildInfix negBase rest
  842       mTypeSig <- MP.optional (expectedTok TkReservedDoubleColon *> typeSignatureParser)
  843       let typed = case mTypeSig of
  844             Just ty -> ETypeSig withInfix ty
  845             Nothing -> withInfix
  846       finalExpr <- maybeViewPattern typed
  847       expectedTok closeTok
  848       -- The negation is already embedded in finalExpr (as negBase).
  849       -- With TkPrefixMinus (LexicalNegation), the surrounding parens are
  850       -- just grouping — no EParen wrapper.  Otherwise the parens are part
  851       -- of the negation-section syntax and need an EParen wrapper.
  852       pure $
  853         case lexTokenKind minusTok of
  854           TkPrefixMinus -> finalExpr
  855           _ -> EParen finalExpr
  856 
  857     parenNegateAllowed minusTok =
  858       case lexTokenKind minusTok of
  859         TkPrefixMinus -> True
  860         TkVarSym "-" -> True
  861         TkMinusOperator -> False
  862         _ -> False
  863 
  864     parseBoxedContent closeTok =
  865       MP.try (parseProjectionSection closeTok)
  866         <|> MP.try parseSectionR
  867         <|> do
  868           mBase <- MP.optional (MP.try negateExprParser <|> lexpParser)
  869           case mBase of
  870             Nothing ->
  871               finishBoxed closeTok Nothing
  872             Just base -> do
  873               mOp <- MP.optional infixOperatorParser
  874               case mOp of
  875                 Nothing -> do
  876                   mArrowSection <- MP.optional (MP.try (arrowSectionOperatorParser <* expectedTok closeTok))
  877                   case mArrowSection of
  878                     Just op ->
  879                       pure (EParen (ESectionL base op))
  880                     Nothing -> do
  881                       mTypeSig <- MP.optional (expectedTok TkReservedDoubleColon *> typeSignatureParser)
  882                       let typed = case mTypeSig of
  883                             Just ty -> ETypeSig base ty
  884                             Nothing -> base
  885                       -- View pattern arrow: expr -> expr (inside parentheses)
  886                       finalExpr <- maybeViewPattern typed
  887                       finishBoxed closeTok (Just finalExpr)
  888                 Just op -> do
  889                   mClose <- MP.optional (expectedTok closeTok)
  890                   case mClose of
  891                     Just () ->
  892                       pure (EParen (ESectionL base op))
  893                     Nothing -> do
  894                       rhs <- region "after infix operator" lexpParser
  895                       more <-
  896                         MP.many
  897                           ( MP.try
  898                               ( (,)
  899                                   <$> infixOperatorParser
  900                                   <*> region "after infix operator" lexpParser
  901                               )
  902                           )
  903                       let fullInfix = foldInfixL buildInfix base ((op, rhs) : more)
  904                       mTrailingOp <- MP.optional infixOperatorParser
  905                       case mTrailingOp of
  906                         Just trailOp -> do
  907                           expectedTok closeTok
  908                           pure (EParen (ESectionL fullInfix trailOp))
  909                         Nothing -> do
  910                           mTypeSig <- MP.optional (expectedTok TkReservedDoubleColon *> typeSignatureParser)
  911                           let typed = case mTypeSig of
  912                                 Just ty -> ETypeSig fullInfix ty
  913                                 Nothing -> fullInfix
  914                           -- View pattern arrow: expr -> expr (inside parentheses)
  915                           finalExpr <- maybeViewPattern typed
  916                           finishBoxed closeTok (Just finalExpr)
  917       where
  918         parseProjectionSection tok = do
  919           recordDotEnabled <- isExtensionEnabled OverloadedRecordDot
  920           guard recordDotEnabled
  921           fields <- MP.some (expectedTok TkRecordDot *> recordFieldNameParser)
  922           expectedTok tok
  923           pure (EParen (EGetFieldProjection fields))
  924 
  925         parseSectionR = do
  926           op <- reservedAtSectionOperatorParser <|> infixOperatorParser <|> arrowSectionOperatorParser
  927           rhs <- exprParser
  928           expectedTok closeTok
  929           pure (EParen (ESectionR op rhs))
  930 
  931         reservedAtSectionOperatorParser = do
  932           expectedTok TkReservedAt
  933           pure (qualifyName Nothing (mkUnqualifiedName NameVarSym "@"))
  934 
  935         arrowSectionOperatorParser =
  936           tokenSatisfy "operator" $ \tok ->
  937             case lexTokenKind tok of
  938               TkArrowTail -> Just (qualifyName Nothing (mkUnqualifiedName NameVarSym "-<"))
  939               TkDoubleArrowTail -> Just (qualifyName Nothing (mkUnqualifiedName NameVarSym "-<<"))
  940               _ -> Nothing
  941 
  942     finishBoxed closeTok mFirst = do
  943       mComma <- MP.optional (expectedTok TkSpecialComma)
  944       case (mFirst, mComma) of
  945         (Just e, Nothing) -> do
  946           expectedTok closeTok
  947           pure (EParen e)
  948         (_, Just ()) -> do
  949           rest <- parseTupleElems closeTok
  950           pure (ETuple Boxed (mFirst : rest))
  951         (Nothing, Nothing) ->
  952           fail "expected expression or closing paren"
  953 
  954     parseTupleOrParen tupleFlavor closeTok = do
  955       first <- MP.optional texprParser
  956       mComma <- MP.optional (expectedTok TkSpecialComma)
  957       case (first, mComma) of
  958         (Just e, Nothing) ->
  959           case tupleFlavor of
  960             Boxed -> do
  961               expectedTok closeTok
  962               pure (EParen e)
  963             Unboxed -> do
  964               mPipe <- MP.optional (expectedTok TkReservedPipe)
  965               case mPipe of
  966                 Just () -> do
  967                   trailingBars <- MP.many (expectedTok TkReservedPipe)
  968                   expectedTok closeTok
  969                   let arity = 2 + length trailingBars
  970                   pure (EUnboxedSum 0 arity e)
  971                 Nothing -> do
  972                   expectedTok closeTok
  973                   pure (ETuple Unboxed [Just e])
  974         (_, Just ()) -> do
  975           rest <- parseTupleElems closeTok
  976           pure (ETuple tupleFlavor (first : rest))
  977         (Nothing, Nothing) ->
  978           fail "expected expression or closing paren"
  979 
  980     parseTupleElems closeTok = do
  981       e <- MP.optional texprParser
  982       mComma <- MP.optional (expectedTok TkSpecialComma)
  983       case mComma of
  984         Just () -> (e :) <$> parseTupleElems closeTok
  985         Nothing -> do
  986           expectedTok closeTok
  987           pure [e]
  988 
  989     parseUnboxedSumExprLeadingBars closeTok = do
  990       _ <- expectedTok TkReservedPipe
  991       leadingBars <- MP.many (expectedTok TkReservedPipe)
  992       let altIdx = 1 + length leadingBars
  993       inner <- texprParser
  994       trailingBars <- MP.many (expectedTok TkReservedPipe)
  995       expectedTok closeTok
  996       let arity = altIdx + 1 + length trailingBars
  997       pure (EUnboxedSum altIdx arity inner)
