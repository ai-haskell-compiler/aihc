   23 isHaskellWhitespace :: Char -> Bool
   24 isHaskellWhitespace c =
   25   c == ' ' || c == '\t' || c == '\r' || c == '\f' || c == '\v' || isSpace c

   27 tryConsumeLineDirective :: LexerState -> Maybe (Maybe LexToken, LexerState)
   28 tryConsumeLineDirective st
   29   | not (lexerAtLineStart st) = Nothing
   30   | otherwise =
   31       let inp = lexerInput st
   32           (spaces, rest) = T.span (\c -> c == ' ' || c == '\t') inp
   33           -- CPP #line directives require '#' at column 1 (no leading
   34           -- whitespace).  GHC enforces the same rule: an indented '#' is
   35           -- never a line directive.  Shebangs, however, are allowed with
   36           -- optional leading whitespace.
   37           --
   38           -- Note: skipTrivia may have already consumed leading whitespace
   39           -- before calling us, so we check lexerCol rather than whether
   40           -- 'spaces' is empty.  The column after consuming 'spaces' is
   41           -- where '#' would sit.
   42           hashCol = lexerCol st + T.length spaces
   43           atColumn1 = hashCol == 1
   44        in case rest of
   45             '#' :< more ->
   46               let lineText = "#" <> takeLineRemainder more
   47                   consumed = spaces <> lineText
   48                   isFirstLine = lexerLine st == 1
   49                in case classifyHashLineTrivia atColumn1 isFirstLine lineText of
   50                     Just (HashLineDirective update) ->
   51                       Just (Nothing, applyDirectiveAdvance consumed update st)
   52                     Just HashLineShebang ->
   53                       let st' = advanceChars consumed st
   54                        in Just (Nothing, st')
   55                     Just HashLineMalformed ->
   56                       let st' = advanceChars consumed st
   57                        in Just (Just (mkToken st st' consumed (TkError "malformed line directive")), st')
   58                     Nothing -> Nothing
   59             _ -> Nothing
   60 
   61 tryConsumeControlPragma :: LexerState -> Maybe (Maybe LexToken, LexerState)
   62 tryConsumeControlPragma st =
   63   let inp = lexerInput st
   64    in case parseControlPragma inp of
   65         Just (consumedT, Right update0) ->
   66           let (consumedT', update) =
   67                 case directiveLine update0 of
   68                   Just lineNo ->
   69                     case T.drop (T.length consumedT) (lexerInput st) of
   70                       '\n' :< _ ->
   71                         (consumedT <> "\n", update0 {directiveLine = Just lineNo, directiveCol = Just 1})
   72                       _ -> (consumedT, update0)
   73                   Nothing -> (consumedT, update0)
   74            in Just (Nothing, applyDirectiveAdvance consumedT' update st)
   75         Just (consumedT, Left msg) ->
   76           let st' = advanceChars consumedT st
   77            in Just (Just (mkToken st st' consumedT (TkError msg)), st')
   78         Nothing -> Nothing
   79 
   80 consumeLineCommentToken :: LexerState -> (LexToken, LexerState)
   81 consumeLineCommentToken st =
   82   let inp = lexerInput st
   83       rest = T.drop 2 inp
   84       consumed = "--" <> T.takeWhile (/= '\n') rest
   85       st' = advanceChars consumed st
   86    in (mkToken st st' consumed TkLineComment, st')

   88 consumeBlockCommentToken :: LexerState -> Maybe (LexToken, LexerState)
   89 consumeBlockCommentToken st =
   90   case scanNestedBlockComment 1 (T.drop 2 (lexerInput st)) of
   91     Just consumedTail ->
   92       let consumed = "{-" <> consumedTail
   93           st' = advanceChars consumed st
   94           st'' =
   95             if T.any (== '\n') consumed
   96               then st'
   97               else st' {lexerAtLineStart = lexerAtLineStart st}
   98        in Just (mkToken st st'' consumed TkBlockComment, st'')
   99     Nothing -> Nothing
  100 
  101 consumeBlockCommentTokenOrError :: LexerState -> Either (LexToken, LexerState) (LexToken, LexerState)
  102 consumeBlockCommentTokenOrError st =
  103   case consumeBlockCommentToken st of
  104     Just result -> Right result
  105     Nothing ->
  106       let consumed = lexerInput st
  107           st' = advanceChars consumed st
  108           tok = mkToken st st' consumed (TkError "unterminated block comment")
  109        in Left (tok, st')

  111 scanNestedBlockComment :: Int -> Text -> Maybe Text
  112 scanNestedBlockComment depth0 input = go depth0 0 input
  113   where
  114     -- Skip characters that can't start a nesting change in bulk, then
  115     -- inspect the stopping character.  Allocation is O(nesting changes)
  116     -- instead of O(length).
  117     go depth !n remaining
  118       | depth <= 0 = Just (T.take n input)
  119       | otherwise =
  120           let (prefix, rest0) = T.span (\c -> c /= '{' && c /= '-') remaining
  121               n' = n + T.length prefix
  122            in case T.uncons rest0 of
  123                 Nothing -> Nothing
  124                 Just (c, rest1) ->
  125                   case T.uncons rest1 of
  126                     Nothing -> Nothing -- truncated escape sequence, unterminated
  127                     Just (c2, rest2)
  128                       | c == '{' && c2 == '-' -> go (depth + 1) (n' + 2) rest2
  129                       | c == '-' && c2 == '}' ->
  130                           if depth == 1
  131                             then Just (T.take (n' + 2) input)
  132                             else go (depth - 1) (n' + 2) rest2
  133                       | otherwise ->
  134                           -- c was '{' or '-' but not a nesting pair; advance by 1
  135                           -- and re-examine rest1 (which still starts with c2)
  136                           go depth (n' + 1) rest1

  138 applyDirectiveAdvance :: Text -> DirectiveUpdate -> LexerState -> LexerState
  139 applyDirectiveAdvance consumed update st =
  140   let hasTrailingNewline = T.isSuffixOf "\n" consumed
  141       st' = advanceChars consumed st
  142    in st'
  143         { lexerLogicalSourceName = fromMaybe (lexerLogicalSourceName st') (directiveSourceName update),
  144           lexerLine = maybe (lexerLine st') (max 1) (directiveLine update),
  145           lexerCol = maybe (lexerCol st') (max 1) (directiveCol update),
  146           lexerAtLineStart = hasTrailingNewline || (Just 1 == directiveCol update)
  147         }

  149 -- | Classify a line beginning with @#@.
  150 --
  151 -- @atColumn1@ is 'True' when @#@ sits at column 1 of the physical source
  152 -- line (no leading whitespace).  @isFirstLine@ is 'True' on line 1.
  153 -- Shebangs are accepted at column 1 on any line (for mid-file shebangs
  154 -- as in some polyglot scripts) or anywhere on line 1 (GHC also accepts
  155 -- an optional leading space before the initial shebang).
  156 -- An indented @#!@ past line 1 is left for the operator lexer.
  157 -- CPP @#line@ directives are only recognised at column 1, matching GHC.
  158 classifyHashLineTrivia :: Bool -> Bool -> Text -> Maybe HashLineTrivia
  159 classifyHashLineTrivia atColumn1 isFirstLine raw
  160   | (atColumn1 || isFirstLine) && isHashBangLine raw = Just HashLineShebang
  161   | not atColumn1 = Nothing
  162   | looksLikeHashLineDirective raw =
  163       case parseHashLineDirective raw of
  164         Just update -> Just (HashLineDirective update)
  165         Nothing -> Just HashLineMalformed
  166   | otherwise = Nothing
  167 
  168 parseHashLineDirective :: Text -> Maybe DirectiveUpdate
  169 parseHashLineDirective raw =
  170   let trimmed = T.dropWhile isSpace (T.drop 1 (T.dropWhile isSpace raw))
  171       trimmed' =
  172         case T.stripPrefix "line" trimmed of
  173           Just afterLine -> T.dropWhile isSpace afterLine
  174           Nothing -> trimmed
  175       (digits, rest) = T.span isDigit trimmed'
  176    in if T.null digits
  177         then Nothing
  178         else do
  179           lineNo <- readBoundedInt digits
  180           Just
  181             DirectiveUpdate
  182               { directiveLine = Just lineNo,
  183                 directiveCol = Just 1,
  184                 directiveSourceName = parseDirectiveSourceName rest
  185               }
  186 
  187 isHashBangLine :: Text -> Bool
  188 isHashBangLine raw =
  189   "#!" `T.isPrefixOf` T.dropWhile isSpace raw
  190 
  191 looksLikeHashLineDirective :: Text -> Bool
  192 looksLikeHashLineDirective raw =
  193   let afterHash = T.dropWhile isSpace (T.drop 1 (T.dropWhile isSpace raw))
  194    in case afterHash of
  195         c :< _ | isDigit c -> True
  196         _ -> "line" `T.isPrefixOf` afterHash

  198 parseDirectiveSourceName :: Text -> Maybe FilePath
  199 parseDirectiveSourceName rest =
  200   let rest' = T.dropWhile isSpace rest
  201    in case rest' of
  202         '"' :< more ->
  203           let (name, trailing) = T.break (== '"') more
  204            in case trailing of
  205                 '"' :< _ -> Just (T.unpack name)
  206                 _ -> Nothing
  207         _ -> Nothing
  208 
  209 takeLineRemainder :: Text -> Text
  210 takeLineRemainder chars =
  211   let (prefix, rest) = T.break (== '\n') chars
  212    in case rest of
  213         '\n' :< _ -> prefix <> "\n"
  214         _ -> prefix

  216 isLineComment :: Text -> Bool
  217 isLineComment rest =
  218   case rest of
  219     c :< _
  220       | c == '-' -> isLineComment (T.dropWhile (== '-') rest)
  221       | isSymbolicOpChar c -> False
  222       | otherwise -> True
  223     _ -> True

  225 readBoundedInt :: Text -> Maybe Int
  226 readBoundedInt txt = do
  227   n <- readMaybe (T.unpack txt) :: Maybe Integer
  228   if n >= fromIntegral (minBound :: Int) && n <= fromIntegral (maxBound :: Int)
  229     then Just (fromInteger n)
  230     else Nothing
