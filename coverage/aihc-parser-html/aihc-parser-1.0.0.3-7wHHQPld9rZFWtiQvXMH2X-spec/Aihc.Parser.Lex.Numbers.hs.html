   23 lexIntBase :: LexerEnv -> LexerState -> Maybe (LexToken, LexerState)
   24 lexIntBase env st =
   25   case lexerInput st of
   26     '0' :< (base :< rest)
   27       | base `elem` ("xXoObB" :: String) ->
   28           let allowUnderscores = hasExt NumericUnderscores env
   29               isDigitChar
   30                 | base `elem` ("xX" :: String) = isHexDigit
   31                 | base `elem` ("oO" :: String) = isOctDigit
   32                 | otherwise = (`elem` ("01" :: String))
   33               (digitsRaw, _) = takeDigitsWithUnderscores allowUnderscores isDigitChar rest
   34            in if T.null digitsRaw
   35                 then Nothing
   36                 else
   37                   let raw = "0" <> T.singleton base <> digitsRaw
   38                       n
   39                         | base `elem` ("xX" :: String) = readHexLiteral raw
   40                         | base `elem` ("oO" :: String) = readOctLiteral raw
   41                         | otherwise = readBinLiteral raw
   42                       (tokTxt, tokKind, st') =
   43                         lexIntSuffix env st raw n
   44                    in Just (mkToken st st' tokTxt tokKind, st')
   45     _ -> Nothing

   47 lexHexFloat :: LexerEnv -> LexerState -> Maybe (LexToken, LexerState)
   48 lexHexFloat env st =
   49   case lexerInput st of
   50     '0' :< (x :< rest1)
   51       | x `elem` ("xX" :: String),
   52         let (intDigits, rest2) = T.span isHexDigit rest1,
   53         not (T.null intDigits) -> do
   54           let (mFracDigits, rest3) =
   55                 case rest2 of
   56                   '.' :< more ->
   57                     let (frac, rest') = T.span isHexDigit more
   58                      in if T.null frac
   59                           then (Nothing, rest2)
   60                           else (Just frac, rest')
   61                   _ -> (Nothing, rest2)
   62           case (mFracDigits, takeHexExponent rest3) of
   63             (Nothing, Nothing) -> Nothing
   64             (mFrac, mExpo) ->
   65               let dotAndFrac = maybe "" ("." <>) mFrac
   66                   expo = fromMaybe "" mExpo
   67                   fracDigits = fromMaybe "" mFrac
   68                   raw = "0" <> T.singleton x <> intDigits <> dotAndFrac <> expo
   69                   value = parseHexFloatLiteral (T.unpack intDigits) (T.unpack fracDigits) (T.unpack expo)
   70                   (tokTxt, tokKind, st') =
   71                     lexFloatSuffix env st raw value
   72                in Just (mkToken st st' tokTxt tokKind, st')
   73     _ -> Nothing

   75 lexFloat :: LexerEnv -> LexerState -> Maybe (LexToken, LexerState)
   76 lexFloat env st =
   77   let allowUnderscores = hasExt NumericUnderscores env
   78       input = lexerInput st
   79       startsWithUnderscore =
   80         case input of
   81           '_' :< _ -> True
   82           _ -> False
   83       (lhsRaw, rest) = takeDigitsWithUnderscores allowUnderscores isDigit input
   84    in if T.null lhsRaw || startsWithUnderscore
   85         then Nothing
   86         else case rest of
   87           '.' :< dotRest@(d :< _)
   88             | isDigit d ->
   89                 let (rhsRaw, rest') = takeDigitsWithUnderscores allowUnderscores isDigit dotRest
   90                     (expo, _) = takeExponent allowUnderscores rest'
   91                     raw = lhsRaw <> "." <> rhsRaw <> expo
   92                     normalized = T.filter (/= '_') raw
   93                     value = parseDecimalFloatLiteral normalized
   94                     (tokTxt, tokKind, st') =
   95                       lexFloatSuffix env st raw value
   96                  in Just (mkToken st st' tokTxt tokKind, st')
   97           _ ->
   98             case takeExponent allowUnderscores rest of
   99               (expo, _)
  100                 | T.null expo -> Nothing
  101                 | otherwise ->
  102                     let raw = lhsRaw <> expo
  103                         normalized = T.filter (/= '_') raw
  104                         value = parseDecimalFloatLiteral normalized
  105                         (tokTxt, tokKind, st') =
  106                           lexFloatSuffix env st raw value
  107                      in Just (mkToken st st' tokTxt tokKind, st')

  109 lexInt :: LexerEnv -> LexerState -> Maybe (LexToken, LexerState)
  110 lexInt env st =
  111   let allowUnderscores = hasExt NumericUnderscores env
  112       input = lexerInput st
  113       startsWithUnderscore =
  114         case input of
  115           '_' :< _ -> True
  116           _ -> False
  117       (digitsRaw, _) = takeDigitsWithUnderscores allowUnderscores isDigit input
  118    in if T.null digitsRaw || startsWithUnderscore
  119         then Nothing
  120         else
  121           let digits = T.filter (/= '_') digitsRaw
  122               n = read (T.unpack digits) :: Integer
  123               (tokTxt, tokKind, st') =
  124                 lexIntSuffix env st digitsRaw n
  125            in Just (mkToken st st' tokTxt tokKind, st')

  127 -- | Parse optional MagicHash suffix (# or ##) and return the hash text and new state.
  128 magicHashSuffix :: LexerEnv -> LexerState -> Maybe (Text, LexerState)
  129 magicHashSuffix env st =
  130   case lexerInput st of
  131     '#' :< rest
  132       | hasExt MagicHash env ->
  133           case rest of
  134             '#' :< _ -> Just ("##", advanceN 2 st)
  135             _ -> Just ("#", advanceN 1 st)
  136     _ -> Nothing
  137 
  138 -- | Parse optional MagicHash and ExtendedLiterals suffix for integers.
  139 -- Handles: (nothing), #, ##, #Int8, #Int16, #Int32, #Int64, #Int, #Word8, #Word16, #Word32, #Word64, #Word
  140 lexIntSuffix :: LexerEnv -> LexerState -> Text -> Integer -> (Text, LexTokenKind, LexerState)
  141 lexIntSuffix env st raw n =
  142   let st' = advanceChars raw st
  143       input = lexerInput st'
  144       withMagicHash =
  145         case magicHashSuffix env st' of
  146           Just ("##", st'') -> (raw <> "##", TkInteger n TWordHash, st'')
  147           Just ("#", st'') -> (raw <> "#", TkInteger n TIntHash, st'')
  148           _ -> (raw, TkInteger n TInteger, st')
  149    in case input of
  150         '#' :< rest
  151           | hasExt ExtendedLiterals env ->
  152               case parseExtendedIntSuffix rest of
  153                 Just (typeName, suffixLen) ->
  154                   let fullRaw = raw <> T.take (1 + suffixLen) input
  155                       st'' = advanceN (1 + suffixLen) st'
  156                    in (fullRaw, TkInteger n typeName, st'')
  157                 Nothing -> withMagicHash
  158           | otherwise -> withMagicHash
  159         _ -> (raw, TkInteger n TInteger, st')

  161 -- | Parse optional MagicHash and ExtendedLiterals suffix for floats.
  162 -- Handles: (nothing), #, ##
  163 lexFloatSuffix :: LexerEnv -> LexerState -> Text -> Rational -> (Text, LexTokenKind, LexerState)
  164 lexFloatSuffix env st raw value =
  165   let st' = advanceChars raw st
  166    in case magicHashSuffix env st' of
  167         Just ("##", st'') -> (raw <> "##", TkFloat value TDoubleHash, st'')
  168         Just ("#", st'') -> (raw <> "#", TkFloat value TFloatHash, st'')
  169         _ -> (raw, TkFloat value TFractional, st')

  171 -- | Parse an ExtendedLiterals type suffix after the initial '#'.
  172 -- Returns (NumericType, length of type name without the '#').
  173 -- e.g. "Word8" -> (TWord8Hash, 5), "Int" -> (TIntHash, 3)
  174 parseExtendedIntSuffix :: Text -> Maybe (NumericType, Int)
  175 parseExtendedIntSuffix input =
  176   case matchType input of
  177     Just ("Int8", len) -> Just (TInt8Hash, len)
  178     Just ("Int16", len) -> Just (TInt16Hash, len)
  179     Just ("Int32", len) -> Just (TInt32Hash, len)
  180     Just ("Int64", len) -> Just (TInt64Hash, len)
  181     Just ("Int", len) -> Just (TIntHash, len)
  182     Just ("Word8", len) -> Just (TWord8Hash, len)
  183     Just ("Word16", len) -> Just (TWord16Hash, len)
  184     Just ("Word32", len) -> Just (TWord32Hash, len)
  185     Just ("Word64", len) -> Just (TWord64Hash, len)
  186     Just ("Word", len) -> Just (TWordHash, len)
  187     _ -> Nothing

  189 -- | Try to match one of the known type names at the start of the text.
  190 -- Returns (matched name, length) if the match is followed by a non-identifier char
  191 -- or end of input.
  192 matchType :: Text -> Maybe (Text, Int)
  193 matchType input =
  194   let candidates = ["Int8", "Int16", "Int32", "Int64", "Int", "Word8", "Word16", "Word32", "Word64", "Word"]
  195    in firstMatch candidates input
  196 
  197 firstMatch :: [Text] -> Text -> Maybe (Text, Int)
  198 firstMatch [] _ = Nothing
  199 firstMatch (name : rest) input =
  200   let len = T.length name
  201    in case T.stripPrefix name input of
  202         Just remaining
  203           | T.null remaining || not (isIdentChar (T.head remaining)) ->
  204               Just (name, len)
  205         _ -> firstMatch rest input
  206 
  207 isIdentChar :: Char -> Bool
  208 isIdentChar c = isDigit c || isAsciiUpper c || c == '_' || isAsciiLower c

  210 -- Scan ([_]*[digit])* and return a zero-copy split.
  211 takeDigitsWithUnderscores :: Bool -> (Char -> Bool) -> Text -> (Text, Text)
  212 takeDigitsWithUnderscores False isDigitChar = T.span isDigitChar
  213 takeDigitsWithUnderscores True isDigitChar = \chars ->
  214   let (consumed, rest) = T.span (\c -> isDigitChar c || c == '_') chars
  215    in if T.null consumed || T.last consumed /= '_'
  216         then (consumed, rest)
  217         else
  218           let trimmed = T.dropWhileEnd (== '_') consumed
  219            in (trimmed, T.drop (T.length trimmed) chars)

  221 takeExponent :: Bool -> Text -> (Text, Text)
  222 takeExponent allowUnderscores chars =
  223   case chars of
  224     '_' :< _
  225       | allowUnderscores ->
  226           let (_allUnderscores, rest') = T.span (== '_') chars
  227            in case rest' of
  228                 marker :< rest2
  229                   | marker `elem` ("eE" :: String) ->
  230                       let (_signPart, rest3) =
  231                             case rest2 of
  232                               sign :< more | sign `elem` ("+-" :: String) -> (T.singleton sign, more)
  233                               _ -> ("", rest2)
  234                           digitsStartWithUnderscore =
  235                             case rest3 of
  236                               '_' :< _ -> True
  237                               _ -> False
  238                           (digits, rest4) = takeDigitsWithUnderscores allowUnderscores isDigit rest3
  239                        in if T.null digits || digitsStartWithUnderscore
  240                             then ("", chars)
  241                             else
  242                               let consumed = T.take (T.length chars - T.length rest4) chars
  243                                in (consumed, rest4)
  244                 _ -> ("", chars)
  245     marker :< rest
  246       | marker `elem` ("eE" :: String) ->
  247           let (_signPart, rest1) =
  248                 case rest of
  249                   sign :< more | sign `elem` ("+-" :: String) -> (T.singleton sign, more)
  250                   _ -> ("", rest)
  251               digitsStartWithUnderscore =
  252                 case rest1 of
  253                   '_' :< _ -> True
  254                   _ -> False
  255               (digits, rest2) = takeDigitsWithUnderscores allowUnderscores isDigit rest1
  256            in if T.null digits || digitsStartWithUnderscore then ("", chars) else let consumed = T.take (T.length chars - T.length rest2) chars in (consumed, rest2)
  257     _ -> ("", chars)
  258 
  259 takeHexExponent :: Text -> Maybe Text
  260 takeHexExponent chars =
  261   case chars of
  262     marker :< rest
  263       | marker `elem` ("pP" :: String) ->
  264           let (_signPart, rest1) =
  265                 case rest of
  266                   sign :< more | sign `elem` ("+-" :: String) -> (T.singleton sign, more)
  267                   _ -> ("", rest)
  268               (digits, _) = T.span isDigit rest1
  269            in if T.null digits then Nothing else Just (T.take (T.length chars - T.length rest1 + T.length digits) chars)
  270     _ -> Nothing

  272 readBaseLiteral :: String -> (String -> [(Integer, String)]) -> Text -> Integer
  273 readBaseLiteral label parser txt =
  274   case parser (T.unpack (T.filter (/= '_') (T.drop 2 txt))) of
  275     [(n, "")] -> n
  276     _ -> error ("invalid " ++ label ++ " literal: " ++ T.unpack txt)

  278 readHexLiteral :: Text -> Integer
  279 readHexLiteral = readBaseLiteral "hex" readHex

  281 readOctLiteral :: Text -> Integer
  282 readOctLiteral = readBaseLiteral "octal" readOct

  284 readBinLiteral :: Text -> Integer
  285 readBinLiteral = readBaseLiteral "binary" (readInt 2 (`elem` ("01" :: String)) digitToInt)

  287 parseDecimalFloatLiteral :: Text -> Rational
  288 parseDecimalFloatLiteral txt =
  289   case parseSignedDecimalRational txt of
  290     Just value -> value
  291     Nothing -> error ("invalid decimal float literal: " ++ T.unpack txt)
  292 
  293 parseSignedDecimalRational :: Text -> Maybe Rational
  294 parseSignedDecimalRational txt = do
  295   let (sign, unsigned) =
  296         case T.uncons txt of
  297           Just ('-', rest) -> (-1, rest)
  298           Just ('+', rest) -> (1, rest)
  299           _ -> (1, txt)
  300       (mantissaTxt, exponentTxt0) = T.break (`elem` ['e', 'E']) unsigned
  301       exponentTxt = T.drop 1 exponentTxt0
  302   mantissa <- parseDecimalMantissa mantissaTxt
  303   exponentN <-
  304     if T.null exponentTxt0
  305       then Just 0
  306       else parseSignedInteger exponentTxt
  307   pure (applyDecimalExponent (fromInteger sign * mantissa) exponentN)
  308 
  309 parseDecimalMantissa :: Text -> Maybe Rational
  310 parseDecimalMantissa txt = do
  311   let (whole, frac0) = T.break (== '.') txt
  312       frac = T.drop 1 frac0
  313   if T.null whole && T.null frac
  314     then Nothing
  315     else do
  316       wholeDigits <- parseDecimalDigits whole
  317       fracDigits <- parseDecimalDigits frac
  318       let fracScale = 10 ^ T.length frac
  319       pure ((wholeDigits * fracScale + fracDigits) % fracScale)
  320 
  321 parseDecimalDigits :: Text -> Maybe Integer
  322 parseDecimalDigits digits
  323   | T.null digits = Just 0
  324   | T.all isDigit digits = readMaybe (T.unpack digits)
  325   | otherwise = Nothing
  326 
  327 parseSignedInteger :: Text -> Maybe Integer
  328 parseSignedInteger digits =
  329   case T.uncons digits of
  330     Just ('+', rest) -> parseDecimalDigits rest
  331     Just ('-', rest) -> negate <$> parseDecimalDigits rest
  332     _ -> parseDecimalDigits digits
  333 
  334 applyDecimalExponent :: Rational -> Integer -> Rational
  335 applyDecimalExponent value exponentN
  336   | exponentN >= 0 = value * fromInteger (10 ^ exponentN)
  337   | otherwise = value / fromInteger (10 ^ negate exponentN)
  338 
  339 parseHexFloatLiteral :: String -> String -> String -> Rational
  340 parseHexFloatLiteral intDigits fracDigits expo =
  341   (parseHexDigits intDigits + parseHexFraction fracDigits) * (2 ^^ exponentValue expo)
  342 
  343 parseHexDigits :: String -> Rational
  344 parseHexDigits = foldl (\acc d -> acc * 16 + fromIntegral (digitToInt d)) 0
  345 
  346 parseHexFraction :: String -> Rational
  347 parseHexFraction ds =
  348   sum [fromIntegral (digitToInt d) % (16 ^ i) | (d, i) <- zip ds [1 :: Integer ..]]
  349 
  350 exponentValue :: String -> Int
  351 exponentValue expo =
  352   case expo of
  353     _ : '-' : ds | not (null ds) -> negate (fromMaybe 0 (readMaybe ds))
  354     _ : '+' : ds | not (null ds) -> fromMaybe 0 (readMaybe ds)
  355     _ : ds | not (null ds) -> fromMaybe 0 (readMaybe ds)
  356     _ -> 0

  358 withOptionalMagicHashSuffix ::
  359   Int ->
  360   LexerEnv ->
  361   LexerState ->
  362   Text ->
  363   LexTokenKind ->
  364   (Text -> LexTokenKind) ->
  365   (Text, LexTokenKind, LexerState)
  366 withOptionalMagicHashSuffix maxHashes env st raw plainKind hashKind =
  367   let st' = advanceChars raw st
  368       hashCount =
  369         if hasExt MagicHash env
  370           then min maxHashes (T.length (T.takeWhile (== '#') (lexerInput st')))
  371           else 0
  372    in case hashCount of
  373         0 -> (raw, plainKind, st')
  374         _ ->
  375           let hashes = T.replicate hashCount "#"
  376               rawHash = raw <> hashes
  377            in (rawHash, hashKind rawHash, advanceChars hashes st')
