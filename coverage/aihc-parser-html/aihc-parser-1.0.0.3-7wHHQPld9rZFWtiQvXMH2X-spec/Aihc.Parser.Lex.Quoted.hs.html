   21 -- | Scan a quoted string body (after the opening delimiter) until the
   22 -- unescaped closing character.  Returns @Right (body, rest)@ on success or
   23 -- @Left body@ if the input ends without a closing delimiter.
   24 scanQuoted :: Char -> Text -> Either Text (Text, Text)
   25 scanQuoted endCh input = go 0 input
   26   where
   27     go consumed rest =
   28       case T.findIndex (\c -> c == '\\' || c == endCh) rest of
   29         Nothing -> Left (T.take (consumed + T.length rest) input)
   30         Just i ->
   31           let consumed' = consumed + i
   32               special = T.drop i rest
   33            in if T.null special
   34                 then Left (T.take consumed' input)
   35                 else case T.head special of
   36                   c
   37                     | c == endCh -> Right (T.take consumed' input, T.drop 1 special)
   38                     | otherwise ->
   39                         case consumedEscapeTail (T.drop 1 special) of
   40                           Just tailLen -> go (consumed' + 1 + tailLen) (T.drop tailLen (T.drop 1 special))
   41                           Nothing -> Left (T.take (consumed' + 1) input)

   43 -- | Scan a multiline string body (after the opening @\"\"\"@) until an
   44 -- unescaped closing @\"\"\"@.  Returns @Right (body, rest)@ on success or
   45 -- @Left body@ if the input ends without a closing delimiter.
   46 scanMultilineString :: Text -> Either Text (Text, Text)
   47 scanMultilineString input = go 0 input
   48   where
   49     go consumed rest =
   50       case T.findIndex (\c -> c == '\\' || c == '"') rest of
   51         Nothing -> Left (T.take (consumed + T.length rest) input)
   52         Just i ->
   53           let consumed' = consumed + i
   54               special = T.drop i rest
   55            in if T.null special
   56                 then Left (T.take consumed' input)
   57                 else case T.head special of
   58                   '\\' ->
   59                     case consumedEscapeTail (T.drop 1 special) of
   60                       Just tailLen -> go (consumed' + 1 + tailLen) (T.drop tailLen (T.drop 1 special))
   61                       Nothing -> Left (T.take (consumed' + 1) input)
   62                   '"' ->
   63                     if "\"\"\"" `T.isPrefixOf` special
   64                       then Right (T.take consumed' input, T.drop 3 special)
   65                       else
   66                         if "\"\"" `T.isPrefixOf` special
   67                           then go (consumed' + 2) (T.drop 2 special)
   68                           else go (consumed' + 1) (T.drop 1 special)
   69                   _ -> error "unreachable: findIndex only returns backslash or quote"

   71 -- | Determine how much text after a backslash belongs to the current
   72 -- escape-like sequence for delimiter scanning purposes.
   73 --
   74 -- We intentionally recognize string gaps here so a gap-closing backslash does
   75 -- not incorrectly escape the following quote.
   76 consumedEscapeTail :: Text -> Maybe Int
   77 consumedEscapeTail rest = do
   78   (_, _, consumed) <- parseEscapeWithLength rest
   79   pure consumed

   81 -- | Decode the body of a Haskell string literal (content between the quotes,
   82 -- without the surrounding @\"@ characters) natively on 'Text', avoiding the
   83 -- round-trip through 'String'.  Returns 'Nothing' if the body contains an
   84 -- invalid escape sequence, in which case the caller should fall back to the
   85 -- raw body.
   86 decodeStringBody :: Text -> Maybe Text
   87 decodeStringBody inp
   88   | not ('\\' `T.elem` inp) = Just inp -- fast path: no escapes, no allocation
   89   | otherwise = TL.toStrict . TLB.toLazyText <$> go mempty inp
   90   where
   91     go :: TLB.Builder -> Text -> Maybe TLB.Builder
   92     go !acc t =
   93       let (plain, rest) = T.break (== '\\') t
   94           acc' = acc <> TLB.fromText plain
   95        in case T.uncons rest of
   96             Nothing -> Just acc'
   97             Just ('\\', after) -> case parseEscape after of
   98               Nothing -> Nothing
   99               Just (mc, rest') ->
  100                 go (maybe acc' (\c -> acc' <> TLB.singleton c) mc) rest'
  101             _ -> Just acc' -- unreachable: T.break stops at '\\'

  103 isOctDigit :: Char -> Bool
  104 isOctDigit c = c >= '0' && c <= '7'
  105 
  106 parseEscape :: Text -> Maybe (Maybe Char, Text)
  107 parseEscape = fmap (\(decoded, rest, _) -> (decoded, rest)) . parseEscapeWithLength
  108 
  109 parseEscapeWithLength :: Text -> Maybe (Maybe Char, Text, Int)
  110 parseEscapeWithLength t = case T.uncons t of
  111   Nothing -> Nothing
  112   Just (c, rest) -> case c of
  113     'a' -> consumedOne (Just '\a') rest
  114     'b' -> consumedOne (Just '\b') rest
  115     'f' -> consumedOne (Just '\f') rest
  116     'n' -> consumedOne (Just '\n') rest
  117     'r' -> consumedOne (Just '\r') rest
  118     't' -> consumedOne (Just '\t') rest
  119     'v' -> consumedOne (Just '\v') rest
  120     '\\' -> consumedOne (Just '\\') rest
  121     '"' -> consumedOne (Just '"') rest
  122     '\'' -> consumedOne (Just '\'') rest
  123     '&' -> consumedOne Nothing rest -- empty escape
  124     '^' -> case T.uncons rest of -- control character \^X
  125       Just (cc, rest')
  126         | cc >= '@' && cc <= '_' ->
  127             Just (Just (chr (ord cc - 64)), rest', 2)
  128       _ -> Nothing
  129     'x' ->
  130       -- hex escape \xNN  (use Integer to prevent Int overflow on long inputs)
  131       let (digits, rest') = T.span isHexDigit rest
  132        in if T.null digits
  133             then Nothing
  134             else decodeNumericEscape 16 1 digits rest'
  135     'o' ->
  136       -- octal escape \oNN  (use Integer to prevent Int overflow on long inputs)
  137       let (digits, rest') = T.span isOctDigit rest
  138        in if T.null digits
  139             then Nothing
  140             else decodeNumericEscape 8 1 digits rest'
  141     _
  142       | isDigit c -> -- decimal escape \NNN  (use Integer to prevent Int overflow on long inputs)
  143           let (moreDigits, rest') = T.span isDigit rest
  144               digits = T.cons c moreDigits
  145            in decodeNumericEscape 10 0 digits rest'
  146       | isSpace c -> -- gap escape \ whitespace \
  147           let (spaces, rest') = T.span isSpace rest
  148            in case T.uncons rest' of
  149                 Just ('\\', rest'') -> Just (Nothing, rest'', 2 + T.length spaces)
  150                 _ -> Nothing
  151       | otherwise -> parseNamedEscapeWithLength t
  152   where
  153     consumedOne decoded rest' = Just (decoded, rest', 1)
  154 
  155     decodeNumericEscape :: Integer -> Int -> Text -> Text -> Maybe (Maybe Char, Text, Int)
  156     decodeNumericEscape !base prefixLength digits rest' =
  157       let n = T.foldl' (\a d -> a * base + toInteger (digitToInt d)) (0 :: Integer) digits
  158           consumed = prefixLength + T.length digits
  159        in if n > 0x10FFFF then Nothing else Just (Just (chr (fromIntegral n)), rest', consumed)
  160 
  161 parseNamedEscapeWithLength :: Text -> Maybe (Maybe Char, Text, Int)
  162 parseNamedEscapeWithLength t = foldr tryMatch Nothing namedEscapeTable
  163   where
  164     tryMatch (name, ch) fallback =
  165       case T.stripPrefix name t of
  166         Just rest -> Just (Just ch, rest, T.length name)
  167         Nothing -> fallback

  209 processMultilineString :: String -> String
  210 processMultilineString =
  211   resolveEscapes
  212     . stripTrailingNewline
  213     . stripLeadingNewline
  214     . List.intercalate "\n"
  215     . map blankToEmpty
  216     . stripCommonIndent
  217     . map expandLeadingTabs
  218     . splitMultilineNewlines
  219     . collapseStringGaps
  220 
  221 collapseStringGaps :: String -> String
  222 collapseStringGaps [] = []
  223 collapseStringGaps ('\\' : rest)
  224   | not (null ws), '\\' : rest'' <- rest' = collapseStringGaps rest''
  225   where
  226     (ws, rest') = span (\c -> c == ' ' || c == '\t' || c == '\n' || c == '\r' || c == '\f') rest
  227 collapseStringGaps (c : rest) = c : collapseStringGaps rest
  228 
  229 splitMultilineNewlines :: String -> [String]
  230 splitMultilineNewlines = go []
  231   where
  232     go acc [] = [reverse acc]
  233     go acc ('\r' : '\n' : rest) = reverse acc : go [] rest
  234     go acc ('\r' : rest) = reverse acc : go [] rest
  235     go acc ('\n' : rest) = reverse acc : go [] rest
  236     go acc ('\f' : rest) = reverse acc : go [] rest
  237     go acc (c : rest) = go (c : acc) rest
  238 
  239 expandLeadingTabs :: String -> String
  240 expandLeadingTabs = go 0
  241   where
  242     go col ('\t' : rest) =
  243       let spaces = 8 - (col `mod` 8)
  244        in replicate spaces ' ' ++ go (col + spaces) rest
  245     go col (' ' : rest) = ' ' : go (col + 1) rest
  246     go _ rest = rest

  248 stripCommonIndent :: [String] -> [String]
  249 stripCommonIndent lns =
  250   case mapMaybe indentOf nonBlank of
  251     [] -> lns
  252     indents -> map (dropPrefix (minimum indents)) lns
  253   where
  254     nonBlank = filter (not . all isSpace) (drop 1 lns)
  255     indentOf s = Just (length (takeWhile isSpace s))
  256     dropPrefix = drop
  257 
  258 blankToEmpty :: String -> String
  259 blankToEmpty s
  260   | all isSpace s = ""
  261   | otherwise = s
  262 
  263 stripLeadingNewline :: String -> String
  264 stripLeadingNewline ('\n' : rest) = rest
  265 stripLeadingNewline s = s
  266 
  267 stripTrailingNewline :: String -> String
  268 stripTrailingNewline s
  269   | not (null s) && last s == '\n' = init s
  270   | otherwise = s
  271 
  272 resolveEscapes :: String -> String
  273 resolveEscapes s =
  274   case reads ('"' : s ++ "\"") of
  275     [(str, "")] -> str
  276     _ -> s
  277 
  278 readMaybeChar :: Text -> Maybe Char
  279 readMaybeChar raw =
  280   case reads (T.unpack raw) of
  281     [(c, "")] -> Just c
  282     _ -> Nothing
