   18 -- | Parse a command (the body of a @proc@ abstraction).
   19 --
   20 -- Grammar (simplified):
   21 --
   22 -- @
   23 -- cmd   = exp10 -\< exp | exp10 -\<\< exp | cmd0
   24 -- cmd0  = cmd10 (op cmd10)*
   25 -- cmd10 = do { cstmts } | if … | case … | let … | \\pats -> cmd | fcmd
   26 -- fcmd  = fcmd aexp | (cmd)
   27 -- @
   28 cmdParser :: TokParser Cmd
   29 cmdParser = do
   30   tok <- lookAhead anySingle
   31   case lexTokenKind tok of
   32     TkKeywordDo -> cmd0Parser
   33     TkKeywordIf -> cmd0Parser
   34     TkKeywordCase -> cmd0Parser
   35     TkKeywordLet -> cmd0Parser
   36     TkReservedBackslash -> cmd0Parser
   37     TkSpecialLParen -> MP.try cmd0Parser <|> cmdArrAppParser
   38     _ -> MP.try cmdArrAppParser <|> cmd0Parser
   39 
   40 cmd0Parser :: TokParser Cmd
   41 cmd0Parser = do
   42   lhs <- cmd10Parser
   43   rest <-
   44     MP.many
   45       ( (,)
   46           <$> infixOperatorParser
   47           <*> cmd10Parser
   48       )
   49   pure (foldl buildCmdInfix lhs rest)
   50   where
   51     buildCmdInfix l (op, r) = CmdInfix l op r
   52 
   53 cmd10Parser :: TokParser Cmd
   54 cmd10Parser = do
   55   tok <- lookAhead anySingle
   56   case lexTokenKind tok of
   57     TkKeywordDo -> cmdDoParser
   58     TkKeywordIf -> cmdIfParser
   59     TkKeywordCase -> cmdCaseParser
   60     TkKeywordLet ->
   61       -- 'let decls in cmd' is a command; 'let decls' (without 'in') inside
   62       -- a do-block is a statement, not handled here.
   63       cmdLetParser
   64     TkReservedBackslash -> cmdLamParser
   65     _ -> cmdFcmdParser
   66 
   67 cmdArrAppParser :: TokParser Cmd
   68 cmdArrAppParser = do
   69   expr <- cmdArrAppLhsParser
   70   (appType, rhs) <- cmdArrTailParser
   71   pure (CmdArrApp expr appType rhs)
   72 
   73 cmdFcmdParser :: TokParser Cmd
   74 cmdFcmdParser = do
   75   headCmd <- cmdParenParser
   76   args <- MP.many atomExprParser
   77   pure (foldl CmdApp headCmd args)
   78 
   79 -- | Parse an arrow tail operator in command context, returning the
   80 -- application type and the right-hand expression.
   81 cmdArrTailParser :: TokParser (ArrAppType, Expr)
   82 cmdArrTailParser = do
   83   appType <- tokenSatisfy "arrow operator" $ \tok ->
   84     case lexTokenKind tok of
   85       TkArrowTail -> Just HsFirstOrderApp
   86       TkDoubleArrowTail -> Just HsHigherOrderApp
   87       _ -> Nothing
   88   rhs <- exprParser
   89   pure (appType, rhs)
   90 
   91 -- | Parse a command do-block: @do { cstmt ; ... }@
   92 cmdDoParser :: TokParser Cmd
   93 cmdDoParser = withSpanAnn (CmdAnn . mkAnnotation) $ do
   94   expectedTok TkKeywordDo
   95   CmdDo <$> bracedSemiSep1 cmdStmtParser
   96 
   97 -- | Parse a command if-then-else: @if exp then cmd else cmd@
   98 cmdIfParser :: TokParser Cmd
   99 cmdIfParser = withSpanAnn (CmdAnn . mkAnnotation) $ do
  100   expectedTok TkKeywordIf
  101   cond <- exprParser
  102   skipSemicolons
  103   expectedTok TkKeywordThen
  104   yes <- cmdParser
  105   skipSemicolons
  106   expectedTok TkKeywordElse
  107   CmdIf cond yes <$> cmdParser
  108 
  109 -- | Parse a command case: @case exp of { calts }@
  110 cmdCaseParser :: TokParser Cmd
  111 cmdCaseParser = withSpanAnn (CmdAnn . mkAnnotation) $ do
  112   expectedTok TkKeywordCase
  113   scrut <- region "while parsing case scrutinee" exprParser
  114   expectedTok TkKeywordOf
  115   alts <- bracedSemiSep1 cmdCaseAltParser
  116   pure (CmdCase scrut alts)
  117 
  118 cmdCaseAltParser :: TokParser (CaseAlt Cmd)
  119 cmdCaseAltParser = withSpan $ do
  120   pat <- caseAltPatternParser
  121   rhs <- caseRhsParserWithBodyParser cmdParser
  122   pure (\span' -> CaseAlt [mkAnnotation span'] pat rhs)
  123 
  124 -- | Parse a command let: @let decls in cmd@
  125 cmdLetParser :: TokParser Cmd
  126 cmdLetParser = withSpanAnn (CmdAnn . mkAnnotation) $ do
  127   decls <- parseLetDeclsParser
  128   expectedTok TkKeywordIn
  129   CmdLet decls <$> cmdParser
  130 
  131 -- | Parse a command lambda: @\\pats -> cmd@
  132 cmdLamParser :: TokParser Cmd
  133 cmdLamParser = withSpanAnn (CmdAnn . mkAnnotation) $ do
  134   expectedTok TkReservedBackslash
  135   pats <- MP.some apatParser
  136   expectedTok TkReservedRightArrow
  137   CmdLam pats <$> cmdParser
  138 
  139 -- | Parse a parenthesised command: @( cmd )@
  140 cmdParenParser :: TokParser Cmd
  141 cmdParenParser =
  142   withSpanAnn (CmdAnn . mkAnnotation) $
  143     CmdPar <$> parens cmdParser
  144 
  145 -- | Parse a do-statement in command context (arrow do).
  146 cmdStmtParser :: TokParser (DoStmt Cmd)
  147 cmdStmtParser = do
  148   tok <- lookAhead anySingle
  149   case lexTokenKind tok of
  150     TkKeywordLet -> MP.try cmdLetStmtParser <|> cmdBodyStmtParser
  151     TkKeywordRec -> cmdRecStmtParser
  152     -- Keyword commands: parse as command body statements.
  153     TkKeywordDo -> cmdBodyStmtParser
  154     TkKeywordIf -> cmdBodyStmtParser
  155     TkKeywordCase -> cmdBodyStmtParser
  156     TkReservedBackslash -> cmdBodyStmtParser
  157     TkSpecialLParen -> MP.try cmdBindOrBodyStmtParser <|> MP.try cmdBindStmtParser <|> cmdBodyStmtParser
  158     _ -> do
  159       isPatternBind <- startsWithPatternBind
  160       if isPatternBind
  161         then cmdBindStmtParser
  162         else cmdBindOrBodyStmtParser
  163 
  164 startsWithPatternBind :: TokParser Bool
  165 startsWithPatternBind =
  166   fmap (either (const False) (const True)) . MP.observing . MP.try . MP.lookAhead $ do
  167     _ <- patternParser
  168     expectedTok TkReservedLeftArrow
  169 
  170 -- | Parse a command do-statement: @cmd@ or @pat <- cmd@.
  171 -- Uses the expression-first approach: parse as expression, check for @<-@.
  172 cmdBindOrBodyStmtParser :: TokParser (DoStmt Cmd)
  173 cmdBindOrBodyStmtParser = withSpanAnn (DoAnn . mkAnnotation) $ do
  174   -- Arrow tails (-<, -<<) belong to the command level, not the expression.
  175   expr <- exprParser
  176   mArrow <- MP.optional (expectedTok TkReservedLeftArrow)
  177   case mArrow of
  178     Just () -> DoBind <$> liftCheck (checkPattern expr) <*> cmdParser
  179     Nothing -> do
  180       -- No bind arrow: this is a body statement.  Check for arrow tail.
  181       mArrTail <- MP.optional cmdArrTailParser
  182       case mArrTail of
  183         Just (appType, rhs) ->
  184           pure (DoExpr (CmdArrApp expr appType rhs))
  185         Nothing -> do
  186           mTok <- MP.optional (lookAhead anySingle)
  187           MP.customFailure
  188             UnexpectedTokenExpecting
  189               { unexpectedFound = mkFoundToken <$> mTok,
  190                 unexpectedExpecting = "arrow command (-< or -<<) in do statement",
  191                 unexpectedContext = []
  192               }
  193 
  194 -- | Parse a command bind statement where the pattern is unambiguously a
  195 -- pattern (starts with !, ~, or x@).
  196 cmdBindStmtParser :: TokParser (DoStmt Cmd)
  197 cmdBindStmtParser = withSpanAnn (DoAnn . mkAnnotation) $ do
  198   pat <- patternParser
  199   expectedTok TkReservedLeftArrow
  200   cmd <- region "while parsing '<-' binding" cmdParser
  201   pure (DoBind pat cmd)
  202 
  203 -- | Parse a body-only command statement (fallback from cmdStmtParser).
  204 cmdBodyStmtParser :: TokParser (DoStmt Cmd)
  205 cmdBodyStmtParser =
  206   withSpanAnn (DoAnn . mkAnnotation) $
  207     DoExpr <$> cmdParser
  208 
  209 -- | Parse a command let-statement: @let decls@
  210 cmdLetStmtParser :: TokParser (DoStmt Cmd)
  211 cmdLetStmtParser =
  212   withSpanAnn (DoAnn . mkAnnotation) $
  213     DoLetDecls <$> parseLetDeclsStmtParser
  214 
  215 -- | Parse a command rec-statement: @rec { cstmts }@
  216 cmdRecStmtParser :: TokParser (DoStmt Cmd)
  217 cmdRecStmtParser = withSpanAnn (DoAnn . mkAnnotation) $ do
  218   expectedTok TkKeywordRec
  219   DoRecStmt <$> bracedSemiSep1 cmdStmtParser
