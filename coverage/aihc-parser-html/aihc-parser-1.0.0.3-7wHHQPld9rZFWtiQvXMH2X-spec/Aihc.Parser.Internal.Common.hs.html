   94 label :: Text -> TokParser a -> TokParser a
   95 label expected parser = do
   96   outcome <- MP.observing parser
   97   case outcome of
   98     Right parsed -> pure parsed
   99     Left err ->
  100       case err of
  101         MPE.TrivialError off _ _ -> do
  102           mTok <- MP.optional (lookAhead anySingle)
  103           let mFound = mkFoundToken <$> mTok
  104           MP.parseError $
  105             MPE.FancyError
  106               off
  107               ( Set.singleton
  108                   ( MPE.ErrorCustom
  109                       UnexpectedTokenExpecting
  110                         { unexpectedFound = mFound,
  111                           unexpectedExpecting = expected,
  112                           unexpectedContext = []
  113                         }
  114                   )
  115               )
  116         _ -> MP.parseError err

  118 region :: Text -> TokParser a -> TokParser a
  119 region context =
  120   MP.region addContextToError
  121   where
  122     addContextToError err =
  123       case err of
  124         MPE.FancyError off fancySet ->
  125           MPE.FancyError off (Set.map appendContext fancySet)
  126         _ -> err
  127     appendContext fancyErr =
  128       case fancyErr of
  129         MPE.ErrorCustom custom ->
  130           case custom of
  131             UnexpectedTokenExpecting found expecting contexts ->
  132               MPE.ErrorCustom (UnexpectedTokenExpecting found expecting (contexts <> [context]))
  133         _ -> fancyErr

  135 -- | Match a specific token kind exactly.
  136 expectedTok :: LexTokenKind -> TokParser ()
  137 expectedTok expected =
  138   tokenSatisfy (renderTokenKind expected) $ \tok ->
  139     if lexTokenKind tok == expected then Just () else Nothing
  140 {-# INLINE expectedTok #-}

  142 -- | Match the end-of-file token.
  143 --
  144 -- The lexer emits a 'TkEOF' token at the end of input. This parser consumes
  145 -- that token, ensuring the entire input has been processed.
  146 eofTok :: TokParser ()
  147 eofTok =
  148   tokenSatisfy "end of input" $ \tok ->
  149     if lexTokenKind tok == TkEOF then Just () else Nothing

  151 -- | Match a specific variable identifier (contextual keyword).
  152 varIdTok :: Text -> TokParser ()
  153 varIdTok expected =
  154   tokenSatisfy ("identifier '" <> T.unpack expected <> "'") $ \tok ->
  155     case lexTokenKind tok of
  156       TkVarId ident | ident == expected -> Just ()
  157       _ -> Nothing

  175   TkReservedEquals -> "operator '='"

  224 tokenSatisfy :: String -> (LexToken -> Maybe a) -> TokParser a
  225 tokenSatisfy expectedLabel f =
  226   MP.token f expectedItems
  227   where
  228     expectedItems =
  229       Set.singleton $
  230         if null expectedLabel
  231           then MPE.EndOfInput
  232           else MPE.Label (NE.fromList expectedLabel)
  233 {-# INLINE tokenSatisfy #-}

  235 hiddenPragma :: String -> (Pragma -> Maybe a) -> TokParser a
  236 hiddenPragma expectedLabel f = do
  237   mResult <- optionalHiddenPragma f
  238   case mResult of
  239     Just result -> pure result
  240     Nothing -> fail expectedLabel
  241 
  242 optionalHiddenPragma :: (Pragma -> Maybe a) -> TokParser (Maybe a)
  243 optionalHiddenPragma f = do
  244   pst <- MP.getParserState
  245   case spanNoMatch (tokStreamPendingPragmas (MP.stateInput pst)) of
  246     (ignored, pragmaTok : rest)
  247       | Just result <- f pragmaTok -> do
  248           MP.updateParserState $ \st ->
  249             st
  250               { MP.stateInput =
  251                   (MP.stateInput st)
  252                     { tokStreamPendingPragmas = ignored <> rest
  253                     }
  254               }
  255           pure (Just result)
  256       | otherwise -> pure Nothing
  257     _ -> pure Nothing
  258   where
  259     spanNoMatch pragmas =
  260       case pragmas of
  261         pragmaTok : rest
  262           | Just _ <- f pragmaTok -> ([], pragmaTok : rest)
  263           | otherwise ->
  264               let (ignored, remaining) = spanNoMatch rest
  265                in (pragmaTok : ignored, remaining)
  266         [] -> ([], [])

  268 moduleNameParser :: TokParser Text
  269 moduleNameParser =
  270   label "module name" $
  271     tokenSatisfy "module name" $ \tok ->
  272       case lexTokenKind tok of
  273         TkConId ident | isModuleName ident -> Just ident

  277 identifierNameWithTokenParser :: TokParser (LexToken, Name)
  278 identifierNameWithTokenParser =
  279   tokenSatisfy "identifier" $ \tok ->
  280     case lexTokenKind tok of
  281       TkVarId ident -> Just (tok, qualifyName Nothing (mkUnqualifiedNameAt tok NameVarId ident))
  282       TkConId ident -> Just (tok, qualifyName Nothing (mkUnqualifiedNameAt tok NameConId ident))
  283       TkQVarId modName ident -> Just (tok, mkNameAt tok (Just modName) NameVarId ident)
  284       TkQConId modName ident -> Just (tok, mkNameAt tok (Just modName) NameConId ident)
  285       _ -> Nothing

  386 -- | Parse an infix operator name (varop) for function definitions.
  387 -- Per Haskell Report section 4.4.3, funlhs uses 'varop' which is:
  388 --   varop → varsym | ` varid `
  389 -- This excludes constructor operators (consym) and qualified operators.
  390 -- Note: Whitespace-sensitive lexing (GHC proposal 0229) now distinguishes
  391 -- TkVarSym "!" (infix operator) from TkPrefixBang (bang pattern), so we
  392 -- can accept all VarSym operators here.
  393 infixOperatorNameParser :: TokParser UnqualifiedName
  394 infixOperatorNameParser =
  395   symbolicOperatorParser <|> backtickIdentifierParser
  396   where
  397     symbolicOperatorParser =
  398       tokenSatisfy "variable operator" $ \tok ->
  399         case lexTokenKind tok of
  400           TkVarSym op -> Just (mkUnqualifiedNameAt tok NameVarSym op)
  401           _ -> Nothing
  402     backtickIdentifierParser = do
  403       expectedTok TkSpecialBacktick
  404       op <- varIdNameParser
  405       expectedTok TkSpecialBacktick
  406       pure op
  407     varIdNameParser =
  408       tokenSatisfy "variable identifier" $ \tok ->
  409         case lexTokenKind tok of
  410           TkVarId name -> Just (mkUnqualifiedNameAt tok NameVarId name)
  411           _ -> Nothing
  412 
  413 -- | Parse an infix constructor operator name (conop) for pattern synonym where clauses.
  414 -- Per Haskell Report, pattern synonym where-clause equations use the constructor
  415 -- name in infix position: @pat ConOp pat = expr@.
  416 -- This is the constructor counterpart of 'infixOperatorNameParser'.
  417 --   conop → consym | ` conid `
  418 constructorInfixOperatorNameParser :: TokParser UnqualifiedName
  419 constructorInfixOperatorNameParser =
  420   symbolicConstructorOperatorParser <|> backtickConstructorIdentifierParser
  421   where
  422     symbolicConstructorOperatorParser =
  423       tokenSatisfy "constructor operator" $ \tok ->
  424         case lexTokenKind tok of
  425           TkConSym op -> Just (mkUnqualifiedNameAt tok NameConSym op)
  426           TkReservedColon -> Just (mkUnqualifiedNameAt tok NameConSym ":")
  427           _ -> Nothing
  428     backtickConstructorIdentifierParser = do
  429       expectedTok TkSpecialBacktick
  430       op <- constructorIdentifierNameParser
  431       expectedTok TkSpecialBacktick
  432       pure op
  433     constructorIdentifierNameParser =
  434       tokenSatisfy "constructor identifier" $ \tok ->
  435         case lexTokenKind tok of
  436           TkConId name -> Just (mkUnqualifiedNameAt tok NameConId name)
  437           _ -> Nothing

  458 withSpanAnn :: (SourceSpan -> a -> a) -> TokParser a -> TokParser a
  459 withSpanAnn f parser = do
  460   startInput <- MP.getInput
  461   out <- parser
  462   endInput <- MP.getInput
  463   let startSpan = inputStartSpan startInput
  464       endSpan = maybe noSourceSpan lexTokenSpan (tokStreamPrevToken endInput)
  465       parserSpan = mergeSourceSpans startSpan endSpan
  466   pure $ f parserSpan out

  481 inputStartSpan :: TokStream -> SourceSpan
  482 inputStartSpan ts
  483   | tokStreamEOFEmitted ts = noSourceSpan
  484   | tok : _ <- tokStreamBuffer ts = lexTokenSpan tok
  485   | rawTok : _ <- tokStreamRawTokens ts = lexTokenSpan rawTok
  486   | otherwise = noSourceSpan

  489 optionalSuffix :: TokParser b -> (a -> b -> a) -> TokParser a -> TokParser a
  490 optionalSuffix suffixParser attach parser = do
  491   base <- parser
  492   mSuffix <- MP.optional suffixParser
  493   pure $
  494     case mSuffix of
  495       Just suffix -> attach base suffix
  496       Nothing -> base

  511     expectedTok closeTok

  514 -- | Expect a @}@ token, closing implicit layout contexts if needed.
  515 -- This implements the parse-error rule for closing braces: if @}@ is not found
  516 -- but there is an implicit layout context, close it (which buffers a virtual @}@)
  517 -- and consume that virtual @}@.
  518 closeAndExpectRBrace :: TokParser ()
  519 closeAndExpectRBrace =
  520   expectedTok TkSpecialRBrace <|> do
  521     closed <- closeImplicitLayout
  522     if closed then expectedTok TkSpecialRBrace else MP.empty

  552 contextItemParserWith :: TokParser Type -> TokParser Type -> TokParser Type
  553 contextItemParserWith typeParser typeAtomParser =
  554   withSpanAnn (TAnn . mkAnnotation) $
  555     MP.try parenthesizedContextItemParser <|> MP.try kindSigContextItemParser <|> bareContextItemParser
  556   where
  557     bareContextItemParser =
  558       do
  559         name <- implicitParamNameParser
  560         expectedTok TkReservedDoubleColon
  561         TImplicitParam name <$> typeParser
  562         <|> do
  563           expectedTok TkKeywordUnderscore
  564           pure TWildcard
  565         <|> constraintTypeParser
  566     parenthesizedContextItemParser = do
  567       expectedTok TkSpecialLParen
  568       item <- contextItemParserWith typeParser typeAtomParser
  569       expectedTok TkSpecialRParen
  570       guardNotFollowedByConstraintInfixOp
  571       pure (TParen item)
  572       where
  573         guardNotFollowedByConstraintInfixOp = do
  574           isFollowed <-
  575             fmap (either (const False) (const True))
  576               . MP.observing
  577               . MP.try
  578               . MP.lookAhead
  579               $ constraintTypeInfixOperatorParser
  580           guard (not isFollowed)
  581     -- \| Parse a type followed by `::` and another type (kind annotation).
  582     -- This handles cases like `(c :: Type -> Constraint)` in superclass contexts,
  583     -- both as standalone parenthesized constraints and as items in comma-separated lists.
  584     -- Uses lookahead to check for `::` at top bracket depth to avoid ambiguity.
  585     -- IMPORTANT: Uses `constraintTypeAppParser` (not `typeParser`) for the left side
  586     -- to avoid a parsing cycle: typeParser -> contextTypeParser -> constraintsParserWith
  587     -- -> constraintParserWith -> kindSigConstraintParser -> typeParser.
  588     kindSigContextItemParser :: TokParser Type
  589     kindSigContextItemParser = do
  590       guard =<< hasKindSignatureAtTopLevel
  591       ty <- constraintTypeAppParser
  592       expectedTok TkReservedDoubleColon
  593       TKindSig ty <$> kindTypeParser
  594 
  595     -- \| Lookahead: check if there's a `::` at the top bracket depth.
  596     -- This avoids ambiguity with the bare constraint parser.
  597     hasKindSignatureAtTopLevel :: TokParser Bool
  598     hasKindSignatureAtTopLevel = MP.lookAhead (go 0)
  599       where
  600         go :: Int -> TokParser Bool
  601         go depth = do
  602           tok <- anySingle
  603           case lexTokenKind tok of
  604             TkEOF -> pure False
  605             TkReservedDoubleColon | depth == 0 -> pure True
  606             TkReservedRightArrow | depth == 0 -> pure False
  607             TkSpecialComma | depth == 0 -> pure False
  608             TkSpecialLParen -> go (depth + 1)
  609             TkSpecialRParen
  610               | depth > 0 -> go (depth - 1)
  611               | otherwise -> pure False
  612             TkSpecialUnboxedLParen -> go (depth + 1)
  613             TkSpecialUnboxedRParen
  614               | depth > 0 -> go (depth - 1)
  615               | otherwise -> pure False
  616             TkSpecialLBracket -> go (depth + 1)
  617             TkSpecialRBracket
  618               | depth > 0 -> go (depth - 1)
  619               | otherwise -> pure False
  620             _ -> go depth
  621     constraintTypeParser = do
  622       first <- constraintTypeAppParser
  623       rest <- MP.many ((,) <$> constraintTypeInfixOperatorParser <*> constraintTypeAppParser)
  624       pure (foldInfixR buildInfixType first rest)
  625     constraintTypeAppParser = do
  626       first <- typeAtomParser
  627       rest <- MP.many constraintTypeAppArgParser
  628       pure (foldl applyConstraintAppArg first rest)
  629     constraintTypeAppArgParser =
  630       (Left <$> MP.try (expectedTok TkTypeApp *> (typeAtomParser >>= rejectBareConstraintImplicitParam)))
  631         <|> (Right <$> (typeAtomParser >>= rejectBareConstraintImplicitParam))
  632     applyConstraintAppArg fn (Left ty) = TTypeApp fn ty
  633     applyConstraintAppArg fn (Right ty) = TApp fn ty
  634     rejectBareConstraintImplicitParam ty =
  635       case peelTypeAnn ty of
  636         TImplicitParam {} -> fail "implicit parameter type must be parenthesized"
  637         _ -> pure ty
  638     -- \| Parse a type expression that can appear as a kind annotation.
  639     -- Handles function types (e.g., Type -> Constraint) and type application,
  640     -- but NOT context types (C a => ...) to avoid parsing cycles.
  641     kindTypeParser = do
  642       first <- constraintTypeAppParser
  643       rest <- MP.many ((,) <$> constraintTypeInfixOperatorParser <*> constraintTypeAppParser)
  644       let baseType = foldInfixR buildInfixType first rest
  645       mRhs <- MP.optional (expectedTok TkReservedRightArrow *> kindTypeParser)
  646       case mRhs of
  647         Just rhs ->
  648           pure (TFun ArrowUnrestricted baseType rhs)
  649         Nothing -> pure baseType
  650     buildInfixType lhs ((op, promoted), rhs) =
  651       TInfix lhs op promoted rhs
  652     constraintTypeInfixOperatorParser =
  653       MP.try promotedInfixOperatorParser <|> backtickConstraintOperatorParser <|> unpromotedInfixOperatorParser
  654     backtickConstraintOperatorParser = MP.try $ do
  655       expectedTok TkSpecialBacktick
  656       op <- constraintOperatorIdentifierParser
  657       expectedTok TkSpecialBacktick
  658       pure (op, Unpromoted)
  659     constraintOperatorIdentifierParser =
  660       tokenSatisfy "constraint operator identifier" $ \tok ->
  661         case lexTokenKind tok of
  662           TkVarId name -> Just (qualifyName Nothing (mkUnqualifiedNameAt tok NameVarId name))
  663           TkConId name -> Just (qualifyName Nothing (mkUnqualifiedNameAt tok NameConId name))
  664           _ -> Nothing
  665     unpromotedInfixOperatorParser =
  666       tokenSatisfy "type infix operator" $ \tok ->
  667         case lexTokenKind tok of
  668           TkVarSym op
  669             | op /= "."
  670                 && op /= "!" ->
  671                 Just (qualifyName Nothing (mkUnqualifiedNameAt tok NameVarSym op), Unpromoted)
  672           TkConSym op -> Just (qualifyName Nothing (mkUnqualifiedNameAt tok NameConSym op), Unpromoted)
  673           TkQVarSym modName op ->
  674             Just (mkNameAt tok (Just modName) NameVarSym op, Unpromoted)
  675           TkQConSym modName op -> Just (mkNameAt tok (Just modName) NameConSym op, Unpromoted)
  676           _ -> Nothing
  677     promotedInfixOperatorParser = do
  678       expectedTok (TkVarSym "'")
  679       expectedTok TkReservedColon
  680       pure (qualifyName Nothing (mkUnqualifiedName NameConSym ":"), Promoted)

  682 contextItemsParserWith :: TokParser Type -> TokParser Type -> TokParser [Type]

  699           $ constraintInfixOpStartParser

  707                 Just ()

  709           TkQVarSym _ _ -> Just ()
  710           TkQConSym _ _ -> Just ()

  726 -- | Parse either a plain type signature or a typed binding that must be

  728 typedBindingOrSignatureParser ::
  729   TokParser ty ->
  730   ([UnqualifiedName] -> ty -> a) ->
  731   (UnqualifiedName -> ty -> TokParser a) ->
  732   String ->

  746 functionHeadParserWithBinder :: TokParser UnqualifiedName -> TokParser UnqualifiedName -> TokParser Pattern -> TokParser Pattern -> TokParser (MatchHeadForm, UnqualifiedName, [Pattern])
  747 functionHeadParserWithBinder binderParser infixOpParser infixPatternParser prefixPatternParser = do
  748   (firstKind, secondKind) <-
  749     lookAhead $ do
  750       first <- lexTokenKind <$> anySingle
  751       second <- lexTokenKind <$> anySingle
  752       pure (first, second)
  753   case firstKind of
  754     TkVarId {}
  755       | startsPrefixHead secondKind -> prefixHeadParser
  756       | otherwise -> MP.try infixHeadParser <|> prefixHeadParser
  757     TkSpecialLParen ->
  758       MP.try parenthesizedInfixHeadParser
  759         <|> MP.try infixHeadParser
  760         <|> prefixHeadParser
  761     _ -> infixHeadParser
  762   where
  763     startsPrefixHead kind =
  764       case kind of
  765         TkVarId {} -> True
  766         TkConId {} -> True
  767         TkQConId {} -> True
  768         TkInteger {} -> True
  769         TkFloat {} -> True
  770         TkChar {} -> True
  771         TkCharHash {} -> True
  772         TkString {} -> True
  773         TkStringHash {} -> True
  774         TkSpecialLParen -> True
  775         TkSpecialUnboxedLParen -> True
  776         TkSpecialLBracket -> True
  777         TkKeywordUnderscore -> True
  778         TkPrefixBang -> True
  779         TkPrefixTilde -> True
  780         TkPrefixMinus -> True
  781         TkQuasiQuote {} -> True
  782         TkTypeApp -> True
  783         TkReservedEquals -> True
  784         TkReservedPipe -> True
  785         _ -> False
  786 
  787     prefixHeadParser = do
  788       name <- binderParser
  789       pats <- MP.many prefixPatternParser
  790       pure (MatchHeadPrefix, name, pats)

  792     infixHeadParser = do
  793       lhsPat <- infixPatternParser
  794       op <- infixOpParser
  795       rhsPat <- infixPatternParser
  796       pure (MatchHeadInfix, op, [lhsPat, rhsPat])
  797 
  798     parenthesizedInfixHeadParser = do
  799       expectedTok TkSpecialLParen
  800       lhsPat <- infixPatternParser
  801       op <- infixOpParser
  802       rhsPat <- infixPatternParser
  803       expectedTok TkSpecialRParen
  804       tailPats <- MP.many prefixPatternParser
  805       pure (MatchHeadInfix, op, [lhsPat, rhsPat] <> tailPats)

  833         { matchAnns = [],

  835           matchPats = pats,

  837         }
  838     ]

  892 -- | Signal to the layout engine that a virtual close brace should be inserted.
  893 -- This implements the parse-error rule: when the parser encounters a token that
  894 -- is illegal in the current context but @}@ would be legal, it calls this to
  895 -- close the innermost implicit layout context.
  896 --
  897 -- Returns @True@ if a layout was closed, @False@ if there was no implicit
  898 -- layout context to close.
  899 closeImplicitLayout :: TokParser Bool
  900 closeImplicitLayout = do
  901   pst <- MP.getParserState
  902   let ts = MP.stateInput pst
  903   case closeImplicitLayoutContext (tokStreamLayoutState ts) of
  904     Nothing -> pure False
  905     Just laySt' -> do
  906       let inserted = layoutBuffer laySt'
  907           laySt'' = laySt' {layoutBuffer = []}
  908       MP.updateParserState
  909         ( \s ->
  910             let input = MP.stateInput s
  911              in s
  912                   { MP.stateInput =
  913                       input
  914                         { tokStreamLayoutState = laySt'',
  915                           tokStreamBuffer = inserted <> tokStreamBuffer input
  916                         }
  917                   }
  918         )
  919       pure True

  921 -- | Like Megaparsec's 'MP.sepEndBy' but implements the parse-error rule for

  925 -- @R { f = case y of A -> 1, g = 2 }@

  927 -- where the comma is a record field separator but appears inside the implicit

  950 -- | Try to match a separator token. If that fails, try closing an implicit
  951 -- layout context and then matching the separator. This implements the
  952 -- parse-error rule: if a token is illegal in the current context but would
  953 -- be legal after inserting a virtual @}@, insert the @}@ and retry.

  966   st <- MP.getParserState

  971 -- | Non-consuming lookahead dispatch for optional context types.
  972 -- Uses scanning to probe for @=>@ at top bracket depth.
  973 -- Returns 'True' when the input looks like a context.
  974 startsWithContextType :: TokParser Bool
  975 startsWithContextType = MP.lookAhead (go [])
  976   where
  977     go :: [LexTokenKind] -> TokParser Bool
  978     go [] = do
  979       tok <- anySingle
  980       case lexTokenKind tok of
  981         TkEOF -> pure False
  982         TkReservedDoubleArrow -> pure True
  983         TkReservedDoubleColon -> pure False
  984         TkReservedRightArrow -> pure False
  985         TkReservedEquals -> pure False
  986         TkSpecialComma -> pure False
  987         TkSpecialSemicolon -> pure False
  988         TkReservedPipe -> pure False
  989         TkSpecialRParen -> pure False
  990         TkSpecialUnboxedRParen -> pure False
  991         TkSpecialRBracket -> pure False
  992         TkSpecialRBrace -> pure False
  993         TkTHExpQuoteClose -> pure False
  994         TkTHTypedQuoteClose -> pure False
  995         TkSpecialLParen -> go [TkSpecialRParen]
  996         TkSpecialUnboxedLParen -> go [TkSpecialUnboxedRParen]
  997         TkSpecialLBracket -> go [TkSpecialRBracket]
  998         TkTHExpQuoteOpen -> go [TkTHExpQuoteClose]
  999         TkTHTypedQuoteOpen -> go [TkTHTypedQuoteClose]
 1000         TkTHDeclQuoteOpen -> go [TkTHExpQuoteClose]
 1001         TkTHTypeQuoteOpen -> go [TkTHExpQuoteClose]
 1002         TkTHPatQuoteOpen -> go [TkTHExpQuoteClose]
 1003         TkSpecialLBrace -> go [TkSpecialRBrace]
 1004         -- Keywords that cannot appear inside a type expression: stop scanning.
 1005         -- This also prevents an enclosing expression form (such as if/then/else)
 1006         -- from being mistaken for a later top-level context arrow.
 1007         TkKeywordThen -> pure False
 1008         TkKeywordElse -> pure False
 1009         TkKeywordOf -> pure False
 1010         TkKeywordIn -> pure False
 1011         TkKeywordInstance -> pure False
 1012         TkKeywordWhere -> pure False
 1013         TkKeywordDeriving -> pure False
 1014         TkKeywordClass -> pure False
 1015         TkKeywordData -> pure False
 1016         TkKeywordNewtype -> pure False
 1017         _ -> go []
 1018     go stack@(expectedClose : rest) = do
 1019       tok <- anySingle
 1020       case lexTokenKind tok of
 1021         TkEOF -> pure False
 1022         kind
 1023           | kind == expectedClose ->
 1024               case rest of
 1025                 [] -> go []
 1026                 _ -> go rest
 1027         TkSpecialLParen -> go (TkSpecialRParen : stack)
 1028         TkSpecialUnboxedLParen -> go (TkSpecialUnboxedRParen : stack)
 1029         TkSpecialLBracket -> go (TkSpecialRBracket : stack)
 1030         TkTHExpQuoteOpen -> go (TkTHExpQuoteClose : stack)
 1031         TkTHTypedQuoteOpen -> go (TkTHTypedQuoteClose : stack)
 1032         TkTHDeclQuoteOpen -> go (TkTHExpQuoteClose : stack)
 1033         TkTHTypeQuoteOpen -> go (TkTHExpQuoteClose : stack)
 1034         TkTHPatQuoteOpen -> go (TkTHExpQuoteClose : stack)
 1035         TkSpecialLBrace -> go (TkSpecialRBrace : stack)
 1036         _ -> go stack

 1117 -- | Build a left-associated infix chain from a left operand and a list
 1118 -- of @(operator, operand)@ pairs.  Given @lhs@ and
 1119 -- @[(op1, a), (op2, b), (op3, c)]@ this produces
 1120 -- @((lhs \`op1\` a) \`op2\` b) \`op3\` c@.
 1121 --
 1122 -- This matches GHC's parsed expression AST before any later fixity
 1123 -- reassociation pass has run.
 1124 foldInfixL :: (a -> (op, a) -> a) -> a -> [(op, a)] -> a
 1125 foldInfixL = foldl

 1127 -- | Build a right-associated infix chain from a left operand and a list
 1128 -- of @(operator, operand)@ pairs.  Given @lhs@ and
 1129 -- @[(op1, a), (op2, b), (op3, c)]@ this produces
 1130 -- @lhs \`op1\` (a \`op2\` (b \`op3\` c))@.
 1131 foldInfixR :: (a -> (op, a) -> a) -> a -> [(op, a)] -> a
 1132 foldInfixR _ lhs [] = lhs
 1133 foldInfixR build lhs ((op, rhs) : rest) =
 1134   build lhs (op, foldInfixR build rhs rest)
