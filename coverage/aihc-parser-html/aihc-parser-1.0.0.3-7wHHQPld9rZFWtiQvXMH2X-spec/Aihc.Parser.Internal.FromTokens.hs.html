   38 parseFromTokens :: TokParser a -> FilePath -> [LexToken] -> ParseResult a
   39 parseFromTokens parser sourceName toks =
   40   case runParser (parser <* eofTok) sourceName (mkTokStreamFromTokens toks) of
   41     Left bundle -> ParseErr (parseErrorBundleToSpannedText bundle)
   42     Right parsed -> ParseOk parsed
   43
