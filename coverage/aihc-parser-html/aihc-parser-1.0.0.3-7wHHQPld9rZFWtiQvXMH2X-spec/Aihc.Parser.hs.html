   59 -- | Default parser configuration.
   60 --
   61 -- * 'parserSourceName' is set to @\"\<input\>\"@
   62 -- * 'parserExtensions' is empty (no extensions enabled by default)
   63 --
   64 -- >>> parserSourceName defaultConfig
   65 -- "<input>"
   66 --
   67 -- >>> parserExtensions defaultConfig
   68 -- []
   69 defaultConfig :: ParserConfig
   70 defaultConfig =
   71   ParserConfig
   72     { parserSourceName = "<input>",
   73       parserExtensions = []
   74     }

   76 -- | Parse a Haskell expression.
   77 --
   78 -- >>> shorthand $ parseExpr defaultConfig "1 + 2"
   79 -- ParseOk (EInfix (EInt 1 TInteger) "+" (EInt 2 TInteger))
   80 --
   81 -- >>> shorthand $ parseExpr defaultConfig "\\x -> x + 1"
   82 -- ParseOk (ELambdaPats [PVar "x"] (EInfix (EVar "x") "+" (EInt 1 TInteger)))
   83 --
   84 -- Parse errors are returned as 'ParseErr':
   85 --
   86 -- >>> case parseExpr defaultConfig "1 +" of { ParseErr _ -> "error"; ParseOk _ -> "ok" }
   87 -- "error"
   88 parseExpr :: ParserConfig -> Text -> ParseResult Expr
   89 parseExpr cfg input =
   90   let ts = mkTokStream (parserSourceName cfg) (applyImpliedExtensions (parserExtensions cfg)) input
   91    in case runParser (exprParser <* eofTok) (parserSourceName cfg) ts of
   92         Left bundle -> ParseErr (parseErrorBundleToSpannedText bundle)
   93         Right expr -> ParseOk expr
   94 
   95 -- | Parse a Haskell pattern.
   96 --
   97 -- >>> shorthand $ parsePattern defaultConfig "(x, y)"
   98 -- ParseOk (PTuple [PVar "x", PVar "y"])
   99 --
  100 -- >>> shorthand $ parsePattern defaultConfig "Just x"
  101 -- ParseOk (PCon "Just" [PVar "x"])
  102 parsePattern :: ParserConfig -> Text -> ParseResult Pattern
  103 parsePattern cfg input =
  104   let ts = mkTokStream (parserSourceName cfg) (applyImpliedExtensions (parserExtensions cfg)) input
  105    in case runParser (patternParser <* eofTok) (parserSourceName cfg) ts of
  106         Left bundle -> ParseErr (parseErrorBundleToSpannedText bundle)
  107         Right pat -> ParseOk pat

  109 -- | Parse a Haskell signature type.
  110 --
  111 -- This is the context used by top-level type signatures. Unlike 'parseType',
  112 -- it rejects an unparenthesized outer kind signature:
  113 --
  114 -- >>> case parseSignatureType defaultConfig "_ :: _" of { ParseErr _ -> "error"; ParseOk _ -> "ok" }
  115 -- "error"
  116 parseSignatureType :: ParserConfig -> Text -> ParseResult Type
  117 parseSignatureType cfg input =
  118   let ts = mkTokStream (parserSourceName cfg) (applyImpliedExtensions (parserExtensions cfg)) input
  119    in case runParser (typeSignatureParser <* eofTok) (parserSourceName cfg) ts of
  120         Left bundle -> ParseErr (parseErrorBundleToSpannedText bundle)
  121         Right ty -> ParseOk ty

  123 -- | Parse a Haskell type in the general declaration RHS context.
  124 --
  125 -- >>> shorthand $ parseType defaultConfig "Int -> Bool"
  126 -- ParseOk (TFun (TCon "Int") (TCon "Bool"))
  127 --
  128 -- >>> shorthand $ parseType defaultConfig "Maybe a"
  129 -- ParseOk (TApp (TCon "Maybe") (TVar "a"))
  130 --
  131 -- >>> shorthand $ parseType defaultConfig "_ :: _"
  132 -- ParseOk (TKindSig (TWildcard) (TWildcard))
  133 parseType :: ParserConfig -> Text -> ParseResult Type
  134 parseType cfg input =
  135   let ts = mkTokStream (parserSourceName cfg) (applyImpliedExtensions (parserExtensions cfg)) input
  136    in case runParser (typeParser <* eofTok) (parserSourceName cfg) ts of
  137         Left bundle -> ParseErr (parseErrorBundleToSpannedText bundle)
  138         Right ty -> ParseOk ty

  140 -- | Parse a single Haskell declaration.
  141 --
  142 -- >>> shorthand $ parseDecl defaultConfig "f x = x + 1"
  143 -- ParseOk (DeclValue (FunctionBind "f" [Match {MatchHeadPrefix, [PVar "x"], EInfix (EVar "x") "+" (EInt 1 TInteger)}]))
  144 parseDecl :: ParserConfig -> Text -> ParseResult Decl
  145 parseDecl cfg input =
  146   let ts = mkTokStream (parserSourceName cfg) (applyImpliedExtensions (parserExtensions cfg)) input
  147    in case runParser (declParser <* eofTok) (parserSourceName cfg) ts of
  148         Left bundle -> ParseErr (parseErrorBundleToSpannedText bundle)
  149         Right decl -> ParseOk decl

  151 -- | Parse a complete Haskell module.
  152 --
  153 -- Returns any recovered parse errors alongside a (possibly partial) 'Module'.
  154 -- When individual declarations fail to parse, the parser recovers and continues,
  155 -- returning the error and the successfully parsed declarations.
  156 --
  157 -- >>> shorthand $ snd $ parseModule defaultConfig "module Main where\nmain = putStrLn \"Hello\""
  158 -- Module {ModuleHead {"Main"}, [DeclValue (PatternBind (PVar "main") (EApp (EVar "putStrLn") (EString "Hello")))]}
  159 --
  160 -- Modules without a header are also supported:
  161 --
  162 -- >>> case parseModule defaultConfig "x = 1" of { (_, m) -> moduleName m }
  163 -- Nothing
  164 parseModule :: ParserConfig -> Text -> ([(SourceSpan, Text)], Module)
  165 parseModule cfg input =
  166   let ts = mkTokStreamModule (parserSourceName cfg) (applyImpliedExtensions (parserExtensions cfg)) input
  167       parser = do
  168         modu <- moduleParser
  169         _ <- eofTok
  170         errs <- drainParseErrors
  171         pure (errs, modu)
  172    in case runParser parser (parserSourceName cfg) ts of
  173         Left bundle ->
  174           ( parseErrorBundleToSpannedText bundle,
  175             Module
  176               { moduleAnns = [],
  177                 moduleHead = Nothing,
  178                 moduleLanguagePragmas = [],
  179                 moduleImports = [],
  180                 moduleDecls = []
  181               }
  182           )
  183         Right (errs, modu) ->
  184           (parseErrorsToSpannedText errs, modu)

  186 -- | Pretty-print a list of spanned parse errors with source context.
  187 formatParseErrors :: FilePath -> Maybe Text -> [(SourceSpan, Text)] -> String
  188 formatParseErrors sourceName mSource errs =
  189   let opts = defaultLayoutOptions
  190       blocks =
  191         map
  192           ( \(srcSpan, msg) ->
  193               renderString
  194                 ( layoutPretty opts $
  195                     case (srcSpan, mSource) of
  196                       (ss@SourceSpan {}, Just source) ->
  197                         vcat [renderSourceReference sourceName source ss, pretty msg]
  198                       _ ->
  199                         vcat [pretty sourceName, pretty msg]
  200                 )
  201           )
  202           errs
  203    in List.intercalate "\n\n" blocks

  205 -- renderSourceReference "<input>" "x = 1" (SourceSpan 1 5 1 6) = """
  206 -- <input>:1:5:
  207 -- 1 | x = 1
  208 --   |     ^
  209 -- """
  210 -- renderSourceReference "<input>" "module where" (SourceSpan 1 8 1 13) = """
  211 -- <input>:1:5:
  212 -- 1 | module where
  213 --   |        ^^^^^
  214 -- """
  215 renderSourceReference :: String -> Text -> SourceSpan -> Doc ann
  216 renderSourceReference origin source srcSpan =
  217   let (renderedOrigin, lineNo, colNo, endCol, srcLine) = case srcSpan of
  218         SourceSpan {sourceSpanSourceName, sourceSpanStartLine, sourceSpanStartCol, sourceSpanEndCol, sourceSpanStartOffset} ->
  219           ( sourceSpanSourceName,
  220             sourceSpanStartLine,
  221             sourceSpanStartCol,
  222             sourceSpanEndCol,
  223             extractSourceLineByOffset source sourceSpanStartOffset
  224           )
  225         NoSourceSpan -> (origin, 1, 1, 1, "")
  226       lineNoText = show lineNo
  227       markerPrefix = replicate (length lineNoText) ' ' ++ " | "
  228       markerStart = max 0 (colNo - 1)
  229       markerLen = max 1 (endCol - colNo)
  230       marker = replicate markerStart ' ' ++ replicate markerLen '^'
  231       header =
  232         pretty renderedOrigin <> colon <> pretty lineNo <> colon <> pretty colNo <> colon
  233    in vcat
  234         [ header,
  235           pretty (lineNoText ++ " | " ++ srcLine),
  236           pretty (markerPrefix ++ marker)
  237         ]

  239 extractSourceLineByOffset :: Text -> Int -> String
  240 extractSourceLineByOffset source offset =
  241   let bytes = TE.encodeUtf8 source
  242       len = BS.length bytes
  243       anchor = max 0 (min len offset)
  244       lineStart = scanBackward bytes anchor
  245       lineEnd = scanForward bytes anchor
  246    in T.unpack (TE.decodeUtf8 (BS.take (lineEnd - lineStart) (BS.drop lineStart bytes)))
  247 
  248 scanBackward :: BS.ByteString -> Int -> Int
  249 scanBackward bytes = go
  250   where
  251     go idx
  252       | idx <= 0 = 0
  253       | isLineBreak (BS.index bytes (idx - 1)) = idx
  254       | otherwise = go (idx - 1)
  255 
  256 scanForward :: BS.ByteString -> Int -> Int
  257 scanForward bytes = go
  258   where
  259     len = BS.length bytes
  260     go idx
  261       | idx >= len = len
  262       | isLineBreak (BS.index bytes idx) = idx
  263       | otherwise = go (idx + 1)
  264 
  265 isLineBreak :: Word8 -> Bool
  266 isLineBreak w = w == 10 || w == 13
