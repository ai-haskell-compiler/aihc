   38 -- | Report core:
   39 --
   40 -- > decl -> gendecl
   41 -- >      | (funlhs | pat) rhs
   42 declParser :: TokParser Decl
   43 declParser = pragmaDeclParser <|> ordinaryDeclParser

   45 ordinaryDeclParser :: TokParser Decl
   46 ordinaryDeclParser = do
   47   (tok, nextTok) <- lookAhead ((,) <$> anySingle <*> anySingle)
   48   let tokKind = lexTokenKind tok
   49       nextTokKind = lexTokenKind nextTok
   50       valueDecl = do
   51         exprFallback <- exprDeclEnabled
   52         if exprFallback
   53           then MP.try valueDeclParser <|> exprDeclParser
   54           else valueDeclParser
   55       sigOrValueDecl = do
   56         typeSigPrefix <- startsWithTypeSig
   57         if typeSigPrefix
   58           then typeSigOrPatternTypeSigDeclParser
   59           else valueDecl
   60       fallbackDecl = do
   61         typeSigPrefix <- startsWithTypeSig
   62         exprFallback <- exprDeclEnabled
   63         (if typeSigPrefix then typeSigOrPatternTypeSigDeclParser else MP.empty)
   64           <|> MP.try valueDeclParser
   65           <|> patternBindDeclParser
   66           <|> (if exprFallback then exprDeclParser else MP.empty)
   67   case tokKind of
   68     TkKeywordData ->
   69       case nextTokKind of
   70         TkVarFamily -> dataFamilyDeclParser
   71         TkKeywordInstance -> dataFamilyInstParser
   72         _ -> dataDeclParser
   73     TkKeywordClass -> classDeclParser
   74     TkKeywordDefault -> defaultDeclParser
   75     TkKeywordDeriving -> standaloneDerivingDeclParser
   76     TkKeywordForeign -> foreignDeclParser
   77     TkKeywordInfix -> fixityDeclParser
   78     TkKeywordInfixl -> fixityDeclParser
   79     TkKeywordInfixr -> fixityDeclParser
   80     TkKeywordInstance -> instanceDeclParser
   81     TkKeywordNewtype
   82       | nextTokKind == TkKeywordInstance -> newtypeFamilyInstParser
   83     TkKeywordNewtype -> newtypeDeclParser
   84     TkKeywordType ->
   85       case nextTokKind of
   86         TkVarRole -> roleAnnotationDeclParser
   87         TkVarFamily -> typeFamilyDeclParser
   88         TkKeywordData -> typeDataDeclParser
   89         TkKeywordInstance -> typeFamilyInstParser
   90         _ -> typeDeclarationParser
   91     TkKeywordPattern -> patternSynonymParser
   92     TkVarId {} ->
   93       case nextTokKind of
   94         TkReservedDoubleColon -> sigOrValueDecl
   95         TkSpecialComma -> sigOrValueDecl
   96         TkReservedEquals -> valueDecl
   97         _ -> nonBareVarPatternBindDeclParser <|> valueDecl
   98     _ -> fallbackDecl

  100 -- | Like 'patternBindDeclParser' but rejects bare variable patterns.
  101 -- When the leading token is a variable identifier, a bare @x = 5@ must be
  102 -- parsed as a zero-argument function bind, not a pattern bind.  This parser
  103 -- detects that case early (after parsing the pattern) and fails, letting
  104 -- 'valueDeclParser' handle it instead.
  105 nonBareVarPatternBindDeclParser :: TokParser Decl
  106 nonBareVarPatternBindDeclParser = MP.try $ withSpanAnn (DeclAnn . mkAnnotation) $ do
  107   pat <- region "while parsing pattern binding" patternParser
  108   case pat of
  109     PVar {} -> fail "bare variable bindings are parsed as function declarations"
  110     _ -> do
  111       DeclValue . PatternBind NoMultiplicityTag pat <$> equationRhsParser

  117 -- | Check whether the expression-as-declaration fallback is enabled.
  118 -- GHC allows top-level expressions under TemplateHaskell (bare splices),
  119 -- TemplateHaskellQuotes, and QuasiQuotes (e.g. @[qq|...|]@ at the top level).
  120 exprDeclEnabled :: TokParser Bool
  121 exprDeclEnabled = do
  122   th <- isExtensionEnabled TemplateHaskell
  123   thq <- isExtensionEnabled TemplateHaskellQuotes
  124   qq <- isExtensionEnabled QuasiQuotes
  125   pure (th || thq || qq)

  127 -- | Parse a top-level expression as a declaration.
  128 --
  129 -- GHC allows arbitrary expressions at the top level under TemplateHaskell
  130 -- (interpreted as declaration splices). This parser wraps the expression in
  131 -- 'DeclSplice'. The expression parser itself handles extension-specific
  132 -- constructs (e.g. @$expr@, @$(expr)@ via TH, @[qq|...|]@ via QuasiQuotes),
  133 -- so no special dispatch is needed here.
  134 exprDeclParser :: TokParser Decl
  135 exprDeclParser = DeclSplice <$> exprParser

  137 -- | Parse a @type@ declaration after the @type@ keyword has been consumed.
  138 -- Uses 'typeDeclHeadParser' to handle both prefix and infix type heads,
  139 -- then dispatches based on the next token:
  140 -- - @::@ → standalone kind signature (must have zero type parameters)
  141 -- - @=@ → type synonym
  142 --
  143 -- > topdecl -> 'type' simpletype '=' type
  144 typeDeclarationParser :: TokParser Decl
  145 typeDeclarationParser = withSpanAnn (DeclAnn . mkAnnotation) $ do
  146   expectedTok TkKeywordType
  147   typeHead <- typeDeclHeadParser
  148   nextTok <- anySingle
  149   case lexTokenKind nextTok of
  150     TkReservedDoubleColon -> do
  151       -- Standalone kind signature: cannot have type parameters
  152       if null (binderHeadParams typeHead)
  153         then
  154           DeclStandaloneKindSig (binderHeadName typeHead) <$> typeParser
  155         else
  156           fail "Standalone kind signatures cannot have type parameters."
  157     TkReservedEquals -> do
  158       body <- typeParser
  159       pure $
  160         DeclTypeSyn
  161           TypeSynDecl
  162             { typeSynHead = typeHead,
  163               typeSynBody = body
  164             }
  165     _ ->
  166       MP.customFailure
  167         UnexpectedTokenExpecting
  168           { unexpectedFound = Just (mkFoundToken nextTok),
  169             unexpectedExpecting = "'::' or '=' after type declaration head",
  170             unexpectedContext = []
  171           }

  196 -- | Parse an optional declaration context, returning @[]@ on absence.
  197 -- Uses 'MP.try' so inputs like @data T :: C => ()@ can backtrack from a
  198 -- failed context parse and treat @::@ as an inline kind signature instead.
  199 contextPrefixDispatchList :: TokParser [Type]
  200 contextPrefixDispatchList = MP.option [] (MP.try (declContextParser <* expectedTok TkReservedDoubleArrow))

  202 -- | Parse a declaration head with an optional leading context, but prefer an
  203 -- inline @::@ result kind when the head is immediately followed by @::@.
  204 --
  205 -- Without this preference, inputs like @data T :: C => K@ are ambiguous:
  206 -- @T :: C@ is also a valid context item. GHC resolves that form as a head with
  207 -- an inline result kind, so we do the same.
  208 declHeadPreferringInlineKind :: TokParser head -> TokParser (Maybe [Type], head, Maybe Type)
  209 declHeadPreferringInlineKind headParser =
  210   MP.try
  211     ( do
  212         head' <- headParser
  213         inlineKind <- expectedTok TkReservedDoubleColon *> typeParser
  214         pure (Nothing, head', Just inlineKind)
  215     )
  216     <|> do
  217       (context, head') <- declHeadWithOptionalContext headParser
  218       inlineKind <- MP.optional (expectedTok TkReservedDoubleColon *> typeParser)
  219       pure (context, head', inlineKind)
  220 
  221 -- | Parse a declaration head that may be preceded by a context.
  222 --
  223 -- The @MP.try@ must cover both the context and the following head parser so
  224 -- inputs like @data T :: C => ()@ can backtrack and treat @::@ as an inline
  225 -- kind signature rather than committing to a datatype context too early.
  226 declHeadWithOptionalContext :: TokParser head -> TokParser (Maybe [Type], head)
  227 declHeadWithOptionalContext headParser =
  228   MP.try ((,) . Just <$> (declContextParser <* expectedTok TkReservedDoubleArrow) <*> headParser)
  229     <|> ((Nothing,) <$> headParser)

  245 -- | Parse an optional type family result signature. GHC admits either an unnamed
  246 -- @:: Kind@ annotation or a named result variable with optional injectivity annotation,
  247 -- such as @= r@, @= r | r -> a@, or @= (r :: Type) | r -> a where ...@.
  248 --
  249 -- The 'Bool' parameter indicates whether the @family@ keyword was explicitly
  250 -- present.  When it was /not/ present (i.e. the shorthand @type T a …@ form
  251 -- inside a class body), @= r@ is syntactically ambiguous with a default type
  252 -- instance (@type T a = r@).  GHC resolves this by treating @= binder |@ as a
  253 -- named result signature with injectivity, while @= expr@ without @|@ is
  254 -- always a default type instance.  We mirror that behaviour: without an
  255 -- explicit @family@ keyword, @namedSigParser@ requires the @|@ injectivity
  256 -- annotation that follows the binder.
  257 typeFamilyResultSigParser :: Bool -> TokParser (Maybe TypeFamilyResultSig)
  258 typeFamilyResultSigParser explicitFamily =
  259   MP.optional (kindSigParser <|> namedSigParser)
  260   where
  261     kindSigParser =
  262       TypeFamilyKindSig <$> (expectedTok TkReservedDoubleColon *> typeParser)
  263 
  264     namedSigParser = do
  265       expectedTok TkReservedEquals
  266       result <- namedResultBinderParser
  267       mInjectivity <- MP.optional typeFamilyInjectivityParser
  268       case mInjectivity of
  269         Just injectivity -> pure $ TypeFamilyInjectiveSig result injectivity
  270         Nothing
  271           -- With an explicit @family@ keyword, @= r@ on its own is
  272           -- unambiguously a named result signature.
  273           | explicitFamily -> pure $ TypeFamilyTyVarSig result
  274           -- Without @family@, @= r@ (no injectivity @|@) is ambiguous with a
  275           -- default type instance.  Fail so the caller can backtrack and try
  276           -- the default-instance parser instead.
  277           | otherwise -> fail "named result sig without injectivity requires explicit 'family' keyword"
  278 
  279     namedResultBinderParser =
  280       withSpan $
  281         ( do
  282             ident <- lowerIdentifierParser
  283             pure (\span' -> TyVarBinder [mkAnnotation span'] ident Nothing TyVarBSpecified TyVarBVisible)
  284         )
  285           <|> ( do
  286                   expectedTok TkSpecialLParen
  287                   ident <- lowerIdentifierParser
  288                   expectedTok TkReservedDoubleColon
  289                   kind <- typeParser
  290                   expectedTok TkSpecialRParen
  291                   pure (\span' -> TyVarBinder [mkAnnotation span'] ident (Just kind) TyVarBSpecified TyVarBVisible)
  292               )
  293 
  294 typeFamilyInjectivityParser :: TokParser TypeFamilyInjectivity
  295 typeFamilyInjectivityParser = withSpan $ do
  296   expectedTok TkReservedPipe
  297   result <- lowerIdentifierParser
  298   expectedTok TkReservedRightArrow
  299   determined <- MP.some lowerIdentifierParser
  300   pure $ \span' ->
  301     TypeFamilyInjectivity
  302       { typeFamilyInjectivityAnns = [mkAnnotation span'],
  303         typeFamilyInjectivityResult = result,
  304         typeFamilyInjectivityDetermined = determined
  305       }

  543 -- | Parse a type signature or a pattern-typed equation.
  544 --
  545 -- With @ScopedTypeVariables@, @f :: Int = 0@ is valid Haskell meaning the same
  546 -- as @(f :: Int) = 0@: a pattern bind whose LHS carries a type annotation.
  547 -- GHC parses @name :: Type@ and then, if @=@ (or a guard @|@) follows,
  548 -- reinterprets the construct as a 'PatternBind' with a 'PTypeSig' pattern.
  549 --
  550 -- This parser mirrors that behaviour at the top level.  It first parses
  551 -- @name(s) :: Type@ (the type-signature prefix), then peeks at the next
  552 -- token.  If the next token begins an equation RHS (@=@ or @|@), the result is
  553 -- reinterpreted as a pattern-typed equation; otherwise a plain type signature
  554 -- is returned.
  555 typeSigOrPatternTypeSigDeclParser :: TokParser Decl
  556 typeSigOrPatternTypeSigDeclParser =
  557   withSpanAnn (DeclAnn . mkAnnotation) $
  558     typedBindingOrSignatureParser
  559       typeSignatureParser
  560       DeclTypeSig
  561       ( \name ty -> do
  562           rhs <- equationRhsParser
  563           let pat = PTypeSig (PVar name) ty
  564           pure (DeclValue (PatternBind NoMultiplicityTag pat rhs))
  565       )
  566       "typed pattern bindings with '=' require exactly one binder"

 1007 dataConDeclParser :: TokParser DataConDecl
 1008 dataConDeclParser = withSpan $ do
 1009   (forallVars, context) <- dataConQualifiersParser
 1010   tok <- lookAhead anySingle
 1011   case lexTokenKind tok of
 1012     -- `(#` is either the LHS arg of an infix constructor (e.g. @(# #) :. Int@) or a
 1013     -- standalone unboxed tuple/sum constructor (e.g. @(# Int, Bool #)@).
 1014     -- Try infix first to match the original priority; fall back to unboxed.
 1015     TkSpecialUnboxedLParen ->
 1016       MP.try (dataConInfixParser forallVars context)
 1017         <|> unboxedConDeclParser forallVars context
 1018     -- `[ ]` is the list-con; any other `[…` must be a type in infix position.
 1019     TkSpecialLBracket ->
 1020       MP.try (listConDeclParser forallVars context)
 1021         <|> dataConInfixParser forallVars context
 1022     -- General case: skip unboxed/list (wrong leading token), try infix then boxed-tuple then prefix/record.
 1023     _ ->
 1024       MP.try (dataConInfixParser forallVars context)
 1025         <|> MP.try (boxedTupleConDeclParser forallVars context)
 1026         <|> dataConRecordOrPrefixParser forallVars context

 1034 boxedTupleConDeclParser :: [TyVarBinder] -> [Type] -> TokParser (SourceSpan -> DataConDecl)
 1035 boxedTupleConDeclParser forallVars context = do
 1036   expectedTok TkSpecialLParen
 1037   mClose <- MP.optional (expectedTok TkSpecialRParen)
 1038   case mClose of
 1039     Just () ->
 1040       pure $ \span' -> DataConAnn (mkAnnotation span') (TupleCon forallVars context Boxed [])
 1041     Nothing -> do
 1042       firstField <- constructorArgParser
 1043       -- A comma is mandatory: boxed 1-tuples don't exist in Haskell
 1044       -- (e.g. @data C = (Int)@ is invalid). Without this, a

 1046       -- would be misinterpreted as a 1-tuple constructor.
 1047       expectedTok TkSpecialComma
 1048       rest <- constructorArgParser `MP.sepBy1` expectedTok TkSpecialComma
 1049       expectedTok TkSpecialRParen

 1052 unboxedConDeclParser :: [TyVarBinder] -> [Type] -> TokParser (SourceSpan -> DataConDecl)
 1053 unboxedConDeclParser forallVars context = do
 1054   expectedTok TkSpecialUnboxedLParen
 1055   mClose <- MP.optional (expectedTok TkSpecialUnboxedRParen)
 1056   case mClose of
 1057     Just () ->
 1058       pure $ \span' -> DataConAnn (mkAnnotation span') (TupleCon forallVars context Unboxed [])
 1059     Nothing -> do
 1060       leadingPipes <- MP.many (MP.try (expectedTok TkReservedPipe))
 1061       if not (null leadingPipes)
 1062         then do
 1063           field <- constructorArgParser
 1064           trailingPipes <- MP.many (expectedTok TkReservedPipe)
 1065           expectedTok TkSpecialUnboxedRParen
 1066           let pos = length leadingPipes + 1
 1067               arity = length leadingPipes + 1 + length trailingPipes
 1068           pure $ \span' -> DataConAnn (mkAnnotation span') (UnboxedSumCon forallVars context pos arity field)
 1069         else do
 1070           firstField <- constructorArgParser
 1071           mSep <- MP.optional (MP.try ((Left () <$ expectedTok TkSpecialComma) <|> (Right () <$ expectedTok TkReservedPipe)))
 1072           case mSep of
 1073             Nothing -> do
 1074               expectedTok TkSpecialUnboxedRParen
 1075               pure $ \span' -> DataConAnn (mkAnnotation span') (TupleCon forallVars context Unboxed [firstField])
 1076             Just (Left ()) -> do
 1077               rest <- constructorArgParser `MP.sepBy1` expectedTok TkSpecialComma
 1078               expectedTok TkSpecialUnboxedRParen
 1079               pure $ \span' -> DataConAnn (mkAnnotation span') (TupleCon forallVars context Unboxed (firstField : rest))
 1080             Just (Right ()) -> do
 1081               trailingPipes <- MP.many (expectedTok TkReservedPipe)
 1082               expectedTok TkSpecialUnboxedRParen
 1083               let arity = 1 + 1 + length trailingPipes
 1084               pure $ \span' -> DataConAnn (mkAnnotation span') (UnboxedSumCon forallVars context 1 arity firstField)
