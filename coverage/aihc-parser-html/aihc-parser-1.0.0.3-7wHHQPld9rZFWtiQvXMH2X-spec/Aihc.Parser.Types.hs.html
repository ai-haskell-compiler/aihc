   72 -- | Token stream backed by a shared lazy raw token list and a layout overlay.
   73 --
   74 -- The raw token list (@tokStreamRawTokens@) is produced lazily by scanning the
   75 -- input once. Backtracking never causes characters to be re-scanned — it only
   76 -- restores a pointer into this shared list. The layout overlay
   77 -- (@tokStreamLayoutState@) runs 'layoutTransition' on each raw token to produce
   78 -- virtual @{@, @;@, @}@ tokens. This allows the parser to influence layout
   79 -- decisions (e.g., by closing implicit layout contexts via
   80 -- 'closeImplicitLayoutContext').
   81 data TokStream = TokStream
   82   { -- | Shared lazy list of raw tokens (never re-scanned on backtrack).
   83     tokStreamRawTokens :: [LexToken],
   84     -- | Layout engine state (context stack and pending layout).
   85     tokStreamLayoutState :: LayoutState,
   86     -- | Tokens ready for the parser, including virtual layout tokens.
   87     tokStreamBuffer :: [LexToken],
   88     -- | Hidden pragmas that appeared before the next source token.
   89     -- Parsers may inspect these explicitly, but they never participate in the
   90     -- ordinary token stream.
   91     tokStreamPendingPragmas :: [Pragma],
   92     tokStreamPrevToken :: Maybe LexToken,
   93     tokStreamExtensionSet :: ExtensionSet,
   94     -- | Whether this stream has already emitted TkEOF.
   95     -- After EOF is emitted, 'take1_' returns Nothing.
   96     tokStreamEOFEmitted :: !Bool
   97   }
   98 
   99 -- -- Manual Eq instance — we skip tokStreamRawTokens since list position is not
  100 -- -- efficiently comparable and Megaparsec tracks offset separately.
  101 -- instance Eq TokStream where

  104 --       && tokStreamPendingPragmas a == tokStreamPendingPragmas b
  105 --       && tokStreamPrevToken a == tokStreamPrevToken b
  106 --       && tokStreamExtensions a == tokStreamExtensions b
  107 --       && tokStreamEOFEmitted a == tokStreamEOFEmitted b

  110 -- instance Ord TokStream where
  111 --   compare a b =
  112 --     compare (tokStreamEOFEmitted a) (tokStreamEOFEmitted b)

  114 -- Manual Show instance
  115 instance Show TokStream where
  116   show ts =
  117     "TokStream { eofEmitted = "
  118       <> show (tokStreamEOFEmitted ts)
  119       <> ", pendingPragmas = "
  120       <> show (length (tokStreamPendingPragmas ts))
  121       <> ", prevToken = "
  122       <> show (tokStreamPrevToken ts)
  123       <> ", extensionSet = "
  124       <> show (tokStreamExtensionSet ts)
  125       <> " }"

  127 -- Manual NFData instance — don't force the lazy raw token list.
  128 instance NFData TokStream where
  129   rnf ts =
  130     rnf (tokStreamPendingPragmas ts) `seq`
  131       rnf (tokStreamPrevToken ts) `seq`
  132         rnf (tokStreamExtensionSet ts) `seq`
  133           rnf (tokStreamEOFEmitted ts)

  135 -- | Create a TokStream for parsing expressions/declarations (no module layout).
  136 mkTokStream :: FilePath -> [Extension] -> Text -> TokStream
  137 mkTokStream sourceName exts input =
  138   let (env, lexSt) = mkInitialLexerState sourceName exts input
  139    in normalizeTokStream
  140         TokStream
  141           { tokStreamRawTokens = scanAllTokens env lexSt,
  142             tokStreamLayoutState = mkInitialLayoutState False exts,
  143             tokStreamBuffer = [],
  144             tokStreamPendingPragmas = [],
  145             tokStreamPrevToken = Nothing,
  146             tokStreamExtensionSet = mkExtensionSet exts,
  147             tokStreamEOFEmitted = False
  148           }
  149 
  150 -- | Create a TokStream for parsing full modules (with module-body layout).
  151 -- Also bootstraps LANGUAGE pragma extensions from the module header.
  152 mkTokStreamModule :: FilePath -> [Extension] -> Text -> TokStream
  153 mkTokStreamModule sourceName baseExts input =
  154   let (env, lexSt) = mkInitialLexerState sourceName effectiveExts input
  155    in normalizeTokStream
  156         TokStream
  157           { tokStreamRawTokens = scanAllTokens env lexSt,
  158             tokStreamLayoutState = mkInitialLayoutState True effectiveExts,
  159             tokStreamBuffer = [],
  160             tokStreamPendingPragmas = [],
  161             tokStreamPrevToken = Nothing,
  162             tokStreamExtensionSet = mkExtensionSet effectiveExts,
  163             tokStreamEOFEmitted = False
  164           }
  165   where
  166     headerSettings = readModuleHeaderExtensions input
  167     effectiveExts = applyImpliedExtensions (foldr applyExtensionSetting baseExts headerSettings)

  185 normalizeTokStream :: TokStream -> TokStream
  186 normalizeTokStream ts0
  187   | tokStreamEOFEmitted ts0 = ts0
  188   | otherwise =
  189       normalizeTokStreamParts
  190         (tokStreamRawTokens ts0)
  191         (tokStreamLayoutState ts0)
  192         (tokStreamBuffer ts0)
  193         (tokStreamPendingPragmas ts0)
  194         (tokStreamPrevToken ts0)
  195         (tokStreamExtensionSet ts0)
  196         False

  198 -- | Advance through layout and hidden tokens until the stream is ready for
  199 -- 'stepOne'. Keeping the stream fields separate lets a token step normalize
  200 -- its successor without first allocating an intermediate 'TokStream'.
  201 normalizeTokStreamParts :: [LexToken] -> LayoutState -> [LexToken] -> [Pragma] -> Maybe LexToken -> ExtensionSet -> Bool -> TokStream
  202 normalizeTokStreamParts rawTokens layoutState buffer pendingPragmas prevToken extensionSet eofEmitted
  203   | eofEmitted = finish rawTokens layoutState buffer pendingPragmas
  204   | otherwise = go rawTokens layoutState buffer pendingPragmas
  205   where
  206     finish rawTokens' layoutState' buffer' pendingPragmas' =
  207       TokStream
  208         { tokStreamRawTokens = rawTokens',
  209           tokStreamLayoutState = layoutState',
  210           tokStreamBuffer = buffer',
  211           tokStreamPendingPragmas = pendingPragmas',
  212           tokStreamPrevToken = prevToken,
  213           tokStreamExtensionSet = extensionSet,
  214           tokStreamEOFEmitted = eofEmitted
  215         }
  216 
  217     go rawTokens' layoutState' buffer' pendingPragmas' =
  218       case buffer' of
  219         tok : rest ->
  220           case lexTokenKind tok of
  221             TkPragma pragma' ->
  222               go rawTokens' layoutState' rest (pendingPragmas' <> [pragma'])
  223             TkLineComment -> go rawTokens' layoutState' rest pendingPragmas'
  224             TkBlockComment -> go rawTokens' layoutState' rest pendingPragmas'
  225             _ -> finish rawTokens' layoutState' buffer' pendingPragmas'
  226         [] ->
  227           case rawTokens' of
  228             [] -> finish [] layoutState' [] pendingPragmas'
  229             rawTok : rawRest ->
  230               let (allToks, laySt') = layoutTransition layoutState' rawTok
  231                in go rawRest laySt' allToks pendingPragmas'
  232 {-# INLINE normalizeTokStreamParts #-}
  233 
  234 -- | Step one token from the stream. This is the core primitive used by all
  235 -- Stream methods.
  236 --
  237 -- Tokens are produced in two phases:
  238 --
  239 -- 1. Drain any buffered tokens from the layout state (virtual braces/semicolons
  240 --    from 'layoutTransition' or 'closeImplicitLayoutContext').
  241 -- 2. Fetch the next raw token from the shared lazy list and run
  242 --    'layoutTransition' to produce virtual layout tokens and the raw token
  243 --    itself. The first token is returned; the rest are buffered.
  244 --
  245 -- Because the raw token list is a shared lazy list, backtracking to an earlier
  246 -- stream position is O(1) and never re-scans characters.
  247 stepOne :: TokStream -> Maybe (LexToken, TokStream)
  248 stepOne ts
  249   | tokStreamEOFEmitted ts = Nothing
  250   | otherwise =
  251       case tokStreamBuffer ts of
  252         -- Drain buffered tokens first (virtual braces/semicolons)
  253         tok : rest ->
  254           let kind = lexTokenKind tok
  255               isEOF = case kind of
  256                 TkEOF -> True
  257                 _ -> False
  258               pendingPragmas =
  259                 case lexTokenOrigin tok of
  260                   InsertedLayout -> tokStreamPendingPragmas ts
  261                   FromSource
  262                     | TkSpecialSemicolon <- kind -> tokStreamPendingPragmas ts
  263                     | otherwise -> []
  264            in Just
  265                 ( tok,
  266                   normalizeTokStreamParts
  267                     (tokStreamRawTokens ts)
  268                     (tokStreamLayoutState ts)
  269                     rest
  270                     pendingPragmas
  271                     (Just tok)
  272                     (tokStreamExtensionSet ts)
  273                     isEOF
  274                 )
  275         [] ->
  276           Nothing
  277 {-# INLINE stepOne #-}

  279 instance Stream TokStream where
  280   type Token TokStream = LexToken
  281   type Tokens TokStream = [LexToken]
  282 
  283   tokenToChunk _ tok = [tok]
  284   tokensToChunk _ = id
  285   chunkToTokens _ = id
  286   chunkLength _ = length
  287   chunkEmpty _ = null
  288 
  289   take1_ = stepOne
  290 
  291   takeN_ n ts
  292     | n <= 0 = Just ([], ts)
  293     | otherwise = go n [] ts
  294     where
  295       go 0 acc s = Just (reverse acc, s)
  296       go k acc s =
  297         case stepOne s of
  298           Nothing
  299             | null acc -> Nothing
  300             | otherwise -> Just (reverse acc, s)
  301           Just (tok, s') -> go (k - 1) (tok : acc) s'
  302 
  303   takeWhile_ f =
  304     go []
  305     where
  306       go acc s =
  307         case stepOne s of
  308           Nothing -> (reverse acc, s)
  309           Just (tok, s')
  310             | f tok -> go (tok : acc) s'
  311             | otherwise ->
  312                 -- Put the non-matching token back by using the pre-step state
  313                 (reverse acc, s)
