   30 -- | Parse a Template Haskell type splice: $typ or $(typ)
   31 thSpliceTypeParser :: TokParser Type
   32 thSpliceTypeParser = withSpanAnn (TAnn . mkAnnotation) $ do
   33   expectedTok TkTHSplice
   34   body <- parenSpliceBody <|> bareSpliceBody
   35   pure (TSplice body)
   36   where
   37     parenSpliceBody = withSpanAnn (EAnn . mkAnnotation) $ do
   38       body <- parens exprParser
   39       pure (EParen body)
   40     bareSpliceBody = withSpanAnn (EAnn . mkAnnotation) $ do
   41       EVar <$> identifierNameParser

   43 -- | Report core plus outer extension forms:
   44 --
   45 -- > type -> btype ['->' type]
   46 typeParser :: TokParser Type
   47 typeParser = label "type" $ forallTypeParser <|> contextOrKindSigTypeParser
   48 
   49 -- | Type syntax accepted after expression signatures. This keeps the same
   50 -- report core while allowing outer @forall@ and context forms.
   51 typeSignatureParser :: TokParser Type
   52 typeSignatureParser = label "type" $ forallSignatureTypeParser <|> contextOrFunSignatureTypeParser
   53 
   54 -- | Extension form wrapping the report core with a kind annotation.
   55 kindSigTypeParser :: TokParser Type
   56 kindSigTypeParser =
   57   optionalSuffix
   58     (expectedTok TkReservedDoubleColon *> typeSignatureParser)
   59     TKindSig
   60     typeFunParser
   61 
   62 -- | Extension form:
   63 --
   64 -- > forall a_1 ... a_n . type
   65 forallSignatureTypeParser :: TokParser Type
   66 forallSignatureTypeParser = withSpanAnn (TAnn . mkAnnotation) $ do
   67   telescope <- forallTelescopeParser
   68   attachForallBody telescope <$> typeSignatureParser
   69 
   70 contextOrFunSignatureTypeParser :: TokParser Type
   71 contextOrFunSignatureTypeParser = do
   72   isContextType <- startsWithContextType
   73   if isContextType then contextSignatureTypeParser else typeFunSignatureParser
   74 
   75 contextSignatureTypeParser :: TokParser Type
   76 contextSignatureTypeParser = do
   77   constraints <- contextItemsParserWith typeSignatureParser typeSignatureContextAtomParser
   78   expectedTok TkReservedDoubleArrow
   79   TContext constraints <$> typeSignatureParser
   80 
   81 -- | Report core plus extension infix-type support:
   82 --
   83 -- > type -> btype ['->' type]
   84 typeFunSignatureParser :: TokParser Type
   85 typeFunSignatureParser = do
   86   lhs <- typeSignatureInfixParser
   87   mArrow <- MP.optional arrowKindParser
   88   case mArrow of
   89     Nothing -> pure lhs
   90     Just arrowKind -> TFun arrowKind lhs <$> typeSignatureParser
   91 
   92 contextOrKindSigTypeParser :: TokParser Type
   93 contextOrKindSigTypeParser = do
   94   isContextType <- startsWithContextType
   95   if isContextType then contextTypeParser else kindSigTypeParser
   96 
   97 -- | Extension form:
   98 --
   99 -- > forall a_1 ... a_n . type
  100 forallTypeParser :: TokParser Type
  101 forallTypeParser = withSpanAnn (TAnn . mkAnnotation) $ do
  102   telescope <- forallTelescopeParser
  103   attachForallBody telescope <$> typeParser
  104 
  105 attachForallBody :: ForallTelescope -> Type -> Type
  106 attachForallBody telescope body =
  107   case body of
  108     TAnn ann (TKindSig ty kind) -> TAnn ann (TKindSig (TForall telescope ty) kind)
  109     TKindSig ty kind -> TKindSig (TForall telescope ty) kind
  110     _ -> TForall telescope body
  111 
  112 -- | Extension form:
  113 --
  114 -- > forall -> 'forall' binder_1 ... binder_n ('.' | '->')
  115 forallTelescopeParser :: TokParser ForallTelescope
  116 forallTelescopeParser = do
  117   expectedTok TkKeywordForall
  118   binders <- MP.some forallBinderParser
  119   visibility <-
  120     (expectedTok (TkVarSym ".") $> ForallInvisible)
  121       <|> (expectedTok TkReservedRightArrow $> ForallVisible)
  122   pure (ForallTelescope visibility binders)
  123 
  124 -- | Parse a single forall binder: {k} | (k :: *) | k
  125 forallBinderParser :: TokParser TyVarBinder
  126 forallBinderParser =
  127   withSpan $
  128     -- Inferred binder: {k} | {k :: Type}
  129     ( do
  130         expectedTok TkSpecialLBrace
  131         ident <- tyVarNameParser
  132         mKind <- MP.optional (expectedTok TkReservedDoubleColon *> typeParser)
  133         expectedTok TkSpecialRBrace
  134         pure (\span' -> TyVarBinder [mkAnnotation span'] ident mKind TyVarBInferred TyVarBVisible)
  135     )
  136       <|> ( do
  137               expectedTok TkSpecialLParen
  138               ident <- tyVarNameParser
  139               expectedTok TkReservedDoubleColon
  140               kind <- typeParser
  141               expectedTok TkSpecialRParen
  142               pure (\span' -> TyVarBinder [mkAnnotation span'] ident (Just kind) TyVarBSpecified TyVarBVisible)
  143           )
  144       <|> ( do
  145               ident <- tyVarNameParser
  146               pure (\span' -> TyVarBinder [mkAnnotation span'] ident Nothing TyVarBSpecified TyVarBVisible)
  147           )
  148 
  149 contextTypeParser :: TokParser Type
  150 contextTypeParser = contextTypeParserWith typeParser
  151 
  152 contextItemsParser :: TokParser [Type]
  153 contextItemsParser = contextItemsParserWith typeParser typeAtomParser
  154 
  155 contextTypeParserWith :: TokParser Type -> TokParser Type
  156 contextTypeParserWith rhsParser = do
  157   constraints <- contextItemsParserWith rhsParser typeAtomParser
  158   expectedTok TkReservedDoubleArrow
  159   rhs <- rhsParser
  160   pure $
  161     case rhs of
  162       TAnn ann (TKindSig ty kind) -> TAnn ann (TKindSig (TContext constraints ty) kind)
  163       TKindSig ty kind -> TKindSig (TContext constraints ty) kind
  164       _ -> TContext constraints rhs

  166 -- | Report core plus extension infix-type support:
  167 --
  168 -- > type -> btype ['->' type]
  169 -- > btype -> [btype] atype
  170 typeFunParser :: TokParser Type
  171 typeFunParser = typeFunParserWith typeSignatureParser
  172 
  173 typeFunParserWith :: TokParser Type -> TokParser Type
  174 typeFunParserWith rhsParser = do
  175   lhs <- typeInfixParser
  176   mArrow <- MP.optional arrowKindParser
  177   case mArrow of
  178     Nothing -> pure lhs
  179     Just arrowKind -> TFun arrowKind lhs <$> rhsParser

  181 -- | Parse an arrow annotation and consume the @->@ token.
  182 -- Handles @->@ (unrestricted), @⊸@ (linear), @%1 ->@ (linear), and @%m ->@
  183 -- (explicit multiplicity).  The @%@-prefixed forms are only attempted when
  184 -- @LinearTypes@ is enabled.
  185 arrowKindParser :: TokParser ArrowKind
  186 arrowKindParser = do
  187   linearEnabled <- isExtensionEnabled LinearTypes
  188   (expectedTok TkReservedRightArrow $> ArrowUnrestricted)
  189     <|> (if linearEnabled then linearArrowKindParser else MP.empty)
  190 
  191 linearArrowKindParser :: TokParser ArrowKind
  192 linearArrowKindParser =
  193   (expectedTok TkLinearArrow $> ArrowLinear)
  194     <|> MP.try multiplicityAnnotatedArrowParser
  195 
  196 -- | Parse @%<mult> ->@, consuming all three tokens.
  197 -- Only matches when @%@ is a prefix operator (no space between @%@ and the
  198 -- multiplicity expression), which is how GHC distinguishes @a %1 -> b@ (linear)
  199 -- from @a % 1 -> b@ (type operator applied to @a@ and @1@, then unrestricted arrow).
  200 multiplicityAnnotatedArrowParser :: TokParser ArrowKind
  201 multiplicityAnnotatedArrowParser = do
  202   expectedTok TkPrefixPercent
  203   mult <- typeAtomParser
  204   expectedTok TkReservedRightArrow
  205   pure (multiplicityToArrowKind mult)
  206 
  207 multiplicityToArrowKind :: Type -> ArrowKind
  208 multiplicityToArrowKind ty =
  209   case peelTypeAnn ty of
  210     TTypeLit (TypeLitInteger 1 _) -> ArrowLinear
  211     _ -> ArrowExplicit ty

  213 -- | Extension layer over the report's @btype@ chain:
  214 --
  215 -- > btype -> [btype] atype
  216 typeInfixParser :: TokParser Type
  217 typeInfixParser = do
  218   lhs <- typeAppParser
  219   rest <- MP.many ((,) <$> typeInfixOperatorParser <*> typeInfixRhsParser)
  220   pure (foldInfixR buildInfixType lhs rest)
  221 
  222 typeInfixRhsParser :: TokParser Type
  223 typeInfixRhsParser = do
  224   rhs <- typeAppParser
  225   rejectBareTypeImplicitParam rhs
  226 
  227 typeSignatureInfixParser :: TokParser Type
  228 typeSignatureInfixParser = do
  229   lhs <- typeSignatureAppParser
  230   rest <- MP.many ((,) <$> typeInfixOperatorParser <*> typeSignatureInfixRhsParser)
  231   pure (foldInfixR buildInfixType lhs rest)
  232 
  233 typeSignatureInfixRhsParser :: TokParser Type
  234 typeSignatureInfixRhsParser = do
  235   rhs <- typeSignatureAppParser
  236   rejectBareSignatureImplicitParam rhs
  237 
  238 -- | Parse a type head that may contain infix operators but NOT type applications.
  239 -- Used for type family heads like @type family l `And` r@ where the head is
  240 -- an infix type, but we don't want to consume trailing type parameters.
  241 typeHeadInfixParser :: TokParser Type
  242 typeHeadInfixParser = do
  243   lhs <- typeAtomParser
  244   foldInfixR buildInfixType lhs <$> typeHeadInfixLoopParser
  245 
  246 typeHeadInfixLoopParser :: TokParser [((Name, TypePromotion), Type)]
  247 typeHeadInfixLoopParser = MP.many $ MP.try $ do
  248   op <- typeInfixOperatorParser
  249   atom <- typeAtomParser
  250   pure (op, atom)
  251 
  252 buildInfixType :: Type -> ((Name, TypePromotion), Type) -> Type
  253 buildInfixType lhs ((op, promoted), rhs) = TInfix lhs op promoted rhs
  254 
  255 typeInfixOperatorParser :: TokParser (Name, TypePromotion)
  256 typeInfixOperatorParser =
  257   promotedInfixOperatorParser
  258     <|> backtickTypeOperatorParser
  259     <|> unpromotedInfixOperatorParser
  260   where
  261     unpromotedInfixOperatorParser =
  262       tokenSatisfy "type infix operator" $ \tok ->
  263         case lexTokenKind tok of
  264           TkReservedColon -> Just (qualifyName Nothing (mkUnqualifiedNameAt tok NameConSym ":"), Unpromoted)
  265           TkVarSym op
  266             | op /= "."
  267                 && op /= "!"
  268                 && op /= "'" ->
  269                 Just (qualifyName Nothing (mkUnqualifiedNameAt tok NameVarSym op), Unpromoted)
  270           TkConSym op -> Just (qualifyName Nothing (mkUnqualifiedNameAt tok NameConSym op), Unpromoted)
  271           TkQVarSym modName op -> Just (mkNameAt tok (Just modName) NameVarSym op, Unpromoted)
  272           TkQConSym modName op -> Just (mkNameAt tok (Just modName) NameConSym op, Unpromoted)
  273           _ -> Nothing
  274 
  275     backtickTypeOperatorParser = MP.try $ do
  276       expectedTok TkSpecialBacktick
  277       op <- typeOperatorIdentifierParser
  278       expectedTok TkSpecialBacktick
  279       pure (op, Unpromoted)
  280 
  281     typeOperatorIdentifierParser =
  282       tokenSatisfy "type operator identifier" $ \tok ->
  283         case lexTokenKind tok of
  284           TkVarId name -> Just (qualifyName Nothing (mkUnqualifiedNameAt tok NameVarId name))
  285           TkConId name -> Just (qualifyName Nothing (mkUnqualifiedNameAt tok NameConId name))
  286           TkQVarId modName name -> Just (mkNameAt tok (Just modName) NameVarId name)
  287           TkQConId modName name -> Just (mkNameAt tok (Just modName) NameConId name)
  288           _ -> Nothing
  289 
  290     promotedInfixOperatorParser = MP.try $ do
  291       -- Accept both TkVarSym "'" and TkTHQuoteTick for promoted operators
  292       expectedTok (TkVarSym "'") <|> expectedTok TkTHQuoteTick
  293       -- After the quote, accept any symbolic infix operator (e.g., ': for promoted cons,
  294       -- or ':$$: for a promoted user-defined type operator)
  295       tokenSatisfy "promoted type infix operator" $ \tok ->
  296         case lexTokenKind tok of
  297           TkReservedColon -> Just (qualifyName Nothing (mkUnqualifiedNameAt tok NameConSym ":"), Promoted)
  298           TkVarSym sym
  299             | sym /= "." && sym /= "!" ->
  300                 Just (qualifyName Nothing (mkUnqualifiedNameAt tok NameVarSym sym), Promoted)
  301           TkConSym sym -> Just (qualifyName Nothing (mkUnqualifiedNameAt tok NameConSym sym), Promoted)
  302           TkQVarSym modQual sym -> Just (mkNameAt tok (Just modQual) NameVarSym sym, Promoted)
  303           TkQConSym modQual sym -> Just (mkNameAt tok (Just modQual) NameConSym sym, Promoted)
  304           _ -> Nothing

  306 -- | Report core:
  307 --
  308 -- > btype -> [btype] atype
  309 typeAppParser :: TokParser Type
  310 typeAppParser = do
  311   first <- typeAtomParser
  312   rest <- MP.many typeAppArgParser
  313   pure (foldl applyTypeAppArg first rest)
  314 
  315 typeSignatureAppParser :: TokParser Type
  316 typeSignatureAppParser = do
  317   first <- typeSignatureAtomParser
  318   rest <- MP.many typeSignatureAppArgParser
  319   pure (foldl applyTypeAppArg first rest)

  321 buildTypeApp :: Type -> Type -> Type
  322 buildTypeApp = TApp
  323 
  324 typeAppArgParser :: TokParser (Either Type Type)
  325 typeAppArgParser =
  326   (Left <$> MP.try invisibleTypeAppParser)
  327     <|> (Right <$> (typeAtomParser >>= rejectBareTypeImplicitParam))
  328 
  329 invisibleTypeAppParser :: TokParser Type
  330 invisibleTypeAppParser = do
  331   expectedTok TkTypeApp
  332   typeAtomParser >>= rejectBareTypeImplicitParam
  333 
  334 typeSignatureAppArgParser :: TokParser (Either Type Type)
  335 typeSignatureAppArgParser =
  336   (Left <$> MP.try invisibleSignatureTypeAppParser)
  337     <|> (Right <$> (typeSignatureAtomParser >>= rejectBareSignatureImplicitParam))
  338 
  339 typeSignatureContextAtomParser :: TokParser Type
  340 typeSignatureContextAtomParser = typeSignatureAtomParser >>= rejectBareSignatureImplicitParam
  341 
  342 rejectBareSignatureImplicitParam :: Type -> TokParser Type
  343 rejectBareSignatureImplicitParam = rejectBareTypeImplicitParam
  344 
  345 rejectBareTypeImplicitParam :: Type -> TokParser Type
  346 rejectBareTypeImplicitParam ty =
  347   case peelTypeAnn ty of
  348     TImplicitParam {} -> fail "implicit parameter type must be parenthesized"
  349     _ -> pure ty
  350 
  351 invisibleSignatureTypeAppParser :: TokParser Type
  352 invisibleSignatureTypeAppParser = do
  353   expectedTok TkTypeApp
  354   typeSignatureAtomParser >>= rejectBareSignatureImplicitParam
  355 
  356 applyTypeAppArg :: Type -> Either Type Type -> Type
  357 applyTypeAppArg fn (Left ty) = TTypeApp fn ty
  358 applyTypeAppArg fn (Right ty) = TApp fn ty

  360 -- | Report core:
  361 --
  362 -- > atype -> gtycon
  363 -- >       | tyvar
  364 -- >       | '(' type_1 ',' ... ',' type_k ')'
  365 -- >       | '[' type ']'
  366 -- >       | '(' type ')'
  367 --
  368 -- This parser also admits extension atoms such as promoted types,
  369 -- quasi-quotes, splices, wildcards, and implicit-parameter types.
  370 typeAtomParser :: TokParser Type
  371 typeAtomParser = typeAtomParserByToken
  372 
  373 typeSignatureAtomParser :: TokParser Type
  374 typeSignatureAtomParser = typeAtomParserByToken
  375 
  376 typeAtomParserByToken :: TokParser Type
  377 typeAtomParserByToken = do
  378   tok <- MP.lookAhead MP.anySingle
  379   case lexTokenKind tok of
  380     TkInteger {} -> typeLiteralTypeParser
  381     TkString {} -> typeLiteralTypeParser
  382     TkChar {} -> typeLiteralTypeParser
  383     TkQuasiQuote {} -> typeQuasiQuoteParser
  384     TkTHSplice -> do
  385       thAny <- thAnyEnabled
  386       if thAny then thSpliceTypeParser else typeAtomParserAlternatives False False
  387     TkImplicitParam {} -> do
  388       ipEnabled <- isExtensionEnabled ImplicitParams
  389       if ipEnabled then typeImplicitParamParser else typeAtomParserAlternatives False False
  390     TkSpecialLBracket -> typeListParser
  391     TkSpecialLParen -> MP.try typeParenOperatorParser <|> typeParenOrTupleParser
  392     TkSpecialUnboxedLParen -> typeParenOrTupleParser
  393     TkKeywordUnderscore -> typeWildcardParser
  394     TkVarId {} -> typeIdentifierParser
  395     TkConId {} -> typeIdentifierParser
  396     TkQVarId {} -> typeIdentifierParser
  397     TkQConId {} -> typeIdentifierParser
  398     _ -> do
  399       thAny <- thAnyEnabled
  400       ipEnabled <- isExtensionEnabled ImplicitParams
  401       typeAtomParserAlternatives thAny ipEnabled
  402 
  403 typeAtomParserAlternatives :: Bool -> Bool -> TokParser Type
  404 typeAtomParserAlternatives thAny ipEnabled =
  405   MP.try promotedTypeParser
  406     <|> typeLiteralTypeParser
  407     <|> typeQuasiQuoteParser
  408     <|> (if thAny then thSpliceTypeParser else MP.empty)
  409     <|> (if ipEnabled then typeImplicitParamParser else MP.empty)
  410     <|> typeListParser
  411     <|> MP.try typeParenOperatorParser
  412     <|> typeParenOrTupleParser
  413     <|> typeStarParser
  414     <|> typeWildcardParser
  415     <|> typeIdentifierParser

  417 -- | Parse an implicit parameter type.
  418 --
  419 -- The body uses 'typeSignatureParser' so @?x :: (T :: K)@ is accepted
  420 -- through the parenthesized atom grammar, but @?x :: T :: K@ is left for an
  421 -- outer kind-signature parser in a plain type context and rejected in a
  422 -- signature context.
  423 typeImplicitParamParser :: TokParser Type
  424 typeImplicitParamParser = withSpanAnn (TAnn . mkAnnotation) $ do
  425   name <- implicitParamNameParser
  426   expectedTok TkReservedDoubleColon
  427   TImplicitParam name <$> typeSignatureParser
  428 
  429 typeWildcardParser :: TokParser Type
  430 typeWildcardParser =
  431   tokenSatisfy "wildcard" $ \tok ->
  432     case lexTokenKind tok of
  433       TkKeywordUnderscore -> Just (TAnn (mkAnnotation (lexTokenSpan tok)) TWildcard)
  434       _ -> Nothing
  435 
  436 typeLiteralTypeParser :: TokParser Type
  437 typeLiteralTypeParser =
  438   tokenSatisfy "type literal" $ \tok ->
  439     case lexTokenKind tok of
  440       TkInteger n _ -> Just (TAnn (mkAnnotation (lexTokenSpan tok)) (TTypeLit (TypeLitInteger n (lexTokenText tok))))
  441       TkString s -> Just (TAnn (mkAnnotation (lexTokenSpan tok)) (TTypeLit (TypeLitSymbol s (lexTokenText tok))))
  442       TkChar c -> Just (TAnn (mkAnnotation (lexTokenSpan tok)) (TTypeLit (TypeLitChar c (lexTokenText tok))))
  443       _ -> Nothing
  444 
  445 promotedTypeParser :: TokParser Type
  446 promotedTypeParser = withSpanAnn (TAnn . mkAnnotation) $ do
  447   -- Accept both TkVarSym "'" and TkTHQuoteTick for promoted types
  448   -- This handles ambiguity between TH value quotes and promoted types
  449   expectedTok (TkVarSym "'") <|> expectedTok TkTHQuoteTick
  450   ty <- typeAtomParser
  451   maybe (fail "promoted type") pure (markTypePromoted ty)
  452 
  453 typeParenOperatorParser :: TokParser Type
  454 typeParenOperatorParser = withSpanAnn (TAnn . mkAnnotation) $ do
  455   expectedTok TkSpecialLParen
  456   starIsType <- isExtensionEnabled StarIsType
  457   unicodeSyntax <- isExtensionEnabled UnicodeSyntax
  458   op <- tokenSatisfy "type operator" $ \tok ->
  459     case lexTokenKind tok of
  460       TkVarSym sym | not (isStarTypeSymbol starIsType unicodeSyntax sym) -> Just (qualifyName Nothing (mkUnqualifiedNameAt tok NameVarSym sym))
  461       TkConSym sym | not (isStarTypeSymbol starIsType unicodeSyntax sym) -> Just (qualifyName Nothing (mkUnqualifiedNameAt tok NameConSym sym))
  462       TkQVarSym modQual sym -> Just (mkNameAt tok (Just modQual) NameVarSym sym)
  463       TkQConSym modQual sym -> Just (mkNameAt tok (Just modQual) NameConSym sym)
  464       -- Handle reserved operators that can be used as type constructors
  465       TkReservedRightArrow -> Just (qualifyName Nothing (mkUnqualifiedNameAt tok NameVarSym "->"))
  466       TkReservedColon -> Just (qualifyName Nothing (mkUnqualifiedNameAt tok NameConSym ":"))
  467       -- Note: ~ is now lexed as TkVarSym "~" so TkVarSym case handles it
  468       _ -> Nothing
  469   expectedTok TkSpecialRParen
  470   pure $
  471     case (nameQualifier op, nameType op, nameText op) of
  472       (Nothing, NameVarSym, "->") -> TBuiltinCon TBuiltinArrow
  473       (Nothing, NameConSym, ":") -> TBuiltinCon TBuiltinCons
  474       _ -> TCon op Unpromoted
  475 
  476 typeQuasiQuoteParser :: TokParser Type
  477 typeQuasiQuoteParser =
  478   tokenSatisfy "type quasi quote" $ \tok ->
  479     case lexTokenKind tok of
  480       TkQuasiQuote quoter body -> Just (typeAnnSpan (lexTokenSpan tok) (TQuasiQuote quoter body))
  481       _ -> Nothing
  482 
  483 typeIdentifierParser :: TokParser Type
  484 typeIdentifierParser = do
  485   (tok, name) <- identifierNameWithTokenParser
  486   pure $
  487     TAnn (mkAnnotation (lexTokenSpan tok)) $
  488       case (nameQualifier name, nameType name, T.uncons (nameText name)) of
  489         (Nothing, NameVarId, Just (c, _))
  490           | isLower c || c == '_' ->
  491               TVar (nameToUnqualified name)
  492         _ -> TCon name Unpromoted
  493 
  494 typeStarParser :: TokParser Type
  495 typeStarParser = do
  496   starIsType <- isExtensionEnabled StarIsType
  497   unicodeSyntax <- isExtensionEnabled UnicodeSyntax
  498   if starIsType
  499     then tokenSatisfy "star kind" $ \tok ->
  500       case lexTokenKind tok of
  501         TkVarSym sym | isStarTypeSymbol starIsType unicodeSyntax sym -> Just (TAnn (mkAnnotation (lexTokenSpan tok)) (TStar (lexTokenText tok)))
  502         TkConSym sym | isStarTypeSymbol starIsType unicodeSyntax sym -> Just (TAnn (mkAnnotation (lexTokenSpan tok)) (TStar (lexTokenText tok)))
  503         _ -> Nothing
  504     else MP.empty
  505 
  506 isStarTypeSymbol :: Bool -> Bool -> T.Text -> Bool
  507 isStarTypeSymbol starIsType unicodeSyntax sym =
  508   starIsType && (sym == "*" || (unicodeSyntax && sym == "★"))
  509 
  510 typeListParser :: TokParser Type
  511 typeListParser = withSpanAnn (TAnn . mkAnnotation) $ do
  512   expectedTok TkSpecialLBracket
  513   mClosed <- MP.optional (expectedTok TkSpecialRBracket)
  514   case mClosed of
  515     Just () -> pure (TBuiltinCon TBuiltinList)
  516     Nothing -> do
  517       elems <- typeParser `MP.sepBy1` expectedTok TkSpecialComma
  518       expectedTok TkSpecialRBracket
  519       pure (TList Unpromoted elems)
  520 
  521 typeParenOrTupleParser :: TokParser Type
  522 typeParenOrTupleParser = withSpanAnn (TAnn . mkAnnotation) $ do
  523   (tupleFlavor, closeTok) <- tupleDelimsParser
  524   mClosed <- MP.optional (expectedTok closeTok)
  525   case mClosed of
  526     Just () -> pure (TTuple tupleFlavor Unpromoted [])
  527     Nothing -> do
  528       MP.try (tupleConstructorParser tupleFlavor closeTok) <|> parenthesizedTypeOrTupleParser tupleFlavor closeTok
  529   where
  530     tupleConstructorParser tupleFlavor closeTok = do
  531       _ <- expectedTok TkSpecialComma
  532       moreCommas <- MP.many (expectedTok TkSpecialComma)
  533       expectedTok closeTok
  534       let arity = 2 + length moreCommas
  535       case tupleFlavor of
  536         Boxed -> pure (TBuiltinCon (TBuiltinTuple arity))
  537         Unboxed -> do
  538           let tupleConName = "(#" <> T.replicate (arity - 1) "," <> "#)"
  539           pure (TCon (qualifyName Nothing (mkUnqualifiedName NameConId tupleConName)) Unpromoted)
  540 
  541     parenthesizedTypeOrTupleParser tupleFlavor closeTok = do
  542       first <- typeParser
  543       mKind <- if tupleFlavor == Boxed then MP.optional (expectedTok TkReservedDoubleColon *> typeParser) else pure Nothing
  544       case mKind of
  545         Just kind -> do
  546           expectedTok closeTok
  547           pure (TParen (TKindSig first kind))
  548         Nothing -> do
  549           mComma <- MP.optional (expectedTok TkSpecialComma)
  550           case mComma of
  551             Nothing -> do
  552               -- Check for pipe (unboxed sum type)
  553               mPipe <- if tupleFlavor == Unboxed then MP.optional (expectedTok TkReservedPipe) else pure Nothing
  554               case mPipe of
  555                 Just () -> do
  556                   -- (# Type1 | Type2 | ... #) - unboxed sum type
  557                   rest <- typeParser `MP.sepBy1` expectedTok TkReservedPipe
  558                   expectedTok closeTok
  559                   pure (TUnboxedSum (first : rest))
  560                 Nothing -> do
  561                   expectedTok closeTok
  562                   case tupleFlavor of
  563                     Boxed -> pure (TParen first)
  564                     Unboxed -> pure (TTuple Unboxed Unpromoted [first])
  565             Just () -> do
  566               second <- typeParser
  567               more <- MP.many (expectedTok TkSpecialComma *> typeParser)
  568               expectedTok closeTok
  569               pure (TTuple tupleFlavor Unpromoted (first : second : more))
  570 
  571 markTypePromoted :: Type -> Maybe Type
  572 markTypePromoted ty =
  573   case ty of
  574     TAnn ann sub
  575       | Just inner <- markTypePromoted sub ->
  576           Just (TAnn ann inner)
  577     TCon name _ -> Just (TCon name Promoted)
  578     TBuiltinCon con -> Just (promoteBuiltinCon con)
  579     TList _ elems -> Just (TList Promoted elems)
  580     TTuple tupleFlavor _ elems -> Just (TTuple tupleFlavor Promoted elems)
  581     TTypeApp fn arg -> TTypeApp <$> markTypePromoted fn <*> pure arg
  582     _ -> Nothing
  583 
  584 promoteBuiltinCon :: TypeBuiltinCon -> Type
  585 promoteBuiltinCon con =
  586   TCon
  587     ( qualifyName Nothing $
  588         case con of
  589           TBuiltinTuple arity -> mkUnqualifiedName NameConId ("(" <> T.replicate (max 0 (arity - 1)) "," <> ")")
  590           TBuiltinArrow -> mkUnqualifiedName NameVarSym "->"
  591           TBuiltinList -> mkUnqualifiedName NameConId "[]"
  592           TBuiltinCons -> mkUnqualifiedName NameConSym ":"
  593     )
  594     Promoted
