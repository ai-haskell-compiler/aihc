    6 -- Description : Compact pretty-printing for debugging/inspection
    7 --
    8 -- This module provides a compact, human-readable representation of parsed
    9 -- AST structures via the 'Shorthand' typeclass. Key features:
   10 --
   11 -- * Source spans are omitted to reduce noise
   12 -- * Empty fields (Nothing, [], False, etc.) and field labels are omitted
   13 -- * Remaining constructor/value tokens are a subsequence of the derived 'Show' output
   14 -- * Output is on a single line by default
   15 -- * Uses the prettyprinter library for consistent formatting
   16 --
   17 -- Example:
   18 --
   19 -- >>> shorthand $ snd $ parseModule defaultConfig "module Demo where x = 1"
   20 -- Module {ModuleHead {"Demo"}, [DeclValue (PatternBind (PVar "x") (EInt 1 TInteger))]}

   23   )

   46 -- | Typeclass for compact, human-readable AST representations.
   47 --
   48 -- The 'shorthand' method produces a 'Doc' that can be rendered to text
   49 -- or shown as a string. This is useful for debugging and golden tests.
   50 --
   51 -- Use 'show' on the result of 'shorthand' to get a 'String':
   52 --
   53 -- @
   54 -- show (shorthand expr) :: String
   55 -- @
   56 class Shorthand a where
   57   shorthand :: a -> Doc ()
   58 
   59 -- ParseResult
   60 
   61 instance (Shorthand a) => Shorthand (ParseResult a) where
   62   shorthand (ParseOk a) = "ParseOk" <+> parens (shorthand a)
   63   shorthand (ParseErr _) = "ParseErr"
   64 
   65 -- Module
   66 
   67 instance Shorthand Module where
   68   shorthand modu =
   69     "Module" <+> braces (hsep (punctuate comma fields))
   70     where
   71       fields =
   72         optionalField docModuleHead (moduleHead modu)
   73           <> listField docExtensionSetting (moduleLanguagePragmas modu)
   74           <> listField docImportDecl (moduleImports modu)
   75           <> listField docDecl (moduleDecls modu)
   76 
   77 instance Shorthand Decl where
   78   shorthand = docDecl
   79 
   80 instance Shorthand Expr where
   81   shorthand = docExpr
   82 
   83 instance Shorthand Pattern where
   84   shorthand = docPattern
   85 
   86 instance Shorthand Type where
   87   shorthand = docType
   88 
   89 instance Shorthand LexToken where
   90   shorthand = docToken
   91 
   92 instance Shorthand LexTokenKind where
   93   shorthand = docTokenKind

   95 docModuleHead :: ModuleHead -> Doc ann
   96 docModuleHead head' =
   97   "ModuleHead" <+> braces (hsep (punctuate comma fields))
   98   where
   99     fields =
  100       [docText (moduleHeadName head')]
  101         <> optionalField docPragma (moduleHeadWarningPragma head')
  102         <> optionalField (brackets . hsep . punctuate comma . map docExportSpec) (moduleHeadExports head')
  103 
  104 docExtensionSetting :: ExtensionSetting -> Doc ann
  105 docExtensionSetting setting =
  106   case setting of
  107     EnableExtension ext -> pretty (extensionName ext)
  108     DisableExtension ext -> "DisableExtension" <+> pretty (extensionName ext)

  110 docExportSpec :: ExportSpec -> Doc ann
  111 docExportSpec spec =
  112   case spec of
  113     ExportAnn _ sub -> docExportSpec sub
  114     ExportModule mWarning name ->
  115       "ExportModule" <> braces (hsep (punctuate comma (optionalField docPragma mWarning <> [docText name])))
  116     ExportVar mWarning mNamespace name ->
  117       "ExportVar" <> braces (hsep (punctuate comma (optionalField docPragma mWarning <> optionalField docIENamespace mNamespace <> [docName name])))
  118     ExportAbs mWarning mNamespace name ->
  119       "ExportAbs" <> braces (hsep (punctuate comma (optionalField docPragma mWarning <> optionalField docIENamespace mNamespace <> [docName name])))
  120     ExportAll mWarning mNamespace name ->
  121       "ExportAll" <> braces (hsep (punctuate comma (optionalField docPragma mWarning <> optionalField docIENamespace mNamespace <> [docName name])))
  122     ExportWith mWarning mNamespace name members ->
  123       "ExportWith" <> braces (hsep (punctuate comma fields))
  124       where
  125         fields =
  126           optionalField docPragma mWarning
  127             <> optionalField docIENamespace mNamespace
  128             <> [docName name, brackets (hsep (punctuate comma (map docExportMember members)))]
  129     ExportWithAll mWarning mNamespace name wildcardIndex members ->
  130       "ExportWithAll" <> braces (hsep (punctuate comma fields))
  131       where
  132         fields =
  133           optionalField docPragma mWarning
  134             <> optionalField docIENamespace mNamespace
  135             <> [docName name, pretty wildcardIndex, brackets (hsep (punctuate comma (map docExportMember members)))]

  182     ImportItemAllWith mNamespace name wildcardIndex members ->

  208 docDecl :: Decl -> Doc ann
  209 docDecl decl =
  210   case decl of
  211     DeclAnn _ sub -> docDecl sub
  212     DeclValue vdecl -> "DeclValue" <+> parens (docValueDecl vdecl)
  213     DeclTypeSig names ty -> "DeclTypeSig" <+> brackets (hsep (punctuate comma (map docUnqualifiedNameText names))) <+> parens (docType ty)
  214     DeclPatSyn ps -> "DeclPatSyn" <+> parens (docPatSynDecl ps)
  215     DeclPatSynSig names ty -> "DeclPatSynSig" <+> brackets (hsep (punctuate comma (map docUnqualifiedName names))) <+> parens (docType ty)
  216     DeclStandaloneKindSig name kind -> "DeclStandaloneKindSig" <+> parens (docUnqualifiedName name) <+> parens (docType kind)
  217     DeclFixity assoc mNamespace mPrec ops -> "DeclFixity" <+> docFixityAssoc assoc <+> maybe "Nothing" docIENamespace mNamespace <+> maybe "Nothing" pretty mPrec <+> brackets (hsep (punctuate comma (map docUnqualifiedName ops)))
  218     DeclRoleAnnotation ann -> "DeclRoleAnnotation" <+> parens (docRoleAnnotation ann)
  219     DeclTypeSyn syn -> "DeclTypeSyn" <+> parens (docTypeSynDecl syn)
  220     DeclData dd -> "DeclData" <+> parens (docDataDecl dd)
  221     DeclTypeData dd -> "DeclTypeData" <+> parens (docDataDecl dd)
  222     DeclNewtype nd -> "DeclNewtype" <+> parens (docNewtypeDecl nd)
  223     DeclClass cd -> "DeclClass" <+> parens (docClassDecl cd)
  224     DeclInstance inst -> "DeclInstance" <+> parens (docInstanceDecl inst)
  225     DeclStandaloneDeriving sd -> "DeclStandaloneDeriving" <+> parens (docStandaloneDerivingDecl sd)
  226     DeclDefault tys -> "DeclDefault" <+> brackets (hsep (punctuate comma (map docType tys)))
  227     DeclForeign fd -> "DeclForeign" <+> parens (docForeignDecl fd)
  228     DeclSplice body -> "DeclSplice" <+> parens (docExpr body)
  229     DeclTypeFamilyDecl tf -> "DeclTypeFamilyDecl" <+> parens (docTypeFamilyDecl tf)
  230     DeclDataFamilyDecl df -> "DeclDataFamilyDecl" <+> parens (docDataFamilyDecl df)
  231     DeclTypeFamilyInst tfi -> "DeclTypeFamilyInst" <+> parens (docTypeFamilyInst tfi)
  232     DeclDataFamilyInst dfi -> "DeclDataFamilyInst" <+> parens (docDataFamilyInst dfi)
  233     DeclPragma pragma -> "DeclPragma" <+> docPragma pragma

  235 docValueDecl :: ValueDecl -> Doc ann
  236 docValueDecl vdecl =
  237   case vdecl of
  238     FunctionBind name matches -> "FunctionBind" <+> docUnqualifiedNameText name <+> brackets (hsep (punctuate comma (map docMatch matches)))
  239     PatternBind multTag pat rhs -> "PatternBind" <+> hsep (docPatternBindFields multTag pat rhs)

  241 docPatternBindFields :: MultiplicityTag -> Pattern -> Rhs Expr -> [Doc ann]
  242 docPatternBindFields multTag pat rhs =
  243   multiplicityFields <> [parens (docPattern pat), parens (docRhs rhs)]
  244   where
  245     multiplicityFields =
  246       case multTag of
  247         NoMultiplicityTag -> []
  248         _ -> [docMultiplicityTag multTag]

  250 docMultiplicityTag :: MultiplicityTag -> Doc ann
  251 docMultiplicityTag NoMultiplicityTag = "NoMultiplicityTag"
  252 docMultiplicityTag LinearMultiplicityTag = "LinearMultiplicityTag"
  253 docMultiplicityTag (ExplicitMultiplicityTag ty) = "ExplicitMultiplicityTag" <+> parens (docType ty)

  301 docRhsWith :: (body -> Doc ann) -> Rhs body -> Doc ann
  302 docRhsWith docBody rhs =
  303   case rhs of
  304     UnguardedRhs _ body Nothing -> docBody body
  305     UnguardedRhs _ body (Just decls) -> docBody body <+> "Just" <+> brackets (hsep (punctuate comma (map docDecl decls)))
  306     GuardedRhss _ grhss Nothing -> "GuardedRhss" <+> brackets (hsep (punctuate comma (map (docGuardedRhsWith docBody) grhss)))
  307     GuardedRhss _ grhss (Just decls) -> "GuardedRhss" <+> brackets (hsep (punctuate comma (map (docGuardedRhsWith docBody) grhss))) <+> "Just" <+> brackets (hsep (punctuate comma (map docDecl decls)))

  379 docDataConDecl :: DataConDecl -> Doc ann
  380 docDataConDecl dcd =
  381   case dcd of
  382     DataConAnn _ inner -> docDataConDecl inner
  383     PrefixCon forallVars constraints name fields' ->
  384       "PrefixCon" <+> braces (hsep (punctuate comma ([docUnqualifiedName name] <> listField docTyVarBinder forallVars <> listField docType constraints <> listField docBangType fields')))
  385     InfixCon forallVars constraints lhs op rhs ->
  386       "InfixCon" <+> braces (hsep (punctuate comma (listField docTyVarBinder forallVars <> listField docType constraints <> [docBangType lhs, docUnqualifiedName op, docBangType rhs])))
  387     RecordCon forallVars constraints name fields' ->
  388       "RecordCon" <+> hsep (docRecordConFields forallVars constraints name fields')
  389     GadtCon forallBinders constraints names body ->
  390       "GadtCon" <+> braces (hsep (punctuate comma (listField docForallTelescope forallBinders <> listField docType constraints <> listField docUnqualifiedName names <> [docGadtBody body])))
  391     TupleCon forallVars constraints flavor fields' ->
  392       "TupleCon" <+> braces (hsep (punctuate comma (listField docTyVarBinder forallVars <> listField docType constraints <> [pretty (show flavor)] <> listField docBangType fields')))
  393     UnboxedSumCon forallVars constraints pos arity field' ->
  394       "UnboxedSumCon" <+> braces (hsep (punctuate comma (listField docTyVarBinder forallVars <> listField docType constraints <> [pretty (show pos), pretty (show arity), docBangType field'])))
  395     ListCon forallVars constraints ->
  396       "ListCon" <+> braces (hsep (punctuate comma (listField docTyVarBinder forallVars <> listField docType constraints)))

  782         _ ->

  784             <+> docName name

  796     PAs name inner -> "PAs" <+> docUnqualifiedName name <+> parens (docPattern inner)

  798     PIrrefutable inner -> "PIrrefutable" <+> parens (docPattern inner)
  799     PNegLit lit -> "PNegLit" <+> parens (docLiteral lit)
  800     PParen inner -> "PParen" <+> parens (docPattern inner)
  801     PRecord name fields' hasWildcard -> "PRecord" <+> docName name <+> braces (hsep (punctuate comma ([docPatternRecordField recordField | recordField <- fields'] ++ [".." | hasWildcard])))
  802     PTypeSig inner ty -> "PTypeSig" <+> parens (docPattern inner) <+> parens (docType ty)
  803     PSplice body -> "PSplice" <+> parens (docExpr body)
