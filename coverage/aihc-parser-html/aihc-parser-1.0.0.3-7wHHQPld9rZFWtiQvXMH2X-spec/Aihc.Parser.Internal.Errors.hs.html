   22 parseErrorBundleToSpannedText :: ParseErrorBundle -> [(SourceSpan, Text)]
   23 parseErrorBundleToSpannedText bundle =
   24   parseErrorsToSpannedText (NE.toList (MPE.bundleErrors bundle))
   25 
   26 parseErrorsToSpannedText :: [MPE.ParseError TokStream ParserErrorComponent] -> [(SourceSpan, Text)]
   27 parseErrorsToSpannedText errs =
   28   [ (fromMaybe NoSourceSpan mSpan, RText.renderStrict (layoutPretty defaultLayoutOptions doc))
   29   | err <- List.sortOn MP.errorOffset errs,
   30     (mSpan, doc) <- renderParseErrors err
   31   ]
   32 
   33 renderParseErrors :: MPE.ParseError TokStream ParserErrorComponent -> [(Maybe SourceSpan, Doc ann)]
   34 renderParseErrors err =
   35   case err of
   36     MPE.TrivialError _ mUnexpected expected ->
   37       let mSpan = trivialUnexpectedSpan mUnexpected
   38        in [(mSpan, vcat (map pretty (renderTrivialError mUnexpected expected)))]
   39     MPE.FancyError _ fancySet ->
   40       map renderFancyError (Set.toAscList fancySet)
   41   where
   42     trivialUnexpectedSpan :: Maybe (ErrorItem LexToken) -> Maybe SourceSpan
   43     trivialUnexpectedSpan mItem =
   44       case mItem of
   45         Just (Tokens ts) -> Just (lexTokenSpan (NE.head ts))
   46         _ -> Nothing
   47 
   48     renderFancyError :: ErrorFancy ParserErrorComponent -> (Maybe SourceSpan, Doc ann)
   49     renderFancyError fancy =
   50       case fancy of
   51         ErrorCustom custom ->
   52           ( customFoundSpan custom,
   53             vcat (map pretty (customMessageLines custom))
   54           )
   55         ErrorFail message -> (Nothing, pretty message)
   56         _ ->
   57           ( Nothing,
   58             pretty (show fancy)
   59           )
   60 
   61     customFoundSpan :: ParserErrorComponent -> Maybe SourceSpan
   62     customFoundSpan (UnexpectedTokenExpecting (Just found) _ _) =
   63       Just (foundTokenSpan found)
   64     customFoundSpan _ = Nothing
   65 
   66     customMessageLines :: ParserErrorComponent -> [String]
   67     customMessageLines e@(UnexpectedTokenExpecting mFound _ contexts) =
   68       [maybe "unexpected end of input" renderUnexpectedToken mFound, MPE.showErrorComponent e]
   69         <> map (\context -> "context: " <> T.unpack context) contexts
   70 
   71 renderTrivialError :: Maybe (ErrorItem LexToken) -> Set.Set (ErrorItem LexToken) -> [String]
   72 renderTrivialError mUnexpected expected =
   73   maybe [] (\item -> ["unexpected " <> renderErrorItem item]) mUnexpected
   74     <> ["expecting " <> renderExpectedItems (Set.toAscList expected) | not (Set.null expected)]
   75 
   76 renderErrorItem :: ErrorItem LexToken -> String
   77 renderErrorItem item =
   78   case item of
   79     Tokens toks -> unwords (map (T.unpack . lexTokenText) (NE.toList toks))
   80     Label label -> NE.toList label
   81     EndOfInput -> "end of input"
   82 
   83 renderExpectedItems :: [ErrorItem LexToken] -> String
   84 renderExpectedItems items =
   85   case map renderErrorItem items of
   86     [] -> ""
   87     [item] -> item
   88     [itemA, itemB] -> itemA <> " or " <> itemB
   89     rendered -> List.intercalate ", " (init rendered) <> ", or " <> last rendered
   90 
   91 renderUnexpectedToken :: FoundToken -> String
   92 renderUnexpectedToken found =
   93   "unexpected " <> tokenDescriptor found
   94 
   95 tokenDescriptor :: FoundToken -> String
   96 tokenDescriptor found =
   97   case foundTokenOrigin found of
   98     InsertedLayout -> "end of input"
   99     FromSource ->
  100       "'" <> T.unpack (foundTokenText found) <> "'"
