   25 languagePragmaParser :: TokParser [ExtensionSetting]
   26 languagePragmaParser =
   27   hiddenPragma "LANGUAGE pragma" $ \p -> case pragmaType p of
   28     PragmaLanguage names -> Just names
   29     _ -> Nothing
   30 
   31 moduleHeaderParser :: TokParser ModuleHead
   32 moduleHeaderParser = withSpan $ do
   33   expectedTok TkKeywordModule
   34   name <- moduleNameParser
   35   mWarning <- MP.optional warningPragmaParser
   36   exports <- MP.optional exportSpecListParser
   37   expectedTok TkKeywordWhere
   38   pure $ \span' ->
   39     ModuleHead
   40       { moduleHeadAnns = [mkAnnotation span'],
   41         moduleHeadName = name,
   42         moduleHeadWarningPragma = mWarning,
   43         moduleHeadExports = exports
   44       }
   45 
   46 warningPragmaParser :: TokParser Pragma
   47 warningPragmaParser =
   48   hiddenPragma "warning pragma" $ \p -> case pragmaType p of
   49     PragmaWarning _ -> Just p
   50     PragmaDeprecated _ -> Just p
   51     _ -> Nothing
   52 
   53 exportSpecListParser :: TokParser [ExportSpec]
   54 exportSpecListParser = parens $ exportSpecParser `MP.sepEndBy` expectedTok TkSpecialComma
   55 
   56 exportSpecParser :: TokParser ExportSpec
   57 exportSpecParser = withSpanAnn (ExportAnn . mkAnnotation) $ do
   58   mWarning <- MP.optional warningPragmaParser
   59   exportModuleParser mWarning <|> exportNameParser mWarning
   60 
   61 exportModuleParser :: Maybe Pragma -> TokParser ExportSpec
   62 exportModuleParser mWarning = do
   63   expectedTok TkKeywordModule
   64   ExportModule mWarning <$> moduleNameParser
   65 
   66 exportNameParser :: Maybe Pragma -> TokParser ExportSpec
   67 exportNameParser mWarning = do
   68   namespace <- MP.optional exportImportNamespaceParser
   69   name <- identifierNameParser <|> parens operatorNameParser
   70   members <- MP.optional exportMembersParser
   71   pure $
   72     case members of
   73       Just MembersAll -> ExportAll mWarning namespace name
   74       Just (MembersList names) -> ExportWith mWarning namespace name names
   75       Just (MembersListAll wildcardIndex names) -> ExportWithAll mWarning namespace name wildcardIndex names
   76       Nothing
   77         | namespace == Just IEEntityNamespaceType || isTypeName name ->
   78             ExportAbs mWarning namespace name
   79         | otherwise ->
   80             ExportVar mWarning namespace name
   81 
   82 data MembersResult
   83   = MembersAll
   84   | MembersList [IEBundledMember]
   85   | MembersListAll Int [IEBundledMember]
   86 
   87 exportMembersParser :: TokParser MembersResult
   88 exportMembersParser = membersParser
   89 
   90 importMembersParser :: TokParser MembersResult
   91 importMembersParser = membersParser
   92 
   93 membersParser :: TokParser MembersResult
   94 membersParser =
   95   parens (parseDotDotFirst <|> parseMemberList <|> emptyMembers)
   96   where
   97     emptyMembers = pure (MembersList [])
   98 
   99     parseDotDotFirst = do
  100       expectedTok TkReservedDotDot
  101       MP.optional (expectedTok TkSpecialComma) >>= \case
  102         Nothing -> pure MembersAll
  103         Just _ -> do
  104           trailingMembers <- memberNameParser `MP.sepBy` expectedTok TkSpecialComma
  105           pure (MembersListAll 0 trailingMembers)
  106 
  107     parseMemberList = do
  108       firstMember <- memberNameParser
  109       parseMemberSegments [firstMember]
  110 
  111     parseMemberSegments members =
  112       MP.optional (expectedTok TkSpecialComma) >>= \case
  113         Nothing -> pure (MembersList members)
  114         Just _ ->
  115           (expectedTok TkReservedDotDot >> parseWildcardTail members)
  116             <|> do
  117               nextMember <- memberNameParser
  118               parseMemberSegments (members <> [nextMember])
  119 
  120     parseWildcardTail members =
  121       MP.optional (expectedTok TkSpecialComma) >>= \case
  122         Nothing -> pure (MembersListAll (length members) members)
  123         Just _ -> do
  124           trailingMembers <- memberNameParser `MP.sepBy` expectedTok TkSpecialComma
  125           pure (MembersListAll (length members) (members <> trailingMembers))
  126 
  127     memberNameParser = do
  128       namespace <- MP.optional bundledNamespaceParser
  129       name <- identifierNameParser <|> parens operatorNameParser
  130       pure (IEBundledMember namespace name)
  131 
  132 -- | Checks if a name refers to a type/class (as opposed to a variable/function).
  133 -- In Haskell:
  134 -- - Identifiers starting with uppercase letters are type constructors/classes
  135 -- - Symbolic operators starting with ':' are constructor operators (type-level)
  136 isTypeName :: Name -> Bool
  137 isTypeName name =
  138   case T.uncons (nameText name) of
  139     Just (c, _) -> isUpper c || c == ':'
  140     Nothing -> False

  142 importDeclParser :: TokParser ImportDecl
  143 importDeclParser = withSpan $ do
  144   expectedTok TkKeywordImport
  145   importedSafe <-
  146     MP.option False (expectedTok TkVarSafe >> pure True)
  147   importedSource <- optionalHiddenPragma $ \p -> case pragmaType p of
  148     PragmaSource {} -> Just p
  149     _ -> Nothing
  150   preQualified <-
  151     MP.option False (expectedTok TkVarQualified >> pure True)
  152   importedLevel <- MP.optional importLevelParser
  153   importedPackage <- MP.optional packageNameParser
  154   importedModule <- moduleNameParser
  155   postQualified <-
  156     MP.optional $
  157       tokenSatisfy "'qualified'" $ \tok ->
  158         if lexTokenKind tok == TkVarQualified then Just (mkFoundToken tok) else Nothing
  159   when (preQualified && isJust postQualified) $
  160     MP.customFailure
  161       UnexpectedTokenExpecting
  162         { unexpectedFound = postQualified,
  163           unexpectedExpecting = "import declaration without duplicate 'qualified'",
  164           unexpectedContext = []
  165         }
  166   importAlias <- MP.optional (expectedTok TkVarAs *> moduleNameParser)
  167   importSpec <- MP.optional importSpecParser
  168   let isQualified = preQualified || isJust postQualified
  169   pure $ \span' ->
  170     ImportDecl
  171       { importDeclAnns = [mkAnnotation span'],
  172         importDeclLevel = importedLevel,
  173         importDeclPackage = importedPackage,
  174         importDeclSourcePragma = importedSource,
  175         importDeclSafe = importedSafe,
  176         importDeclQualified = isQualified,
  177         importDeclQualifiedPost = isJust postQualified,
  178         importDeclModule = importedModule,
  179         importDeclAs = importAlias,
  180         importDeclSpec = importSpec
  181       }
  182 
  183 importLevelParser :: TokParser ImportLevel
  184 importLevelParser =
  185   (varIdTok "quote" >> pure ImportLevelQuote)
  186     <|> (varIdTok "splice" >> pure ImportLevelSplice)
  187 
  188 packageNameParser :: TokParser Text
  189 packageNameParser = stringTextParser
  190 
  191 importSpecParser :: TokParser ImportSpec
  192 importSpecParser = withSpan $ do
  193   isHiding <-
  194     MP.option False (expectedTok TkVarHiding >> pure True)
  195   items <- parens $ importItemParser `MP.sepEndBy` expectedTok TkSpecialComma
  196   pure $ \span' ->
  197     ImportSpec
  198       { importSpecAnns = [mkAnnotation span'],
  199         importSpecHiding = isHiding,
  200         importSpecItems = items
  201       }
  202 
  203 importItemParser :: TokParser ImportItem
  204 importItemParser = withSpanAnn (ImportAnn . mkAnnotation) $ do
  205   namespace <- MP.optional exportImportNamespaceParser
  206   itemName <- identifierUnqualifiedNameParser <|> parens importOperatorParser
  207   -- When there's no explicit namespace, we still need to try parsing members
  208   -- for type constructors and type classes (uppercase names or parenthesized operators)
  209   let shouldTryMembers = case namespace of
  210         Just _ -> True
  211         Nothing -> isTypeName (qualifyName Nothing itemName)
  212   members <- if shouldTryMembers then MP.optional importMembersParser else pure Nothing
  213   let effectiveNamespace = namespace
  214   pure $
  215     case members of
  216       Just MembersAll -> ImportItemAll effectiveNamespace itemName
  217       Just (MembersList names) -> ImportItemWith effectiveNamespace itemName names
  218       Just (MembersListAll wildcardIndex names) -> ImportItemAllWith effectiveNamespace itemName wildcardIndex names
  219       Nothing
  220         | effectiveNamespace == Just IEEntityNamespaceType || isTypeName (qualifyName Nothing itemName) -> ImportItemAbs effectiveNamespace itemName
  221         | otherwise -> ImportItemVar effectiveNamespace itemName
  222 
  223 importOperatorParser :: TokParser UnqualifiedName
  224 importOperatorParser = operatorUnqualifiedNameParser
  225 
  226 exportImportNamespaceParser :: TokParser IEEntityNamespace
  227 exportImportNamespaceParser =
  228   (expectedTok TkKeywordType >> pure IEEntityNamespaceType)
  229     <|> (expectedTok TkKeywordData >> pure IEEntityNamespaceData)
  230     <|> patternNamespaceParser
  231   where
  232     patternNamespaceParser = do
  233       patSynEnabled <- isExtensionEnabled PatternSynonyms
  234       if patSynEnabled
  235         then expectedTok TkKeywordPattern >> pure IEEntityNamespacePattern
  236         else MP.empty
  237 
  238 bundledNamespaceParser :: TokParser IEBundledNamespace
  239 bundledNamespaceParser =
  240   (expectedTok TkKeywordType >> pure IEBundledNamespaceType)
  241     <|> (expectedTok TkKeywordData >> pure IEBundledNamespaceData)
