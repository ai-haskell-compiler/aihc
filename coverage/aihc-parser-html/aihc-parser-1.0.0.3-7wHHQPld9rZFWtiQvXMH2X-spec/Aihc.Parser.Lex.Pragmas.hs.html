   26 -- | Single entry point for pragma parsing.
   27 -- Scans "{-# ... #-}" once, then parses the body to determine which Pragma it is.
   28 tryParsePragma :: LexerState -> Maybe (LexToken, LexerState)
   29 tryParsePragma st = do
   30   (rawBody, consumedLen) <- extractPragmaBody (lexerInput st)
   31   let fullText = "{-#" <> rawBody <> "#-}"
   32       pragma = Pragma {pragmaType = parsePragmaType rawBody, pragmaRawText = fullText}
   33       st' = advanceN consumedLen st
   34   Just (mkToken st st' fullText (TkPragma pragma), st')
   35 
   36 -- | Extract the raw body text between "{-#" and "#-}".
   37 -- Returns (body_text, total_consumed_length) where total includes "{-#" and "#-}".
   38 extractPragmaBody :: Text -> Maybe (Text, Int)
   39 extractPragmaBody input = do
   40   rest0 <- T.stripPrefix "{-#" input
   41   let (body, marker) = T.breakOn "#-}" rest0
   42   if T.null marker
   43     then Nothing
   44     else
   45       let consumedLen = T.length input - T.length (T.drop 3 marker)
   46        in Just (body, consumedLen)

   48 -- | Parse pragma body text into a structured PragmaType.
   49 -- The body is the text between "{-#" and "#-}" (not including the delimiters).
   50 parsePragmaType :: Text -> PragmaType
   51 parsePragmaType rawBody =
   52   let trimmed = T.strip rawBody
   53       upperBody = T.toUpper trimmed
   54    in case tryParseNamedPragma trimmed upperBody of
   55         Just pt -> pt
   56         Nothing -> PragmaUnknown ("{-#" <> rawBody <> "#-}")
   57 
   58 tryParseNamedPragma :: Text -> Text -> Maybe PragmaType
   59 tryParseNamedPragma body upperBody
   60   | Just pt <- parseLanguagePragma body upperBody = Just pt
   61   | Just pt <- parseInstanceOverlapPragma body upperBody = Just pt
   62   | Just pt <- parseOptionsPragma body upperBody = Just pt
   63   | Just pt <- parseWarningPragma body upperBody = Just pt
   64   | Just pt <- parseDeprecatedPragma body upperBody = Just pt
   65   | Just pt <- parseInlinePragma body upperBody = Just pt
   66   | Just pt <- parseUnpackPragma body upperBody = Just pt
   67   | Just pt <- parseSourcePragma body upperBody = Just pt
   68   | Just pt <- parseSCCPragma body upperBody = Just pt
   69   | otherwise = Nothing

   71 parseLanguagePragma :: Text -> Text -> Maybe PragmaType
   72 parseLanguagePragma body upperBody
   73   | Just rest <- T.stripPrefix "LANGUAGE" upperBody,
   74     isPragmaBodyEnd rest =
   75       let names = parseLanguagePragmaNames (dropPragmaName "LANGUAGE" body)
   76        in Just (PragmaLanguage names)
   77   | otherwise = Nothing

   79 parseInstanceOverlapPragma :: Text -> Text -> Maybe PragmaType
   80 parseInstanceOverlapPragma _body upperBody
   81   | Just rest <- T.stripPrefix "OVERLAPPING" upperBody,
   82     isPragmaBodyEnd rest =
   83       Just (PragmaInstanceOverlap Overlapping)
   84   | Just rest <- T.stripPrefix "OVERLAPPABLE" upperBody,
   85     isPragmaBodyEnd rest =
   86       Just (PragmaInstanceOverlap Overlappable)
   87   | Just rest <- T.stripPrefix "OVERLAPS" upperBody,
   88     isPragmaBodyEnd rest =
   89       Just (PragmaInstanceOverlap Overlaps)
   90   | Just rest <- T.stripPrefix "INCOHERENT" upperBody,
   91     isPragmaBodyEnd rest =
   92       Just (PragmaInstanceOverlap Incoherent)
   93   | otherwise = Nothing

   98     isPragmaBodyEnd rest =
   99       let settings = parseOptionsPragmaSettings (dropPragmaName "OPTIONS_GHC" body)
  100        in Just (PragmaLanguage settings)
  101   | Just rest <- T.stripPrefix "OPTIONS" upperBody,
  102     isPragmaBodyEnd rest =
  103       let settings = parseOptionsPragmaSettings (dropPragmaName "OPTIONS" body)
  104        in Just (PragmaLanguage settings)
  105   | otherwise = Nothing

  127 parseInlinePragma :: Text -> Text -> Maybe PragmaType

  129   | Just rest <- T.stripPrefix "INLINEABLE" upperBody,

  132        in Just (PragmaInline "INLINEABLE" fullBody)
  133   | Just rest <- T.stripPrefix "INLINABLE" upperBody,

  135       let fullBody = T.strip (dropPragmaName "INLINABLE" body)
  136        in Just (PragmaInline "INLINABLE" fullBody)

  138     isPragmaBodyEnd rest =
  139       let fullBody = T.strip (dropPragmaName "NOINLINEABLE" body)
  140        in Just (PragmaInline "NOINLINEABLE" fullBody)
  141   | Just rest <- T.stripPrefix "NOINLINABLE" upperBody,
  142     isPragmaBodyEnd rest =
  143       let fullBody = T.strip (dropPragmaName "NOINLINABLE" body)
  144        in Just (PragmaInline "NOINLINABLE" fullBody)
  145   | Just rest <- T.stripPrefix "INLINE" upperBody,
  146     isPragmaBodyEnd rest =
  147       let fullBody = T.strip (dropPragmaName "INLINE" body)

  185 dropPragmaName :: Text -> Text -> Text
  186 dropPragmaName name = T.drop (T.length name)

  188 -- | Check if the remaining text after a pragma name is valid (whitespace or end).
  189 isPragmaBodyEnd :: Text -> Bool
  190 isPragmaBodyEnd text =
  191   case text of
  192     Empty -> True
  193     c :< _ -> isSpace c

  195 -- | Extract message text from pragma body, handling quoted strings.
  196 extractPragmaMessage :: Text -> Text
  197 extractPragmaMessage txt =
  198   case txt of
  199     '"' :< _ ->
  200       case reads (T.unpack txt) of
  201         [(decoded, "")] -> T.pack decoded
  202         _ -> txt
  203     _ -> txt

  205 -- | Parse the body of a LANGUAGE pragma into extension settings.
  206 parseLanguagePragmaNames :: Text -> [ExtensionSetting]
  207 parseLanguagePragmaNames body =
  208   mapMaybe (parseExtensionSettingName . T.strip . T.takeWhile (/= '#')) (T.splitOn "," body)

  210 -- | Parse the body of an OPTIONS pragma into extension settings.
  211 parseOptionsPragmaSettings :: Text -> [ExtensionSetting]
  212 parseOptionsPragmaSettings body = go (pragmaWords body)
  213   where
  214     go ws =
  215       case ws of
  216         [] -> []
  217         "-cpp" : rest -> EnableExtension CPP : go rest
  218         "-fffi" : rest -> EnableExtension ForeignFunctionInterface : go rest
  219         "-fglasgow-exts" : rest -> glasgowExtsSettings <> go rest
  220         opt : rest
  221           | Just ext <- T.stripPrefix "-X" opt,
  222             not (T.null ext) ->
  223               case parseExtensionSettingName ext of
  224                 Just setting -> setting : go rest
  225                 Nothing -> go rest
  226         _ : rest -> go rest

  266 -- | Split pragma text into words, respecting quoted strings.
  267 pragmaWords :: Text -> [Text]
  268 pragmaWords txt = go [] [] Nothing (T.unpack txt)
  269   where
  270     go acc current quote chars =
  271       case chars of
  272         [] ->
  273           let acc' = pushCurrent acc current
  274            in reverse acc'
  275         c : rest ->
  276           case quote of
  277             Just q
  278               | c == q -> go acc current Nothing rest
  279               | c == '\\' ->
  280                   case rest of
  281                     escaped : rest' -> go acc (escaped : current) quote rest'
  282                     [] -> go acc current quote []
  283               | otherwise -> go acc (c : current) quote rest
  284             Nothing
  285               | c == '"' || c == '\'' -> go acc current (Just c) rest
  286               | c == '\\' ->
  287                   case rest of
  288                     escaped : rest' -> go acc (escaped : current) Nothing rest'
  289                     [] -> go acc current Nothing []
  290               | c `elem` [' ', '\n', '\r', '\t'] ->
  291                   let acc' = pushCurrent acc current
  292                    in go acc' [] Nothing rest
  293               | otherwise -> go acc (c : current) Nothing rest
  294 
  295     pushCurrent acc current =
  296       case reverse current of
  297         [] -> acc
  298         token -> T.pack token : acc

  300 parseControlPragma :: Text -> Maybe (Text, Either Text DirectiveUpdate)
  301 parseControlPragma input = do
  302   (body, _) <- extractPragmaBody input
  303   let trimmed = T.strip body
  304       upperBody = T.toUpper trimmed
  305   case () of
  306     _
  307       | Just rest <- T.stripPrefix "LINE" upperBody,
  308         isPragmaBodyEnd rest ->
  309           let bodyAfter = dropPragmaName "LINE" trimmed
  310               ws = T.words bodyAfter
  311            in case ws of
  312                 lineNo : _
  313                   | T.all isDigit lineNo ->
  314                       case readBoundedInt lineNo of
  315                         Just parsedLine ->
  316                           Just
  317                             ( "{-#" <> body <> "#-}",
  318                               Right
  319                                 DirectiveUpdate
  320                                   { directiveLine = Just parsedLine,
  321                                     directiveCol = Just 1,
  322                                     directiveSourceName = parseDirectiveSourceName (T.dropWhile isSpace (T.drop (T.length lineNo) bodyAfter))
  323                                   }
  324                             )
  325                         Nothing -> Just ("{-#" <> body <> "#-}", Left "malformed LINE pragma")
  326                 _ -> Just ("{-#" <> body <> "#-}", Left "malformed LINE pragma")
  327     _
  328       | Just rest <- T.stripPrefix "COLUMN" upperBody,
  329         isPragmaBodyEnd rest ->
  330           let bodyAfter = dropPragmaName "COLUMN" trimmed
  331               ws = T.words bodyAfter
  332            in case ws of
  333                 colNo : _
  334                   | T.all isDigit colNo ->
  335                       case readBoundedInt colNo of
  336                         Just parsedCol ->
  337                           Just
  338                             ( "{-#" <> body <> "#-}",
  339                               Right DirectiveUpdate {directiveLine = Nothing, directiveCol = Just parsedCol, directiveSourceName = Nothing}
  340                             )
  341                         Nothing -> Just ("{-#" <> body <> "#-}", Left "malformed COLUMN pragma")
  342                 _ -> Just ("{-#" <> body <> "#-}", Left "malformed COLUMN pragma")
  343     _ -> Nothing

  345 parseDirectiveSourceName :: Text -> Maybe FilePath
  346 parseDirectiveSourceName rest =
  347   let rest' = T.dropWhile isSpace rest
  348    in case rest' of
  349         '"' :< more ->
  350           let (name, trailing) = T.break (== '"') more
  351            in case trailing of
  352                 '"' :< _ -> Just (T.unpack name)
  353                 _ -> Nothing
  354         _ -> Nothing

  356 readBoundedInt :: Text -> Maybe Int
  357 readBoundedInt txt = do
  358   n <- readMaybe (T.unpack txt) :: Maybe Integer
  359   if n >= fromIntegral (minBound :: Int) && n <= fromIntegral (maxBound :: Int)
  360     then Just (fromInteger n)
  361     else Nothing
