    4 -- |
    5 -- Module      : Aihc.Parser.Parens
    6 -- Description : Parenthesization pass for AST
    7 --
    8 -- This module provides a parenthesization pass that inserts 'EParen', 'PParen',
    9 -- 'TParen', and 'CmdPar' nodes at all required positions in the AST. After this
   10 -- pass, the pretty-printer can format code without worrying about parentheses.
   11 --
   12 -- The pass is __idempotent__: calling it twice produces the same result as
   13 -- calling it once. It never adds unnecessary parentheses, so parsed code
   14 -- (which already has explicit paren nodes from the source) is not modified.
   15 --
   16 -- __Entry points:__
   17 --
   18 -- * 'addModuleParens'  — parenthesise an entire module
   19 -- * 'addDeclParens'    — parenthesise a declaration
   20 -- * 'addExprParens'    — parenthesise an expression
   21 -- * 'addPatternParens' — parenthesise a pattern
   22 -- * 'addSignatureTypeParens' — parenthesise a signature type
   23 -- * 'addTypeParens'          — parenthesise a plain type

   25   ( addModuleParens,
   26     addDeclParens,
   27     addExprParens,
   28     addPatternParens,

   49 -- | Wrap an expression in 'EParen' if the predicate holds, unless it is
   50 -- already parenthesised.
   51 wrapExpr :: Bool -> Expr -> Expr
   52 wrapExpr True (EAnn ann sub) = EAnn ann (wrapExpr True sub)
   53 wrapExpr True e@EParen {} = e
   54 wrapExpr True e = EParen e
   55 wrapExpr False e = e
   56 
   57 wrapCmd :: Bool -> Cmd -> Cmd
   58 wrapCmd True (CmdAnn ann sub) = CmdAnn ann (wrapCmd True sub)
   59 wrapCmd True c@CmdPar {} = c
   60 wrapCmd True c = CmdPar c
   61 wrapCmd False c = c
   62 
   63 -- | Wrap a pattern in 'PParen' if the predicate holds, unless already wrapped.
   64 wrapPat :: Bool -> Pattern -> Pattern
   65 wrapPat True (PAnn ann sub) = PAnn ann (wrapPat True sub)
   66 wrapPat True p@PParen {} = p
   67 wrapPat True p = PParen p
   68 wrapPat False p = p
   69 
   70 -- | Wrap a type in 'TParen' if the predicate holds, unless already wrapped.
   71 wrapTy :: Bool -> Type -> Type
   72 wrapTy True (TAnn ann sub) = TAnn ann (wrapTy True sub)
   73 wrapTy True t@TParen {} = t
   74 wrapTy True t = TParen t
   75 wrapTy False t = t

   81 -- | Check if an expression is a "block expression" that can appear without
   82 -- parentheses as a function argument when BlockArguments is enabled.
   83 isBlockExpr :: Expr -> Bool
   84 isBlockExpr = \case
   85   EAnn _ sub -> isBlockExpr sub
   86   EIf {} -> True
   87   EMultiWayIf {} -> True
   88   ECase {} -> True
   89   EDo {} -> True
   90   ELambdaPats {} -> True
   91   ELambdaCase {} -> True
   92   ELambdaCases {} -> True
   93   ELetDecls {} -> True
   94   EProc {} -> True
   95   _ -> False
   96 
   97 -- | Check if an expression is "greedy" - i.e., it could consume trailing syntax.
   98 isGreedyExpr :: Expr -> Bool
   99 isGreedyExpr = \case
  100   EAnn _ sub -> isGreedyExpr sub
  101   ECase {} -> True
  102   EIf {} -> True
  103   EMultiWayIf {} -> True
  104   ELambdaPats {} -> True
  105   ELambdaCase {} -> True
  106   ELambdaCases {} -> True
  107   ELetDecls {} -> True
  108   EDo {} -> True
  109   EProc {} -> True
  110   EApp _ arg | isBlockExpr arg -> isOpenEnded arg
  111   EApp _ arg -> isGreedyExpr arg
  112   _ -> False
  113 
  114 -- | Check if an expression is "open-ended" - its rightmost component can
  115 -- capture a trailing where clause.
  116 isOpenEnded :: Expr -> Bool
  117 isOpenEnded = \case
  118   EAnn _ sub -> isOpenEnded sub
  119   ECase _ alts -> not (null alts)
  120   EIf {} -> True
  121   EMultiWayIf {} -> True
  122   ELambdaPats {} -> True
  123   ELetDecls {} -> True
  124   EProc {} -> True
  125   ENegate inner -> isOpenEnded inner
  126   EInfix _ _ rhs -> isOpenEnded rhs
  127   EApp _ arg | isBlockExpr arg -> isOpenEnded arg
  128   _ -> False

  130 -- | Does the pretty-printed form of an expression end with @:: Type@?
  131 endsWithTypeSig :: Expr -> Bool
  132 endsWithTypeSig = \case
  133   EAnn _ sub -> endsWithTypeSig sub
  134   ETypeSig {} -> True
  135   ELetDecls _ body -> endsWithTypeSig body
  136   ELambdaPats _ body -> endsWithTypeSig body
  137   EMultiWayIf rhss ->
  138     case reverse rhss of
  139       grhs : _ -> endsWithTypeSig (guardedRhsBody grhs)
  140       [] -> False
  141   ENegate inner -> endsWithTypeSig inner
  142   EInfix _ _ rhs -> endsWithTypeSig rhs
  143   EApp _ arg -> endsWithTypeSig arg
  144   EIf _ _ no -> endsWithTypeSig no
  145   _ -> False

  147 -- | Check whether an expression needs parenthesization before a record dot.
  148 -- Qualified variables (e.g., @A.x@), ambiguous TH name quotes, TH splices
  149 -- (@$x@, @$$x@), primitive literals, MagicHash literals, and identifiers ending in @#@ all
  150 -- need parens to prevent ambiguity:
  151 -- - Qualified names: @A.x.field@ looks like the qualified name @A.x.field@
  152 -- - TH quotes: @'M.x.field@ looks like quoting the qualified name @M.x.field@
  153 -- - TH splices: @$x.field@ and @$$x.field@ parse as a splice followed by an
  154 --   unexpected @.@
  155 -- - Primitive numeric literals: @10#.field@ makes @#.@ part of an operator
  156 -- - MagicHash: @'c'#.field@ or @x#.field@ — the @#.@ merges into an operator
  157 needsParensBeforeDot :: Expr -> Bool
  158 needsParensBeforeDot = \case
  159   EAnn _ sub -> needsParensBeforeDot sub
  160   ENegate inner -> startsWithPrimitiveLiteral inner
  161   EVar name -> nameNeedsParensBeforeDot name
  162   ETHSplice body -> spliceBodyNeedsParensBeforeDot body
  163   ETHTypedSplice body -> spliceBodyNeedsParensBeforeDot body
  164   -- Most TH value name quotes are already delimited enough for record dot:
  165   -- @'x.field@, @'().field@, and @'(M.+).field@ parse as field access.
  166   -- Qualified identifiers are different: @'M.x.field@ quotes @M.x.field@.
  167   -- Hash-ending names also need grouping because @'x#.field@ does not parse
  168   -- as record-dot access on the quoted name.
  169   ETHNameQuote body -> thNameQuoteNeedsParensBeforeDot body
  170   -- TH type name quotes need the same distinction.  Built-in type
  171   -- constructors are accepted as @''().field@ and @''[].field@, while
  172   -- @''T.field@ and @''(T).field@ are rejected unless the quote is grouped.
  173   ETHTypeNameQuote ty -> thTypeNameQuoteNeedsParensBeforeDot ty
  174   EInt _ nt _ -> nt /= TInteger
  175   EFloat _ ft _ -> ft /= TFractional
  176   -- MagicHash literals: the trailing # merges with . to form an operator
  177   ECharHash {} -> True
  178   EStringHash {} -> True
  179   _ -> False
  180 
  181 spliceBodyNeedsParensBeforeDot :: Expr -> Bool
  182 spliceBodyNeedsParensBeforeDot = \case
  183   EAnn _ sub -> spliceBodyNeedsParensBeforeDot sub
  184   EVar name ->
  185     T.isSuffixOf "#" (nameText name)
  186       || ( isJust (nameQualifier name)
  187              && case nameType name of
  188                NameVarId -> True
  189                NameConId -> True
  190                NameVarSym -> False
  191                NameConSym -> False
  192          )
  193   EInt _ nt _ -> nt /= TInteger
  194   EFloat _ ft _ -> ft /= TFractional
  195   ECharHash {} -> True
  196   EStringHash {} -> True
  197   _ -> False
  198 
  199 nameNeedsParensBeforeDot :: Name -> Bool
  200 nameNeedsParensBeforeDot name =
  201   T.isSuffixOf "#" (nameText name)
  202     || ( isJust (nameQualifier name)
  203            && case nameType name of
  204              NameVarId -> True
  205              NameConId -> True
  206              NameVarSym -> False
  207              NameConSym -> False
  208        )
  209 
  210 thNameQuoteNeedsParensBeforeDot :: Expr -> Bool
  211 thNameQuoteNeedsParensBeforeDot = \case
  212   EAnn _ sub -> thNameQuoteNeedsParensBeforeDot sub
  213   EVar name ->
  214     T.isSuffixOf "#" (nameText name)
  215       || ( isJust (nameQualifier name)
  216              && case nameType name of
  217                NameVarId -> True
  218                NameConId -> True
  219                NameVarSym -> False
  220                NameConSym -> False
  221          )
  222   _ -> False
  223 
  224 thTypeNameQuoteNeedsParensBeforeDot :: Type -> Bool
  225 thTypeNameQuoteNeedsParensBeforeDot = \case
  226   TAnn _ sub -> thTypeNameQuoteNeedsParensBeforeDot sub
  227   TCon name _ ->
  228     case nameType name of
  229       NameVarSym -> False
  230       NameConSym -> False
  231       _ -> True
  232   TTuple {} -> False
  233   TBuiltinCon {} -> False
  234   TList _ [] -> False
  235   _ -> True

  237 -- | Check whether an expression's pretty-printed form starts with a block form.
  238 startsWithBlockExpr :: Expr -> Bool
  239 startsWithBlockExpr = \case
  240   EAnn _ sub -> startsWithBlockExpr sub
  241   EInfix lhs _ _ -> startsWithBlockExpr lhs
  242   EApp fn _ -> startsWithBlockExpr fn
  243   ERecordUpd base _ -> startsWithBlockExpr base
  244   EGetField base _ -> startsWithBlockExpr base
  245   ETypeSig inner _ -> startsWithBlockExpr inner
  246   ETypeApp fn _ -> startsWithBlockExpr fn
  247   expr -> isBlockExpr expr

  249 -- | Check whether an expression starts with a primitive (unboxed) numeric
  250 -- literal.  This matters before record-dot syntax: @10#.field@ is tokenized
  251 -- differently from @(10#).field@.  Negation itself prints with a separating
  252 -- space, so @- 10#@ does not merge into a negative primitive literal token.
  253 startsWithPrimitiveLiteral :: Expr -> Bool
  254 startsWithPrimitiveLiteral = go
  255   where
  256     go (EAnn _ sub) = go sub
  257     go (EInt _ nt _) = nt /= TInteger
  258     go (EFloat _ ft _) = ft /= TFractional
  259     go (EApp fn _) = go fn
  260     go (ERecordUpd base _) = go base
  261     go (ETypeApp fn _) = go fn
  262     go (ETypeSig inner _) = go inner
  263     go _ = False

  269 data ExprCtx
  270   = CtxInfixRhs Bool
  271   | CtxInfixLhs
  272   | CtxAppFun
  273   | CtxAppArg
  274   | CtxAppArgNoParens
  275   | CtxAppArgGreedy
  276   | CtxSectionRhs
  277   | CtxTypeSigBody
  278   | CtxGuarded
  279 
  280 needsExprParens :: ExprCtx -> Expr -> Bool
  281 needsExprParens ctx (EAnn _ sub) = needsExprParens ctx sub
  282 needsExprParens ctx expr =
  283   case ctx of
  284     CtxInfixRhs protectOpenEnded ->
  285       case expr of
  286         -- The expression parser builds left-associated chains, so a nested
  287         -- infix expression on the RHS only appears when the source had
  288         -- explicit parentheses.
  289         EInfix {} -> True
  290         ETypeSig {} -> True
  291         _ | protectOpenEnded && absorbsFollowingInfix expr -> True
  292         -- ENegate does NOT need parenthesization in infix RHS position.
  293         -- GHC already rejects `x + - 1` (precedence >= 6) at parse time,
  294         -- so any ENegate appearing as the RHS of an infix operator in a
  295         -- successfully parsed module is guaranteed to be valid without
  296         -- parentheses. Wrapping in EParen would introduce a spurious HsPar
  297         -- in the GHC AST, causing roundtrip fingerprint mismatches.
  298         _ -> False
  299     CtxInfixLhs ->
  300       case expr of
  301         ETypeSig {} -> True
  302         EMultiWayIf {} -> False
  303         -- ENegate needs parenthesization when its inner expression is
  304         -- open-ended (e.g. ends with a `let` body).  Without parens, the
  305         -- parser re-reads `- f let {} in x \`op\` y` as
  306         -- `-(f (let {} in (x \`op\` y)))` because the `let` body greedily
  307         -- consumes the infix.  For non-greedy inner expressions (plain
  308         -- variables, literals, etc.) no parens are needed: `-x + 1` is
  309         -- correctly re-parsed as `(-x) + 1`.
  310         ENegate inner -> isOpenEnded inner
  311         ECase {} -> False
  312         _ -> isOpenEnded expr
  313     CtxAppFun ->
  314       case expr of
  315         ENegate {} -> True
  316         _ -> False
  317     CtxAppArg ->
  318       case expr of
  319         _ | endsWithTypeSig expr -> True
  320         _ | isBlockExpr expr -> False
  321         -- A pragma as a function argument needs parens: `fn {-# P #-} x` is
  322         -- ambiguous; `fn ({-# P #-} x)` is unambiguous.
  323         EPragma {} -> True
  324         _ -> False
  325     CtxAppArgNoParens ->
  326       False
  327     CtxAppArgGreedy ->
  328       case expr of
  329         ECase {} -> False
  330         EPragma {} -> True
  331         _ -> isGreedyExpr expr
  332     CtxSectionRhs ->
  333       case expr of
  334         ETypeSig {} -> True
  335         _ -> False
  336     CtxTypeSigBody ->
  337       case expr of
  338         ETypeSig {} -> True
  339         ENegate inner -> isGreedyExpr inner || isOpenEnded inner || endsWithTypeSig inner
  340         ELambdaPats {} -> True
  341         ECase {} -> False
  342         EMultiWayIf {} -> False
  343         EDo {} -> False
  344         ELambdaCase {} -> False
  345         ELambdaCases {} -> False
  346         _ -> isOpenEnded expr
  347     CtxGuarded -> isGreedyExpr expr
  348 
  349 exprCtxPrec :: ExprCtx -> Expr -> Int
  350 exprCtxPrec ctx expr =
  351   case ctx of
  352     CtxInfixRhs _
  353       | isGreedyExpr expr -> 0
  354       | otherwise -> 1
  355     CtxInfixLhs
  356       | isBlockExpr expr -> 0
  357       | otherwise -> 1
  358     CtxAppFun -> 2
  359     CtxAppArg
  360       | isBlockExpr expr -> 0
  361       | otherwise -> 3
  362     CtxAppArgNoParens -> 0
  363     CtxAppArgGreedy
  364       | isBlockExpr expr -> 0
  365       | otherwise -> 3
  366     CtxSectionRhs -> 0
  367     CtxTypeSigBody
  368       | ECase {} <- peelExprAnn expr -> 0
  369       | EMultiWayIf {} <- peelExprAnn expr -> 0
  370       | EDo {} <- peelExprAnn expr -> 0
  371       | otherwise -> 1
  372     CtxGuarded -> 0

  374 -- | Parenthesize a left section operand while protecting any rightmost infix
  375 -- RHS from reassociating with the section operator.
  376 addSectionLhsParens :: Expr -> Expr
  377 addSectionLhsParens expr =
  378   case peelExprAnn expr of
  379     EInfix lhs op rhs ->
  380       EInfix
  381         (addExprParensIn CtxInfixLhs lhs)
  382         op
  383         (addSectionInfixRhsParens rhs)
  384     ENegate inner
  385       | isGreedyExpr inner || isOpenEnded inner ->
  386           wrapExpr True (addExprParens expr)
  387     _ -> addExprParensPrec 1 expr
  388   where
  389     addSectionInfixRhsParens rhs =
  390       case peelExprAnn rhs of
  391         EInfix {} -> wrapExpr True (addSectionLhsParens rhs)
  392         _ | isOpenEnded rhs -> wrapExpr True (addExprParens rhs)
  393         _ -> addExprParensIn (CtxInfixRhs False) rhs

  538 data GuardArrow = GuardArrow | GuardEquals

  540 guardExprNeedsParensWith :: (Type -> Bool) -> GuardArrow -> Expr -> Bool
  541 guardExprNeedsParensWith typeSigNeedsParens arrow = \case
  542   EAnn _ sub -> guardExprNeedsParensWith typeSigNeedsParens arrow sub
  543   ETypeSig _ ty ->
  544     case arrow of
  545       GuardArrow -> typeSigNeedsParens ty
  546       GuardEquals -> False
  547   EApp _ arg | isBlockExpr arg -> guardExprNeedsParensWith typeSigNeedsParens arrow arg
  548   expr ->
  549     case arrow of
  550       GuardArrow -> trailingTypeSigNeedsParens typeSigNeedsParens expr
  551       GuardEquals -> False
  552 
  553 trailingTypeSigNeedsParens :: (Type -> Bool) -> Expr -> Bool
  554 trailingTypeSigNeedsParens typeSigNeedsParens expr =
  555   case expr of
  556     EAnn _ sub -> trailingTypeSigNeedsParens typeSigNeedsParens sub
  557     ETypeSig _ ty -> typeSigNeedsParens ty
  558     ELetDecls _ body -> trailingTypeSigNeedsParens typeSigNeedsParens body
  559     ELambdaPats _ body -> trailingTypeSigNeedsParens typeSigNeedsParens body
  560     EInfix _ _ rhs -> trailingTypeSigNeedsParens typeSigNeedsParens rhs
  561     EApp _ arg -> trailingTypeSigNeedsParens typeSigNeedsParens arg
  562     EIf _ _ no -> trailingTypeSigNeedsParens typeSigNeedsParens no
  563     _ -> False

  565 typeHasFunctionArrow :: Type -> Bool
  566 typeHasFunctionArrow ty =
  567   case ty of
  568     TAnn _ sub -> typeHasFunctionArrow sub
  569     TParen sub -> typeHasFunctionArrow sub
  570     TForall _ inner -> typeHasFunctionArrow inner
  571     TContext _ inner -> typeHasFunctionArrow inner
  572     TKindSig lhs rhs -> typeHasFunctionArrow lhs || typeHasFunctionArrow rhs
  573     TFun {} -> True
  574     _ -> False

  580 -- | Parenthesise an entire module.
  581 --
  582 -- >>> shorthand $ addModuleParens (Module [] Nothing [] [] [DeclTypeSig ["f"] (TKindSig TWildcard TWildcard)])
  583 -- Module {[DeclTypeSig ["f"] (TParen (TKindSig (TWildcard) (TWildcard)))]}
  584 --
  585 -- >>> shorthand $ addModuleParens (Module [] Nothing [] [] [DeclTypeSig ["f"] (TCon "Int" Unpromoted)])
  586 -- Module {[DeclTypeSig ["f"] (TCon "Int")]}
  587 addModuleParens :: Module -> Module
  588 addModuleParens modu =
  589   modu
  590     { moduleDecls = map addDeclParens (moduleDecls modu)
  591     }

  597 -- | Parenthesise a declaration.
  598 --
  599 -- >>> shorthand $ addDeclParens (DeclTypeSig ["f"] (TKindSig TWildcard TWildcard))
  600 -- DeclTypeSig ["f"] (TParen (TKindSig (TWildcard) (TWildcard)))
  601 --
  602 -- >>> shorthand $ addDeclParens (DeclTypeSig ["f"] (TCon "Int" Unpromoted))
  603 -- DeclTypeSig ["f"] (TCon "Int")
  604 addDeclParens :: Decl -> Decl
  605 addDeclParens decl =
  606   case decl of
  607     DeclAnn ann sub -> DeclAnn ann (addDeclParens sub)
  608     DeclValue vdecl -> DeclValue (addValueDeclParens vdecl)
  609     DeclTypeSig names ty -> DeclTypeSig names (addSignatureTypeParens ty)
  610     DeclPatSyn ps -> DeclPatSyn (addPatSynDeclParens ps)
  611     DeclPatSynSig names ty -> DeclPatSynSig names (addSignatureTypeParens ty)
  612     DeclStandaloneKindSig name kind -> DeclStandaloneKindSig name (addStandaloneKindSigParens kind)
  613     DeclFixity {} -> decl
  614     DeclRoleAnnotation {} -> decl

  641 addPatternBindLhsParens :: Pattern -> Rhs Expr -> Pattern
  642 addPatternBindLhsParens pat rhs =
  643   case pat of
  644     PAnn ann sub -> PAnn ann (addPatternBindLhsParens sub rhs)
  645     PTypeSig inner ty ->
  646       wrapPat
  647         (typedPatternBindLhsNeedsParens inner)
  648         (PTypeSig (addPatternInfixOperandParens inner) (addSignatureTypeParens ty))
  649     _ -> addPatternParens pat

  651 typedPatternBindLhsNeedsParens :: Pattern -> Bool
  652 typedPatternBindLhsNeedsParens (PAnn _ sub) = typedPatternBindLhsNeedsParens sub
  653 typedPatternBindLhsNeedsParens (PCon name typeArgs args) =
  654   not (null typeArgs) || not (null args) || isNothing (nameQualifier name)

  657 addMatchParens :: UnqualifiedName -> Match -> Match
  658 addMatchParens name match =
  659   match
  660     { matchPats = addFunctionHeadPats name (matchHeadForm match) (matchPats match),
  661       matchRhs = addRhsParens (matchRhs match)
  662     }

  664 addFunctionHeadPats :: UnqualifiedName -> MatchHeadForm -> [Pattern] -> [Pattern]
  665 addFunctionHeadPats _name headForm pats =
  666   case headForm of
  667     MatchHeadPrefix -> map addFunctionHeadPatternAtomParens pats
  668     MatchHeadInfix ->
  669       case pats of
  670         lhs : rhs : tailPats ->
  671           let lhs' = addInfixFunctionHeadPatternAtomParens lhs
  672               rhs' = addInfixFunctionHeadPatternAtomParens rhs
  673            in case tailPats of
  674                 [] -> [lhs', rhs']
  675                 _ ->
  676                   -- When infix head has tail pats, the infix part is wrapped in parens
  677                   -- and tail pats use function head atom rules
  678                   lhs' : rhs' : map addFunctionHeadPatternAtomParens tailPats
  679         _ -> map addFunctionHeadPatternAtomParens pats

  681 addRhsParens :: Rhs Expr -> Rhs Expr
  682 addRhsParens rhs =
  683   case rhs of
  684     UnguardedRhs sp body whereDecls ->
  685       UnguardedRhs sp (addExprParens body) (fmap (map addDeclParens) whereDecls)
  686     GuardedRhss sp guards whereDecls ->
  687       GuardedRhss sp (map (addGuardedRhsParens GuardEquals) guards) (fmap (map addDeclParens) whereDecls)

  690 addGuardedRhsParens arrow grhs =
  691   grhs
  692     { guardedRhsGuards = addGuardQualifiersParens arrow (guardedRhsGuards grhs),
  693       guardedRhsBody = addExprParens (guardedRhsBody grhs)

  696 addGuardQualifiersParens :: GuardArrow -> [GuardQualifier] -> [GuardQualifier]
  697 addGuardQualifiersParens arrow quals =
  698   case quals of
  699     [] -> []
  700     [qual] -> [addGuardQualifierParens arrow True qual]
  701     qual : rest -> addGuardQualifierParens arrow False qual : addGuardQualifiersParens arrow rest

  703 addGuardQualifierParens :: GuardArrow -> Bool -> GuardQualifier -> GuardQualifier
  704 addGuardQualifierParens arrow isLast qual =
  705   case qual of

  804 -- | Determine whether a type, when directly following a bang (!) or lazy (~)
  805 -- prefix, needs to be parenthesized to avoid lexer ambiguity. This happens
  806 -- when the type's rendering starts with a symbol character that would fuse
  807 -- with the ! or ~ to form a single operator token.
  808 bangTypeNeedsPrefixParens :: Type -> Bool
  809 bangTypeNeedsPrefixParens (TAnn _ sub) = bangTypeNeedsPrefixParens sub
  810 bangTypeNeedsPrefixParens (TParen _) = False
  811 bangTypeNeedsPrefixParens TStar {} = True
  812 bangTypeNeedsPrefixParens (TCon _ Promoted) = True
  813 bangTypeNeedsPrefixParens (TBuiltinCon TBuiltinCons) = True
  814 bangTypeNeedsPrefixParens (TBuiltinCon _) = False
  815 bangTypeNeedsPrefixParens TImplicitParam {} = True
  816 bangTypeNeedsPrefixParens TSplice {} = True
  817 -- Compound types: the first rendered character comes from the head/lhs.
  818 bangTypeNeedsPrefixParens (TApp f _) = bangTypeNeedsPrefixParens f
  819 bangTypeNeedsPrefixParens (TTypeApp f _) = bangTypeNeedsPrefixParens f
  820 bangTypeNeedsPrefixParens (TFun _ a _) = bangTypeNeedsPrefixParens a
  821 bangTypeNeedsPrefixParens (TKindSig ty _) = bangTypeNeedsPrefixParens ty
  822 bangTypeNeedsPrefixParens (TContext cs _) = case cs of
  823   [] -> False
  824   (c : _) -> bangTypeNeedsPrefixParens c
  825 bangTypeNeedsPrefixParens (TForall {}) = True
  826 -- Promoted tuples/lists start with a tick mark.
  827 bangTypeNeedsPrefixParens (TTuple _ promoted _) = promoted == Promoted
  828 bangTypeNeedsPrefixParens (TList promoted _) = promoted == Promoted
  829 -- Infix types render lhs first.
  830 bangTypeNeedsPrefixParens (TInfix lhs _ _ _) = bangTypeNeedsPrefixParens lhs
  831 bangTypeNeedsPrefixParens _ = False
  832 
  833 addPrefixConBangTypeParens :: BangType -> BangType
  834 addPrefixConBangTypeParens bt =

  841 addInfixConBangTypeParens :: BangType -> BangType
  842 addInfixConBangTypeParens bt =
  843   bt
  844     { bangType =
  845         wrapTy
  846           ( ( (bangStrict bt || bangLazy bt)
  847                 && bangTypeNeedsPrefixParens (bangType bt)
  848             )
  849               || needsTypeParens CtxTypeFunArg (bangType bt)
  850               || infixConOperandNeedsParens (bangType bt)
  851           )
  852           (addTypeIn CtxTypeFunArg (bangType bt))
  853     }

  855 -- | Types that would be misinterpreted as data constructor declarations
  856 -- (unboxed sum-con or list-con) when used as an infix constructor operand.
  857 -- Tuple operands are left bare so the raw surface syntax is preserved. The
  858 -- data-con parsers are tried before the infix parser, so the remaining types
  859 -- must be parenthesized to prevent ambiguity.
  860 infixConOperandNeedsParens :: Type -> Bool
  861 infixConOperandNeedsParens (TAnn _ sub) = infixConOperandNeedsParens sub
  862 infixConOperandNeedsParens (TParen _) = False
  863 infixConOperandNeedsParens (TTuple Boxed _ _) = False
  864 infixConOperandNeedsParens (TTuple Unboxed _ _) = False
  865 infixConOperandNeedsParens (TUnboxedSum {}) = True
  866 infixConOperandNeedsParens (TList _ []) = True
  867 infixConOperandNeedsParens (TBuiltinCon TBuiltinList) = True
  868 infixConOperandNeedsParens (TInfix {}) = True
  869 -- Application head determines what the parser sees first.
  870 infixConOperandNeedsParens (TApp f _) = infixConOperandNeedsParens f
  871 infixConOperandNeedsParens (TTypeApp _ TSplice {}) = True
  872 infixConOperandNeedsParens (TTypeApp f _) = infixConOperandNeedsParens f
  873 infixConOperandNeedsParens _ = False

  876 prefixConBangTypeNeedsParens (TAnn _ sub) = prefixConBangTypeNeedsParens sub

 1048 -- | Add parentheses to an expression at all required positions.
 1049 --
 1050 -- >>> shorthand $ addExprParens (EApp (EVar "f") (EInfix (EVar "x") "+" (EVar "y")))
 1051 -- EApp (EVar "f") (EParen (EInfix (EVar "x") "+" (EVar "y")))
 1052 --
 1053 -- >>> shorthand $ addExprParens (EApp (EVar "f") (EVar "x"))
 1054 -- EApp (EVar "f") (EVar "x")
 1055 addExprParens :: Expr -> Expr
 1056 addExprParens = addExprParensPrec 0
 1057 
 1058 addExprParensIn :: ExprCtx -> Expr -> Expr
 1059 addExprParensIn ctx@(CtxInfixRhs protectOpenEnded) expr@EInfix {} =
 1060   wrapExpr (needsExprParens ctx expr) (addInfixRhsParens protectOpenEnded expr)
 1061 addExprParensIn ctx expr =
 1062   wrapExpr (needsExprParens ctx expr) (addExprParensPrec (exprCtxPrec ctx expr) expr)

 1064 -- | Flatten a left-nested application chain.
 1065 flattenApps :: Expr -> (Expr, [Expr])
 1066 flattenApps = go []
 1067   where
 1068     go args (EAnn _ x) = go args x
 1069     go args (EApp fn arg) = go (arg : args) fn
 1070     go args root = (root, args)
 1071 
 1072 addExprParensPrec :: Int -> Expr -> Expr
 1073 addExprParensPrec prec expr =
 1074   case expr of
 1075     EApp {} -> addAppsChainPrec prec expr
 1076     ETypeApp fn ty ->
 1077       let fn' = wrapExpr (isGreedyExpr fn) (addExprParensIn CtxAppFun fn)
 1078        in wrapExpr (prec > 2) (ETypeApp fn' (addTypeIn CtxTypeAppVisibleArg ty))
 1079     ETypeSyntax form ty -> wrapExpr (prec > 2) (ETypeSyntax form (addSignatureTypeParens ty))
 1080     EVar {} -> expr
 1081     EInt {} -> expr
 1082     EFloat {} -> expr
 1083     EChar {} -> expr
 1084     ECharHash {} -> expr
 1085     EString {} -> expr
 1086     EStringHash {} -> expr
 1087     EOverloadedLabel {} -> expr
 1088     EQuasiQuote {} -> expr
 1089     ETHExpQuote body -> ETHExpQuote (addExprParens body)
 1090     ETHTypedQuote body -> ETHTypedQuote (addExprParens body)
 1091     ETHDeclQuote decls -> ETHDeclQuote (map addDeclParens decls)
 1092     ETHTypeQuote ty -> ETHTypeQuote (addTypeParens ty)
 1093     ETHPatQuote pat -> ETHPatQuote (addTHPatQuoteParens pat)
 1094     ETHNameQuote body -> ETHNameQuote (addExprParens body)
 1095     ETHTypeNameQuote ty -> ETHTypeNameQuote (addSignatureTypeParens ty)
 1096     ETHSplice body -> ETHSplice (addSpliceBodyParens body)
 1097     ETHTypedSplice body -> ETHTypedSplice (addSpliceBodyParens body)
 1098     EIf cond yes no ->
 1099       wrapExpr
 1100         (prec > 0)
 1101         (EIf (addExprParens cond) (addExprParens yes) (addExprParens no))
 1102     EMultiWayIf rhss ->
 1103       wrapExpr (prec > 0) (EMultiWayIf (map (addGuardedRhsParens GuardArrow) rhss))
 1104     ELambdaPats pats body ->
 1105       wrapExpr (prec > 0) (ELambdaPats (map addArrowBndrPatternParens pats) (addExprParens body))
 1106     ELambdaCase alts ->
 1107       wrapExpr (prec > 0) (ELambdaCase (map addCaseAltParens alts))
 1108     ELambdaCases alts ->
 1109       wrapExpr (prec > 0) (ELambdaCases (map addLambdaCaseAltParens alts))
 1110     EInfix lhs op rhs ->
 1111       wrapExpr
 1112         (prec > 1)
 1113         ( EInfix
 1114             (addTopLevelInfixLhsParens prec lhs)
 1115             op
 1116             (addExprParensIn (CtxInfixRhs (prec == 1)) rhs)
 1117         )
 1118     ENegate inner ->
 1119       wrapExpr (prec > 2) (ENegate (addNegateParens inner))
 1120     ESectionL lhs op ->
 1121       -- Sections always require surrounding parens in source syntax.
 1122       -- Wrap in EParen so the AST matches what the parser produces.
 1123       let lhs' =
 1124             if isGreedyExpr lhs || isTypeSig lhs
 1125               then wrapExpr True (addExprParens lhs)
 1126               else addSectionLhsParens lhs
 1127        in EParen (ESectionL lhs' op)
 1128     ESectionR op rhs ->
 1129       -- Right sections accept ordinary expressions as operands. Only explicit
 1130       -- type signatures need extra grouping to avoid @(`op` x :: T)@.
 1131       EParen (ESectionR op (addExprParensIn CtxSectionRhs rhs))
 1132     ELetDecls decls body ->
 1133       wrapExpr (prec > 0) (ELetDecls (map addDeclParens decls) (addExprParens body))
 1134     ECase scrutinee alts ->
 1135       wrapExpr (prec > 0) (ECase (addExprParens scrutinee) (map addCaseAltParens alts))
 1136     EDo stmts flavor ->
 1137       wrapExpr (prec > 0) (EDo (map addDoStmtParens stmts) flavor)
 1138     EListComp body quals ->
 1139       EListComp (addExprParens body) (map addCompStmtParens quals)
 1140     EListCompParallel body qualifierGroups ->
 1141       EListCompParallel (addExprParens body) (map (map addCompStmtParens) qualifierGroups)
 1142     EArithSeq seqInfo -> EArithSeq (addArithSeqParens seqInfo)
 1143     ERecordCon name fields hasWildcard ->
 1144       ERecordCon name [field {recordFieldValue = addExprParens (recordFieldValue field)} | field <- fields] hasWildcard
 1145     ERecordUpd base fields ->
 1146       ERecordUpd (addExprParensPrec 3 base) [field {recordFieldValue = addExprParens (recordFieldValue field)} | field <- fields]
 1147     EGetField base field ->
 1148       -- Qualified names (A.a) must be parenthesized because A.a.field would be
 1149       -- parsed as the qualified name A.a.field rather than field access on A.a.
 1150       let base' = addExprParensPrec 3 base
 1151        in EGetField (wrapExpr (needsParensBeforeDotField base' field) base') field
 1152     EGetFieldProjection {} -> EParen expr
 1153     ETypeSig inner ty ->
 1154       wrapExpr (prec > 1) (ETypeSig (addTypeSigBodyParens inner) (addSignatureTypeParens ty))
 1155     EParen inner ->
 1156       -- If inner is a section or projection, addExprParens(inner) already produces EParen(section/projection).
 1157       -- Delegating avoids double-wrapping and maintains idempotency.
 1158       case peelExprAnn inner of
 1159         ESectionL {} -> addExprParens inner
 1160         ESectionR {} -> addExprParens inner
 1161         EGetFieldProjection {} -> addExprParens inner
 1162         _ -> EParen (addExprParens inner)
 1163     EList values -> EList (map addDelimitedExprParens values)
 1164     ETuple tupleFlavor values -> ETuple tupleFlavor (map (fmap addDelimitedExprParens) values)
 1165     EUnboxedSum altIdx arity inner -> EUnboxedSum altIdx arity (addDelimitedExprParens inner)
 1166     EProc pat body ->
 1167       wrapExpr (prec > 0) (EProc (addArrowBndrPatternParens pat) (addCmdParensIn CtxCmdTop body))
 1168     EPragma pragma inner ->
 1169       -- EPragma is transparent w.r.t. precedence: wrapping decisions are made
 1170       -- by the outer context via needsExprParens, not by a self-prec check.
 1171       EPragma pragma (addExprParens inner)
 1172     EAnn ann sub -> EAnn ann (addExprParensPrec prec sub)
 1173   where
 1174     isTypeSig :: Expr -> Bool
 1175     isTypeSig (EAnn _ sub) = isTypeSig sub
 1176     isTypeSig (ETypeSig {}) = True
 1177     isTypeSig _ = False

 1179 -- | Handle application chains, preserving original EApp spans.
 1180 addAppsChainPrec :: Int -> Expr -> Expr
 1181 addAppsChainPrec prec expr =
 1182   let (root, args) = flattenApps expr
 1183       root' = addExprParensIn CtxAppFun root
 1184       nArgs = length args
 1185       args' =
 1186         [ let isLast = i == nArgs - 1
 1187               canStayBare = isBlockExpr a && not (endsWithTypeSig a) && (isLast || not (isOpenEnded a))
 1188               ctx
 1189                 | canStayBare = CtxAppArgNoParens
 1190                 | isLast = CtxAppArg
 1191                 | isGreedyExpr a = CtxAppArgGreedy
 1192                 | otherwise = CtxAppArg
 1193            in addExprParensIn ctx a
 1194         | (i, a) <- Prelude.zip [0 :: Int ..] args
 1195         ]
 1196       rebuilt = foldl EApp root' args'
 1197    in wrapExpr (prec > 2) rebuilt

 1199 addSpliceBodyParens :: Expr -> Expr
 1200 addSpliceBodyParens body =
 1201   case body of
 1202     EAnn ann sub -> EAnn ann (addSpliceBodyParens sub)
 1203     -- Bare variable names, including qualified names like $A.a, use the
 1204     -- compact splice syntax. Symbolic names still render with their own
 1205     -- operator parentheses, e.g. $(A.+).
 1206     EVar {} -> body
 1207     -- Tuple and unboxed-sum syntax is already delimited, so the splice marker
 1208     -- can attach directly: $(), $$(1, 2), $(# | x #).
 1209     ETuple tupleFlavor values -> ETuple tupleFlavor (map (fmap addExprParens) values)
 1210     EUnboxedSum altIdx arity inner -> EUnboxedSum altIdx arity (addExprParens inner)
 1211     -- For everything else (including sections, which addExprParens now wraps
 1212     -- in EParen): wrap in one outer EParen so the body prints as $(expr).
 1213     -- A body that was already parenthesized keeps exactly one wrapper for
 1214     -- sections/projections, whose addExprParens result is itself an EParen.
 1215     EParen inner ->
 1216       case peelExprAnn inner of
 1217         ESectionL {} -> addExprParens inner
 1218         ESectionR {} -> addExprParens inner

 1220         _ -> EParen (addExprParens inner)

 1227 -- @0x5e.a@ is tokenized by GHC as a hexadecimal fractional literal, not as
 1228 -- record-dot access. Other integer bases, and non-hex field initials, are safe.

 1230 hexIntegerLiteralDotNeedsParens expr field =
 1231   case expr of
 1232     EAnn _ sub -> hexIntegerLiteralDotNeedsParens sub field

 1234     _ -> False
 1235   where
 1236     isHexIntegerLiteral repr =
 1237       "0x" `T.isPrefixOf` repr || "0X" `T.isPrefixOf` repr

 1239       case T.uncons (nameText name) of
 1240         Just (c, _) -> isHexDigit c

 1243 addNegateParens :: Expr -> Expr
 1244 addNegateParens inner =
 1245   case peelExprAnn inner of
 1246     -- `-(518# {}).a` and similar forms must keep the field access grouped;
 1247     -- otherwise `-518# {}.a` is lexed as a negative primitive literal record update.
 1248     EGetField base _ | startsWithPrimitiveLiteral base -> wrapExpr True (addExprParens inner)

 1252     -- `proc`, so `- \x -> x`, `- do ...`, and `- proc ...` do not need an

 1254     _ | isBareNegateOperand inner -> addExprParens inner
 1255     -- Application and type-application bind tighter than negation, so `-f x`

 1487 addSignatureTypeParensShared :: TypeCtx -> Int -> Type -> Type
 1488 addSignatureTypeParensShared ctx prec ty =
 1489   let atom = addSignatureTypeParensShared CtxTypeAtom
 1490    in case ty of
 1491         TAnn ann sub -> TAnn ann (addSignatureTypeParensShared ctx prec sub)
 1492         TVar {} -> ty
 1493         TCon name promoted -> TCon name promoted
 1494         TBuiltinCon {} -> ty
 1495         TImplicitParam name inner -> TImplicitParam name (addImplicitParamBodyParens inner)
 1496         TTypeLit {} -> ty
 1497         TStar {} -> ty
 1498         TQuasiQuote {} -> ty
 1499         TForall telescope inner ->
 1500           -- forallTypeParser uses contextOrFunTypeParser (not typeParser) for its
 1501           -- body, so a bare nested TForall would fail to parse. Wrap it in TParen.
 1502           wrapTy
 1503             (prec > 0)
 1504             ( TForall
 1505                 (telescope {forallTelescopeBinders = map addTyVarBinderParens (forallTelescopeBinders telescope)})
 1506                 (addForallBodyParens inner)
 1507             )
 1508         TInfix lhs op promoted rhs ->
 1509           -- Type operators are right-associative in GHC, so the RHS can contain
 1510           -- nested TInfix without parens (a `op1` b `op2` c = a `op1` (b `op2` c)).
 1511           -- The LHS needs parens for nested TInfix to prevent left-association.
 1512           wrapTy (prec > 0) (TInfix (addTypeIn CtxTypeAppFun lhs) op promoted (addTypeIn CtxTypeInfixRhs rhs))
 1513         TApp f x ->
 1514           wrapTy (prec > 2) (TApp (addTypeIn CtxTypeAppFun f) (addTypeIn CtxTypeAppArg x))
 1515         TTypeApp f x ->
 1516           wrapTy (prec > 2) (TTypeApp (addTypeIn CtxTypeAppFun f) (addTypeIn CtxTypeAppVisibleArg x))
 1517         TFun arrowKind a b ->
 1518           wrapTy (prec > 0) (TFun (addArrowKindParens arrowKind) (addTypeIn CtxTypeFunArg a) (atom 0 b))
 1519         TTuple tupleFlavor promoted elems ->
 1520           TTuple tupleFlavor promoted (map addSignatureTypeParensInner elems)
 1521         TUnboxedSum elems -> TUnboxedSum (map addSignatureTypeParensInner elems)
 1522         TList promoted elems -> TList promoted (map addSignatureTypeParensInner elems)
 1523         -- Inside an explicit TParen, TKindSig does not need an additional
 1524         -- TParen wrapper since the enclosing delimiter already provides it.
 1525         TParen inner -> TParen (addTypeInExplicitParenParens inner)
 1526         -- TKindSig always needs parens in most contexts. The parser absorbs
 1527         -- (ty :: kind) as TKindSig directly, so the TParen is not preserved
 1528         -- through roundtrips. The pretty-printer relies on the TParen wrapper
 1529         -- to produce the required parentheses.
 1530         TKindSig ty' kind ->
 1531           wrapTy True (TKindSig (atom 0 ty') (atom 0 kind))
 1532         TContext constraints inner ->
 1533           wrapTy (prec > 0) (TContext (addContextConstraints constraints) (atom 0 inner))
 1534         TSplice body -> TSplice (addSpliceBodyParens body)
 1535         TWildcard {} -> ty

 1548 -- @forall a. t :: k@ parses as @(forall a. t) :: k@, not
 1549 -- @forall a. (t :: k)@. When the kind-signature subject is itself a forall,
 1550 -- however, @forall a. forall b. t :: k@ preserves the nested forall body and
 1551 -- does not need an extra TParen.
 1552 addForallBodyParens :: Type -> Type
 1553 addForallBodyParens (TAnn ann sub) = TAnn ann (addForallBodyParens sub)
 1554 addForallBodyParens (TKindSig ty kind)
 1555   | typeStartsWithForall ty =
 1556       TKindSig (addSignatureTypeParensShared CtxTypeAtom 0 ty) (addSignatureTypeParensShared CtxTypeAtom 0 kind)
 1557 addForallBodyParens ty@(TForall {}) = addSignatureTypeParensShared CtxTypeAtom 0 ty
 1558 addForallBodyParens ty = addSignatureTypeParensShared CtxTypeAtom 0 ty
 1559 
 1560 typeStartsWithForall :: Type -> Bool
 1561 typeStartsWithForall (TAnn _ sub) = typeStartsWithForall sub
 1562 typeStartsWithForall TForall {} = True
 1563 typeStartsWithForall _ = False

 1570 -- | Process the body to the right of a constraint arrow in a top-level type
 1571 -- RHS. A bare kind signature there applies to the whole constrained type:
 1572 -- @type X = C => T :: K@ parses like @type X = (C => T) :: K@.
 1573 -- Parenthesize inner kind signatures to preserve @C => (T :: K)@.
 1574 addContextBodyParens :: Type -> Type
 1575 addContextBodyParens (TAnn ann sub) = TAnn ann (addContextBodyParens sub)
 1576 addContextBodyParens ty =
 1577   case ty of
 1578     TForall telescope inner ->
 1579       TForall
 1580         (telescope {forallTelescopeBinders = map addTyVarBinderParens (forallTelescopeBinders telescope)})
 1581         (addContextBodyParens inner)
 1582     TKindSig ty' kind ->
 1583       wrapTy
 1584         True
 1585         (TKindSig (addSignatureTypeParensShared CtxTypeAtom 0 ty') (addSignatureTypeParensShared CtxTypeAtom 0 kind))
 1586     TContext constraints inner ->
 1587       TContext (map addContextConstraintDelimitedParens constraints) (addContextBodyParens inner)
 1588     _ -> addSignatureTypeParensShared CtxTypeAtom 0 ty

 1678 -- | Add parentheses to a pattern at all required positions.
 1679 --
 1680 -- >>> shorthand $ addPatternParens (PAs "xs" (PNegLit (LitInt 1 TInteger "1")))
 1681 -- PAs UnqualifiedName {"xs"} (PParen (PNegLit (LitInt 1 TInteger)))
 1682 --
 1683 -- >>> shorthand $ addPatternParens (PAs "xs" (PVar "x"))
 1684 -- PAs UnqualifiedName {"xs"} (PVar "x")
 1685 addPatternParens :: Pattern -> Pattern
 1686 addPatternParens pat =
 1687   case pat of
 1688     PAnn sp sub -> PAnn sp (addPatternParens sub)
 1689     PVar {} -> pat
 1690     PTypeBinder binder -> PTypeBinder (addTyVarBinderParens binder)
 1691     PTypeSyntax form ty -> PTypeSyntax form (addSignatureTypeParens ty)
 1692     PWildcard {} -> pat
 1693     PLit lit -> PLit lit
 1694     PQuasiQuote {} -> pat
 1695     PTuple tupleFlavor elems -> PTuple tupleFlavor (map addPatternInDelimited elems)
 1696     PUnboxedSum altIdx arity inner -> PUnboxedSum altIdx arity (addPatternInUnboxedSum altIdx inner)
 1697     PList elems -> PList (map addPatternInDelimited elems)
 1698     PCon con typeArgs args -> PCon con (map (addTypeIn CtxTypeAtom) typeArgs) (map addPatternAtomParens args)
 1699     PInfix lhs op rhs -> PInfix (addPatternInfixOperandParens lhs) op (addPatternInfixRhsOperandParens rhs)
 1700     PView viewExpr inner ->
 1701       wrapPat True (PView (addViewExprParens viewExpr) (addPatternViewInnerParens inner))
 1702     PAs name inner -> PAs name (addApatParens inner)
 1703     PStrict inner -> PStrict (addApatParens inner)
 1704     PIrrefutable inner -> PIrrefutable (addApatParens inner)
 1705     PNegLit lit -> PNegLit lit
 1706     PParen inner -> PParen (addPatternInDelimited inner)
 1707     PRecord con fields hasWildcard ->
 1708       PRecord con [field {recordFieldValue = addPatternInRecordField (recordFieldValue field)} | field <- fields] hasWildcard
 1709     PTypeSig inner ty -> PTypeSig (addPatternInfixOperandParens inner) (addSignatureTypeParens ty)
 1710     PSplice body -> PSplice (addSpliceBodyParens body)
 1711 
 1712 -- | Add parens for a pattern inside a delimited context (tuples, lists, etc.).
 1713 -- View patterns don't need extra parens there.

 1719   case peelPatternAnn pat of
 1720     PView viewExpr inner ->
 1721       PView
 1722         ((if allowLayoutTypeSig then addViewExprParensAllowLayoutTypeSig else addViewExprParens) viewExpr)
 1723         (addPatternViewInnerParens inner)
 1724     PAs name inner -> PAs name (addApatParens inner)
 1725     PStrict inner -> PStrict (addApatParens inner)
 1726     PIrrefutable inner -> PIrrefutable (addApatParens inner)
 1727     _ -> addPatternParens pat
 1728 
 1729 addPatternInUnboxedSum :: Int -> Pattern -> Pattern
 1730 addPatternInUnboxedSum altIdx = addPatternInDelimitedWith (altIdx > 0)
 1731 
 1732 addPatternInRecordField :: Pattern -> Pattern
 1733 addPatternInRecordField = addPatternInDelimitedWith True

 1745 -- Nested view patterns do not need extra parens.
 1746 addPatternViewInnerParens :: Pattern -> Pattern
 1747 addPatternViewInnerParens pat =
 1748   case pat of
 1749     PAnn sp sub -> PAnn sp (addPatternViewInnerParens sub)
 1750     PView viewExpr inner ->

 1752     _ -> addPatternParens pat

 1754 addViewExprParens :: Expr -> Expr
 1755 addViewExprParens = addViewExprParensWith False
 1756 
 1757 addViewExprParensAllowLayoutTypeSig :: Expr -> Expr
 1758 addViewExprParensAllowLayoutTypeSig = addViewExprParensWith True
 1759 
 1760 addViewExprParensWith :: Bool -> Expr -> Expr
 1761 addViewExprParensWith allowLayoutTypeSig expr =
 1762   wrapExpr (viewExprNeedsOuterParens expr) (addViewExprBodyParens expr)
 1763   where
 1764     -- View-pattern expressions are followed by @-> pat@ inside a pattern, and
 1765     -- that arrow gives the expression parser a hard boundary.  That lets block
 1766     -- arguments stay bare in application chains where ordinary expression
 1767     -- printing would add parens to protect the following syntax.
 1768     --
 1769     -- With MultiWayIf, UnboxedSums, ViewPatterns, and BlockArguments enabled,
 1770     -- this is valid in a pattern context:
 1771     --
 1772     --   f (# [] if | let {} -> [] [] -> _ | #) = ()
 1773     --
 1774     -- and parses like:
 1775     --
 1776     --   f (# ([] (if | let {} -> []) [] -> _) | #) = ()
 1777     --
 1778     -- The same unboxed-sum spelling is not valid as an expression RHS; GHC is
 1779     -- the source of truth, so this exception only applies while printing view
 1780     -- patterns.
 1781     --
 1782     -- Type signatures remain different.  A block argument ending in @:: T@
 1783     -- can still capture the following arrow or change the parsed shape, so it
 1784     -- stays parenthesized:
 1785     --
 1786     --   f (([] (if | [] -> [] :: T)) -> p) = ()
 1787     addViewExprBodyParens e =
 1788       case e of
 1789         EAnn ann sub -> EAnn ann (addViewExprBodyParens sub)
 1790         EInfix {} -> addViewExprInfixParens e
 1791         EApp {} -> addViewExprAppChainParens e
 1792         _ -> addExprParens e
 1793 
 1794     addViewExprInfixParens e =
 1795       case e of
 1796         EAnn ann sub -> EAnn ann (addViewExprInfixParens sub)
 1797         EInfix lhs op rhs ->
 1798           EInfix
 1799             (addViewExprInfixLhsParens lhs)
 1800             op
 1801             (addExprParensIn (CtxInfixRhs False) rhs)
 1802         _ -> addExprParens e
 1803 
 1804     addViewExprInfixLhsParens e =
 1805       case e of
 1806         EAnn ann sub -> EAnn ann (addViewExprInfixLhsParens sub)
 1807         EInfix lhs op rhs ->
 1808           EInfix
 1809             (addViewExprInfixLhsParens lhs)
 1810             op
 1811             (addExprParensIn (CtxInfixRhs True) rhs)
 1812         _
 1813           | viewExprInfixLhsCanStayBare e ->
 1814               addViewExprAppChainParens e
 1815           | otherwise ->
 1816               addExprParensIn CtxInfixLhs e
 1817 
 1818     addViewExprAppChainParens e =
 1819       let (root, args) = flattenApps e
 1820           root' = addExprParensIn CtxAppFun root
 1821           args' = [addExprParensIn (viewExprAppArgCtx isLast arg) arg | (isLast, arg) <- markLast args]
 1822        in foldl EApp root' args'
 1823 
 1824     viewExprAppArgCtx isLast arg
 1825       | viewExprAppArgCanStayBare isLast arg = CtxAppArgNoParens
 1826       | isLast = CtxAppArg
 1827       | isGreedyExpr arg = CtxAppArgGreedy
 1828       | otherwise = CtxAppArg
 1829 
 1830     viewExprAppArgCanStayBare isLast arg =
 1831       isBlockExpr arg
 1832         && not (endsWithTypeSig arg)
 1833         && (isLast || not (viewExprBlockArgAbsorbsFollowingAppArg arg))
 1834 
 1835     viewExprBlockArgAbsorbsFollowingAppArg arg =
 1836       case peelExprAnn arg of
 1837         EIf {} -> True
 1838         ELambdaPats {} -> True
 1839         ELetDecls {} -> True
 1840         EProc {} -> True
 1841         _ -> False
 1842 
 1843     viewExprInfixLhsCanStayBare e =
 1844       case peelExprAnn e of
 1845         EApp {} ->
 1846           case reverse (snd (flattenApps e)) of
 1847             arg : _ -> viewExprBlockArgCanPrecedeInfixOp arg
 1848             [] -> False
 1849         _ -> False
 1850 
 1851     viewExprBlockArgCanPrecedeInfixOp arg =
 1852       case peelExprAnn arg of
 1853         EMultiWayIf {} -> not (endsWithTypeSig arg)
 1854         _ -> False
 1855 
 1856     markLast [] = []
 1857     markLast [x] = [(True, x)]
 1858     markLast (x : xs) = (False, x) : markLast xs
 1859 
 1860     viewExprNeedsOuterParens e =
 1861       isTypeSyntaxExpr e || (endsWithTypeSig e && not (viewExprTypeSigCanStayBare e))
 1862 
 1863     viewExprTypeSigCanStayBare e =
 1864       allowLayoutTypeSig && isBareViewExprBlock e && not (classicIfTypeSigNeedsParens e)
 1865 
 1866     isTypeSyntaxExpr e =
 1867       case peelExprAnn e of
 1868         ETypeSyntax {} -> True
 1869         _ -> False
 1870 
 1871     isBareViewExprBlock e =
 1872       case peelExprAnn e of
 1873         ECase {} -> True
 1874         EIf {} -> True
 1875         EMultiWayIf {} -> True
 1876         ELambdaCase {} -> True
 1877         ELambdaCases {} -> True
 1878         _ -> False
 1879 
 1880     classicIfTypeSigNeedsParens e =
 1881       case peelExprAnn e of
 1882         EIf _ _ no -> elseBranchHasDirectTypeSig no
 1883         _ -> False
 1884 
 1885     elseBranchHasDirectTypeSig branch =
 1886       case peelExprAnn branch of
 1887         ETypeSig {} -> True
 1888         _ -> False

 2000     PIrrefutable {} -> wrapPat True (addPatternParens pat)

 2040       CmdCase (addExprParens scrut) (map addCmdCaseAltParens alts)
 2041     CmdLet decls body ->
 2042       CmdLet (map addDeclParens decls) (addCmdParens body)
 2043     CmdLam pats body ->
 2044       CmdLam (map addPatternAtomParens pats) (addCmdParens body)
 2045     CmdApp c e ->
 2046       CmdApp (addCmdParens c) (addExprParensPrec 3 e)
 2047     CmdPar c ->
 2048       CmdPar (addCmdParens c)
 2049   where
 2050     addCmdInfixLhsParens inner =
 2051       case inner of
 2052         CmdAnn ann sub -> CmdAnn ann (addCmdInfixLhsParens sub)
 2053         CmdInfix l op r ->
 2054           CmdInfix
 2055             (addCmdInfixLhsParens l)
 2056             op
 2057             (addCmdInfixRhsParens True r)
 2058         _ -> wrapCmdInfixLhs (addCmdParens inner)
 2059 
 2060     addCmdInfixRhsParens protectFollowingInfix inner =
 2061       let inner' = addCmdParens inner
 2062        in wrapCmd
 2063             ( case peelCmdAnn inner' of
 2064                 CmdArrApp {} -> True
 2065                 CmdInfix {} -> True
 2066                 _ -> protectFollowingInfix && cmdAbsorbsFollowingInfix inner'
 2067             )
 2068             inner'
 2069 
 2070     wrapCmdInfixLhs inner =
 2071       case peelCmdAnn inner of
 2072         CmdArrApp {} -> CmdPar inner
 2073         CmdIf {} -> CmdPar inner
 2074         CmdLet {} -> CmdPar inner
 2075         CmdLam {} -> CmdPar inner
 2076         _ -> inner
 2077 
 2078     cmdAbsorbsFollowingInfix inner =
 2079       case peelCmdAnn inner of
 2080         CmdLam {} -> True
 2081         CmdLet {} -> True
 2082         CmdIf {} -> True
 2083         CmdCase {} -> True
 2084         CmdDo {} -> True
 2085         CmdInfix _ _ rhs -> cmdAbsorbsFollowingInfix rhs
 2086         _ -> False
 2087 
 2088 addCmdArrAppLhsParens :: Expr -> Expr
 2089 addCmdArrAppLhsParens lhs =
 2090   wrapExpr (cmdArrAppLhsNeedsParens lhs) (addExprParensPrec 1 lhs)
 2091 
 2092 cmdArrAppLhsNeedsParens :: Expr -> Bool
 2093 cmdArrAppLhsNeedsParens expr =
 2094   startsWithBlockExpr expr
 2095     || isOpenEnded expr
 2096     || endsWithTypeSig expr
 2097     || endsWithCmdLayoutTail expr
 2098 
 2099 -- | Arrow tails are parsed outside the expression grammar, so a command lhs that
 2100 -- ends in a layout block needs explicit grouping even when the same expression
 2101 -- can stay bare before ordinary expression suffixes like @::@ or @\@-type apps.
 2102 endsWithCmdLayoutTail :: Expr -> Bool
 2103 endsWithCmdLayoutTail = \case
 2104   EAnn _ sub -> endsWithCmdLayoutTail sub
 2105   expr | endsWithCmdLayoutBlock expr -> True
 2106   EInfix _ _ rhs -> endsWithCmdLayoutTail rhs
 2107   ENegate inner -> endsWithCmdLayoutTail inner
 2108   EPragma _ inner -> endsWithCmdLayoutTail inner
 2109   EApp _ arg -> endsWithCmdLayoutBlock arg
 2110   _ -> False
 2111 
 2112 endsWithCmdLayoutBlock :: Expr -> Bool
 2113 endsWithCmdLayoutBlock = \case
 2114   EAnn _ sub -> endsWithCmdLayoutBlock sub
 2115   EDo {} -> True
 2116   ELambdaCase {} -> True
 2117   ELambdaCases {} -> True
 2118   EApp _ arg -> endsWithCmdLayoutBlock arg
 2119   _ -> False

 2121 addCmdArrAppRhsParens :: CmdCtx -> Expr -> Expr
 2122 addCmdArrAppRhsParens ctx rhs =
 2123   wrapExpr (cmdTopNeedsRhsParens && endsWithTypeSig rhs) (addExprParens rhs)
 2124   where
 2125     cmdTopNeedsRhsParens =
 2126       case ctx of
 2127         CtxCmdTop -> True
 2128         CtxCmdDoStmt -> False

 2130 addCmdDoStmtParens :: DoStmt Cmd -> DoStmt Cmd
 2131 addCmdDoStmtParens stmt =
 2132   case stmt of
 2133     DoAnn ann inner -> DoAnn ann (addCmdDoStmtParens inner)
 2134     DoBind pat cmd' -> DoBind (addPatternParens pat) (addCmdParensIn CtxCmdDoStmt cmd')
 2135     DoLetDecls decls -> DoLetDecls (map addDeclParens decls)
 2136     DoExpr cmd' -> DoExpr (wrapCmd (isLetCmd cmd') (addCmdParensIn CtxCmdDoStmt cmd'))
 2137     DoRecStmt stmts -> DoRecStmt (map addCmdDoStmtParens stmts)
 2138   where
 2139     isLetCmd CmdLet {} = True
 2140     isLetCmd (CmdAnn _ sub) = isLetCmd sub
 2141     isLetCmd _ = False

 2143 addCmdCaseAltParens :: CaseAlt Cmd -> CaseAlt Cmd
 2144 addCmdCaseAltParens (CaseAlt anns pat rhs) =
 2145   CaseAlt anns (addCaseAltPatternParens pat) (addCmdCaseAltRhsParens rhs)
 2146 
 2147 addCmdCaseAltRhsParens :: Rhs Cmd -> Rhs Cmd
 2148 addCmdCaseAltRhsParens rhs =
 2149   case rhs of
 2150     UnguardedRhs sp body whereDecls ->
 2151       UnguardedRhs sp (addCmdParens body) (fmap (map addDeclParens) whereDecls)
 2152     GuardedRhss sp guards whereDecls ->
 2153       GuardedRhss sp (map addCmdGuardedRhsParens guards) (fmap (map addDeclParens) whereDecls)
 2154 
 2155 addCmdGuardedRhsParens :: GuardedRhs Cmd -> GuardedRhs Cmd
 2156 addCmdGuardedRhsParens grhs =
 2157   grhs
 2158     { guardedRhsGuards = addGuardQualifiersParens GuardArrow (guardedRhsGuards grhs),
 2159       guardedRhsBody = addCmdParens (guardedRhsBody grhs)
 2160     }
