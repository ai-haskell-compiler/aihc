    5 -- Module      : Aihc.Parser.Pretty
    6 -- Description : Pretty-printing of AST to Haskell source code
    7 --
    8 -- This module provides pretty-printing of parsed AST back to valid Haskell
    9 -- source code. It is used for round-trip testing and code generation.
   10 --
   11 -- The 'Pretty' instances from 'Prettyprinter' are provided for the main
   12 -- AST types, allowing direct use of 'pretty' from @Prettyprinter@.
   13 --
   14 -- __Parenthesization__ is handled by the 'Aihc.Parser.Parens' module, which
   15 -- inserts 'EParen', 'PParen', 'TParen', and 'CmdPar' nodes at all required
   16 -- positions. The Pretty instances call the paren-insertion pass before
   17 -- formatting, so the formatting code here does not need to worry about
   18 -- operator precedence or context-sensitive parenthesization.
   19 --
   20 -- Import this module to bring the 'Pretty' instances into scope. The exported
   21 -- helpers render AST nodes without first inserting additional parentheses,
   22 -- which is useful for tests that need to inspect parenthesization directly.
   23 --
   24 -- __Provided instances:__ 'Pretty' for 'Module', 'Expr', 'Pattern', 'Type'.
   25 module Aihc.Parser.Pretty
   26   ( prettyExpr,
   27     prettyPattern,
   28     prettyType,
   29   )

   64 -- | Wrap a document in braces with interior spaces: @{ content }@.
   65 -- Unlike 'braces' which produces @{content}@, this version avoids the
   66 -- @{-@ block-comment ambiguity that occurs when the content starts with a
   67 -- minus sign (e.g. a negated literal pattern in a let binding).
   68 spacedBraces :: Doc ann -> Doc ann
   69 spacedBraces d = "{" <+> d <+> "}"

   73   | T.isInfixOf "\n" raw = nesting $ \level -> nest (negate level) (pretty raw)
   74   | otherwise = pretty raw

  105 prettyModuleDoc :: Module -> Doc ann
  106 prettyModuleDoc modu =
  107   vsep (pragmaLines <> headerLines <> importLines <> declLines)
  108   where
  109     pragmaLines =
  110       map
  111         (\ext -> "{-# LANGUAGE" <+> pretty (extensionSettingName ext) <+> "#-}")
  112         (moduleLanguagePragmas modu)
  113     headerLines =
  114       case moduleName modu of
  115         Just name ->
  116           [ hsep
  117               ( ["module", pretty name]
  118                   <> maybe [] (\p -> [prettyPragma p]) (moduleWarningPragma modu)
  119                   <> maybe [] (\specs -> [prettyExportSpecList specs]) (moduleExports modu)
  120                   <> ["where"]
  121               )
  122           ]
  123         Nothing -> []
  124     importLines = map prettyImportDecl (moduleImports modu)
  125     declLines = concatMap prettyDeclLines (moduleDecls modu)
  126 
  127 prettyExportSpecList :: [ExportSpec] -> Doc ann
  128 prettyExportSpecList specs =
  129   parens (hsep (punctuate comma (map prettyExportSpec specs)))
  130 
  131 prettyExportSpec :: ExportSpec -> Doc ann
  132 prettyExportSpec spec =
  133   case spec of
  134     ExportAnn _ sub -> prettyExportSpec sub
  135     ExportModule mWarning modName -> prettyExportWarning mWarning ("module" <+> pretty modName)
  136     ExportVar mWarning namespace name ->
  137       prettyExportWarning mWarning (prettyNamespacePrefix namespace <> prettyName name)
  138     ExportAbs mWarning namespace name ->
  139       prettyExportWarning mWarning (prettyNamespacePrefix namespace <> prettyName name)
  140     ExportAll mWarning namespace name ->
  141       prettyExportWarning mWarning (prettyNamespacePrefix namespace <> prettyName name <> "(..)")
  142     ExportWith mWarning namespace name members ->
  143       prettyExportWarning
  144         mWarning
  145         (prettyNamespacePrefix namespace <> prettyName name <> parens (hsep (punctuate comma (map prettyExportMember members))))
  146     ExportWithAll mWarning namespace name wildcardIndex members ->
  147       prettyExportWarning
  148         mWarning
  149         (prettyNamespacePrefix namespace <> prettyName name <> parens (hsep (punctuate comma (insertWildcard wildcardIndex (map prettyExportMember members)))))
  150 
  151 prettyExportWarning :: Maybe Pragma -> Doc ann -> Doc ann
  152 prettyExportWarning mWarning doc =
  153   case mWarning of
  154     Nothing -> doc
  155     Just p -> hsep [prettyPragma p, doc]

  157 prettyImportDecl :: ImportDecl -> Doc ann
  158 prettyImportDecl decl =
  159   let renderPostQualified =
  160         importDeclQualifiedPost decl
  161           && importDeclQualified decl
  162    in hsep
  163         ( ["import"]
  164             <> ["safe" | importDeclSafe decl]
  165             <> maybe [] (\p -> [prettyPragma p]) (importDeclSourcePragma decl)
  166             <> ["qualified" | importDeclQualified decl && not renderPostQualified]
  167             <> maybe [] (\level -> [prettyImportLevel level]) (importDeclLevel decl)
  168             <> maybe [] (\pkg -> [prettyQuotedText pkg]) (importDeclPackage decl)
  169             <> [pretty (importDeclModule decl)]
  170             <> ["qualified" | importDeclQualified decl && renderPostQualified]
  171             <> maybe [] (\alias -> ["as", pretty alias]) (importDeclAs decl)
  172             <> maybe [] (\spec -> [prettyImportSpec spec]) (importDeclSpec decl)
  173         )
  174 
  175 prettyImportLevel :: ImportLevel -> Doc ann
  176 prettyImportLevel level =
  177   case level of
  178     ImportLevelQuote -> "quote"
  179     ImportLevelSplice -> "splice"

  181 -- | Pretty-print a top-level declaration splice.
  182 -- The expression is printed as-is; explicit TH splices appear as @$expr@ or
  183 -- @$(expr)@ through the normal 'ETHSplice' pretty-printer.
  184 -- Nested continuation lines keep infix operators from starting in top-level
  185 -- layout column one, where GHC can read them as a new declaration.
  186 prettyDeclSpliceExpr :: Expr -> Doc ann
  187 prettyDeclSpliceExpr = nest 2 . prettyExpr

  189 prettyQuotedText :: Text -> Doc ann
  190 prettyQuotedText txt = "\"" <> pretty txt <> "\""

  192 prettyImportSpec :: ImportSpec -> Doc ann
  193 prettyImportSpec spec =
  194   hsep
  195     ( ["hiding" | importSpecHiding spec]
  196         <> [parens (hsep (punctuate comma (map prettyImportItem (importSpecItems spec))))]
  197     )

  199 prettyImportItem :: ImportItem -> Doc ann
  200 prettyImportItem item =
  201   case item of
  202     ImportAnn _ sub -> prettyImportItem sub
  203     ImportItemVar namespace name -> prettyNamespacePrefix namespace <> prettyBinderUName name
  204     ImportItemAbs namespace name -> prettyNamespacePrefix namespace <> prettyConstructorUName name
  205     ImportItemAll namespace name -> prettyNamespacePrefix namespace <> prettyConstructorUName name <> "(..)"
  206     ImportItemWith namespace name members ->
  207       prettyNamespacePrefix namespace <> prettyConstructorUName name <> parens (hsep (punctuate comma (map prettyExportMember members)))
  208     ImportItemAllWith namespace name wildcardIndex members ->
  209       prettyNamespacePrefix namespace <> prettyConstructorUName name <> parens (hsep (punctuate comma (insertWildcard wildcardIndex (map prettyExportMember members))))

  211 insertWildcard :: Int -> [Doc ann] -> [Doc ann]
  212 insertWildcard wildcardIndex members =
  213   let (before, after) = splitAt wildcardIndex members
  214    in before <> [".."] <> after

  245 prettyDeclLines :: Decl -> [Doc ann]
  246 prettyDeclLines decl =
  247   case decl of
  248     DeclAnn _ sub -> prettyDeclLines sub
  249     DeclValue valueDecl -> prettyValueDeclLines valueDecl
  250     DeclTypeSig names ty -> [hsep [hsep (punctuate comma (map prettyBinderName names)), "::", prettyType ty]]
  251     DeclPatSyn patSynDecl -> [prettyPatSynDecl patSynDecl]
  252     DeclPatSynSig names ty -> [hsep ["pattern", hsep (punctuate comma (map prettyConstructorUName names)), "::", prettyType ty]]
  253     DeclStandaloneKindSig name kind -> [hsep ["type", prettyConstructorUName name, "::", prettyType kind]]
  254     DeclFixity assoc mNamespace prec ops ->
  255       [ hsep
  256           ( [prettyFixityAssoc assoc]
  257               <> maybe [] (pure . pretty . show) prec
  258               <> maybe [] (pure . prettyNamespace) mNamespace
  259               <> punctuate comma (map prettyInfixOp ops)
  260           )
  261       ]
  262     DeclRoleAnnotation ann -> [prettyRoleAnnotation ann]
  263     DeclTypeSyn synDecl ->
  264       [hsep (["type"] <> prettyDeclBinderHead [] (typeSynHead synDecl) <> ["=", prettyType (typeSynBody synDecl)])]
  265     DeclData dataDecl -> [prettyDataDecl dataDecl]
  266     DeclTypeData dataDecl -> [prettyTypeDataDecl dataDecl]
  267     DeclNewtype newtypeDecl -> [prettyNewtypeDecl newtypeDecl]
  268     DeclClass classDecl -> [prettyClassDecl classDecl]
  269     DeclInstance instanceDecl -> [prettyInstanceDecl instanceDecl]
  270     DeclStandaloneDeriving derivingDecl -> [prettyStandaloneDeriving derivingDecl]
  271     DeclDefault tys -> ["default" <+> parens (hsep (punctuate comma (map prettyType tys)))]
  272     DeclForeign foreignDecl -> [prettyForeignDecl foreignDecl]
  273     DeclSplice body -> [prettyDeclSpliceExpr body]
  274     DeclTypeFamilyDecl tf -> [prettyTypeFamilyDecl tf]
  275     DeclDataFamilyDecl df -> [prettyDataFamilyDecl df]
  276     DeclTypeFamilyInst tfi -> [prettyTopTypeFamilyInst tfi]
  277     DeclDataFamilyInst dfi -> [prettyTopDataFamilyInst dfi]
  278     DeclPragma pragma -> [prettyPragma pragma]

  280 prettyRoleAnnotation :: RoleAnnotation -> Doc ann
  281 prettyRoleAnnotation ann =
  282   hsep
  283     ( [ "type",
  284         "role",
  285         prettyConstructorUName (roleAnnotationName ann)
  286       ]
  287         <> map prettyRole (roleAnnotationRoles ann)

  290 prettyRole :: Role -> Doc ann

  295     RolePhantom -> "phantom"

  298 prettyValueDeclLines :: ValueDecl -> [Doc ann]
  299 prettyValueDeclLines valueDecl =
  300   case valueDecl of
  301     PatternBind multTag pat rhs -> [prettyMultiplicityTag multTag <> prettyPattern pat <+> prettyRhs rhs]
  302     FunctionBind name matches ->
  303       concatMap (prettyFunctionMatchLines name) matches

  305 -- | Pretty-print a value declaration on a single line.
  306 prettyValueDeclSingleLine :: ValueDecl -> Doc ann
  307 prettyValueDeclSingleLine valueDecl =
  308   case valueDecl of
  309     PatternBind multTag pat rhs -> prettyMultiplicityTag multTag <> prettyPattern pat <+> prettyRhs rhs
  310     FunctionBind name matches ->
  311       hsep (punctuate semi (map (prettyFunctionMatch name) matches))

  313 prettyMultiplicityTag :: MultiplicityTag -> Doc ann
  314 prettyMultiplicityTag NoMultiplicityTag = mempty
  315 prettyMultiplicityTag LinearMultiplicityTag = "%1" <+> mempty
  316 prettyMultiplicityTag (ExplicitMultiplicityTag ty) = "%" <> prettyType ty <+> mempty

  318 -- | Pretty-print a pattern synonym declaration.
  319 prettyPatSynDecl :: PatSynDecl -> Doc ann
  320 prettyPatSynDecl ps =
  321   hsep
  322     ( ["pattern"]
  323         <> prettyPatSynLhs (patSynDeclName ps) (patSynDeclArgs ps)
  324         <> [dirArrow (patSynDeclDir ps)]
  325         <> [prettyPattern (patSynDeclPat ps)]
  326         <> prettyPatSynWhere (patSynDeclName ps) (patSynDeclDir ps)
  327     )
  328   where
  329     dirArrow PatSynBidirectional = "="
  330     dirArrow PatSynUnidirectional = "<-"
  331     dirArrow (PatSynExplicitBidirectional _) = "<-"
  332 
  333 prettyPatSynLhs :: UnqualifiedName -> PatSynArgs -> [Doc ann]
  334 prettyPatSynLhs name args =
  335   case args of
  336     PatSynPrefixArgs vars ->
  337       prettyConstructorUName name : map pretty vars
  338     PatSynInfixArgs lhs rhs ->
  339       [pretty lhs, prettyInfixOp name, pretty rhs]
  340     PatSynRecordArgs fields ->
  341       [prettyConstructorUName name <+> braces (hsep (punctuate comma (map pretty fields)))]
  342 
  343 prettyPatSynWhere :: UnqualifiedName -> PatSynDir -> [Doc ann]
  344 prettyPatSynWhere _ PatSynBidirectional = []
  345 prettyPatSynWhere _ PatSynUnidirectional = []
  346 prettyPatSynWhere _ (PatSynExplicitBidirectional []) = ["where", spacedBraces mempty]
  347 prettyPatSynWhere name (PatSynExplicitBidirectional matches) =
  348   ["where" <> nest 2 (hardline <> vsep (map (prettyFunctionMatch name) matches))]

  350 prettyFunctionMatchLines :: UnqualifiedName -> Match -> [Doc ann]
  351 prettyFunctionMatchLines name match =
  352   case matchRhs match of
  353     UnguardedRhs {} -> [prettyFunctionMatch name match]
  354     GuardedRhss _ grhss mWhereDecls ->
  355       [ prettyFunctionHead name (matchHeadForm match) (matchPats match)
  356           <> nest
  357             2
  358             ( hardline
  359                 <> vsep
  360                   ( map prettyGuardedRhsBlock grhss
  361                       <> [prettyWhereClauseBare mWhereDecls | isJust mWhereDecls]
  362                   )
  363             )
  364       ]
  365 
  366 prettyFunctionMatch :: UnqualifiedName -> Match -> Doc ann
  367 prettyFunctionMatch name match =
  368   prettyFunctionHead name (matchHeadForm match) (matchPats match) <+> prettyRhs (matchRhs match)
  369 
  370 prettyFunctionHead :: UnqualifiedName -> MatchHeadForm -> [Pattern] -> Doc ann
  371 prettyFunctionHead name headForm pats =
  372   case headForm of
  373     MatchHeadPrefix ->
  374       hsep (prettyBinderUName name : map prettyPattern pats)
  375     MatchHeadInfix ->
  376       case pats of
  377         lhs : rhsPat : tailPats ->
  378           let infixHead = prettyPattern lhs <+> prettyInfixOp name <+> prettyPattern rhsPat
  379            in case tailPats of
  380                 [] -> infixHead
  381                 _ -> hsep (parens infixHead : map prettyPattern tailPats)
  382         _ ->
  383           hsep (prettyBinderUName name : map prettyPattern pats)
  384 
  385 prettyRhs :: Rhs Expr -> Doc ann
  386 prettyRhs rhs =
  387   case rhs of
  388     UnguardedRhs _ body whereDecls ->
  389       "="
  390         <+> prettyExpr body
  391         <> prettyIndentedWhereClause whereDecls
  392     GuardedRhss _ guards whereDecls ->
  393       nest 2 (hardline <> prettyGuardedRhssBlock guards whereDecls)
  394 
  395 prettyWhereClauseBare :: Maybe [Decl] -> Doc ann
  396 prettyWhereClauseBare Nothing = mempty
  397 prettyWhereClauseBare (Just []) = "where" <+> spacedBraces mempty
  398 prettyWhereClauseBare (Just decls) =
  399   "where" <> nest 2 (hardline <> vsep (concatMap prettyDeclLines decls))
  400 
  401 prettyIndentedWhereClause :: Maybe [Decl] -> Doc ann
  402 prettyIndentedWhereClause Nothing = mempty
  403 prettyIndentedWhereClause whereDecls = nest 2 (hardline <> prettyWhereClauseBare whereDecls)

  405 prettyTHDeclQuote :: [Decl] -> Doc ann
  406 prettyTHDeclQuote [] = "[d| |]"
  407 prettyTHDeclQuote decls =
  408   "[d|"
  409     <> nest 2 (hardline <> vsep (concatMap prettyDeclLines decls))

  421 -- | Pretty-print a type. The AST is assumed to already have TParen nodes
  422 -- in the correct positions (inserted by 'addTypeParens').
  423 --
  424 -- >>> let ty = TApp (TCon "Proxy" Unpromoted) (TParen (TFun ArrowUnrestricted (TCon "A" Unpromoted) (TCon "B" Unpromoted))) in (renderDoc (prettyType ty), show (shorthand ty))
  425 -- ("Proxy (A -> B)","TApp (TCon \"Proxy\") (TParen (TFun (TCon \"A\") (TCon \"B\")))")
  426 --
  427 -- | Check if a type, when pretty-printed, would start with a
  428 -- promotion tick.  This is used to decide whether a promoted list or
  429 -- tuple needs a space after the opening tick to avoid lexer confusion
  430 -- (e.g.  ''[ '[' ... ]'' is invalid, whereas ''[ '[ ... ]'' is valid).
  431 startsWithTick :: Type -> Bool
  432 startsWithTick (TAnn _ sub) = startsWithTick sub
  433 startsWithTick (TList Promoted _) = True
  434 startsWithTick (TCon _ Promoted) = True
  435 startsWithTick (TTuple _ Promoted _) = True
  436 startsWithTick (TTypeLit (TypeLitChar _ _)) = True
  437 startsWithTick (TApp f _) = startsWithTick f
  438 startsWithTick (TInfix lhs _ _ _) = startsWithTick lhs
  439 startsWithTick _ = False

  441 prettyType :: Type -> Doc ann
  442 prettyType ty =
  443   case ty of
  444     TAnn _ sub -> prettyType sub
  445     TVar name -> pretty name
  446     TCon name promoted ->
  447       let rendered = renderName name
  448           base
  449             | isSymbolicTypeName name = parens (pretty rendered)
  450             | otherwise = pretty rendered
  451           promoteTick
  452             | T.any (== '\'') rendered = "' "
  453             | otherwise = "'"
  454        in if promoted == Promoted then promoteTick <> base else base
  455     TBuiltinCon con -> prettyTypeBuiltinCon con
  456     TImplicitParam name inner -> pretty name <+> "::" <+> prettyType inner
  457     TTypeLit lit -> prettyTypeLiteral lit
  458     TStar spelling -> pretty spelling
  459     TQuasiQuote quoter body -> prettyQuasiQuote quoter body
  460     TForall telescope inner ->
  461       prettyForallTelescope telescope <+> prettyType inner
  462     -- Before infix detection: required grouping from the parser ('TParen',
  463     -- @(a :+: b) -> c@, constraints, nested @(c => t)@).
  464     TParen inner -> parens (prettyType inner)
  465     TInfix lhs op promoted rhs ->
  466       prettyType lhs
  467         <+> (if promoted == Promoted then "'" else mempty)
  468         <> prettyNameInfixOp op
  469         <+> prettyType rhs
  470     TApp f x ->
  471       prettyType f <+> prettyType x
  472     TTypeApp f x ->
  473       prettyType f <+> prettyTypeAppArg x
  474     TFun arrowKind a b ->
  475       prettyType a <+> prettyArrowKind arrowKind <+> prettyType b
  476     TTuple tupleFlavor promoted elems ->
  477       let elemsDoc = hsep (punctuate comma (map prettyType elems))
  478           tupleDoc = prettyTupleBody tupleFlavor elemsDoc
  479        in case (promoted, elems) of
  480             (Promoted, h : _) | startsWithTick h -> "' " <> tupleDoc
  481             (Promoted, _) -> "'" <> tupleDoc
  482             _ -> tupleDoc
  483     TUnboxedSum elems ->
  484       hsep ["(#", hsep (punctuate " |" (map prettyType elems)), "#)"]
  485     TList promoted elems ->
  486       let elemsDoc = hsep (punctuate comma (map prettyType elems))
  487           listDoc = brackets elemsDoc
  488        in case (promoted, elems) of
  489             (Promoted, h : _) | startsWithTick h -> "' " <> listDoc
  490             (Promoted, _) -> "'" <> listDoc
  491             _ -> listDoc
  492     TKindSig ty' kind ->
  493       prettyType ty' <+> "::" <+> prettyType kind
  494     TContext constraints inner ->
  495       case constraints of
  496         [] -> prettyType inner
  497         cs -> prettyContext cs <+> "=>" <+> prettyType inner
  498     TSplice body -> "$" <> prettySpliceBody body
  499     TWildcard -> "_"

  501 prettyArrowKind :: ArrowKind -> Doc ann
  502 prettyArrowKind ArrowUnrestricted = "->"
  503 prettyArrowKind ArrowLinear = "%1" <+> "->"
  504 prettyArrowKind (ArrowExplicit ty) = "%" <> prettyType ty <+> "->"

  510     TBuiltinArrow -> "(->)"
  511     TBuiltinList -> "[]"

  528 -- | Pretty-print a pattern. The AST is assumed to already have PParen nodes
  529 -- in the correct positions (inserted by 'addPatternParens').
  530 --
  531 -- >>> let pat = PAs "xs" (PParen (PNegLit (LitInt 1 TInteger "1"))) in (renderDoc (prettyPattern pat), show (shorthand pat))
  532 -- ("xs@(-1)","PAs UnqualifiedName {\"xs\"} (PParen (PNegLit (LitInt 1 TInteger)))")
  533 prettyPattern :: Pattern -> Doc ann
  534 prettyPattern pat =
  535   case pat of
  536     PAnn _ sub -> prettyPattern sub
  537     PVar name -> prettyBinderUName name
  538     PTypeBinder binder -> prettyTyVarBinder binder
  539     PTypeSyntax TypeSyntaxExplicitNamespace ty -> "type" <+> prettyType ty
  540     PTypeSyntax TypeSyntaxInTerm ty -> prettyType ty
  541     PWildcard -> "_"
  542     PLit lit -> prettyLiteral lit
  543     PQuasiQuote quoter body -> prettyQuasiQuote quoter body
  544     PTuple Unboxed [] -> "(# #)"
  545     PTuple tupleFlavor elems -> prettyTupleBody tupleFlavor (hsep (punctuate comma (map prettyPattern elems)))
  546     PUnboxedSum altIdx arity inner ->
  547       let slots = [if i == altIdx then prettyPattern inner else mempty | i <- [0 .. arity - 1]]
  548        in hsep ["(#", prettyBarSeparated slots, "#)"]
  549     PList elems -> brackets (hsep (punctuate comma (map prettyPattern elems)))
  550     PCon con typeArgs args -> hsep ([prettyPrefixName con] <> map prettyInvisibleTypeArg typeArgs <> map prettyPattern args)
  551     PInfix lhs op rhs -> prettyPattern lhs <+> prettyNameInfixOp op <+> prettyPattern rhs
  552     PView viewExpr inner ->
  553       prettyExpr viewExpr <> nest 1 (hardline <> "->" <+> prettyPattern inner)
  554     PAs name inner -> prettyBinderUName name <> "@" <> prettyPattern inner
  555     PStrict inner -> "!" <> prettyPattern inner
  556     PIrrefutable inner -> "~" <> prettyPattern inner
  557     PNegLit lit -> "-" <> prettyLiteral lit
  558     PParen inner -> parens (prettyPattern inner)
  559     PRecord con fields hasWildcard ->
  560       prettyPrefixName con
  561         <+> braces
  562           ( hsep
  563               ( punctuate
  564                   comma
  565                   ( [prettyPatternFieldBinding field | field <- fields]
  566                       ++ [".." | hasWildcard]
  567                   )
  568               )

  591 prettyDataDecl :: DataDecl -> Doc ann
  592 prettyDataDecl decl =
  593   case dataDeclConstructors decl of
  594     ctors
  595       | any isGadtCon ctors ->
  596           headDoc <+> prettyGadtConBlock ctors (dataDeclDeriving decl)
  597     _ ->
  598       hsep (headParts <> ctorPart <> derivingParts (dataDeclDeriving decl))
  599   where
  600     headParts =
  601       ["data"]
  602         <> maybe [] (pure . prettyPragma) (dataDeclCTypePragma decl)
  603         <> prettyDeclBinderHead (dataDeclContext decl) (dataDeclHead decl)
  604         <> kindPart
  605     headDoc = hsep headParts
  606     kindPart = maybe [] (\k -> ["::", prettyType k]) (dataDeclKind decl)
  607     ctorPart =
  608       case dataDeclConstructors decl of
  609         [] -> []
  610         ctors -> ["=", hsep (punctuate " |" (map prettyDataCon ctors))]

  619     headParts =
  620       ["type data"]
  621         <> prettyDeclBinderHead (dataDeclContext decl) (dataDeclHead decl)
  622         <> kindPart

  656 derivingParts :: [DerivingClause] -> [Doc ann]

  659 prettyGadtConBlock :: [DataConDecl] -> [DerivingClause] -> Doc ann
  660 prettyGadtConBlock ctors derivingClauses =
  661   "where"
  662     <> nest 2 (hardline <> vsep (map prettyDataCon ctors <> derivingParts derivingClauses))

  664 derivingPart :: DerivingClause -> [Doc ann]
  665 derivingPart (DerivingClause strategy classes) =
  666   ["deriving"] <> strategyPart strategy <> classesPart classes <> viaPart strategy
  667   where
  668     strategyPart Nothing = []
  669     strategyPart (Just (DerivingVia _)) = []
  670     strategyPart (Just strategy') = [prettyDerivingStrategy strategy']
  671 
  672     classesPart (Left name) = [prettyName name]
  673     classesPart (Right []) = ["()"]
  674     classesPart (Right classTypes) = [parens (hsep (punctuate comma (map prettyType classTypes)))]
  675 
  676     viaPart (Just (DerivingVia ty)) = ["via", prettyType ty]
  677     viaPart _ = []

  679 prettyDeclBinderHead :: [Type] -> BinderHead UnqualifiedName -> [Doc ann]
  680 prettyDeclBinderHead constraints head' =
  681   contextPrefix constraints
  682     <> prettyDeclHeadNameAndParams head'
  683   where
  684     prettyDeclHeadNameAndParams binderHead =
  685       case binderHead of
  686         PrefixBinderHead name params ->
  687           [prettyConstructorUName name] <> map prettyTyVarBinder params
  688         InfixBinderHead lhs name rhs tailPrms ->
  689           let infixHead = prettyTyVarBinder lhs <+> prettyInfixOp name <+> prettyTyVarBinder rhs
  690            in case tailPrms of
  691                 [] -> [infixHead]
  692                 _ -> parens infixHead : map prettyTyVarBinder tailPrms

  694 prettyTyVarBinder :: TyVarBinder -> Doc ann
  695 prettyTyVarBinder binder =
  696   visibleDoc
  697   where
  698     coreDoc =
  699       case (tyVarBinderSpecificity binder, tyVarBinderKind binder) of
  700         (TyVarBInferred, Nothing) -> braces (pretty (tyVarBinderName binder))
  701         (TyVarBInferred, Just kind) -> braces (pretty (tyVarBinderName binder) <+> "::" <+> prettyType kind)
  702         (TyVarBSpecified, Nothing) -> pretty (tyVarBinderName binder)
  703         (TyVarBSpecified, Just kind) -> parens (pretty (tyVarBinderName binder) <+> "::" <+> prettyType kind)
  704     visibleDoc =
  705       case tyVarBinderVisibility binder of
  706         TyVarBVisible -> coreDoc
  707         TyVarBInvisible -> "@" <> coreDoc

  712     [] -> []

  742 prettyDataCon :: DataConDecl -> Doc ann
  743 prettyDataCon ctor =
  744   case ctor of
  745     DataConAnn _ inner -> prettyDataCon inner
  746     PrefixCon forallVars constraints name fields ->
  747       hsep (dataConQualifierPrefix forallVars constraints <> [prettyConstructorUName name] <> map prettyBangType fields)
  748     InfixCon forallVars constraints lhs op rhs ->
  749       hsep
  750         ( dataConQualifierPrefix forallVars constraints
  751             <> [prettyBangType lhs, prettyInfixOp op, prettyBangType rhs]
  752         )
  753     RecordCon forallVars constraints name fields ->
  754       hsep (dataConQualifierPrefix forallVars constraints <> [prettyConstructorUName name])
  755         <+> braces (prettyRecordFields fields)
  756     GadtCon foralls constraints names body ->
  757       prettyGadtCon foralls constraints names body
  758     TupleCon forallVars constraints Boxed fields ->
  759       hsep (dataConQualifierPrefix forallVars constraints)
  760         <+> parens (hsep (punctuate comma (map prettyBangType fields)))
  761     TupleCon forallVars constraints Unboxed fields ->
  762       hsep (dataConQualifierPrefix forallVars constraints)
  763         <+> "(#"
  764         <+> hsep (punctuate comma (map prettyBangType fields))
  765         <+> "#)"
  766     UnboxedSumCon forallVars constraints pos arity field ->
  767       hsep (dataConQualifierPrefix forallVars constraints)
  768         <+> "(#"
  769         <+> hsep (replicate (pos - 1) "|" <> [prettyBangType field] <> replicate (arity - pos) "|")
  770         <+> "#)"
  771     ListCon forallVars constraints ->
  772       hsep (dataConQualifierPrefix forallVars constraints <> ["[]"])

  774 prettyGadtCon :: [ForallTelescope] -> [Type] -> [UnqualifiedName] -> GadtBody -> Doc ann
  775 prettyGadtCon forallBinders constraints names body =
  776   hsep
  777     ( [hsep (punctuate comma (map prettyConstructorUName names)), "::"]
  778         <> forallPart
  779         <> contextPart
  780         <> [prettyGadtBody body]
  781     )
  782   where
  783     forallPart = map prettyForallTelescope forallBinders
  784     contextPart
  785       | null constraints = []
  786       | otherwise = [prettyContext constraints, "=>"]

  788 prettyGadtBody :: GadtBody -> Doc ann
  789 prettyGadtBody body =
  790   case body of
  791     GadtPrefixBody args resultTy ->
  792       case args of
  793         [] -> prettyType resultTy
  794         _ ->
  795           hsep $
  796             concatMap (\(bt, ak) -> [prettyBangType bt, prettyArrowKind ak]) args
  797               ++ [prettyType resultTy]
  798     GadtRecordBody fields resultTy ->
  799       braces (prettyRecordFields fields) <+> "->" <+> prettyType resultTy

  801 prettyRecordFields :: [FieldDecl] -> Doc ann

  805         comma
  806         [ hsep $

  812         | fld <- fields

 1141 -- | Pretty-print an expression. The AST is assumed to already have EParen
 1142 -- nodes in the correct positions (inserted by 'addExprParens').
 1143 --
 1144 -- >>> let expr = EApp (EVar "f") (EParen (EInfix (EVar "x") "+" (EVar "y"))) in (renderDoc (prettyExpr expr), show (shorthand expr))
 1145 -- ("f\n  (x\n   + y)","EApp (EVar \"f\") (EParen (EInfix (EVar \"x\") \"+\" (EVar \"y\")))")
 1146 prettyExpr :: Expr -> Doc ann
 1147 prettyExpr expr =
 1148   case expr of
 1149     EApp {} -> prettyApp expr
 1150     ETypeApp fn ty ->
 1151       nest 2 (prettyExpr fn) <> nest 1 (hardline <> " " <> prettyTypeAppArg ty)
 1152     EVar name -> prettyName name
 1153     ETypeSyntax TypeSyntaxExplicitNamespace ty -> "type" <+> prettyType ty
 1154     ETypeSyntax TypeSyntaxInTerm ty -> prettyType ty
 1155     EInt _ _ repr -> pretty repr
 1156     EFloat _ _ repr -> pretty repr
 1157     EChar _ repr -> pretty repr
 1158     ECharHash _ repr -> pretty repr
 1159     EString _ repr -> prettyRawText repr
 1160     EStringHash _ repr -> prettyRawText repr
 1161     EOverloadedLabel _ raw -> pretty (" " <> raw)
 1162     EQuasiQuote quoter body -> prettyQuasiQuote quoter body

 1214       prettyExpr base <+> braces (hsep (punctuate comma (map prettyBinding fields)))

 1219     ETypeSig inner ty ->

 1222       ECase scrutinee alts ->

 1224       -- ESectionR with a '#'-starting op renders as "(# ...)" which the lexer

 1425   case cmd of
 1426     CmdArrApp lhs HsFirstOrderApp rhs ->
 1427       prettyExpr lhs <+> "-<" <+> prettyExpr rhs
 1428     CmdArrApp lhs HsHigherOrderApp rhs ->
 1429       prettyExpr lhs <+> "-<<" <+> prettyExpr rhs
 1430     CmdInfix l op r ->
 1431       nest 2 (prettyCmd l) <> nest 1 (hardline <> " " <> prettyNameInfixOp op <+> prettyCmd r)

 1433       "do" <> nest 2 (prettyDoLayout prettyCmdStmt stmts)

 1437       "case" <+> prettyExpr scrut <+> "of" <> prettyCaseLayout (map prettyCmdCaseAlt alts)

 1443       "\\" <+> hsep (map prettyPattern pats) <+> "->" <+> prettyCmd body
